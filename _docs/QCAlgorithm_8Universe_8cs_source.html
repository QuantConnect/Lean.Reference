60  /// <summary>
61  /// Invoked at the end of every time step. This allows the algorithm
62  /// to process events before advancing to the next time step.
63  /// </summary>
64  [DocumentationAttribute(HandlingData)]
65  public void OnEndOfTimeStep()
66  {
67  // rewrite securities w/ derivatives to be in raw mode
68  lock (_pendingUniverseAdditionsLock)
69  {
70  if (!_pendingUniverseAdditions && _pendingUserDefinedUniverseSecurityAdditions.Count == 0)
71  {
72  // no point in looping through everything if there's no pending changes
73  return;
74  }
75 
76  var requiredHistoryRequests = new Dictionary<Security, Resolution>();
77 
78  foreach (var security in Securities.Select(kvp => kvp.Value).Union(
79  _pendingUserDefinedUniverseSecurityAdditions.Select(x => x.Security)))
80  {
81  // check for any derivative securities and mark the underlying as raw
82  if (Securities.Any(skvp => skvp.Key.SecurityType != SecurityType.Base && skvp.Key.HasUnderlyingSymbol(security.Symbol)))
83  {
84  // set data mode raw and default volatility model
85  ConfigureUnderlyingSecurity(security);
86  }
87 
88  var configs = SubscriptionManager.SubscriptionDataConfigService
89  .GetSubscriptionDataConfigs(security.Symbol);
90  if (security.Symbol.HasUnderlying && security.Symbol.SecurityType != SecurityType.Base)
91  {
92  Security underlyingSecurity;
93  var underlyingSymbol = security.Symbol.Underlying;
94  var resolution = configs.GetHighestResolution();
95 
96  // create the underlying security object if it doesn't already exist
97  if (!Securities.TryGetValue(underlyingSymbol, out underlyingSecurity))
98  {
99  underlyingSecurity = AddSecurity(underlyingSymbol.SecurityType,
100  underlyingSymbol.Value,
101  resolution,
102  underlyingSymbol.ID.Market,
103  configs.IsFillForward(),
104  Security.NullLeverage,
105  configs.IsExtendedMarketHours(),
106  dataNormalizationMode: DataNormalizationMode.Raw);
107  }
108 
109  // set data mode raw and default volatility model
110  ConfigureUnderlyingSecurity(underlyingSecurity);
111 
112  if (LiveMode && underlyingSecurity.GetLastData() == null)
113  {
114  if (requiredHistoryRequests.ContainsKey(underlyingSecurity))
115  {
116  // lets request the higher resolution
117  var currentResolutionRequest = requiredHistoryRequests[underlyingSecurity];
118  if (currentResolutionRequest != Resolution.Minute // Can not be less than Minute
119  && resolution < currentResolutionRequest)
120  {
121  requiredHistoryRequests[underlyingSecurity] = (Resolution)Math.Max((int)resolution, (int)Resolution.Minute);
122  }
123  }
124  else
125  {
126  requiredHistoryRequests.Add(underlyingSecurity, (Resolution)Math.Max((int)resolution, (int)Resolution.Minute));
127  }
128  }
129  // set the underlying security on the derivative -- we do this in two places since it's possible
130  // to do AddOptionContract w/out the underlying already added and normalized properly
131  var derivative = security as IDerivativeSecurity;
132  if (derivative != null)
133  {
134  derivative.Underlying = underlyingSecurity;
135  }
136  }
137  }
138 
139  if (!requiredHistoryRequests.IsNullOrEmpty())
140  {
141  // Create requests
142  var historyRequests = Enumerable.Empty<HistoryRequest>();
143  foreach (var byResolution in requiredHistoryRequests.GroupBy(x => x.Value))
144  {
145  historyRequests = historyRequests.Concat(
146  CreateBarCountHistoryRequests(byResolution.Select(x => x.Key.Symbol), 3, byResolution.Key));
147  }
148  // Request data
149  var historicLastData = History(historyRequests);
150  historicLastData.PushThrough(x =>
151  {
152  var security = requiredHistoryRequests.Keys.FirstOrDefault(y => y.Symbol == x.Symbol);
153  security?.Cache.AddData(x);
154  });
155  }
156 
157  // add subscriptionDataConfig to their respective user defined universes
158  foreach (var userDefinedUniverseAddition in _pendingUserDefinedUniverseSecurityAdditions)
159  {
160  foreach (var subscriptionDataConfig in userDefinedUniverseAddition.SubscriptionDataConfigs)
161  {
162  userDefinedUniverseAddition.Universe.Add(subscriptionDataConfig);
163  }
164  }
165 
166  // finally add any pending universes, this will make them available to the data feed
167  // The universe will be added at the end of time step, same as the AddData user defined universes.
168  // This is required to be independent of the start and end date set during initialize
169  UniverseManager.ProcessChanges();
170 
171  _pendingUniverseAdditions = false;
172  _pendingUserDefinedUniverseSecurityAdditions.Clear();
173  }
174 
175  if (!_rawNormalizationWarningSymbols.IsNullOrEmpty())
176  {
177  // Log our securities being set to raw price mode
178  Debug($"Warning: The following securities were set to raw price normalization mode to work with options: " +
179  $"{string.Join(", ", _rawNormalizationWarningSymbols.Take(_rawNormalizationWarningSymbolsMaxCount).Select(x => x.Value))}...");
180 
181  // Set our warning list to null to stop emitting these warnings after its done once
182  _rawNormalizationWarningSymbols = null;
183  }
184  }

505  /// <summary>
506  /// Adds a new universe that creates options of the security by monitoring any changes in the Universe the provided security is in.
507  /// Additionally, a filter can be applied to the options generated when the universe of the security changes.
508  /// </summary>
509  /// <param name="underlyingSymbol">Underlying Symbol to add as an option. For Futures, the option chain constructed will be per-contract, as long as a canonical Symbol is provided.</param>
510  /// <param name="optionFilter">User-defined filter used to select the options we want out of the option chain provided.</param>
511  /// <exception cref="InvalidOperationException">The underlying Symbol's universe is not found.</exception>
512  [DocumentationAttribute(Universes)]
513  public void AddUniverseOptions(Symbol underlyingSymbol, Func<OptionFilterUniverse, OptionFilterUniverse> optionFilter)
514  {
515  // We need to load the universe associated with the provided Symbol and provide that universe to the option filter universe.
516  // The option filter universe will subscribe to any changes in the universe of the underlying Symbol,
517  // ensuring that we load the option chain for every asset found in the underlying's Universe.
518  Universe universe;
519  if (!UniverseManager.TryGetValue(underlyingSymbol, out universe))
520  {
521  underlyingSymbol = AddSecurity(underlyingSymbol).Symbol;
522 
523  // Recheck again, we should have a universe addition pending for the provided Symbol
524  if (!UniverseManager.TryGetValue(underlyingSymbol, out universe))
525  {
526  // Should never happen, but it could be that the subscription
527  // created with AddSecurity is not aligned with the Symbol we're using.
528  throw new InvalidOperationException($"Universe not found for underlying Symbol: {underlyingSymbol}.");
529  }
530  }
531 
532  // Allow all option contracts through without filtering if we're provided a null filter.
533  AddUniverseOptions(universe, optionFilter ?? (_ => _));
534  }

551  /// <param name="security">The security to add</param>
552  /// <param name="configurations">The <see cref="SubscriptionDataConfig"/> instances we want to add</param>
553  private Security AddToUserDefinedUniverse(
554  Security security,
555  List<SubscriptionDataConfig> configurations)
556  {
557  var subscription = configurations.First();
558  // if we are adding a non-internal security which already has an internal feed, we remove it first
559  if (Securities.TryGetValue(security.Symbol, out var existingSecurity))
560  {
561  if (!subscription.IsInternalFeed && existingSecurity.IsInternalFeed())
562  {
563  var securityUniverse = UniverseManager.Select(x => x.Value).OfType<UserDefinedUniverse>().FirstOrDefault(x => x.Members.ContainsKey(security.Symbol));
564  securityUniverse?.Remove(security.Symbol);
565 
566  Securities.Remove(security.Symbol);
567  Securities.Add(security);
568  }
569  else
570  {
571  var isTradable = security.IsTradable;
572  // We will reuse existing so we return it to the user.
573  // We will use the IsTradable flag of the new security, since existing could of been set to false when removed
574  security = existingSecurity;
575  security.IsTradable = isTradable;
576  }
577  }
578  else
579  {
580  Securities.Add(security);
581  }
582 
583  // add this security to the user defined universe
584  Universe universe;
585  var universeSymbol = UserDefinedUniverse.CreateSymbol(security.Type, security.Symbol.ID.Market);
586  if (!UniverseManager.TryGetValue(universeSymbol, out universe))
587  {
588  if (universe == null)
589  {
590  // create a new universe, these subscription settings don't currently get used
591  // since universe selection proper is never invoked on this type of universe
592  var uconfig = new SubscriptionDataConfig(subscription, symbol: universeSymbol, isInternalFeed: true, fillForward: false,
593  exchangeTimeZone: DateTimeZone.Utc,
594  dataTimeZone: DateTimeZone.Utc);
595 
596  // this is the universe symbol, has no real entry in the mhdb, will default to market and security type
597  // set entry in market hours database for the universe subscription to match the config
598  var symbolString = MarketHoursDatabase.GetDatabaseSymbolKey(uconfig.Symbol);
599  MarketHoursDatabase.SetEntry(uconfig.Market, symbolString, uconfig.SecurityType,
600  SecurityExchangeHours.AlwaysOpen(uconfig.ExchangeTimeZone), uconfig.DataTimeZone);
601 
602  universe = new UserDefinedUniverse(uconfig,
603  new UniverseSettings(
604  subscription.Resolution,
605  security.Leverage,
606  subscription.FillDataForward,
607  subscription.ExtendedMarketHours,
608  TimeSpan.Zero),
609  QuantConnect.Time.MaxTimeSpan,
610  new List<Symbol>());
611 
612  AddUniverse(universe);
613  }
614  }
615 
616  var userDefinedUniverse = universe as UserDefinedUniverse;
617  if (userDefinedUniverse != null)
618  {
619  lock (_pendingUniverseAdditionsLock)
620  {
621  _pendingUserDefinedUniverseSecurityAdditions.Add(
622  new UserDefinedUniverseAddition(userDefinedUniverse, configurations, security));
623  }
624  }
625  else
626  {
627  // should never happen, someone would need to add a non-user defined universe with this symbol
628  throw new Exception($"Expected universe with symbol '{universeSymbol.Value}' to be of type {nameof(UserDefinedUniverse)} but was {universe.GetType().Name}.");
629  }
630 
631  return security;
632  }

634  /// <summary>
635  /// Configures the security to be in raw data mode and ensures that a reasonable default volatility model is supplied
636  /// </summary>
637  /// <param name="security">The underlying security</param>
638  private void ConfigureUnderlyingSecurity(Security security)
639  {
640  // force underlying securities to be raw data mode
641  var configs = SubscriptionManager.SubscriptionDataConfigService
642  .GetSubscriptionDataConfigs(security.Symbol);
643  if (configs.DataNormalizationMode() != DataNormalizationMode.Raw)
644  {
645  // Add this symbol to our set of raw normalization warning symbols to alert the user at the end
646  // Set a hard limit to avoid growing this collection unnecessarily large
647  if (_rawNormalizationWarningSymbols != null && _rawNormalizationWarningSymbols.Count <= _rawNormalizationWarningSymbolsMaxCount)
648  {
649  _rawNormalizationWarningSymbols.Add(security.Symbol);
650  }
651 
652  configs.SetDataNormalizationMode(DataNormalizationMode.Raw);
653  // For backward compatibility we need to refresh the security DataNormalizationMode Property
654  security.RefreshDataNormalizationModeProperty();
655  }

658  /// <summary>
659  /// Helper method to create the configuration of a custom universe
660  /// </summary>
661  private SubscriptionDataConfig GetCustomUniverseConfiguration(Type dataType, string name, string market, Resolution? resolution = null)
662  {
663  if (dataType == typeof(CoarseFundamental) || dataType == typeof(FineFundamental))
664  {
665  dataType = typeof(FundamentalUniverse);
666  }
667 
668  if (!TryGetUniverseSymbol(dataType, market, out var universeSymbol))
669  {
670  market ??= Market.USA;
671  if (string.IsNullOrEmpty(name))
672  {
673  name = $"{dataType.Name}-{market}-{Guid.NewGuid()}";
674  }
675  // same as 'AddData<>' 'T' type will be treated as custom/base data type with always open market hours
676  universeSymbol = QuantConnect.Symbol.Create(name, SecurityType.Base, market, baseDataType: dataType);
677  }
678  var marketHoursDbEntry = MarketHoursDatabase.GetEntry(universeSymbol, new[] { dataType });
679  var dataTimeZone = marketHoursDbEntry.DataTimeZone;
680  var exchangeTimeZone = marketHoursDbEntry.ExchangeHours.TimeZone;
681  return new SubscriptionDataConfig(dataType, universeSymbol, resolution ?? Resolution.Daily, dataTimeZone, exchangeTimeZone, false, false, true, true, isFilteredSubscription: false);
682  }
