16 using System;
17 using System.Linq;
18 using System.Drawing;
19 using Newtonsoft.Json;
20 using Newtonsoft.Json.Linq;
21 using System.Collections.Generic;

23 namespace QuantConnect.Util
24 {
25  /// <summary>
26  /// Json Converter for Series which handles special Pie Series serialization case
27  /// </summary>
28  public class SeriesJsonConverter : JsonConverter
29  {
30  /// <summary>
31  /// Write Series to Json
32  /// </summary>
33  /// <param name="writer">The Json Writer to use</param>
34  /// <param name="value">The value to written to Json</param>
35  /// <param name="serializer">The Json Serializer to use</param>
36  public override void WriteJson(JsonWriter writer, object value, JsonSerializer serializer)
37  {
38  var baseSeries = value as BaseSeries;
39  if (baseSeries == null)
40  {
41  return;
42  }
43 
44  writer.WriteStartObject();
45 
46  writer.WritePropertyName("Name");
47  writer.WriteValue(baseSeries.Name);
48  writer.WritePropertyName("Unit");
49  writer.WriteValue(baseSeries.Unit);
50  writer.WritePropertyName("Index");
51  writer.WriteValue(baseSeries.Index);
52  writer.WritePropertyName("SeriesType");
53  writer.WriteValue(baseSeries.SeriesType);
54 
55  if (baseSeries.ZIndex.HasValue)
56  {
57  writer.WritePropertyName("ZIndex");
58  writer.WriteValue(baseSeries.ZIndex.Value);
59  }
60 
61  if (baseSeries.IndexName != null)
62  {
63  writer.WritePropertyName("IndexName");
64  writer.WriteValue(baseSeries.IndexName);
65  }
66 
67  if (baseSeries.Tooltip != null)
68  {
69  writer.WritePropertyName("Tooltip");
70  writer.WriteValue(baseSeries.Tooltip);
71  }
72 
73  switch (value)
74  {
75  case Series series:
76  var values = series.Values;
77  if (series.SeriesType == SeriesType.Pie)
78  {
79  values = new List<ISeriesPoint>();
80  var dataPoint = series.ConsolidateChartPoints();
81  if (dataPoint != null)
82  {
83  values.Add(dataPoint);
84  }
85  }
86 
87  // have to add the converter we want to use, else will use default
88  serializer.Converters.Add(new ColorJsonConverter());
89 
90  writer.WritePropertyName("Values");
91  serializer.Serialize(writer, values);
92  writer.WritePropertyName("Color");
93  serializer.Serialize(writer, series.Color);
94  writer.WritePropertyName("ScatterMarkerSymbol");
95  serializer.Serialize(writer, series.ScatterMarkerSymbol);
96  break;
97 
98  default:
99  writer.WritePropertyName("Values");
100  serializer.Serialize(writer, (value as BaseSeries).Values);
101  break;
102  }
103 
104  writer.WriteEndObject();
105  }
106 
107  /// <summary>
108  /// Reads series from Json
109  /// </summary>
110  public override object ReadJson(JsonReader reader, Type objectType, object existingValue, JsonSerializer serializer)
111  {
112  var jObject = JObject.Load(reader);
113 
114  var name = jObject["Name"].Value<string>();
115  var unit = jObject["Unit"].Value<string>();
116  var index = jObject["Index"].Value<int>();
117  var seriesType = (SeriesType)jObject["SeriesType"].Value<int>();
118  var values = (JArray)jObject["Values"];
119 
120  var zindex = jObject.TryGetPropertyValue<int?>("ZIndex");
121  var indexName = jObject.TryGetPropertyValue<string>("IndexName");
122  var tooltip = jObject.TryGetPropertyValue<string>("Tooltip");
123 
124  if (seriesType == SeriesType.Candle)
125  {
126  return new CandlestickSeries()
127  {
128  Name = name,
129  Unit = unit,
130  Index = index,
131  ZIndex = zindex,
132  Tooltip = tooltip,
133  IndexName = indexName,
134  SeriesType = seriesType,
135  Values = values.ToObject<List<Candlestick>>(serializer).Where(x => x != null).Cast<ISeriesPoint>().ToList()
136  };
137  }
138 
139  var result = new Series()
140  {
141  Name = name,
142  Unit = unit,
143  Index = index,
144  ZIndex = zindex,
145  Tooltip = tooltip,
146  IndexName = indexName,
147  SeriesType = seriesType,
148  Color = jObject["Color"].ToObject<Color>(serializer),
149  ScatterMarkerSymbol = jObject["ScatterMarkerSymbol"].ToObject<ScatterMarkerSymbol>(serializer)
150  };
151 
152  if (seriesType == SeriesType.Scatter)
153  {
154  result.Values = values.ToObject<List<ScatterChartPoint>>(serializer).Where(x => x != null).Cast<ISeriesPoint>().ToList();
155  }
156  else
157  {
158  result.Values = values.ToObject<List<ChartPoint>>(serializer).Where(x => x != null).Cast<ISeriesPoint>().ToList();
159  }
160  return result;
161  }
162 
163  /// <summary>
164  /// Determine if this Converter can convert this type
165  /// </summary>
166  /// <param name="objectType">Type that we would like to convert</param>
167  /// <returns>True if <see cref="Series"/></returns>
168  public override bool CanConvert(Type objectType)
169  {
170  return typeof(BaseSeries).IsAssignableFrom(objectType);
171  }
172  }
