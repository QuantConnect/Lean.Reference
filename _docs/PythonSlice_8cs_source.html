24  /// <summary>
25  /// Provides a data structure for all of an algorithm's data at a single time step
26  /// </summary>
27  public class PythonSlice : Slice
28  {
29  private Slice _slice;
30  private static readonly PyObject _converter;
31 
32  static PythonSlice()
33  {
34  // Python Data class: Converts custom data (PythonData) into a python object'''
35  _converter = PyModule.FromString("converter",
36  "class Data(object):\n" +
37  " def __init__(self, data):\n" +
38  " self.data = data\n" +
39  " members = [attr for attr in dir(data) if not callable(attr) and not attr.startswith(\"__\")]\n" +
40  " for member in members:\n" +
41  " setattr(self, member, getattr(data, member))\n" +
42  " for kvp in data.GetStorageDictionary():\n" +
43  " name = kvp.Key.replace('-',' ').replace('.',' ').title().replace(' ', '')\n" +
44  " value = kvp.Value if isinstance(kvp.Value, float) else kvp.Value\n" +
45  " setattr(self, name, value)\n" +
46 
47  " def __str__(self):\n" +
48  " return self.data.ToString()");
49  }
50 
51  /// <summary>
52  /// Initializes a new instance of the <see cref="PythonSlice"/> class
53  /// </summary>
54  /// <param name="slice">slice object to wrap</param>
55  public PythonSlice(Slice slice)
56  : base(slice)
57  {
58  _slice = slice;
59  }
60 
61  /// <summary>
62  /// Gets the data of the specified symbol and type.
63  /// </summary>
64  /// <param name="type">The type of data we seek</param>
65  /// <param name="symbol">The specific symbol was seek</param>
66  /// <returns>The data for the requested symbol</returns>
67  public dynamic Get(PyObject type, Symbol symbol)
68  {
69  return GetImpl(type.CreateType(), _slice)[symbol];
70  }
71 
72  /// <summary>
73  /// Gets the data of the specified symbol and type.
74  /// </summary>
75  /// <param name="type">The type of data we seek</param>
76  /// <returns>The data for the requested symbol</returns>
77  public PyObject Get(PyObject type)
78  {
79  var result = GetImpl(type.CreateType(), _slice) as object;
80  using (Py.GIL())
81  {
82  return result.ToPython();
83  }
84  }

110  /// <summary>
111  /// Gets the data corresponding to the specified symbol. If the requested data
112  /// is of <see cref="MarketDataType.Tick"/>, then a <see cref="List{Tick}"/> will
113  /// be returned, otherwise, it will be the subscribed type, for example, <see cref="TradeBar"/>
114  /// or event <see cref="UnlinkedData"/> for custom data.
115  /// </summary>
116  /// <param name="symbol">The data's symbols</param>
117  /// <returns>The data for the specified symbol</returns>
118  public override dynamic this[Symbol symbol]
119  {
120  get
121  {
122  var data = _slice[symbol];
123 
124  var dynamicData = data as DynamicData;
125  if (dynamicData != null)
126  {
127  try
128  {
129  using (Py.GIL())
130  {
131  return _converter.InvokeMethod("Data", new[] { dynamicData.ToPython() });
132  }
133  }
134  catch
135  {
136  // NOP
137  }
138  }
139  return data;
140  }
141  }
142 
143  /// <summary>
144  /// Determines whether this instance contains data for the specified symbol
145  /// </summary>
146  /// <param name="symbol">The symbol we seek data for</param>
147  /// <returns>True if this instance contains data for the symbol, false otherwise</returns>
148  public override bool ContainsKey(Symbol symbol)
149  {
150  return _slice.ContainsKey(symbol);
151  }
152 
153  /// <summary>
154  /// Gets the data associated with the specified symbol
155  /// </summary>
156  /// <param name="symbol">The symbol we want data for</param>
157  /// <param name="data">The data for the specifed symbol, or null if no data was found</param>
158  /// <returns>True if data was found, false otherwise</returns>
159  public override bool TryGetValue(Symbol symbol, out dynamic data)
160  {
161  return _slice.TryGetValue(symbol, out data);
162  }
163  }
