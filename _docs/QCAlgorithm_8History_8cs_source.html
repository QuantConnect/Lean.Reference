157  /// <summary>
158  /// Gets the history requests required for provide warm up data for the algorithm
159  /// </summary>
160  /// <returns></returns>
161  [DocumentationAttribute(HistoricalData)]
162  private bool TryGetWarmupHistoryStartTime(out DateTime result)
163  {
164  result = Time;
165 
166  if (_warmupBarCount.HasValue)
167  {
168  var symbols = Securities.Keys;
169  if (symbols.Count != 0)
170  {
171  var startTimeUtc = CreateBarCountHistoryRequests(symbols, _warmupBarCount.Value, Settings.WarmupResolution)
172  .DefaultIfEmpty()
173  .Min(request => request == null ? default : request.StartTimeUtc);
174  if(startTimeUtc != default)
175  {
176  result = startTimeUtc.ConvertFromUtc(TimeZone);
177  return true;
178  }
179  }
180 
181  var defaultResolutionToUse = UniverseSettings.Resolution;
182  if (Settings.WarmupResolution.HasValue)
183  {
184  defaultResolutionToUse = Settings.WarmupResolution.Value;
185  }
186 
187  // if the algorithm has no added security, let's take a look at the universes to determine
188  // what the start date should be used. Defaulting to always open
189  result = Time - _warmupBarCount.Value * defaultResolutionToUse.ToTimeSpan();
190 
191  foreach (var universe in _pendingUniverseAdditions.Concat(UniverseManager.Values))
192  {
193  var config = universe.Configuration;
194  var resolution = universe.Configuration.Resolution;
195  if (Settings.WarmupResolution.HasValue)
196  {
197  resolution = Settings.WarmupResolution.Value;
198  }
199  var exchange = MarketHoursDatabase.GetExchangeHours(config);
200  var start = _historyRequestFactory.GetStartTimeAlgoTz(config.Symbol, _warmupBarCount.Value, resolution, exchange, config.DataTimeZone);
201  // we choose the min start
202  result = result < start ? result : start;
203  }
204  return true;
205  }
206  if (_warmupTimeSpan.HasValue)
207  {
208  result = Time - _warmupTimeSpan.Value;
209  return true;
210  }
211 
212  return false;
213  }

274  /// <summary>
275  /// Gets the historical data for the specified symbols. The exact number of bars will be returned for
276  /// each symbol. This may result in some data start earlier/later than others due to when various
277  /// exchanges are open. The symbols must exist in the Securities collection.
278  /// </summary>
279  /// <typeparam name="T">The data type of the symbols</typeparam>
280  /// <param name="symbols">The symbols to retrieve historical data for</param>
281  /// <param name="periods">The number of bars to request</param>
282  /// <param name="resolution">The resolution to request</param>
283  /// <returns>An enumerable of slice containing the requested historical data</returns>

285  public IEnumerable<DataDictionary<T>> History<T>(IEnumerable<Symbol> symbols, int periods, Resolution? resolution = null)
286  where T : IBaseData
287  {
288  var requests = symbols.Select(x =>
289  {
290  var res = GetResolution(x, resolution);
291  if (res == Resolution.Tick)
292  {
293  throw new ArgumentException("History functions that accept a 'periods' parameter can not be used with Resolution.Tick");
294  }
295 
296  var config = GetMatchingSubscription(x, typeof(T));
297  if (config == null) return null;
298 
299  var exchange = GetExchangeHours(x);
300  var start = _historyRequestFactory.GetStartTimeAlgoTz(x, periods, res, exchange, config.DataTimeZone);
301  return _historyRequestFactory.CreateHistoryRequest(config, start, Time, exchange, res);
302  });
303 
304  return History(requests.Where(x => x != null)).Get<T>().Memoize();
305  }

533  /// <summary>
534  /// Yields data to warmup a security for all it's subscribed data types
535  /// </summary>
536  /// <param name="symbol">The symbol we want to get seed data for</param>
537  /// <returns>Securities historical data</returns>
538  [DocumentationAttribute(AddingData)]
539  [DocumentationAttribute(HistoricalData)]
540  public IEnumerable<BaseData> GetLastKnownPrices(Symbol symbol)
541  {
542  if (!HistoryRequestValid(symbol) || HistoryProvider == null)
543  {
544  return Enumerable.Empty<BaseData>();
545  }
546 
547  var result = new Dictionary<TickType, BaseData>();
548  Resolution? resolution = null;
549  Func<int, bool> requestData = period =>
550  {
551  var historyRequests = CreateBarCountHistoryRequests(new[] { symbol }, period)
552  .Select(request =>
553  {
554  // For speed and memory usage, use Resolution.Minute as the minimum resolution
555  request.Resolution = (Resolution)Math.Max((int)Resolution.Minute, (int)request.Resolution);
556  // force no fill forward behavior
557  request.FillForwardResolution = null;
558 
559  resolution = request.Resolution;
560  return request;
561  })
562  // request only those tick types we didn't get the data we wanted
563  .Where(request => !result.ContainsKey(request.TickType))
564  .ToList();
565  foreach (var slice in History(historyRequests))
566  {
567  for (var i = 0; i < historyRequests.Count; i++)
568  {
569  var historyRequest = historyRequests[i];
570  var data = slice.Get(historyRequest.DataType);
571  if (data.ContainsKey(symbol))
572  {
573  // keep the last data point per tick type
574  result[historyRequest.TickType] = (BaseData)data[symbol];
575  }
576  }
577  }
578  // true when all history requests tick types have a data point
579  return historyRequests.All(request => result.ContainsKey(request.TickType));
580  };
581 
582  if (!requestData(5))
583  {
584  if (resolution.HasValue)
585  {
586  // If the first attempt to get the last know price returns null, it maybe the case of an illiquid security.
587  // We increase the look-back period for this case accordingly to the resolution to cover 3 trading days
588  var periods =
589  resolution.Value == Resolution.Daily ? 3 :
590  resolution.Value == Resolution.Hour ? 24 : 1440;
591  requestData(periods);
592  }
593  else
594  {
595  // this shouldn't happen but just in case
596  QuantConnect.Logging.Log.Error(

639  [DocumentationAttribute(HistoricalData)]
640  private IEnumerable<Slice> History(IEnumerable<HistoryRequest> requests, DateTimeZone timeZone)
641  {
642  var sentMessage = false;
643  // filter out any universe securities that may have made it this far
644  var filteredRequests = requests.Where(hr => HistoryRequestValid(hr.Symbol)).ToList();
645  for (var i = 0; i < filteredRequests.Count; i++)
646  {
647  var request = filteredRequests[i];
648  // prevent future requests
649  if (request.EndTimeUtc > UtcTime)
650  {
651  var endTimeUtc = UtcTime;
652  var startTimeUtc = request.StartTimeUtc;
653  if (request.StartTimeUtc > request.EndTimeUtc)
654  {
655  startTimeUtc = request.EndTimeUtc;
656  }
657 
658  filteredRequests[i] = new HistoryRequest(startTimeUtc, endTimeUtc,
659  request.DataType, request.Symbol, request.Resolution, request.ExchangeHours,
660  request.DataTimeZone, request.FillForwardResolution, request.IncludeExtendedMarketHours,
661  request.IsCustomData, request.DataNormalizationMode, request.TickType, request.DataMappingMode,
662  request.ContractDepthOffset);
663 
664  if (!sentMessage)
665  {
666  sentMessage = true;
667  Debug("Request for future history modified to end now.");
668  }
669  }
670  }
671 
672  // filter out future data to prevent look ahead bias
673  return ((IAlgorithm)this).HistoryProvider.GetHistory(filteredRequests, timeZone);
674  }

760  private IEnumerable<SubscriptionDataConfig> GetMatchingSubscriptions(Symbol symbol, Type type, Resolution? resolution = null)
761  {
762  var matchingSubscriptions = SubscriptionManager.SubscriptionDataConfigService
763  // we add internal subscription so that history requests are covered, this allows us to warm them up too
764  .GetSubscriptionDataConfigs(symbol, includeInternalConfigs:true)
765  // find all subscriptions matching the requested type with a higher resolution than requested
766  .OrderByDescending(s => s.Resolution)
767  // lets make sure to respect the order of the data types
768  .ThenByDescending(config => GetTickTypeOrder(config.SecurityType, config.TickType))
769  .Where(s => SubscriptionDataConfigTypeFilter(type, s.Type));
770 
771  var internalConfig = new List<SubscriptionDataConfig>();
772  var userConfig = new List<SubscriptionDataConfig>();
773  foreach (var config in matchingSubscriptions)
774  {
775  if (config.IsInternalFeed)
776  {
777  internalConfig.Add(config);
778  }
779  else
780  {
781  userConfig.Add(config);
782  }
783  }
784 
785  // if we have any user defined subscription configuration we use it, else we use internal ones if any
786  List<SubscriptionDataConfig> configs = null;
787  if(userConfig.Count != 0)
788  {
789  configs = userConfig;
790  }
791  else if (internalConfig.Count != 0)
792  {
793  configs = internalConfig;
794  }
795 
796  // we use the subscription manager registered configurations here, we can not rely on the Securities collection
797  // since this might be called when creating a security and warming it up
798  if (configs != null && configs.Count != 0)
799  {
800  if (resolution.HasValue
801  && (resolution == Resolution.Daily || resolution == Resolution.Hour)
802  && symbol.SecurityType == SecurityType.Equity)
803  {
804  // for Daily and Hour resolution, for equities, we have to
805  // filter out any existing subscriptions that could be of Quote type
806  // This could happen if they were Resolution.Minute/Second/Tick
807  return configs.Where(s => s.TickType != TickType.Quote);
808  }
809 
810  return configs;
811  }
812  else
813  {
814  var entry = MarketHoursDatabase.GetEntry(symbol, new []{ type });
815  resolution = GetResolution(symbol, resolution);
816 
817  return SubscriptionManager
818  .LookupSubscriptionConfigDataTypes(symbol.SecurityType, resolution.Value, symbol.IsCanonical())
819  .Where(tuple => SubscriptionDataConfigTypeFilter(type, tuple.Item1))
820  .Select(x => new SubscriptionDataConfig(
821  x.Item1,
822  symbol,
823  resolution.Value,
824  entry.DataTimeZone,
825  entry.ExchangeHours.TimeZone,
826  UniverseSettings.FillForward,
827  UniverseSettings.ExtendedMarketHours,
828  true,
829  false,
830  x.Item2,
831  true,
832  UniverseSettings.GetUniverseNormalizationModeOrDefault(symbol.SecurityType)));
833  }
834  }

867  private Resolution GetResolution(Symbol symbol, Resolution? resolution)
868  {
869  Security security;
870  if (Securities.TryGetValue(symbol, out security))
871  {
872  if (resolution != null)
873  {
874  return resolution.Value;
875  }
876 
877  Resolution? result = null;
878  var hasNonInternal = false;
879  foreach (var config in SubscriptionManager.SubscriptionDataConfigService
880  .GetSubscriptionDataConfigs(symbol, includeInternalConfigs: true)
881  // we process non internal configs first
882  .OrderBy(config => config.IsInternalFeed ? 1 : 0))
883  {
884  if (!config.IsInternalFeed || !hasNonInternal)
885  {
886  // once we find a non internal config we ignore internals
887  hasNonInternal |= !config.IsInternalFeed;
888  if (!result.HasValue || config.Resolution < result)
889  {
890  result = config.Resolution;
891  }
892  }
893  }
894 
895  return result ?? UniverseSettings.Resolution;
896  }
897  else
898  {
899  return resolution ?? UniverseSettings.Resolution;
900  }
901  }
