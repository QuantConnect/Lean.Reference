16 using System;
17 using QuantConnect.Securities;
18 
19 namespace QuantConnect.Algorithm.Framework.Alphas.Analysis
20 {
21  /// <summary>
22  /// Contains security values required by insight analysis components
23  /// </summary>
24  /// <remarks>
25  /// The main purpose here is providing an ACL against the algorithm to remove the dependencies
26  /// </remarks>
27  public class SecurityValues
28  {
29  /// <summary>
30  /// Gets the symbol these values are for
31  /// </summary>
32  public Symbol Symbol { get; }
33 
34  /// <summary>
35  /// Gets the utc time these values were sampled
36  /// </summary>
37  public DateTime TimeUtc { get; }
38 
39  /// <summary>
40  /// Gets the security price as of <see cref="TimeUtc"/>
41  /// </summary>
42  public decimal Price { get; }
43 
44  /// <summary>
45  /// Gets the security's volatility as of <see cref="TimeUtc"/>
46  /// </summary>
47  public decimal Volatility { get; }
48 
49  /// <summary>
50  /// Gets the volume traded in the security during this time step
51  /// </summary>
52  public decimal Volume { get; }
53 
54  /// <summary>
55  /// Gets the conversion rate for the quote currency of the security
56  /// </summary>
57  public decimal QuoteCurrencyConversionRate { get; }
58 
59  /// <summary>
60  /// Gets the exchange hours for the security
61  /// </summary>
62  public SecurityExchangeHours ExchangeHours { get; }
63 
64  /// <summary>
65  /// Initializes a new instance of the <see cref="SecurityValues"/> class
66  /// </summary>
67  /// <param name="symbol">The symbol of the security</param>
68  /// <param name="timeUtc">The time these values were sampled</param>
69  /// <param name="exchangeHours">The security's exchange hours</param>
70  /// <param name="price">The security price</param>
71  /// <param name="volatility">The security's volatility</param>
72  /// <param name="volume">The volume traded at this time step</param>
73  /// <param name="quoteCurrencyConversionRate">The conversion rate for the quote currency of the security</param>
74  public SecurityValues(Symbol symbol, DateTime timeUtc, SecurityExchangeHours exchangeHours, decimal price, decimal volatility, decimal volume, decimal quoteCurrencyConversionRate)
75  {
76  Symbol = symbol;
77  Price = price;
78  Volume = volume;
79  TimeUtc = timeUtc;
80  Volatility = volatility;
81  ExchangeHours = exchangeHours;
82  QuoteCurrencyConversionRate = quoteCurrencyConversionRate;
83  }
84 
85  /// <summary>
86  /// Gets the security value corresponding to the specified insight type
87  /// </summary>
88  /// <param name="type">The insight type</param>
89  /// <returns>The security value for the specified insight type</returns>
90  public decimal Get(InsightType type)
91  {
92  switch (type)
93  {
94  case InsightType.Price:
95  return Price;
96 
97  case InsightType.Volatility:
98  return Volatility;
99 
100  default:
101  throw new ArgumentOutOfRangeException(nameof(type), type, null);
102  }
103  }
104  }
105 }
