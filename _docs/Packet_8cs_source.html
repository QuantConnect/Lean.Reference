25  public class Packet

27  /// <summary>

32 
33  /// <summary>
34  /// User unique specific channel endpoint to send the packets
35  /// </summary>
36  [JsonProperty(PropertyName = "sChannel")]
37  public virtual string Channel { get; set; } = "";
38 
39  /// <summary>
40  /// Initialize the base class and setup the packet type.
41  /// </summary>
42  /// <param name="type">PacketType for the class.</param>
43  public Packet(PacketType type)
44  {
45  Channel = "";
46  Type = type;
47  }
48  }
