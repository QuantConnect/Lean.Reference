225  /// <summary>
226  /// Adds the specified order to be processed
227  /// </summary>
228  /// <param name="request">The order to be processed</param>
229  public OrderTicket Process(OrderRequest request)
230  {
231  if (_algorithm.LiveMode)
232  {
233  Log.Trace("BrokerageTransactionHandler.Process(): " + request);
234 
235  _algorithm.Portfolio.LogMarginInformation(request);
236  }
237 
238  switch (request.OrderRequestType)
239  {
240  case OrderRequestType.Submit:
241  return AddOrder((SubmitOrderRequest)request);
242 
243  case OrderRequestType.Update:
244  return UpdateOrder((UpdateOrderRequest)request);
245 
246  case OrderRequestType.Cancel:
247  return CancelOrder((CancelOrderRequest)request);
248 
249  default:
250  throw new ArgumentOutOfRangeException();
251  }
252  }

254  /// <summary>
255  /// Add an order to collection and return the unique order id or negative if an error.
256  /// </summary>
257  /// <param name="request">A request detailing the order to be submitted</param>
258  /// <returns>New unique, increasing orderid</returns>
259  public OrderTicket AddOrder(SubmitOrderRequest request)
260  {
261  var response = !_algorithm.IsWarmingUp
262  ? OrderResponse.Success(request)
263  : OrderResponse.WarmingUp(request);
264 
265  var shortable = true;
266  if (request.Quantity < 0)
267  {
268  shortable = _algorithm.Shortable(request.Symbol, request.Quantity);
269  }
270 
271  if (!shortable)
272  {
273  var message = GetShortableErrorMessage(request.Symbol, request.Quantity);
274  if (_algorithm.LiveMode)
275  {
276  // in live mode we send a warning but we wont block the order being sent to the brokerage
277  _algorithm.Debug($"Warning: {message}");
278  }
279  else
280  {
281  response = OrderResponse.Error(request, OrderResponseErrorCode.ExceedsShortableQuantity, message);
282  }
283  }
284 
285  request.SetResponse(response);
286  var ticket = new OrderTicket(_algorithm.Transactions, request);
287 
288  Interlocked.Increment(ref _totalOrderCount);
289  // send the order to be processed after creating the ticket
290  if (response.IsSuccess)
291  {
292  _openOrderTickets.TryAdd(ticket.OrderId, ticket);
293  _completeOrderTickets.TryAdd(ticket.OrderId, ticket);
294  _orderRequestQueue.Add(request);
295 
296  // wait for the transaction handler to set the order reference into the new order ticket,
297  // so we can ensure the order has already been added to the open orders,
298  // before returning the ticket to the algorithm.
299  WaitForOrderSubmission(ticket);
300  }
301  else
302  {
303  // add it to the orders collection for recall later
304  var order = Order.CreateOrder(request);
305  var orderTag = response.ErrorCode == OrderResponseErrorCode.AlgorithmWarmingUp
306  ? "Algorithm warming up."
307  : response.ErrorMessage;
308 
309  // ensure the order is tagged with a currency
310  var security = _algorithm.Securities[order.Symbol];
311  order.PriceCurrency = security.SymbolProperties.QuoteCurrency;
312 
313  order.Status = OrderStatus.Invalid;
314  order.Tag = orderTag;
315  ticket.SetOrder(order);
316  _completeOrderTickets.TryAdd(ticket.OrderId, ticket);
317  _completeOrders.TryAdd(order.Id, order);
318 
319  HandleOrderEvent(new OrderEvent(order,
320  _algorithm.UtcTime,
321  OrderFee.Zero,
322  orderTag));
323  }
324  return ticket;
325  }

327  /// <summary>
328  /// Wait for the order to be handled by the <see cref="_processingThread"/>
329  /// </summary>
330  /// <param name="ticket">The <see cref="OrderTicket"/> expecting to be submitted</param>
331  protected virtual void WaitForOrderSubmission(OrderTicket ticket)
332  {
333  var orderSetTimeout = Time.OneSecond;
334  if (!ticket.OrderSet.WaitOne(orderSetTimeout))
335  {
336  Log.Error("BrokerageTransactionHandler.WaitForOrderSubmission(): " +
337  $"The order request (Id={ticket.OrderId}) was not submitted within {orderSetTimeout.TotalSeconds} second(s).");
338  }
339  }

341  /// <summary>
342  /// Update an order yet to be filled such as stop or limit orders.
343  /// </summary>
344  /// <param name="request">Request detailing how the order should be updated</param>
345  /// <remarks>Does not apply if the order is already fully filled</remarks>
346  public OrderTicket UpdateOrder(UpdateOrderRequest request)
347  {
348  OrderTicket ticket;
349  if (!_completeOrderTickets.TryGetValue(request.OrderId, out ticket))
350  {
351  return OrderTicket.InvalidUpdateOrderId(_algorithm.Transactions, request);
352  }
353 
354  ticket.AddUpdateRequest(request);
355 
356  try
357  {
358  //Update the order from the behaviour
359  var order = GetOrderByIdInternal(request.OrderId);
360  var orderQuantity = request.Quantity ?? ticket.Quantity;
361 
362  var shortable = true;
363  if (order?.Direction == OrderDirection.Sell || orderQuantity < 0)
364  {
365  shortable = _algorithm.Shortable(ticket.Symbol, orderQuantity, order.Id);
366 
367  if(_algorithm.LiveMode && !shortable)
368  {
369  // let's override and just send warning
370  shortable = true;
371 
372  _algorithm.Debug($"Warning: {GetShortableErrorMessage(ticket.Symbol, ticket.Quantity)}");
373  }
374  }
375 
376  if (order == null)
377  {
378  // can't update an order that doesn't exist!
379  Log.Error("BrokerageTransactionHandler.Update(): Cannot update a null order");
380  request.SetResponse(OrderResponse.UnableToFindOrder(request));
381  }
382  else if (order.Status == OrderStatus.New)
383  {
384  // can't update a pending submit order
385  Log.Error("BrokerageTransactionHandler.Update(): Cannot update a pending submit order with status " + order.Status);
386  request.SetResponse(OrderResponse.InvalidNewStatus(request, order));
387  }
388  else if (order.Status.IsClosed() && !request.IsAllowedForClosedOrder())
389  {
390  // can't update a completed order
391  Log.Error("BrokerageTransactionHandler.Update(): Cannot update closed order with status " + order.Status);
392  request.SetResponse(OrderResponse.InvalidStatus(request, order));
393  }
394  else if (request.Quantity.HasValue && request.Quantity.Value == 0)
395  {
396  request.SetResponse(OrderResponse.ZeroQuantity(request));
397  }
398  else if (_algorithm.IsWarmingUp)
399  {
400  request.SetResponse(OrderResponse.WarmingUp(request));
401  }
402  else if (!shortable)
403  {
404  var shortableResponse = OrderResponse.Error(request, OrderResponseErrorCode.ExceedsShortableQuantity,
405  GetShortableErrorMessage(ticket.Symbol, ticket.Quantity));
406 
407  request.SetResponse(shortableResponse);
408  }
409  else
410  {
411  request.SetResponse(OrderResponse.Success(request), OrderRequestStatus.Processing);
412  _orderRequestQueue.Add(request);
413  }
414  }
415  catch (Exception err)
416  {
417  Log.Error(err);
418  request.SetResponse(OrderResponse.Error(request, OrderResponseErrorCode.ProcessingError, err.Message));
419  }
420 
421  return ticket;
422  }

424  /// <summary>
425  /// Remove this order from outstanding queue: user is requesting a cancel.
426  /// </summary>
427  /// <param name="request">Request containing the specific order id to remove</param>
428  public OrderTicket CancelOrder(CancelOrderRequest request)
429  {
430  OrderTicket ticket;
431  if (!_completeOrderTickets.TryGetValue(request.OrderId, out ticket))
432  {
433  Log.Error("BrokerageTransactionHandler.CancelOrder(): Unable to locate ticket for order.");
434  return OrderTicket.InvalidCancelOrderId(_algorithm.Transactions, request);
435  }
436 
437  try
438  {
439  // if we couldn't set this request as the cancellation then another thread/someone
440  // else is already doing it or it in fact has already been cancelled
441  if (!ticket.TrySetCancelRequest(request))
442  {
443  // the ticket has already been cancelled
444  request.SetResponse(OrderResponse.Error(request, OrderResponseErrorCode.InvalidRequest, "Cancellation is already in progress."));
445  return ticket;
446  }
447 
448  //Error check
449  var order = GetOrderByIdInternal(request.OrderId);
450  if (order != null && request.Tag != null)
451  {
452  order.Tag = request.Tag;
453  }
454  if (order == null)
455  {
456  Log.Error("BrokerageTransactionHandler.CancelOrder(): Cannot find this id.");
457  request.SetResponse(OrderResponse.UnableToFindOrder(request));
458  }
459  else if (order.Status == OrderStatus.New)
460  {
461  Log.Error("BrokerageTransactionHandler.CancelOrder(): Cannot cancel order with status: " + order.Status);
462  request.SetResponse(OrderResponse.InvalidNewStatus(request, order));
463  }
464  else if (order.Status.IsClosed())
465  {
466  Log.Error("BrokerageTransactionHandler.CancelOrder(): Cannot cancel order already " + order.Status);
467  request.SetResponse(OrderResponse.InvalidStatus(request, order));
468  }
469  else if (_algorithm.IsWarmingUp)
470  {
471  request.SetResponse(OrderResponse.WarmingUp(request));
472  }
473  else
474  {
475  _cancelPendingOrders.Set(order.Id, order.Status);
476  // update the order status
477  order.Status = OrderStatus.CancelPending;
478 
479  // notify the algorithm with an order event
480  HandleOrderEvent(new OrderEvent(order,
481  _algorithm.UtcTime,
482  OrderFee.Zero));
483 
484  // send the request to be processed
485  request.SetResponse(OrderResponse.Success(request), OrderRequestStatus.Processing);
486  _orderRequestQueue.Add(request);
487  }
488  }
489  catch (Exception err)
490  {
491  Log.Error(err);
492  request.SetResponse(OrderResponse.Error(request, OrderResponseErrorCode.ProcessingError, err.Message));
493  }
494 
495  return ticket;
496  }

605  /// <summary>
606  /// Primary thread entry point to launch the transaction thread.
607  /// </summary>
608  protected void Run()
609  {
610  try
611  {
612  foreach (var request in _orderRequestQueue.GetConsumingEnumerable(_cancellationTokenSource.Token))
613  {
614  HandleOrderRequest(request);
615  ProcessAsynchronousEvents();
616  }
617  }
618  catch (Exception err)
619  {
620  // unexpected error, we need to close down shop
621  _algorithm.SetRuntimeError(err, "HandleOrderRequest");
622  }
623 
624  if (_processingThread != null)
625  {
626  Log.Trace("BrokerageTransactionHandler.Run(): Ending Thread...");
627  IsActive = false;
628  }
629  }

639  /// <summary>
640  /// Processes all synchronous events that must take place before the next time loop for the algorithm
641  /// </summary>
642  public virtual void ProcessSynchronousEvents()
643  {
644  // how to do synchronous market orders for real brokerages?
645 
646  // in backtesting we need to wait for orders to be removed from the queue and finished processing
647  if (!_algorithm.LiveMode)
648  {
649  if (_orderRequestQueue.IsBusy && !_orderRequestQueue.WaitHandle.WaitOne(Time.OneSecond, _cancellationTokenSource.Token))
650  {
651  Log.Error("BrokerageTransactionHandler.ProcessSynchronousEvents(): Timed out waiting for request queue to finish processing.");
652  }
653  return;
654  }
655 
656  // check if the brokerage should perform cash sync now
657  if (!_algorithm.IsWarmingUp && _brokerage.ShouldPerformCashSync(CurrentTimeUtc))
658  {
659  // only perform cash syncs if we haven't had a fill for at least 10 seconds
660  if (TimeSinceLastFill > TimeSpan.FromSeconds(10))
661  {
662  if (!_brokerage.PerformCashSync(_algorithm, CurrentTimeUtc, () => TimeSinceLastFill))
663  {
664  if (++_failedCashSyncAttempts >= MaxCashSyncAttempts)
665  {
666  throw new Exception("The maximum number of attempts for brokerage cash sync has been reached.");
667  }
668  }
669  }
670  }
671 
672  // we want to remove orders older than 10k records, but only in live mode
673  const int maxOrdersToKeep = 10000;
674  if (_completeOrders.Count < maxOrdersToKeep + 1)
675  {
676  return;
677  }
678 
679  Log.Debug("BrokerageTransactionHandler.ProcessSynchronousEvents(): Start removing old orders...");
680  var max = _completeOrders.Max(x => x.Key);
681  var lowestOrderIdToKeep = max - maxOrdersToKeep;
682  foreach (var item in _completeOrders.Where(x => x.Key <= lowestOrderIdToKeep))
683  {
684  Order value;
685  OrderTicket ticket;
686  _completeOrders.TryRemove(item.Key, out value);
687  _completeOrderTickets.TryRemove(item.Key, out ticket);
688  }
689 
690  Log.Debug($"BrokerageTransactionHandler.ProcessSynchronousEvents(): New order count {_completeOrders.Count}. Exit");
691  }

693  /// <summary>
694  /// Register an already open Order
695  /// </summary>
696  public void AddOpenOrder(Order order, IAlgorithm algorithm)
697  {
698  if (order.Status == OrderStatus.New || order.Status == OrderStatus.None)
699  {
700  // make sure we have a valid order status
701  order.Status = OrderStatus.Submitted;
702  }
703 
704  order.Id = algorithm.Transactions.GetIncrementOrderId();
705

766  /// <summary>
767  /// Handles a request to submit a new order
768  /// </summary>
769  private OrderResponse HandleSubmitOrderRequest(SubmitOrderRequest request)
770  {
771  OrderTicket ticket;
772  var order = Order.CreateOrder(request);
773 
774  // ensure the order is tagged with a currency
775  var security = _algorithm.Securities[order.Symbol];
776  order.PriceCurrency = security.SymbolProperties.QuoteCurrency;
777  if (string.IsNullOrEmpty(order.Tag))
778  {
779  order.Tag = order.GetDefaultTag();
780  }
781 
782  // rounds off the order towards 0 to the nearest multiple of lot size
783  order.Quantity = RoundOffOrder(order, security);
784 
785  if (!_openOrders.TryAdd(order.Id, order) || !_completeOrders.TryAdd(order.Id, order))
786  {
787  Log.Error("BrokerageTransactionHandler.HandleSubmitOrderRequest(): Unable to add new order, order not processed.");
788  return OrderResponse.Error(request, OrderResponseErrorCode.OrderAlreadyExists, "Cannot process submit request because order with id {0} already exists");
789  }
790  if (!_completeOrderTickets.TryGetValue(order.Id, out ticket))
791  {
792  Log.Error("BrokerageTransactionHandler.HandleSubmitOrderRequest(): Unable to retrieve order ticket, order not processed.");
793  return OrderResponse.UnableToFindOrder(request);
794  }
795 
796  // rounds the order prices
797  RoundOrderPrices(order, security);
798 
799  // save current security prices
800  order.OrderSubmissionData = new OrderSubmissionData(security.BidPrice, security.AskPrice, security.Close);
801 
802  // Set order price adjustment mode
803  SetPriceAdjustmentMode(order, _algorithm);
804 
805  // update the ticket's internal storage with this new order reference
806  ticket.SetOrder(order);
807 
808  if (!order.TryGetGroupOrders(TryGetOrder, out var orders))
809  {
810  // an Order of the group is missing
811  return OrderResponse.Success(request);
812  }
813 
814  if (orders.Any(o => o.Quantity == 0))
815  {
816  var response = OrderResponse.ZeroQuantity(request);
817  _algorithm.Error(response.ErrorMessage);
818 
819  InvalidateOrders(orders, response.ErrorMessage);
820  return response;
821  }
822 
823  if (!orders.TryGetGroupOrdersSecurities(_algorithm.Portfolio, out var securities))
824  {
825  var response = OrderResponse.MissingSecurity(request);
826  _algorithm.Error(response.ErrorMessage);
827 
828  InvalidateOrders(orders, response.ErrorMessage);
829  return response;
830  }
831 
832  // check to see if we have enough money to place the order
833  HasSufficientBuyingPowerForOrderResult hasSufficientBuyingPowerResult;
834  try
835  {
836  hasSufficientBuyingPowerResult = _algorithm.Portfolio.HasSufficientBuyingPowerForOrder(orders);
837  }
838  catch (Exception err)
839  {
840  Log.Error(err);
841  _algorithm.Error($"Order Error: id: {order.Id.ToStringInvariant()}, Error executing margin models: {err.Message}");
842  HandleOrderEvent(new OrderEvent(order,
843  _algorithm.UtcTime,
844  OrderFee.Zero,
845  "Error executing margin models"));
846  return OrderResponse.Error(request, OrderResponseErrorCode.ProcessingError, "Error in GetSufficientCapitalForOrder");
847  }
848 
849  if (!hasSufficientBuyingPowerResult.IsSufficient)
850  {
851  var errorMessage = securities.GetErrorMessage(hasSufficientBuyingPowerResult);
852  _algorithm.Error(errorMessage);
853 
854  InvalidateOrders(orders, errorMessage);
855  return OrderResponse.Error(request, OrderResponseErrorCode.InsufficientBuyingPower, errorMessage);
856  }
857 
858  // verify that our current brokerage can actually take the order
859  foreach (var kvp in securities)
860  {
861  if (!_algorithm.BrokerageModel.CanSubmitOrder(kvp.Value, kvp.Key, out var message))
862  {
863  var errorMessage = $"BrokerageModel declared unable to submit order: [{string.Join(",", orders.Select(o => o.Id))}]";
864 
865  // if we couldn't actually process the order, mark it as invalid and bail
866  message ??= new BrokerageMessageEvent(BrokerageMessageType.Warning, "InvalidOrder", string.Empty);
867  var response = OrderResponse.Error(request, OrderResponseErrorCode.BrokerageModelRefusedToSubmitOrder, $"{errorMessage} {message}");
868 
869  InvalidateOrders(orders, response.ErrorMessage);
870  _algorithm.Error(response.ErrorMessage);
871  return response;
872  }
873  }
874 
875  // set the order status based on whether or not we successfully submitted the order to the market
876  bool orderPlaced;
877  try
878  {
879  orderPlaced = orders.All(o => _brokerage.PlaceOrder(o));
880  }
881  catch (Exception err)
882  {
883  Log.Error(err);
884  orderPlaced = false;
885  }
886 
887  if (!orderPlaced)
888  {
889  // we failed to submit the order, invalidate it
890  var errorMessage = $"Brokerage failed to place orders: [{string.Join(",", orders.Select(o => o.Id))}]";
891 
892  InvalidateOrders(orders, errorMessage);
893  _algorithm.Error(errorMessage);
894  return OrderResponse.Error(request, OrderResponseErrorCode.BrokerageFailedToSubmitOrder, errorMessage);
895  }
896 
897  return OrderResponse.Success(request);
898  }

1046  private void HandleOrderEvents(List<OrderEvent> orderEvents)
1047  {
1048  lock (_lockHandleOrderEvent)
1049  {
1050  // Get orders and tickets
1051  var orders = new List<Order>(orderEvents.Count);
1052 
1053  for (var i = 0; i < orderEvents.Count; i++)
1054  {
1055  var orderEvent = orderEvents[i];
1056 
1057  if (orderEvent.Status.IsClosed() && _openOrders.TryRemove(orderEvent.OrderId, out var order))
1058  {
1059  _completeOrders[orderEvent.OrderId] = order;
1060  }
1061  else if (!_completeOrders.TryGetValue(orderEvent.OrderId, out order))
1062  {
1063  Log.Error("BrokerageTransactionHandler.HandleOrderEvents(): Unable to locate open Combo Order with id " + orderEvent.OrderId);
1064  LogOrderEvent(orderEvent);
1065  return;
1066  }
1067  orders.Add(order);
1068 
1069  if (orderEvent.Status.IsClosed() && _openOrderTickets.TryRemove(orderEvent.OrderId, out var ticket))
1070  {
1071  _completeOrderTickets[orderEvent.OrderId] = ticket;
1072  }
1073  else if (!_completeOrderTickets.TryGetValue(orderEvent.OrderId, out ticket))
1074  {
1075  Log.Error("BrokerageTransactionHandler.HandleOrderEvents(): Unable to resolve open ticket: " + orderEvent.OrderId);
1076  LogOrderEvent(orderEvent);
1077  return;
1078  }
1079  orderEvent.Ticket = ticket;
1080  }
1081 
1082  var fillsToProcess = new List<OrderEvent>(orderEvents.Count);
1083 
1084  // now lets update the orders
1085  for (var i = 0; i < orderEvents.Count; i++)
1086  {
1087  var orderEvent = orderEvents[i];
1088  var order = orders[i];
1089  var ticket = orderEvent.Ticket;
1090 
1091  _cancelPendingOrders.UpdateOrRemove(order.Id, orderEvent.Status);
1092  // set the status of our order object based on the fill event except if the order status is filled/cancelled and the event is invalid
1093  // in live trading it can happen that we submit an update which get's rejected by the brokerage because the order is already filled
1094  // so we don't want the invalid update event to set the order status to invalid if it's already filled
1095  if (order.Status != OrderStatus.Filled && order.Status != OrderStatus.Canceled || orderEvent.Status != OrderStatus.Invalid)
1096  {
1097  order.Status = orderEvent.Status;
1098  }
1099 
1100  orderEvent.Id = order.GetNewId();
1101 
1102  // set the modified time of the order to the fill's timestamp
1103  switch (orderEvent.Status)
1104  {
1105  case OrderStatus.Canceled:
1106  order.CanceledTime = orderEvent.UtcTime;
1107  break;
1108 
1109  case OrderStatus.PartiallyFilled:
1110  case OrderStatus.Filled:
1111  order.LastFillTime = orderEvent.UtcTime;
1112 
1113  // append fill message to order tag, for additional information
1114  if (orderEvent.Status == OrderStatus.Filled && !string.IsNullOrWhiteSpace(orderEvent.Message))
1115  {
1116  if (string.IsNullOrWhiteSpace(order.Tag))
1117  {
1118  order.Tag = orderEvent.Message;
1119  }
1120  else
1121  {
1122  order.Tag += " - " + orderEvent.Message;
1123  }
1124  }
1125  break;
1126 
1127  case OrderStatus.UpdateSubmitted:
1128  case OrderStatus.Submitted:
1129  // submit events after the initial submission are all order updates
1130  if (ticket.UpdateRequests.Count > 0)
1131  {
1132  order.LastUpdateTime = orderEvent.UtcTime;
1133  }
1134  break;
1135  }
1136 
1137  // lets always set current Quantity, Limit and Stop prices in the order event so that it's easier for consumers
1138  // to know the current state and detect any update
1139  orderEvent.Quantity = order.Quantity;
1140  switch (order.Type)
1141  {
1142  case OrderType.Limit:
1143  var limit = order as LimitOrder;
1144  orderEvent.LimitPrice = limit.LimitPrice;
1145  break;
1146  case OrderType.ComboLegLimit:
1147  var legLimitOrder = order as ComboLegLimitOrder;
1148  orderEvent.LimitPrice = legLimitOrder.LimitPrice;
1149  break;
1150  case OrderType.StopMarket:
1151  var marketOrder = order as StopMarketOrder;
1152  orderEvent.StopPrice = marketOrder.StopPrice;
1153  break;
1154  case OrderType.StopLimit:
1155  var stopLimitOrder = order as StopLimitOrder;
1156  orderEvent.LimitPrice = stopLimitOrder.LimitPrice;
1157  orderEvent.StopPrice = stopLimitOrder.StopPrice;
1158  break;
1159  case OrderType.TrailingStop:
1160  var trailingStopOrder = order as TrailingStopOrder;
1161  orderEvent.StopPrice = trailingStopOrder.StopPrice;
1162  orderEvent.TrailingAmount = trailingStopOrder.TrailingAmount;
1163  break;
1164  case OrderType.LimitIfTouched:
1165  var limitIfTouchedOrder = order as LimitIfTouchedOrder;
1166  orderEvent.LimitPrice = limitIfTouchedOrder.LimitPrice;
1167  orderEvent.TriggerPrice = limitIfTouchedOrder.TriggerPrice;
1168  break;
1169  }
1170 
1171  // check if the fill currency and the order currency match the symbol currency
1172  if (orderEvent.Status == OrderStatus.Filled || orderEvent.Status == OrderStatus.PartiallyFilled)
1173  {
1174  fillsToProcess.Add(orderEvent);
1175  Interlocked.Exchange(ref _lastFillTimeTicks, CurrentTimeUtc.Ticks);
1176 
1177  var security = _algorithm.Securities[orderEvent.Symbol];
1178 
1179  if (orderEvent.Symbol.SecurityType == SecurityType.Crypto
1180  && order.Direction == OrderDirection.Buy
1181  && CurrencyPairUtil.TryDecomposeCurrencyPair(orderEvent.Symbol, out var baseCurrency, out var quoteCurrency)
1182  && orderEvent.OrderFee.Value.Currency == baseCurrency)
1183  {
1184  // fees are in the base currency, so we have to subtract them from the filled quantity
1185  // else the virtual position will bigger than the real size and we might no be able to liquidate
1186  orderEvent.FillQuantity -= orderEvent.OrderFee.Value.Amount;
1187  orderEvent.OrderFee = new ModifiedFillQuantityOrderFee(orderEvent.OrderFee.Value, quoteCurrency, security.SymbolProperties.ContractMultiplier);
1188 
1189  if (!_loggedFeeAdjustmentWarning)
1190  {
1191  _loggedFeeAdjustmentWarning = true;
1192  const string message = "When buying currency pairs, using Cash account types, fees in base currency" +
1193  " will be deducted from the filled quantity so virtual positions reflect actual holdings.";
1194  Log.Trace($"BrokerageTransactionHandler.HandleOrderEvent(): {message}");
1195  _algorithm.Debug(message);
1196  }
1197  }
1198  }
1199  }
1200 
1201  //Apply the filled orders to our portfolio:
1202  try
1203  {
1204  _algorithm.Portfolio.ProcessFills(fillsToProcess);
1205  }
1206  catch (Exception err)
1207  {
1208  Log.Error(err);
1209  _algorithm.Error($"Fill error: error in TradeBuilder.ProcessFill: {err.Message}");
1210  }
1211 
1212  // Apply the filled orders to the trade builder
1213  for (var i = 0; i < orderEvents.Count; i++)
1214  {
1215  var orderEvent = orderEvents[i];
1216 
1217  if (orderEvent.Status == OrderStatus.Filled || orderEvent.Status == OrderStatus.PartiallyFilled)
1218  {
1219  var security = _algorithm.Securities[orderEvent.Symbol];
1220 
1221  var multiplier = security.SymbolProperties.ContractMultiplier;
1222  var securityConversionRate = security.QuoteCurrency.ConversionRate;
1223  var feeInAccountCurrency = _algorithm.Portfolio.CashBook
1224  .ConvertToAccountCurrency(orderEvent.OrderFee.Value).Amount;
1225 
1226  try
1227  {
1228  _algorithm.TradeBuilder.ProcessFill(
1229  orderEvent,
1230  securityConversionRate,
1231  feeInAccountCurrency,
1232  multiplier);
1233  }
1234  catch (Exception err)
1235  {
1236  Log.Error(err);
1237  }
1238  }
1239 
1240  // update the ticket after we've processed the fill, but before the event, this way everything is ready for user code
1241  orderEvent.Ticket.AddOrderEvent(orderEvent);
1242  }
1243  }
1244 
1245  //We have the events! :) Orders filled, send them in to be handled by algorithm portfolio.
1246  for (var i = 0; i < orderEvents.Count; i++)
1247  {
1248  var orderEvent = orderEvents[i];
1249 
1250  if (orderEvent.Status != OrderStatus.None) //order.Status != OrderStatus.Submitted
1251  {
1252  _orderEvents.Enqueue(orderEvent);
1253 
1254  //Create new order event:
1255  _resultHandler.OrderEvent(orderEvent);
1256 
1257  NewOrderEvent?.Invoke(this, orderEvent);
1258 
1259  try
1260  {
1261  //Trigger our order event handler
1262  _algorithm.OnOrderEvent(orderEvent);
1263  }
1264  catch (Exception err)
1265  {
1266  // unexpected error, we need to close down shop
1267  _algorithm.SetRuntimeError(err, "Order Event Handler");
1268  }
1269  }
1270 
1271  LogOrderEvent(orderEvent);
1272  }
1273  }

1426  /// <summary>
1427  /// Option notification event is received and new order events are generated
1428  /// </summary>
1429  private void HandleOptionNotification(OptionNotificationEventArgs e)
1430  {
1431  if (_algorithm.Securities.TryGetValue(e.Symbol, out var security))
1432  {
1433  // let's take the order event lock, we will be looking at orders and security holdings
1434  // and we don't want them changing mid processing because of an order event coming in at the same time
1435  // for example: DateTime/decimal order attributes are not thread safe by nature!
1436  lock (_lockHandleOrderEvent)
1437  {
1438  if (OptionSymbol.IsOptionContractExpired(e.Symbol, CurrentTimeUtc))
1439  {
1440  if (e.Position == 0)
1441  {
1442  // only log always in live trading, in backtesting log if not 0 holdings
1443  if (_algorithm.LiveMode || security.Holdings.Quantity != 0)
1444  {
1445  Log.Trace(
1446  $"BrokerageTransactionHandler.HandleOptionNotification(): UtcTime: {CurrentTimeUtc} clearing position for expired option holding: " +
1447  $"Symbol: {e.Symbol.Value}, " +
1448  $"Holdings: {security.Holdings.Quantity}");
1449  }
1450 
1451  var quantity = -security.Holdings.Quantity;
1452 
1453  // If the quantity is already 0 for Lean and the brokerage there is nothing else todo here
1454  if (quantity != 0)
1455  {
1456  var exerciseOrder = GenerateOptionExerciseOrder(security, quantity, e.Tag);
1457 
1458  EmitOptionNotificationEvents(security, exerciseOrder);
1459  }
1460  }
1461  else
1462  {
1463  Log.Error("BrokerageTransactionHandler.HandleOptionNotification(): " +
1464  $"unexpected position ({e.Position} instead of zero) " +
1465  $"for expired option contract: {e.Symbol.Value}");
1466  }
1467  }
1468  else
1469  {
1470  // if position is reduced, could be an early exercise or early assignment
1471  if (Math.Abs(e.Position) < security.Holdings.AbsoluteQuantity)
1472  {
1473  Log.Trace("BrokerageTransactionHandler.HandleOptionNotification(): " +
1474  $"Symbol {e.Symbol.Value} EventQuantity {e.Position} Holdings {security.Holdings.Quantity}");
1475 
1476  // if we are long the option and there is an open order, assume it's an early exercise
1477  if (security.Holdings.IsLong)
1478  {
1479  // we only care about open option exercise orders, if it's closed it means we already
1480  // processed it and we wouldn't have a need to handle it here
1481  if (GetOpenOrders(x =>
1482  x.Symbol == e.Symbol &&
1483  x.Type == OrderType.OptionExercise)
1484  .FirstOrDefault() is OptionExerciseOrder exerciseOrder)
1485  {
1486  EmitOptionNotificationEvents(security, exerciseOrder);
1487  }
1488  }
1489 
1490  // if we are short the option and there are no buy orders (open or recently closed), assume it's an early assignment
1491  else if (security.Holdings.IsShort)
1492  {
1493  var nowUtc = CurrentTimeUtc;
1494  // for some brokerages (like IB) there might be a race condition between getting an option
1495  // notification event and lean processing an order event. So if we are here, there are these options:
1496  // A) holdings -10 position 5
1497  // 1) WE just BOUGHT 15 and Lean doesn't know yet
1498  // 2) WE just SOLD 15 and this notification is old
1499  // B) holdings -10 position -5
1500  // 1) WE just BOUGHT 5 and Lean doesn't know yet
1501  // 2) WE just SOLD 5 more and this notification is old
1502  // - Seen this in production already
1503  // 3) Brokerage triggered an early assignment
1504 
1505  // so we get ALL orders for this symbol that were placed or got an update in the last 'orderWindowSeconds'
1506 
1507  const int orderWindowSeconds = 10;
1508  if (!GetOrders(x =>
1509  x.Symbol == e.Symbol
1510  && (x.Status.IsOpen() || x.Status.IsFill() &&
1511  (Math.Abs((x.Time - nowUtc).TotalSeconds) < orderWindowSeconds
1512  || (x.LastUpdateTime.HasValue && Math.Abs((x.LastUpdateTime.Value - nowUtc).TotalSeconds) < orderWindowSeconds)
1513  || (x.LastFillTime.HasValue && Math.Abs((x.LastFillTime.Value - nowUtc).TotalSeconds) < orderWindowSeconds)))).Any())
1514  {
1515  var quantity = e.Position - security.Holdings.Quantity;
1516 
1517  var exerciseOrder = GenerateOptionExerciseOrder(security, quantity, e.Tag);
1518 
1519  EmitOptionNotificationEvents(security, exerciseOrder);
1520  }
1521  }
1522  }
1523  }
1524  }
1525  }
