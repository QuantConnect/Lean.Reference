96  /// <summary>
97  /// Initializes an instance of <see cref="PandasData"/>
98  /// </summary>
99  public PandasData(object data)
100  {
101  if (_pandas == null)
102  {
103  using (Py.GIL())
104  {
105  // Use our PandasMapper class that modifies pandas indexing to support tickers, symbols and SIDs
106  _pandas = Py.Import("PandasMapper");
107  _seriesFactory = _pandas.GetAttr("Series");
108  _dataFrameFactory = _pandas.GetAttr("DataFrame");
109  using var multiIndex = _pandas.GetAttr("MultiIndex");
110  _multiIndexFactory = multiIndex.GetAttr("from_tuples");
111  _empty = new PyString(string.Empty);
112 
113  var time = new PyString("time");
114  var symbol = new PyString("symbol");
115  var expiry = new PyString("expiry");
116  _defaultNames = new PyList(new PyObject[] { expiry, new PyString("strike"), new PyString("type"), symbol, time });
117  _level2Names = new PyList(new PyObject[] { symbol, time });
118  _level3Names = new PyList(new PyObject[] { expiry, symbol, time });
119  }
120  }
121 
122  // in the case we get a list/collection of data we take the first data point to determine the type
123  // but it's also possible to get a data which supports enumerating we don't care about those cases
124  if (data is not IBaseData && data is IEnumerable enumerable)
125  {
126  foreach (var item in enumerable)
127  {
128  data = item;
129  break;
130  }
131  }
132 
133  var type = data.GetType();
134  IsCustomData = type.Namespace != typeof(Bar).Namespace;
135  _symbol = ((IBaseData)data).Symbol;
136 
137  if (_symbol.SecurityType == SecurityType.Future) Levels = 3;
138  if (_symbol.SecurityType.IsOption()) Levels = 5;
139 
140  IEnumerable<string> columns = _standardColumns;
141 
142  if (IsCustomData)
143  {
144  var keys = (data as DynamicData)?.GetStorageDictionary()
145  // if this is a PythonData instance we add in '__typename' which we don't want into the data frame
146  .Where(x => !x.Key.StartsWith("__", StringComparison.InvariantCulture)).ToHashSet(x => x.Key);
147 
148  // C# types that are not DynamicData type
149  if (keys == null)
150  {
151  if (_membersByType.TryGetValue(type, out _members))
152  {
153  keys = _members.ToHashSet(x => x.Name.ToLowerInvariant());
154  }
155  else
156  {
157  var members = type.GetMembers().Where(x => x.MemberType == MemberTypes.Field || x.MemberType == MemberTypes.Property).ToList();
158 
159  var duplicateKeys = members.GroupBy(x => x.Name.ToLowerInvariant()).Where(x => x.Count() > 1).Select(x => x.Key);
160  foreach (var duplicateKey in duplicateKeys)
161  {
162  throw new ArgumentException($"PandasData.ctor(): {Messages.PandasData.DuplicateKey(duplicateKey, type.FullName)}");
163  }
164 
165  // If the custom data derives from a Market Data (e.g. Tick, TradeBar, QuoteBar), exclude its keys
166  keys = members.ToHashSet(x => x.Name.ToLowerInvariant());
167  keys.ExceptWith(_baseDataProperties);
168  keys.ExceptWith(GetPropertiesNames(typeof(QuoteBar), type));
169  keys.ExceptWith(GetPropertiesNames(typeof(TradeBar), type));
170  keys.ExceptWith(GetPropertiesNames(typeof(Tick), type));
171  keys.Add("value");
172 
173  _members = members.Where(x => keys.Contains(x.Name.ToLowerInvariant())).ToList();
174  _membersByType.TryAdd(type, _members);
175  }
176  }
177 
178  var customColumns = new HashSet<string>(columns);
179  customColumns.Add("value");
180  customColumns.UnionWith(keys);
181 
182  columns = customColumns;
183  }
184 
185  _series = columns.ToDictionary(k => k, v => new Serie());
186  }

188  /// <summary>
189  /// Adds security data object to the end of the lists
190  /// </summary>
191  /// <param name="baseData"><see cref="IBaseData"/> object that contains security data</param>
192  public void Add(object baseData)
193  {
194  foreach (var member in _members)
195  {
196  // TODO field/property.GetValue is expensive
197  var key = member.Name.ToLowerInvariant();
198  var endTime = ((IBaseData)baseData).EndTime;
199  var propertyMember = member as PropertyInfo;
200  if (propertyMember != null)
201  {
202  AddToSeries(key, endTime, propertyMember.GetValue(baseData));
203  continue;
204  }
205  var fieldMember = member as FieldInfo;
206  if (fieldMember != null)
207  {
208  AddToSeries(key, endTime, fieldMember.GetValue(baseData));
209  }
210  }
211 
212  var storage = (baseData as DynamicData)?.GetStorageDictionary();
213  if (storage != null)
214  {
215  var endTime = ((IBaseData) baseData).EndTime;
216  var value = ((IBaseData) baseData).Value;
217  AddToSeries("value", endTime, value);
218 
219  foreach (var kvp in storage.Where(x => x.Key != "value"
220  // if this is a PythonData instance we add in '__typename' which we don't want into the data frame
221  && !x.Key.StartsWith("__", StringComparison.InvariantCulture)))
222  {
223  AddToSeries(kvp.Key, endTime, kvp.Value);
224  }
225  }
226  else
227  {
228  var tick = baseData as Tick;
229  if (tick != null)
230  {
231  AddTick(tick);
232  }
233  else
234  {
235  var tradeBar = baseData as TradeBar;
236  var quoteBar = baseData as QuoteBar;
237  Add(tradeBar, quoteBar);
238  }
239  }
240  }

242  /// <summary>
243  /// Adds Lean data objects to the end of the lists
244  /// </summary>
245  /// <param name="tradeBar"><see cref="TradeBar"/> object that contains trade bar information of the security</param>
246  /// <param name="quoteBar"><see cref="QuoteBar"/> object that contains quote bar information of the security</param>
247  public void Add(TradeBar tradeBar, QuoteBar quoteBar)
248  {
249  if (tradeBar != null)
250  {
251  var time = tradeBar.EndTime;
252  GetSerie(Open).Add(time, tradeBar.Open);
253  GetSerie(High).Add(time, tradeBar.High);
254  GetSerie(Low).Add(time, tradeBar.Low);
255  GetSerie(Close).Add(time, tradeBar.Close);
256  GetSerie(Volume).Add(time, tradeBar.Volume);
257  }
258  if (quoteBar != null)
259  {
260  var time = quoteBar.EndTime;
261  if (tradeBar == null)
262  {
263  GetSerie(Open).Add(time, quoteBar.Open);
264  GetSerie(High).Add(time, quoteBar.High);
265  GetSerie(Low).Add(time, quoteBar.Low);
266  GetSerie(Close).Add(time, quoteBar.Close);
267  }
268  if (quoteBar.Ask != null)
269  {
270  GetSerie(AskOpen).Add(time, quoteBar.Ask.Open);
271  GetSerie(AskHigh).Add(time, quoteBar.Ask.High);
272  GetSerie(AskLow).Add(time, quoteBar.Ask.Low);
273  GetSerie(AskClose).Add(time, quoteBar.Ask.Close);
274  GetSerie(AskSize).Add(time, quoteBar.LastAskSize);
275  }
276  if (quoteBar.Bid != null)
277  {
278  GetSerie(BidOpen).Add(time, quoteBar.Bid.Open);
279  GetSerie(BidHigh).Add(time, quoteBar.Bid.High);
280  GetSerie(BidLow).Add(time, quoteBar.Bid.Low);
281  GetSerie(BidClose).Add(time, quoteBar.Bid.Close);
282  GetSerie(BidSize).Add(time, quoteBar.LastBidSize);
283  }
284  }

287  /// <summary>
288  /// Adds a tick data point to this pandas collection
289  /// </summary>
290  /// <param name="tick"><see cref="Tick"/> object that contains tick information of the security</param>
291  public void AddTick(Tick tick)
292  {
293  var time = tick.EndTime;
294 
295  // We will fill some series with null for tick types that don't have a value for that series, so that we make sure
296  // the indices are the same for every tick series.
297 
298  if (tick.TickType == TickType.Quote)
299  {
300  GetSerie(AskPrice).Add(time, tick.AskPrice);
301  GetSerie(AskSize).Add(time, tick.AskSize);
302  GetSerie(BidPrice).Add(time, tick.BidPrice);
303  GetSerie(BidSize).Add(time, tick.BidSize);
304  }
305  else
306  {
307  // Trade and open interest ticks don't have these values, so we'll fill them with null.
308  GetSerie(AskPrice).Add(time, null);
309  GetSerie(AskSize).Add(time, null);
310  GetSerie(BidPrice).Add(time, null);
311  GetSerie(BidSize).Add(time, null);
312  }
313 
314  GetSerie(Exchange).Add(time, tick.Exchange);
315  GetSerie(Suspicious).Add(time, tick.Suspicious);
316  GetSerie(Quantity).Add(time, tick.Quantity);
317 
318  if (tick.TickType == TickType.OpenInterest)
319  {
320  GetSerie(OpenInterest).Add(time, tick.Value);
321  GetSerie(LastPrice).Add(time, null);
322  }
323  else
324  {
325  GetSerie(LastPrice).Add(time, tick.Value);
326  GetSerie(OpenInterest).Add(time, null);
327  }
328  }

330  /// <summary>
331  /// Get the pandas.DataFrame of the current <see cref="PandasData"/> state
332  /// </summary>
333  /// <param name="levels">Number of levels of the multi index</param>
334  /// <returns>pandas.DataFrame object</returns>
335  public PyObject ToPandasDataFrame(int levels = 2)
336  {
337  List<PyObject> list;
338  var symbol = _symbol.ID.ToString().ToPython();
339 
340  // Create the index labels
341  var names = _defaultNames;
342  if (levels == 2)
343  {
344  // symbol, time
345  names = _level2Names;
346  list = new List<PyObject> { symbol, _empty };
347  }
348  else if (levels == 3)
349  {
350  // expiry, symbol, time
351  names = _level3Names;
352  list = new List<PyObject> { _symbol.ID.Date.ToPython(), symbol, _empty };
353  }
354  else
355  {
356  list = new List<PyObject> { _empty, _empty, _empty, symbol, _empty };
357  if (_symbol.SecurityType == SecurityType.Future)
358  {
359  list[0] = _symbol.ID.Date.ToPython();
360  }
361  else if (_symbol.SecurityType.IsOption())
362  {
363  list[0] = _symbol.ID.Date.ToPython();
364  list[1] = _symbol.ID.StrikePrice.ToPython();
365  list[2] = _symbol.ID.OptionRight.ToString().ToPython();
366  }
367  }
368 
369  // creating the pandas MultiIndex is expensive so we keep a cash
370  var indexCache = new Dictionary<List<DateTime>, PyObject>(new ListComparer<DateTime>());
371  // Returns a dictionary keyed by column name where values are pandas.Series objects
372  using var pyDict = new PyDict();
373  foreach (var kvp in _series)
374  {
375  if (kvp.Value.ShouldFilter) continue;
376 
377  if (!indexCache.TryGetValue(kvp.Value.Times, out var index))
378  {
379  using var tuples = kvp.Value.Times.Select(time => CreateTupleIndex(time, list)).ToPyListUnSafe();
380  using var namesDic = Py.kw("names", names);
381 
382  indexCache[kvp.Value.Times] = index = _multiIndexFactory.Invoke(new[] { tuples }, namesDic);
383 
384  foreach (var pyObject in tuples)
385  {
386  pyObject.Dispose();
387  }
388  }
389 
390  // Adds pandas.Series value keyed by the column name
391  using var pyvalues = new PyList();
392  for (var i = 0; i < kvp.Value.Values.Count; i++)
393  {
394  using var pyObject = kvp.Value.Values[i].ToPython();
395  pyvalues.Append(pyObject);
396  }
397  using var series = _seriesFactory.Invoke(pyvalues, index);
398  pyDict.SetItem(kvp.Key, series);
399  }
400  _series.Clear();
401  foreach (var kvp in indexCache)
402  {
403  kvp.Value.Dispose();
404  }
405 
406  for (var i = 0; i < list.Count; i++)
407  {
408  DisposeIfNotEmpty(list[i]);
409  }
410 
411  // Create the DataFrame
412  var result = _dataFrameFactory.Invoke(pyDict);
413 
414  foreach (var item in pyDict)
415  {
416  item.Dispose();
417  }
418 
419  return result;
420  }

433  /// <summary>
434  /// Create a new tuple index
435  /// </summary>
436  private static PyTuple CreateTupleIndex(DateTime index, List<PyObject> list)
437  {
438  DisposeIfNotEmpty(list[list.Count - 1]);
439  list[list.Count - 1] = index.ToPython();
440  return new PyTuple(list.ToArray());

455  private Serie GetSerie(string key)
456  {
457  if (!_series.TryGetValue(key, out var value))
458  {
459  throw new ArgumentException($"PandasData.GetSerie(): {Messages.PandasData.KeyNotFoundInSeries(key)}");
460  }

464  /// <summary>
465  /// Get the lower-invariant name of properties of the type that a another type is assignable from
466  /// </summary>
467  /// <param name="baseType">The type that is assignable from</param>
468  /// <param name="type">The type that is assignable by</param>
469  /// <returns>List of string. Empty list if not assignable from</returns>
470  private static IEnumerable<string> GetPropertiesNames(Type baseType, Type type)
471  {
472  return baseType.IsAssignableFrom(type)
473  ? baseType.GetProperties().Select(x => x.Name.ToLowerInvariant())
474  : Enumerable.Empty<string>();
475  }

477  private class Serie
478  {
479  private static readonly IFormatProvider InvariantCulture = CultureInfo.InvariantCulture;
480  public bool ShouldFilter { get; set; } = true;
481  public List<DateTime> Times { get; set; } = new();
482  public List<object> Values { get; set; } = new();
483 
484  public void Add(DateTime time, object input)
485  {
486  var value = input is decimal ? Convert.ToDouble(input, InvariantCulture) : input;
487  if (ShouldFilter)
488  {
489  // we need at least 1 valid entry for the series not to get filtered
490  if (value is double)
491  {
492  if (!((double)value).IsNaNOrZero())
493  {
494  ShouldFilter = false;
495  }
496  }
497  else if (value is string)
498  {
499  if (!string.IsNullOrWhiteSpace((string)value))
500  {
501  ShouldFilter = false;
502  }
503  }
504  else if (value is bool)
505  {
506  if ((bool)value)
507  {
508  ShouldFilter = false;
509  }
510  }
511  else if (value != null)
512  {
513  ShouldFilter = false;
514  }
515  }
516 
517  Values.Add(value);
518  Times.Add(time);
519  }
520 
521  public void Add(DateTime time, decimal input)
522  {
523  var value = Convert.ToDouble(input, InvariantCulture);
524  if (ShouldFilter && !value.IsNaNOrZero())
525  {
526  ShouldFilter = false;
527  }
528 
529  Values.Add(value);
530  Times.Add(time);
531  }
532  }
533  }
