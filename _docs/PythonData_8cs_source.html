23  /// <summary>
24  /// Dynamic data class for Python algorithms.
25  /// Stores properties of python instances in DynamicData dictionary
26  /// </summary>
27  public class PythonData : DynamicData
28  {
29  private readonly string _pythonTypeName;
30  private readonly dynamic _pythonData;
31  private readonly dynamic _defaultResolution;
32  private readonly dynamic _supportedResolutions;
33  private readonly dynamic _isSparseData;
34  private readonly dynamic _requiresMapping;
35  private DateTime _endTime;
36 
37  /// <summary>
38  /// The end time of this data. Some data covers spans (trade bars)
39  /// and as such we want to know the entire time span covered
40  /// </summary>
41  /// <remarks>
42  /// This property is overriden to allow different values for Time and EndTime
43  /// if they are set in the Reader. In the base implementation EndTime equals Time
44  /// </remarks>
45  public override DateTime EndTime
46  {
47  get
48  {
49  return _endTime == default ? Time : _endTime;
50  }
51  set
52  {
53  _endTime = value;
54  if(Time == default)
55  {
56  // if Time hasn't been set let's set it, like BaseData does. If the user overrides it that's okay
57  Time = value;
58  }
59  }
60  }

70  /// <summary>
71  /// Constructor for initializing the PythonData class with wrapped PyObject
72  /// </summary>
73  /// <param name="pythonData"></param>
74  public PythonData(PyObject pythonData)
75  {
76  _pythonData = pythonData;
77  using (Py.GIL())
78  {
79  // these methods rely on the Symbol so we can call them yet but we can get them
80  _requiresMapping = pythonData.GetPythonMethod("RequiresMapping");
81  _isSparseData = pythonData.GetPythonMethod("IsSparseData");
82  _defaultResolution = pythonData.GetPythonMethod("DefaultResolution");
83  _supportedResolutions = pythonData.GetPythonMethod("SupportedResolutions");
84  _pythonTypeName = pythonData.GetPythonType().GetAssemblyName().Name;
85  }
86  }

88  /// <summary>
89  /// Source Locator for algorithm written in Python.
90  /// </summary>
91  /// <param name="config">Subscription configuration object</param>
92  /// <param name="date">Date of the data file we're looking for</param>
93  /// <param name="isLiveMode">true if we're in live mode, false for backtesting mode</param>
94  /// <returns>STRING API Url.</returns>
95  public override SubscriptionDataSource GetSource(SubscriptionDataConfig config, DateTime date, bool isLiveMode)
96  {
97  using (Py.GIL())
98  {
99  var source = _pythonData.GetSource(config, date, isLiveMode);
100  return (source as PyObject).GetAndDispose<SubscriptionDataSource>();
101  }
102  }
103 
104  /// <summary>
105  /// Generic Reader Implementation for Python Custom Data.
106  /// </summary>
107  /// <param name="config">Subscription configuration</param>
108  /// <param name="line">CSV line of data from the source</param>
109  /// <param name="date">Date of the requested line</param>
110  /// <param name="isLiveMode">true if we're in live mode, false for backtesting mode</param>
111  /// <returns></returns>
112  public override BaseData Reader(SubscriptionDataConfig config, string line, DateTime date, bool isLiveMode)
113  {
114  using (Py.GIL())
115  {
116  var data = _pythonData.Reader(config, line, date, isLiveMode);
117  var result = (data as PyObject).GetAndDispose<BaseData>();
118 
119  (result as PythonData)?.SetProperty("__typename", _pythonTypeName);
120 
121  return result;
122  }
123  }

192  /// <summary>
193  /// Indexes into this PythonData, where index is key to the dynamic property
194  /// </summary>
195  /// <param name="index">the index</param>
196  /// <returns>Dynamic property of a given index</returns>
197  public object this[string index]
198  {
199  get

202  }
203 
204  set

210  /// <summary>
211  /// Helper method to determine if the current instance is of the provided type
212  /// </summary>
213  /// <param name="type">Target type to check against</param>
214  /// <returns>True if this instance is of the provided type</returns>
215  public bool IsOfType(Type type)
216  {
217  if (HasProperty("__typename"))
218  {
219  return (string)GetProperty("__typename") == type.FullName;
220  }
221  return GetType() == type;
222  }
