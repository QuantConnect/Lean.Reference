63  /// <summary>
64  /// <see cref = "AlgorithmPythonWrapper"/> constructor.
65  /// Creates and wraps the algorithm written in python.
66  /// </summary>
67  /// <param name="moduleName">Name of the module that can be found in the PYTHONPATH</param>
68  public AlgorithmPythonWrapper(string moduleName)
69  {
70  try
71  {
72  using (Py.GIL())
73  {
74  Logging.Log.Trace($"AlgorithmPythonWrapper(): Python version {PythonEngine.Version}: Importing python module {moduleName}");
75 
76  var module = Py.Import(moduleName);
77 
78  Logging.Log.Trace($"AlgorithmPythonWrapper(): {moduleName} successfully imported.");
79 
80  var pyList = module.Dir();
81  foreach (var name in pyList)
82  {
83  Type type;
84  var attr = module.GetAttr(name.ToString());
85  var repr = attr.Repr().GetStringBetweenChars('\'', '\'');
86 
87  if (repr.StartsWith(moduleName) && // Must be defined in the module
88  attr.TryConvert(out type, true) && // Must be a Type
89  typeof(QCAlgorithm).IsAssignableFrom(type)) // Must inherit from QCAlgorithm
90  {
91  Logging.Log.Trace("AlgorithmPythonWrapper(): Creating IAlgorithm instance.");
92 
93  _algorithm = attr.Invoke();
94 
95  // Set pandas
96  _algorithm.SetPandasConverter();
97 
98  // IAlgorithm reference for LEAN internal C# calls (without going from C# to Python and back)
99  _baseAlgorithm = _algorithm.AsManagedObject(type);
100 
101  // determines whether OnData method was defined or inherits from QCAlgorithm
102  // If it is not, OnData from the base class will not be called
103  var pyAlgorithm = _algorithm as PyObject;
104  _onData = pyAlgorithm.GetPythonMethod("OnData");
105 
106  _onMarginCall = pyAlgorithm.GetPythonMethod("OnMarginCall");
107 
108  _onOrderEvent = pyAlgorithm.GetAttr("OnOrderEvent");
109 
110  PyObject endOfDayMethod = pyAlgorithm.GetPythonMethod("OnEndOfDay");
111  if (endOfDayMethod != null)
112  {
113  // Since we have a EOD method implemented
114  // Determine which one it is by inspecting its arg count
115  var argCount = endOfDayMethod.GetPythonArgCount();
116  switch (argCount)
117  {
118  case 0: // EOD()
119  IsOnEndOfDayImplemented = true;
120  break;
121  case 1: // EOD(Symbol)
122  IsOnEndOfDaySymbolImplemented = true;
123  break;
124  }
125 
126  // Its important to note that even if both are implemented
127  // python will only use the last implemented, meaning only one will
128  // be used and seen.
129  }
130  }
131  attr.Dispose();
132  }
133  module.Dispose();
134  pyList.Dispose();
135  // If _algorithm could not be set, throw exception
136  if (_algorithm == null)
137  {
138  throw new Exception("Please ensure that one class inherits from QCAlgorithm.");
139  }
140  }
141  }
142  catch (Exception e)
143  {
144  // perform exception interpretation for error in module import
145  var interpreter = StackExceptionInterpreter.CreateFromAssemblies(AppDomain.CurrentDomain.GetAssemblies());
146  e = interpreter.Interpret(e, interpreter);
147 
148  throw new Exception($"AlgorithmPythonWrapper(): {interpreter.GetExceptionMessageHeader(e)}");
149  }
150  }

820  /// <summary>
821  /// Margin call event handler. This method is called right before the margin call orders are placed in the market.
822  /// </summary>
823  /// <param name="requests">The orders to be executed to bring this algorithm within margin limits</param>
824  public void OnMarginCall(List<SubmitOrderRequest> requests)
825  {
826  using (Py.GIL())
827  {
828  var result = _algorithm.OnMarginCall(requests);
829 
830  if (_onMarginCall != null)
831  {
832  var pyRequests = result as PyObject;
833  // If the method does not return or returns a non-iterable PyObject, throw an exception
834  if (pyRequests == null || !pyRequests.IsIterable())
835  {
836  throw new Exception("OnMarginCall must return a non-empty list of SubmitOrderRequest");
837  }
838 
839  requests.Clear();
840 
841  using var iterator = pyRequests.GetIterator();
842  foreach (PyObject pyRequest in iterator)
843  {
844  SubmitOrderRequest request;
845  if (TryConvert(pyRequest, out request))
846  {
847  requests.Add(request);
848  }
849  }
850 
851  // If the PyObject is an empty list or its items are not SubmitOrderRequest objects, throw an exception
852  if (requests.Count == 0)
853  {
854  throw new Exception("OnMarginCall must return a non-empty list of SubmitOrderRequest");
855  }
856  }
857  }

1094  /// </summary>

1096  /// <param name="pyObject">PyObject to be converted</param>
1097  /// <param name="result">C# object that of type T</param>
1098  /// <returns>True if successful conversion</returns>
1099  private bool TryConvert<T>(PyObject pyObject, out T result)
1100  {
1101  result = default(T);
1102  var type = (Type)pyObject.GetPythonType().AsManagedObject(typeof(Type));
1103 
1104  if (type == typeof(T))
1105  {
1106  result = (T)pyObject.AsManagedObject(typeof(T));
1107  }
1108 
1109  return type == typeof(T);
