27  /// <summary>
28  /// Equal weighting alpha streams portfolio construction model that will generate aggregated security targets taking into account all the alphas positions
29  /// and an equal weighting factor for each alpha, which is also factored by the relation of the alphas portfolio value and the current algorithms portfolio value,
30  /// overriding <see cref="GetAlphaWeight"/> allows custom weighting implementations
31  /// </summary>
32  public class EqualWeightingAlphaStreamsPortfolioConstructionModel : AlphaStreamsPortfolioConstructionModel
33  {
34  private bool _rebalance;
35  private Dictionary<Symbol, PortfolioTarget> _targetsPerSymbol;
36  private readonly Queue<Symbol> _removedSymbols = new Queue<Symbol>();
37  private Dictionary<Symbol, decimal> _unitQuantity = new Dictionary<Symbol, decimal>();
38  private Dictionary<Symbol, Dictionary<Symbol, PortfolioTarget>> _targetsPerSymbolPerAlpha = new Dictionary<Symbol, Dictionary<Symbol, PortfolioTarget>>();
39 
40  /// <summary>
41  /// Access the last portfolio state per alpha
42  /// </summary>
43  protected Dictionary<Symbol, AlphaStreamsPortfolioState> LastPortfolioPerAlpha = new Dictionary<Symbol, AlphaStreamsPortfolioState>();
44 
45  /// <summary>
46  /// Create portfolio targets from the specified insights
47  /// </summary>
48  /// <param name="algorithm">The algorithm instance</param>
49  /// <param name="insights">The insights to create portfolio targets from</param>
50  /// <returns>An enumerable of portfolio targets to be sent to the execution model</returns>
51  public override IEnumerable<IPortfolioTarget> CreateTargets(QCAlgorithm algorithm, Insight[] insights)
52  {
53  if (_targetsPerSymbol == null)
54  {
55  _targetsPerSymbol = new Dictionary<Symbol, PortfolioTarget>();
56  foreach (var securityHolding in algorithm.Portfolio.Values.Where(holding => holding.HoldStock))
57  {
58  // we need to liquidate any holdings we have for security which are not hold by any alpha
59  _targetsPerSymbol[securityHolding.Symbol] = new PortfolioTarget(securityHolding.Symbol, 0);
60  }
61  }
62 
63  while (_removedSymbols.TryDequeue(out var removedSymbol))
64  {
65  yield return new PortfolioTarget(removedSymbol, 0);
66  }
67 
68  var updated = false;
69  foreach (var portfolioState in algorithm.CurrentSlice?.Get<AlphaStreamsPortfolioState>().Values ?? Enumerable.Empty<AlphaStreamsPortfolioState>())
70  {
71  if (!_rebalance)
72  {
73  updated |= ProcessPortfolioState(portfolioState, algorithm);
74  }
75  // keep the last state per alpha
76  LastPortfolioPerAlpha[portfolioState.Symbol] = portfolioState;
77  }
78 
79  // if an alpha is removed or added we just rebalance all the targets because the weight changes of each alpha
80  if (_rebalance)
81  {
82  foreach (var portfolioState in LastPortfolioPerAlpha.Values)
83  {
84  updated |= ProcessPortfolioState(portfolioState, algorithm);
85  }
86  _rebalance = false;
87  }
88 
89  if (updated)
90  {
91  foreach (var portfolioTarget in _targetsPerSymbol.Values)
92  {
93  if (algorithm.Securities.ContainsKey(portfolioTarget.Symbol))
94  {
95  yield return portfolioTarget;
96  }
97  else
98  {
99  // this could happen if the algorithm has not yet added a security used by an alpha
100  Log.Error("EqualWeightingAlphaStreamsPortfolioConstructionModel.CreateTargets(): " +
101  $"Skip emitting portfolio target, security not found in algorithm {portfolioTarget.Symbol}");
102  }
103  }
104  }
105  }
106 
107  /// <summary>
108  /// Get's the weight for an alpha
109  /// </summary>
110  /// <param name="alphaId">The algorithm instance that experienced the change in securities</param>
111  /// <returns>The alphas weight</returns>
112  public override decimal GetAlphaWeight(string alphaId)
113  {
114  return 1m / _targetsPerSymbolPerAlpha.Count;
115  }
116 
117  /// <summary>
118  /// Event fired each time the we add/remove securities from the data feed
119  /// </summary>
120  /// <param name="algorithm">The algorithm instance that experienced the change in securities</param>
121  /// <param name="changes">The security additions and removals from the algorithm</param>
122  public override void OnSecuritiesChanged(QCAlgorithm algorithm, SecurityChanges changes)
123  {
124  changes.FilterCustomSecurities = false;
125 
126  foreach (var security in changes.RemovedSecurities)
127  {
128  if (security.Type != SecurityType.Base)
129  {
130  _removedSymbols.Enqueue(security.Symbol);
131  }
132  else if (security.Symbol.IsCustomDataType<AlphaStreamsPortfolioState>())
133  {
134  _rebalance = true;
135  _targetsPerSymbolPerAlpha.Remove(security.Symbol);
136  LastPortfolioPerAlpha.Remove(security.Symbol);
137  }
138  }
139  foreach (var security in changes.AddedSecurities)
140  {
141  if (security.Symbol.IsCustomDataType<AlphaStreamsPortfolioState>())
142  {
143  _rebalance = true;
144  _targetsPerSymbolPerAlpha[security.Symbol] = new Dictionary<Symbol, PortfolioTarget>();
145 
146  var lastState = security.Cache.GetData<AlphaStreamsPortfolioState>();
147  lastState ??= algorithm.GetLastKnownPrices(security).OfType<AlphaStreamsPortfolioState>().LastOrDefault();
148  if (lastState != null)
149  {
150  // keep the last state per alpha
151  LastPortfolioPerAlpha[security.Symbol] = lastState;
152  if(!algorithm.Portfolio.CashBook.ContainsKey(lastState.AccountCurrency))
153  {
154  // ensure we have conversion rates if the alpha has another account currency
155  algorithm.SetCash(lastState.AccountCurrency, 0);
156  }
157  }
158  }
159  }
160  }
161 
162  /// <summary>
163  /// Determines the portfolio weight to give a specific alpha. Default implementation just returns equal weighting
164  /// </summary>
165  /// <param name="portfolioState">The alphas portfolio state to get the weight for</param>
166  /// <param name="totalUsablePortfolioValue">This algorithms usable total portfolio value, removing the free portfolio value</param>
167  /// <param name="cashBook">This algorithms cash book</param>
168  /// <returns>The weight to use on this alphas positions</returns>
169  private decimal GetAlphaWeight(AlphaStreamsPortfolioState portfolioState,
170  decimal totalUsablePortfolioValue,
171  CashBook cashBook)
172  {
173  var alphaPortfolioValueInOurAccountCurrency =
174  cashBook.ConvertToAccountCurrency(portfolioState.TotalPortfolioValue, portfolioState.AccountCurrency);
175 
176  if (alphaPortfolioValueInOurAccountCurrency == 0)
177  {
178  return 0;
179  }
180 
181  return totalUsablePortfolioValue * GetAlphaWeight(portfolioState.AlphaId) / alphaPortfolioValueInOurAccountCurrency;
182  }
183 
184  private bool ProcessPortfolioState(AlphaStreamsPortfolioState portfolioState, QCAlgorithm algorithm)
185  {
186  var alphaId = portfolioState.Symbol;
187 
188  if(!_targetsPerSymbolPerAlpha.TryGetValue(alphaId, out var ourExistingTargets))
189  {
190  _targetsPerSymbolPerAlpha[alphaId] = ourExistingTargets = new Dictionary<Symbol, PortfolioTarget>();
191  }
192 
193  var totalUsablePortfolioValue = algorithm.Portfolio.TotalPortfolioValue - algorithm.Settings.FreePortfolioValue;
194  var alphaWeightFactor = GetAlphaWeight(portfolioState, totalUsablePortfolioValue, algorithm.Portfolio.CashBook);
195 
196  // first we create all the new aggregated targets for the provided portfolio state
197  var newTargets = new Dictionary<Symbol, PortfolioTarget>();
198  foreach (var positionGroup in portfolioState.PositionGroups ?? Enumerable.Empty<PositionGroupState>())
199  {
200  foreach (var position in positionGroup.Positions)
201  {
202  // let's keep the unit quantity so we can round by it
203  _unitQuantity[position.Symbol] = position.UnitQuantity;
204 
205  newTargets.TryGetValue(position.Symbol, out var existingAggregatedTarget);
206  var quantity = position.Quantity * alphaWeightFactor + (existingAggregatedTarget?.Quantity ?? 0);
207  newTargets[position.Symbol] = new PortfolioTarget(position.Symbol, quantity.DiscretelyRoundBy(position.UnitQuantity, MidpointRounding.ToZero));
208  }
209  }
210 
211  var updatedTargets = false;
212 
213  // We adjust the new targets based on what we already have:
214  // - We add any existing targets if any -> other alphas
215  // - But we deduct our own existing target from it if any (previous state)
216  foreach (var ourNewTarget in newTargets.Values)
217  {
218  var symbol = ourNewTarget.Symbol;
219  var newAggregatedTarget = ourNewTarget;
220  if (_targetsPerSymbol.TryGetValue(symbol, out var existingAggregatedTarget))
221  {
222  ourExistingTargets.TryGetValue(symbol, out var ourExistingTarget);
223 
224  var quantity = existingAggregatedTarget.Quantity + ourNewTarget.Quantity - (ourExistingTarget?.Quantity ?? 0);
225  newAggregatedTarget = new PortfolioTarget(symbol, quantity.DiscretelyRoundBy(_unitQuantity[symbol], MidpointRounding.ToZero));
226  }
227  ourExistingTargets[symbol] = ourNewTarget;
228 
229  if (existingAggregatedTarget == null || existingAggregatedTarget.Quantity != newAggregatedTarget.Quantity)
230  {
231  updatedTargets = true;
232  _targetsPerSymbol[symbol] = newAggregatedTarget;
233  }
234  }
235 
236  List<Symbol> symbolsToRemove = null;
237  // We adjust existing targets for symbols that got removed from this alpha
238  foreach (var removedTarget in ourExistingTargets.Values.Where(target => !newTargets.ContainsKey(target.Symbol)))
239  {
240  var symbol = removedTarget.Symbol;
241  var newAggregatedTarget = removedTarget;
242  if (_targetsPerSymbol.TryGetValue(symbol, out var existingAggregatedTarget))
243  {
244  var quantity = existingAggregatedTarget.Quantity - removedTarget.Quantity;
245  newAggregatedTarget = new PortfolioTarget(symbol, quantity.DiscretelyRoundBy(_unitQuantity[symbol], MidpointRounding.ToZero));
246  }
247 
248  symbolsToRemove ??= new List<Symbol>();
249  symbolsToRemove.Add(symbol);
250  if (existingAggregatedTarget == null || existingAggregatedTarget.Quantity != newAggregatedTarget.Quantity)
251  {
252  updatedTargets = true;
253  _targetsPerSymbol[symbol] = newAggregatedTarget;
254  }
255  }
256 
257  if (symbolsToRemove != null)
258  {
259  for (var i = 0; i < symbolsToRemove.Count; i++)
260  {
261  // we can't remove from dictionary while iterating through it
262  ourExistingTargets.Remove(symbolsToRemove[i]);
263  }
264  }
265 
266  return updatedTargets;
267  }
268  }
