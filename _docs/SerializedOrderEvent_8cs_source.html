16 using System.ComponentModel;
17 using Newtonsoft.Json;
18 using Newtonsoft.Json.Converters;
19 
20 namespace QuantConnect.Orders.Serialization
21 {
22  /// <summary>
23  /// Data transfer object used for serializing an <see cref="OrderEvent"/> that was just generated by an algorithm
24  /// </summary>
25  public class SerializedOrderEvent
26  {
27  /// <summary>
28  /// The unique order event id
29  /// </summary>
30  [JsonProperty("id")]
31  public virtual string Id => $"{AlgorithmId}-{OrderId}-{OrderEventId}";
32 
33  /// <summary>
34  /// Algorithm Id, BacktestId or DeployId
35  /// </summary>
36  [JsonProperty("algorithm-id")]
37  public string AlgorithmId { get; set; }
38 
39  /// <summary>
40  /// Id of the order this event comes from.
41  /// </summary>
42  [JsonProperty("order-id")]
43  public int OrderId { get; set; }
44 
45  /// <summary>
46  /// The unique order event id for each order
47  /// </summary>
48  [JsonProperty("order-event-id")]
49  public int OrderEventId { get; set; }
50 
51  /// <summary>
52  /// Easy access to the order symbol associated with this event.
53  /// </summary>
54  [JsonProperty("symbol")]
55  public string Symbol { get; set; }
56 
57  /// <summary>
58  /// The time of this event in unix timestamp
59  /// </summary>
60  [JsonProperty("time")]
61  public double Time { get; set; }
62 
63  /// <summary>
64  /// Status message of the order.
65  /// </summary>
66  [JsonProperty("status"), JsonConverter(typeof(StringEnumConverter), true)]
67  public OrderStatus Status { get; set; }
68 
69  /// <summary>
70  /// The fee amount associated with the order
71  /// </summary>
72  [JsonProperty("order-fee-amount", DefaultValueHandling = DefaultValueHandling.Ignore)]
73  public decimal? OrderFeeAmount { get; set; }
74 
75  /// <summary>
76  /// The fee currency associated with the order
77  /// </summary>
78  [JsonProperty("order-fee-currency", DefaultValueHandling = DefaultValueHandling.Ignore)]
79  public string OrderFeeCurrency { get; set; }
80 
81  /// <summary>
82  /// Fill price information about the order
83  /// </summary>
84  [JsonProperty("fill-price")]
85  public decimal FillPrice { get; set; }
86 
87  /// <summary>
88  /// Currency for the fill price
89  /// </summary>
90  [JsonProperty("fill-price-currency")]
91  public string FillPriceCurrency { get; set; }
92 
93  /// <summary>
94  /// Number of shares of the order that was filled in this event.
95  /// </summary>
96  [JsonProperty("fill-quantity")]
97  public decimal FillQuantity { get; set; }
98 
99  /// <summary>
100  /// Order direction.
101  /// </summary>
102  [JsonProperty("direction"), JsonConverter(typeof(StringEnumConverter), true)]
103  public OrderDirection Direction { get; set; }
104 
105  /// <summary>
106  /// Any message from the exchange.
107  /// </summary>
108  [DefaultValue(""), JsonProperty("message", DefaultValueHandling = DefaultValueHandling.Ignore)]
109  public string Message { get; set; }
110 
111  /// <summary>
112  /// True if the order event is an assignment
113  /// </summary>
114  [JsonProperty("is-assignment")]
115  public bool IsAssignment { get; set; }
116 
117  /// <summary>
118  /// The current order quantity
119  /// </summary>
120  [JsonProperty("quantity")]
121  public decimal Quantity { get; set; }
122 
123  /// <summary>
124  /// The current stop price
125  /// </summary>
126  [JsonProperty("stop-price", DefaultValueHandling = DefaultValueHandling.Ignore)]
127  public decimal? StopPrice { get; set; }
128 
129  /// <summary>
130  /// The current limit price
131  /// </summary>
132  [JsonProperty("limit-price", DefaultValueHandling = DefaultValueHandling.Ignore)]
133  public decimal? LimitPrice { get; set; }
134 
135  /// <summary>
136  /// True if the order event's option is In-The-Money (ITM)
137  /// </summary>
138  [JsonProperty("is-in-the-money", DefaultValueHandling = DefaultValueHandling.Ignore)]
139  public bool IsInTheMoney { get; set; }
140 
141  /// <summary>
142  /// Empty constructor required for JSON converter.
143  /// </summary>
144  public SerializedOrderEvent()
145  {
146  }
147 
148  /// <summary>
149  /// Creates a new instances based on the provided order event and algorithm Id
150  /// </summary>
151  public SerializedOrderEvent(OrderEvent orderEvent, string algorithmId)
152  {
153  AlgorithmId = algorithmId;
154  OrderId = orderEvent.OrderId;
155  OrderEventId = orderEvent.Id;
156  Symbol = orderEvent.Symbol.ID.ToString();
157  Time = QuantConnect.Time.DateTimeToUnixTimeStamp(orderEvent.UtcTime);
158  Status = orderEvent.Status;
159  if (orderEvent.OrderFee.Value.Currency != Currencies.NullCurrency)
160  {
161  OrderFeeAmount = orderEvent.OrderFee.Value.Amount;
162  OrderFeeCurrency = orderEvent.OrderFee.Value.Currency;
163  }
164  FillPrice = orderEvent.FillPrice;
165  FillPriceCurrency = orderEvent.FillPriceCurrency;
166  FillQuantity = orderEvent.FillQuantity;
167  Direction = orderEvent.Direction;
168  Message = orderEvent.Message;
169  IsAssignment = orderEvent.IsAssignment;
170  IsInTheMoney = orderEvent.IsInTheMoney;
171  Quantity = orderEvent.Quantity;
172  StopPrice = orderEvent.StopPrice;
173  LimitPrice = orderEvent.LimitPrice;
174  }
175  }
176 }
