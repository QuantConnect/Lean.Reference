99  /// <summary>
100  /// Return the Subscription Data Source
101  /// </summary>
102  /// <param name="config">Configuration object</param>
103  /// <param name="date">Date of this source file</param>
104  /// <param name="isLiveMode">true if we're in live mode, false for backtesting mode</param>
105  /// <returns>Subscription Data Source.</returns>
106  public override SubscriptionDataSource GetSource(SubscriptionDataConfig config, DateTime date, bool isLiveMode)
107  {
108  var source = Path.Combine(
109  Globals.DataFolder,
110  "alternative",
111  "alphastreams",
112  "portfoliostate",
113  config.Symbol.Value.ToLowerInvariant(),
114  $"{date:yyyyMMdd}.json"
115  );
116  return new SubscriptionDataSource(source, SubscriptionTransportMedium.LocalFile, FileFormat.Csv);
117  }

119  /// <summary>
120  /// Reader converts each line of the data source into BaseData objects.
121  /// </summary>
122  /// <param name="config">Subscription data config setup object</param>
123  /// <param name="line">Content of the source document</param>
124  /// <param name="date">Date of the requested data</param>
125  /// <param name="isLiveMode">true if we're in live mode, false for backtesting mode</param>
126  /// <returns>New data point object</returns>
127  public override BaseData Reader(SubscriptionDataConfig config, string line, DateTime date, bool isLiveMode)
128  {
129  var dataPoint = JsonConvert.DeserializeObject<AlphaStreamsPortfolioState>(line);
130  dataPoint.Symbol = config.Symbol;
131  return dataPoint;
132  }
