107  /// <summary>
108  /// Runs this instance.
109  /// </summary>
110  /// <returns></returns>
111  public bool Run(out ConcurrentDictionary<SecurityIdentifier, List<CoarseFundamental>> coarsePerSecurity, out DateTime[] dates)
112  {
113  var startTime = DateTime.UtcNow;
114  var success = true;
115  Log.Trace($"CoarseUniverseGeneratorProgram.ProcessDailyFolder(): Processing: {_dailyDataFolder.FullName}");
116 
117  var symbolsProcessed = 0;
118  var filesRead = 0;
119  var dailyFilesNotFound = 0;
120  var coarseFilesGenerated = 0;
121 
122  var mapFileResolver = _mapFileProvider.Get(new AuxiliaryDataKey(_market, SecurityType.Equity));
123 
124  var result = coarsePerSecurity = new();
125  dates = Array.Empty<DateTime>();
126 
127  var blackListedTickers = new HashSet<string>();
128  if (_blackListedTickersFile.Exists)
129  {
130  blackListedTickers = File.ReadAllLines(_blackListedTickersFile.FullName).ToHashSet();
131  }
132 
133  var securityIdentifierContexts = PopulateSidContex(mapFileResolver, blackListedTickers);
134  var dailyPricesByTicker = new ConcurrentDictionary<string, List<TradeBar>>();
135  var outputCoarseContent = new ConcurrentDictionary<DateTime, List<CoarseFundamental>>();
136 
137  var parallelOptions = new ParallelOptions { MaxDegreeOfParallelism = Math.Max(1, Environment.ProcessorCount / 2) };
138  try
139  {
140  Parallel.ForEach(securityIdentifierContexts, parallelOptions, sidContext =>
141  {
142  var coarseForSecurity = new List<CoarseFundamental>();
143  var symbol = new Symbol(sidContext.SID, sidContext.LastTicker);
144  var symbolCount = Interlocked.Increment(ref symbolsProcessed);
145  Log.Debug($"CoarseUniverseGeneratorProgram.Run(): Processing {symbol} with tickers: '{string.Join(",", sidContext.Tickers)}'");
146  var factorFile = _factorFileProvider.Get(symbol);
147 
148  // Populate dailyPricesByTicker with all daily data by ticker for all tickers of this security.
149  foreach (var ticker in sidContext.Tickers)
150  {
151  var pathFile = Path.Combine(_dailyDataFolder.FullName, $"{ticker}.zip");
152  var dailyFile = new FileInfo(pathFile);
153  if (!dailyFile.Exists)
154  {
155  Log.Debug($"CoarseUniverseGeneratorProgram.Run(): {dailyFile.FullName} not found, looking for daily data in data folder");
156 
157  dailyFile = new FileInfo(Path.Combine(Globals.DataFolder, "equity", "usa", "daily", $"{ticker}.zip"));
158  if (!dailyFile.Exists)
159  {
160  Log.Error($"CoarseUniverseGeneratorProgram.Run(): {dailyFile} not found!");
161  Interlocked.Increment(ref dailyFilesNotFound);
162  continue;
163  }
164  }
165 
166  if (!dailyPricesByTicker.ContainsKey(ticker))
167  {
168  dailyPricesByTicker.AddOrUpdate(ticker, ParseDailyFile(dailyFile));
169  Interlocked.Increment(ref filesRead);
170  }
171  }
172 
173  // Look for daily data for each ticker of the actual security
174  for (int mapFileRowIndex = sidContext.MapFileRows.Length - 1; mapFileRowIndex >= 1; mapFileRowIndex--)
175  {
176  var ticker = sidContext.MapFileRows[mapFileRowIndex].Item2.ToLowerInvariant();
177  var endDate = sidContext.MapFileRows[mapFileRowIndex].Item1;
178  var startDate = sidContext.MapFileRows[mapFileRowIndex - 1].Item1;
179  List<TradeBar> tickerDailyData;
180  if (!dailyPricesByTicker.TryGetValue(ticker, out tickerDailyData))
181  {
182  Log.Error($"CoarseUniverseGeneratorProgram.Run(): Daily data for ticker {ticker.ToUpperInvariant()} not found!");
183  continue;
184  }
185 
186  // Get daily data only for the time the ticker was
187  foreach (var tradeBar in tickerDailyData.Where(tb => tb.Time >= startDate && tb.Time <= endDate))
188  {
189  var coarseFundamental = GenerateFactorFileRow(ticker, sidContext, factorFile as CorporateFactorProvider, tradeBar);
190  coarseForSecurity.Add(coarseFundamental);
191 
192  outputCoarseContent.AddOrUpdate(tradeBar.Time,
193  new List<CoarseFundamental> { coarseFundamental },
194  (time, list) =>
195  {
196  lock (list)
197  {
198  list.Add(coarseFundamental);
199  return list;
200  }
201  });
202  }
203  }
204 
205  if(coarseForSecurity.Count > 0)
206  {
207  result[sidContext.SID] = coarseForSecurity;
208  }
209  if (symbolCount % 1000 == 0)
210  {
211  var elapsed = DateTime.UtcNow - startTime;
212  Log.Trace($"CoarseUniverseGeneratorProgram.Run(): Processed {symbolCount} in {elapsed:g} at {symbolCount / elapsed.TotalMinutes:F2} symbols/minute ");
213  }
214  });
215 
216  _destinationFolder.Create();
217  var startWriting = DateTime.UtcNow;
218  Parallel.ForEach(outputCoarseContent, coarseByDate =>
219  {
220  var filename = $"{coarseByDate.Key.ToString(DateFormat.EightCharacter, CultureInfo.InvariantCulture)}.csv";
221  var filePath = Path.Combine(_destinationFolder.FullName, filename);
222  Log.Debug($"CoarseUniverseGeneratorProgram.Run(): Saving {filename} with {coarseByDate.Value.Count} entries.");
223  File.WriteAllLines(filePath, coarseByDate.Value.Select(x => CoarseFundamental.ToRow(x)).OrderBy(cr => cr));
224  var filesCount = Interlocked.Increment(ref coarseFilesGenerated);
225  if (filesCount % 1000 == 0)
226  {
227  var elapsed = DateTime.UtcNow - startWriting;
228  Log.Trace($"CoarseUniverseGeneratorProgram.Run(): Processed {filesCount} in {elapsed:g} at {filesCount / elapsed.TotalSeconds:F2} files/second ");
229  }
230  });
231 
232  dates = outputCoarseContent.Keys.OrderBy(x => x).ToArray();
233  Log.Trace($"\n\nTotal of {coarseFilesGenerated} coarse files generated in {DateTime.UtcNow - startTime:g}:\n" +
234  $"\t => {filesRead} daily data files read.\n");
235  }
236  catch (Exception e)
237  {
238  Log.Error(e, $"CoarseUniverseGeneratorProgram.Run(): FAILED!");
239  success = false;
240  }
241 
242  return success;
243  }

245  /// <summary>
246  /// Generates the factor file row.
247  /// </summary>
248  /// <param name="ticker">The ticker.</param>
249  /// <param name="sidContext">The sid context.</param>
250  /// <param name="factorFile">The factor file.</param>
251  /// <param name="tradeBar">The trade bar.</param>
252  /// <param name="fineAvailableDates">The fine available dates.</param>
253  /// <param name="fineFundamentalFolder">The fine fundamental folder.</param>
254  /// <returns></returns>
255  private static CoarseFundamental GenerateFactorFileRow(string ticker, SecurityIdentifierContext sidContext, CorporateFactorProvider factorFile, TradeBar tradeBar)
256  {
257  var date = tradeBar.Time;
258  var factorFileRow = factorFile?.GetScalingFactors(date);
259  var dollarVolume = Math.Truncate((double)(tradeBar.Close * tradeBar.Volume));
260  var priceFactor = factorFileRow?.PriceFactor.Normalize() ?? 1m;
261  var splitFactor = factorFileRow?.SplitFactor.Normalize() ?? 1m;
262  var hasFundamentalData = CheckFundamentalData(date, sidContext.SID);
263 
264  // sid,symbol,close,volume,dollar volume,has fundamental data,price factor,split factor
265  return new CoarseFundamentalSource
266  {
267  Symbol = new Symbol(sidContext.SID, ticker),
268  Value = tradeBar.Close.Normalize(),
269  Time = date,
270  VolumeSetter = decimal.ToInt64(tradeBar.Volume),
271  DollarVolumeSetter = dollarVolume,
272  PriceFactorSetter = priceFactor,
273  SplitFactorSetter = splitFactor,
274  HasFundamentalDataSetter = hasFundamentalData
275  };
276  }

289  /// <summary>
290  /// Parses the daily file.
291  /// </summary>
292  /// <param name="dailyFile">The daily file.</param>
293  /// <returns></returns>
294  private static List<TradeBar> ParseDailyFile(FileInfo dailyFile)
295  {
296  var scaleFactor = 1 / 10000m;
297 
298  var output = new List<TradeBar>();
299  using (var fileStream = dailyFile.OpenRead())
300  using (var stream = Compression.UnzipStreamToStreamReader(fileStream))
301  {
302  while (!stream.EndOfStream)
303  {
304  var tradeBar = new TradeBar
305  {
306  Time = stream.GetDateTime(),
307  Open = stream.GetDecimal() * scaleFactor,
308  High = stream.GetDecimal() * scaleFactor,
309  Low = stream.GetDecimal() * scaleFactor,
310  Close = stream.GetDecimal() * scaleFactor,
311  Volume = stream.GetDecimal()
312  };
313  output.Add(tradeBar);
314  }
315  }
316 
317  return output;
318  }

324  /// <param name="exclusions">The exclusions.</param>
325  /// <returns></returns>
326  private IEnumerable<SecurityIdentifierContext> PopulateSidContex(MapFileResolver mapFileResolver, HashSet<string> exclusions)
327  {
328  Log.Trace("CoarseUniverseGeneratorProgram.PopulateSidContex(): Generating SID context from QuantQuote's map files.");
329  foreach (var mapFile in mapFileResolver)
330  {
331  if (exclusions.Contains(mapFile.Last().MappedSymbol))
332  {
333  continue;
334  }
335 
336  yield return new SecurityIdentifierContext(mapFile, _market);
337  }
338  }
