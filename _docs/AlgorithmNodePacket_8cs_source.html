21 namespace QuantConnect.Packets

23  /// <summary>
24  /// Algorithm Node Packet is a work task for the Lean Engine
25  /// </summary>
26  public class AlgorithmNodePacket : PythonEnvironmentPacket
27  {
28  /// <summary>
29  /// Default constructor for the algorithm node:
30  /// </summary>
31  /// <param name="type"></param>
32  public AlgorithmNodePacket(PacketType type)
33  : base(type)
34  { }
35 
36  /// <summary>
37  /// The host name to use if any
38  /// </summary>
39  [JsonProperty(PropertyName = "sHostName")]
40  public string HostName;
41 
42  /// <summary>
43  /// User Id placing request
44  /// </summary>
45  [JsonProperty(PropertyName = "iUserID")]
46  public int UserId = 0;
47 
48  /// User API Token
49  [JsonProperty(PropertyName = "sUserToken")]
50  public string UserToken = "";
51 
52  /// User Organization Id
53  [JsonProperty(PropertyName = "sOrganizationID")]
54  public string OrganizationId = "";
55 
56  /// <summary>
57  /// Project Id of the request
58  /// </summary>
59  [JsonProperty(PropertyName = "iProjectID")]
60  public int ProjectId = 0;
61 
62  /// <summary>
63  /// Project name of the request
64  /// </summary>
65  [JsonProperty(PropertyName = "sProjectName")]
66  public string ProjectName;
67 
68  /// <summary>
69  /// Algorithm Id - BacktestId or DeployId - Common Id property between packets.
70  /// </summary>
71  [JsonProperty(PropertyName = "sAlgorithmID")]
72  public string AlgorithmId
73  {
74  get
75  {
76  if (Type == PacketType.LiveNode || Type == PacketType.AlphaNode)
77  {
78  return ((LiveNodePacket)this).DeployId;
79  }
80  return ((BacktestNodePacket)this).BacktestId;
81  }
82  }
83 
84  /// <summary>
85  /// User session Id for authentication
86  /// </summary>
87  [JsonProperty(PropertyName = "sSessionID")]
88  public string SessionId = "";
89 
90  /// <summary>
91  /// Language flag: Currently represents IL code or Dynamic Scripted Types.
92  /// </summary>
93  [JsonProperty(PropertyName = "eLanguage")]
94  public Language Language = Language.CSharp;
95 
96  /// <summary>
97  /// Server type for the deployment (512, 1024, 2048)
98  /// </summary>
99  [JsonProperty(PropertyName = "sServerType")]
100  public ServerType ServerType = ServerType.Server512;
101 
102  /// <summary>
103  /// Unique compile id of this backtest
104  /// </summary>
105  [JsonProperty(PropertyName = "sCompileID")]
106  public string CompileId = "";
107 
108  /// <summary>
109  /// Version number identifier for the lean engine.
110  /// </summary>
111  [JsonProperty(PropertyName = "sVersion")]
112  public string Version;
113 
114  /// <summary>
115  /// An algorithm packet which has already been run and is being redelivered on this node.
116  /// In this event we don't want to relaunch the task as it may result in unexpected behaviour for user.
117  /// </summary>
118  [JsonProperty(PropertyName = "bRedelivered")]
119  public bool Redelivered = false;
120 
121  /// <summary>
122  /// Algorithm binary with zip of contents
123  /// </summary>
124  [JsonProperty(PropertyName = "oAlgorithm")]
125  public byte[] Algorithm = new byte[] { };
126 
127  /// <summary>
128  /// Request source - Web IDE or API - for controling result handler behaviour
129  /// </summary>
130  [JsonProperty(PropertyName = "sRequestSource")]
131  public string RequestSource = "WebIDE";
132 
133  /// <summary>
134  /// The maximum amount of RAM (in MB) this algorithm is allowed to utilize
135  /// </summary>
136  [JsonProperty(PropertyName = "iMaxRamAllocation")]
137  public int RamAllocation {
138  get { return Controls.RamAllocation; }
139  }
140 
141  /// <summary>
142  /// Specifies values to control algorithm limits
143  /// </summary>
144  [JsonProperty(PropertyName = "oControls")]
145  public Controls Controls;
146 
147  /// <summary>
148  /// The parameter values used to set algorithm parameters
149  /// </summary>
150  [JsonProperty(PropertyName = "aParameters")]
151  public Dictionary<string, string> Parameters = new Dictionary<string, string>();
152 
153  /// <summary>
154  /// String name of the HistoryProvider we're running with
155  /// </summary>
156  [JsonProperty(PropertyName = "sHistoryProvider")]
157  public string HistoryProvider = "";
158 
159  /// <summary>
160  /// Algorithm running mode.
161  /// </summary>
162  [JsonIgnore]
163  public virtual AlgorithmMode AlgorithmMode { get; } = AlgorithmMode.Backtesting;
164 
165  /// <summary>
166  /// Deployment target, either local or cloud.
167  /// </summary>
168  [JsonIgnore]
169  public DeploymentTarget DeploymentTarget;
170 
171  /// <summary>
172  /// Gets a unique name for the algorithm defined by this packet
173  /// </summary>
174  public string GetAlgorithmName()
175  {
176  return Invariant($"{UserId}-{ProjectId}-{AlgorithmId}");
177  }
178  }
