208  /// <summary>
209  /// Specifies an event should fire on the first tradable date + offset for the specified
210  /// symbol each week
211  /// </summary>
212  /// <param name="symbol">The symbol whose exchange is used to determine the first
213  /// tradeable date of the week</param>
214  /// <param name="daysOffset">The amount of tradable days to offset the first tradable day by</param>
215  /// <returns>A date rule that fires on the first + offset tradable date for the specified
216  /// security each week</returns>
217  public IDateRule WeekStart(Symbol symbol, int daysOffset = 0)
218  {
219  var securitySchedule = GetSecuritySchedule(symbol);
220  var tradingDays = securitySchedule.MarketHours.Values
221  .Where(x => x.IsClosedAllDay == false).OrderBy(x => x.DayOfWeek).ToList();
222 
223  // Limit offsets to securities weekly schedule
224  if (daysOffset > tradingDays.Count - 1)
225  {
226  throw new ArgumentOutOfRangeException(nameof(daysOffset),
227  $"DateRules.WeekStart() : {tradingDays.First().DayOfWeek}+{daysOffset} is out of range for {symbol}'s schedule," +
228  $" please use an offset between 0 - {tradingDays.Count - 1}; Schedule : {string.Join(", ", tradingDays.Select(x => x.DayOfWeek))}");
229  }
230 
231  // Create the new DateRule and return it
232  return new FuncDateRule(GetName(symbol, "WeekStart", daysOffset), (start, end) => WeekIterator(securitySchedule, start, end, daysOffset, true));

309  /// <summary>
310  /// Get the closest trading day to a given DateTime for a given <see cref="SecurityExchangeHours"/>.
311  /// </summary>
312  /// <param name="securityExchangeHours"><see cref="SecurityExchangeHours"/> object with schedule for this Security</param>
313  /// <param name="baseDay">The day to base our search from</param>
314  /// <param name="offset">Amount to offset the schedule by tradable days</param>
315  /// <param name="searchForward">Search into the future for the closest day if true; into the past if false</param>
316  /// <param name="boundary">The boundary DateTime on the resulting day</param>
317  /// <returns></returns>
318  private static DateTime GetScheduledDay(SecurityExchangeHours securityExchangeHours, DateTime baseDay, int offset, bool searchForward, DateTime? boundary = null)
319  {
320  // By default the scheduled date is the given day
321  var scheduledDate = baseDay;
322 
323  // If its not open on this day find the next trading day by searching in the given direction
324  if (!securityExchangeHours.IsDateOpen(scheduledDate))
325  {
326  scheduledDate = searchForward
327  ? securityExchangeHours.GetNextTradingDay(scheduledDate)
328  : securityExchangeHours.GetPreviousTradingDay(scheduledDate);
329  }
330 
331  // Offset the scheduled day accordingly
332  for (var i = 0; i < offset; i++)
333  {
334  scheduledDate = searchForward
335  ? securityExchangeHours.GetNextTradingDay(scheduledDate)
336  : securityExchangeHours.GetPreviousTradingDay(scheduledDate);
337  }
338 
339  // If there is a boundary ensure we enforce it
340  if (boundary.HasValue)
341  {
342  // If we are searching forward and the resulting date is after this boundary we
343  // revert to the last tradable day equal to or less than boundary
344  if (searchForward && scheduledDate > boundary)
345  {
346  scheduledDate = GetScheduledDay(securityExchangeHours, (DateTime)boundary, 0, false);
347  }
348 
349  // If we are searching backward and the resulting date is after this boundary we
350  // revert to the last tradable day equal to or greater than boundary
351  if (!searchForward && scheduledDate < boundary)
352  {
353  scheduledDate = GetScheduledDay(securityExchangeHours, (DateTime)boundary, 0, true);
354  }
355  }
356 
357  return scheduledDate;
358  }

360  private static IEnumerable<DateTime> MonthIterator(SecurityExchangeHours securitySchedule, DateTime start, DateTime end, int offset, bool searchForward)
361  {
362  // No schedule means no security, set to open everyday
363  if (securitySchedule == null)
364  {
365  securitySchedule = SecurityExchangeHours.AlwaysOpen(TimeZones.NewYork);
366  }
367 
368  // Iterate all days between the beginning of "start" month, through end of "end" month.
369  // Necessary to ensure we schedule events in the month we start and end.
370  var beginningOfStartMonth = new DateTime(start.Year, start.Month, 1);
371  var endOfEndMonth = new DateTime(end.Year, end.Month, DateTime.DaysInMonth(end.Year, end.Month));
372 
373  foreach (var date in Time.EachDay(beginningOfStartMonth, endOfEndMonth))
374  {
375  var daysInMonth = DateTime.DaysInMonth(date.Year, date.Month);
376 
377  // Searching forward the first of the month is baseDay, with boundary being the last
378  // Searching backward the last of the month is baseDay, with boundary being the first
379  var baseDate = searchForward? new DateTime(date.Year, date.Month, 1) : new DateTime(date.Year, date.Month, daysInMonth);
380  var boundaryDate = searchForward ? new DateTime(date.Year, date.Month, daysInMonth) : new DateTime(date.Year, date.Month, 1);
381 
382  // Determine the scheduled day for this month
383  if (date == baseDate)
384  {
385  var scheduledDay = GetScheduledDay(securitySchedule, baseDate, offset, searchForward, boundaryDate);
386 
387  // Ensure the date is within our schedules range
388  if (scheduledDay >= start && scheduledDay <= end)
389  {
390  yield return scheduledDay;
391  }
392  }
393  }
394  }

396  private static IEnumerable<DateTime> WeekIterator(SecurityExchangeHours securitySchedule, DateTime start, DateTime end, int offset, bool searchForward)
397  {
398  // Determine the weekly base day and boundary to schedule off of
399  DayOfWeek weeklyBaseDay;
400  DayOfWeek weeklyBoundaryDay;
401  if (securitySchedule == null)
402  {
403  // No schedule means no security, set to open everyday
404  securitySchedule = SecurityExchangeHours.AlwaysOpen(TimeZones.NewYork);
405 
406  // Searching forward Monday is baseDay, with boundary being the following Sunday
407  // Searching backward Friday is baseDay, with boundary being the previous Saturday
408  weeklyBaseDay = searchForward ? DayOfWeek.Monday : DayOfWeek.Friday;
409  weeklyBoundaryDay = searchForward ? DayOfWeek.Saturday + 1 : DayOfWeek.Sunday - 1;
410  }
411  else
412  {
413  // Fetch the securities schedule
414  var weeklySchedule = securitySchedule.MarketHours.Values
415  .Where(x => x.IsClosedAllDay == false).OrderBy(x => x.DayOfWeek).ToList();
416 
417  // Determine our weekly base day and boundary for this security
418  weeklyBaseDay = searchForward ? weeklySchedule.First().DayOfWeek : weeklySchedule.Last().DayOfWeek;
419  weeklyBoundaryDay = searchForward ? weeklySchedule.Last().DayOfWeek : weeklySchedule.First().DayOfWeek;
420  }
421 
422  // Iterate all days between the beginning of "start" week, through end of "end" week.
423  // Necessary to ensure we schedule events in the week we start and end.
424  // Also if we have a sunday for start/end we need to adjust for it being the front of the week when we want it as the end of the week.
425  var startAdjustment = start.DayOfWeek == DayOfWeek.Sunday ? -7 : 0;
426  var beginningOfStartWeek = start.AddDays(-(int)start.DayOfWeek + 1 + startAdjustment); // Date - DayOfWeek + 1
427 
428  var endAdjustment = end.DayOfWeek == DayOfWeek.Sunday ? -7 : 0;
429  var endOfEndWeek = end.AddDays(-(int)end.DayOfWeek + 7 + endAdjustment); // Date - DayOfWeek + 7
430 
431  // Determine the schedule for each week in this range
432  foreach (var date in Time.EachDay(beginningOfStartWeek, endOfEndWeek).Where(x => x.DayOfWeek == weeklyBaseDay))
433  {
434  var boundary = date.AddDays(weeklyBoundaryDay - weeklyBaseDay);
435  var scheduledDay = GetScheduledDay(securitySchedule, date, offset, searchForward, boundary);
436 
437  // Ensure the date is within our schedules range
438  if (scheduledDay >= start && scheduledDay <= end)
439  {
440  yield return scheduledDay;
441  }
442  }
443  }
444  }
