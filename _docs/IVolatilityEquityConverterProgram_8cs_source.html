47  public static void IVolatilityEquityConverter(string sourceDirectory, string sourceMetaDirectory, string destinationDirectory, string resolution)
48  {
49  Console.WriteLine("QuantConnect.ToolBox: IVolatility Converter: ");
50  Console.WriteLine("==============================================");
51  Console.WriteLine(
52  "The IVolatility converter transforms IVolatility orders into the LEAN Algorithmic Trading Engine Data Format.");
53  Console.WriteLine("Parameters required: --source-dir= --source-meta-dir= --destination-dir= --resolution=");
54  Console.WriteLine(" 1> Source archived IVolatility data.");
55  Console.WriteLine(" 2> Source archived IVolatility meta data.");
56  Console.WriteLine(" 3> Destination Directory of LEAN Data Folder. (Typically located under Lean/Data)");
57  Console.WriteLine(" 4> Resolution of your IVolatility data. (min,hour,day)");
58  Console.WriteLine(" ");
59  Console.WriteLine("NOTE: THIS WILL OVERWRITE ANY EXISTING FILES.");
60 
61  if (!(sourceDirectory.IsNullOrEmpty() || sourceMetaDirectory.IsNullOrEmpty()
62  || destinationDirectory.IsNullOrEmpty() || resolution.IsNullOrEmpty()))
63  {
64  _sourceDirectory = sourceDirectory;
65  _sourceMetaDirectory = sourceMetaDirectory;
66  _destinationDirectory = destinationDirectory;
67  _resolution = ParseResolution(resolution);
68  }
69  else
70  {
71  Console.WriteLine("1. Ivolatility equity data source directory: ");
72  _sourceDirectory = (Console.ReadLine() ?? "");
73  Console.WriteLine("2. Ivolatility equity data source directory: ");
74  _sourceMetaDirectory = (Console.ReadLine() ?? "");
75  Console.WriteLine("3. Destination LEAN Data directory: ");
76  _destinationDirectory = (Console.ReadLine() ?? "");
77  Console.WriteLine("4. Enter resolution of source data (Minute, Hour, Daily): ");
78  _resolution = ParseResolution(Console.ReadLine() ?? "");
79  }
80 
81  //Count the total files to process:
82  Console.WriteLine("Counting Files..." + _sourceDirectory);
83  var count = 1;
84  var files = GetFiles(_sourceDirectory);
85  Console.WriteLine("Processing {0} Files ...", files.Length);
86 
87  foreach (var file in files)
88  {
89  Console.WriteLine("Processing {0} of {1} files ...", count, files.Length);
90  var symbol = GetSymbol(file);
91  var streamProvider = StreamProvider.ForExtension(Path.GetExtension(file));
92  var inputStream = streamProvider.Open(file).First();
93  var streamReader = new StreamReader(inputStream);
94  IList<TradeBar> fileTradeBars = new List<TradeBar>();
95 // IList<QuoteBar> fileQuoteBars = new List<QuoteBar>();
96  var tradeDataWriter = new LeanDataWriter(_resolution, symbol, _destinationDirectory);
97 // var quoteDataWriter = new LeanDataWriter(_resolution, symbol, _destinationDirectory, TickType.Quote);
98  string line;
99  while ((line = streamReader.ReadLine()) != null)
100  {
101  var linearray = ParseCsv(line);
102  if (linearray.Length <= 2) continue;
103  var time = DateTime.ParseExact(linearray[TimeField], DateFormat.UI,
104  CultureInfo.InvariantCulture);
105  var priceBid = Parse.Decimal(linearray[PriceBidField]);
106  var sizeBid = Parse.Decimal(linearray[SizeBidField]);
107  var priceAsk = Parse.Decimal(linearray[PriceAskField]);
108  var sizeAsk = Parse.Decimal(linearray[SizeAskField]);
109  var priceLast = Parse.Decimal(linearray[PriceLastField]);
110  //var sizeLast = linearray[sizeLastField]);
111  var volume = Parse.Decimal(linearray[VolumeField]);
112  var tradeBar = new TradeBar(time, symbol, priceLast, priceLast, priceLast, priceLast, 0);
113  tradeBar.Update(priceLast, priceBid, priceAsk, volume, sizeBid, sizeAsk);
114  var bidBar = new Bar(priceBid, priceBid, priceBid, priceBid);
115  var askBar = new Bar(priceAsk, priceAsk, priceAsk, priceAsk);
116  var quoteBar = new QuoteBar(time, symbol, bidBar, sizeBid, askBar, sizeAsk,
117  TimeSpan.FromSeconds(60));
118  fileTradeBars.Add(tradeBar);
119 // fileQuoteBars.Add(quoteBar);
120  }
121  tradeDataWriter.Write(fileTradeBars);
122  //TODO: implement quote bars in LeanDataWriter for minute equity resolution
123 // quoteDataWriter.Write(fileQuoteBars);
124  count++;
125  }
126  }

179  /// <summary>
180  /// Generate a LEAN Factor File from IVol Dividends.csv and Splits.csv files
181  /// </summary>
182  /// <param name="symbol"></param>
183  /// <param name="factorFilePath"></param>
184  private static void GenerateFactorFile(Symbol symbol, string factorFilePath)
185  {
186  var splitFile = _sourceMetaDirectory + Path.DirectorySeparatorChar + symbol.Value + "Splits.csv";
187  var dividendFile = _sourceMetaDirectory + Path.DirectorySeparatorChar + symbol.Value + "Dividends.csv";
188 
189  var list1 = PopulateSplitsAndSpecialDividends(splitFile);
190  var list2 = PopulateRegularDividends(dividendFile);
191  var parsedFactors = list1.Concat(list2).ToList();
192 
193  parsedFactors.OrderByDescending(x => x.Time);
194 
195  var factorFileGenerator = new FactorFileGenerator(symbol, factorFilePath);
196  factorFileGenerator.CreateFactorFile(parsedFactors).WriteToFile(symbol);
197  }

199  /// <summary>
200  /// Parse regular cash dividends from IVol Dividends.csv to LEAN format
201  /// </summary>
202  /// <param name="dividendFile"></param>
203  private static List<BaseData> PopulateRegularDividends(string dividendFile)
204  {
205  var parsed = new List<BaseData>();
206  if (!File.Exists(dividendFile)) return parsed;
207  var dividendsLines = File.ReadAllLines(dividendFile);
208  foreach (var d in dividendsLines.Skip(1))
209  {
210  var line = ParseCsv(d);
211  parsed.Add(new Dividend
212  {
213  Time = DateTime.ParseExact(line[1].Replace("-", String.Empty), DateFormat.EightCharacter,
214  CultureInfo.InvariantCulture),
215  Value = Parse.Decimal(line[3])
216  });
217  }
218  return parsed;
219  }

221  /// <summary>
222  /// Read Ivolatility Split file and convert to Lean BaseData convention.
223  /// IVol split file also contains irregular cash and stock dividends, which should be indexed as dividends, not splits
224  /// </summary>
225  /// <param name="splitFile"></param>
226  /// <exception cref="Exception"></exception>
227  private static List<BaseData> PopulateSplitsAndSpecialDividends(string splitFile)
228  {
229  var parsed = new List<BaseData>();
230 
231  if (!File.Exists(splitFile)) return parsed;
232 
233  var splitsLines = File.ReadAllLines(splitFile);
234  foreach (var s in splitsLines.Skip(1))
235  {
236  var line = ParseCsv(s);
237  var time = DateTime.ParseExact(line[1].Replace("-", String.Empty),
238  DateFormat.EightCharacter, CultureInfo.InvariantCulture);
239  //0 - split, 1 - irregular cash dividend, 2 - stock dividend
240  switch (line[2])
241  {
242  case "0":
243  parsed.Add(new Split
244  {
245  Time = time,
246  Value = ParseSplitAmt(line[3])
247  });
248  break;
249  case "1":
250  case "2":
251  parsed.Add(new Dividend
252  {
253  Time = time,
254  Value = Parse.Decimal(line[4])
255  });
256  break;
257 
258  default:
259  throw new Exception("Unsupported split type: " + line[2]);
260  }
261  }
262  return parsed;
263  }

265  /// <summary>
266  /// Lean uses the inverse of IVol convention.
267  /// Ivol defines factor as split factor (1.5 for 3:2 split, 2 for 2/1 splits etc.),
268  /// LEAN Defines as 0.5 for 2/1 splits
269  /// </summary>
270  /// <param name="amount"></param>
271  /// <returns></returns>
272  private static decimal ParseSplitAmt(string amount)
273  {
274  return 1 / Parse.Decimal(amount);
275  }
