17 using System;
18 using System.Collections.Generic;
19 using QuantConnect.Algorithm.Framework.Alphas;
20 using QuantConnect.Algorithm.Framework.Alphas.Analysis;
21 using QuantConnect.Interfaces;
22 using QuantConnect.Logging;
23 
24 namespace QuantConnect.Lean.Engine.Alphas
25 {
26  /// <summary>
27  /// Manages alpha charting responsibilities.
28  /// </summary>
29  public class ChartingInsightManagerExtension : IInsightManagerExtension
30  {
31  /// <summary>
32  /// The string name used for the Alpha Assets chart
33  /// </summary>
34  public const string AlphaAssets = "Alpha Assets";
35  private readonly bool _liveMode;
36  private readonly StatisticsInsightManagerExtension _statisticsManager;
37 
38  private const int BacktestChartSamples = 1000;
39  private DateTime _lastInsightCountSampleDateUtc;
40  private DateTime _nextChartSampleAlgorithmTimeUtc;
41  private int _dailyInsightCount;
42 
43  // Keep track, we only want to add the charts if the algorithm is producing insights
44  private bool _chartsAdded;
45  private IAlgorithm _algorithm;
46 
47  // Heatmap chart
48  private readonly Chart _totalInsightCountPerSymbolChart = new Chart(AlphaAssets);
49  private readonly Dictionary<Symbol, int> _totalInsightCountPerSymbol = new Dictionary<Symbol, int>();
50 
51  private readonly Chart _totalInsightCountChart = new Chart("Insight Count");
52  private readonly Series _totalInsightCountSeries = new Series("Count", SeriesType.Bar, "#");
53 
54  private readonly Chart _insightScoreChart = new Chart("Alpha");
55  private readonly Dictionary<InsightScoreType, Series> _insightScoreSeriesByScoreType = new Dictionary<InsightScoreType, Series>();
56 
57  /// <summary>
58  /// Gets or sets the interval at which alpha charts are updated. This is in realtion to algorithm time.
59  /// </summary>
60  protected TimeSpan SampleInterval { get; set; } = TimeSpan.FromMinutes(1);
61 
62  /// <summary>
63  /// Initializes a new instance of the <see cref="ChartingInsightManagerExtension"/> class
64  /// </summary>
65  /// <param name="algorithm">The algorithm instance. This is only used for adding the charts
66  /// to the algorithm. We purposefully do not save a reference to avoid potentially inconsistent reads</param>
67  /// <param name="statisticsManager">Statistics manager used to access mean population scores for charting</param>
68  public ChartingInsightManagerExtension(IAlgorithm algorithm, StatisticsInsightManagerExtension statisticsManager)
69  {
70  _algorithm = algorithm;
71  _statisticsManager = statisticsManager;
72  _liveMode = algorithm.LiveMode;
73 
74  // Add our series for average scores over sample period to our "Alpha" chart
75  foreach (var scoreType in InsightManager.ScoreTypes)
76  {
77  var series = new Series($"{scoreType} Score", SeriesType.Line, "%");
78  _insightScoreSeriesByScoreType[scoreType] = series;
79  _insightScoreChart.AddSeries(series);
80  }
81 
82  // Add a series for insight count over sample period to the "Insight Count" chart
83  _totalInsightCountChart.AddSeries(_totalInsightCountSeries);
84  }
85 
86  /// <summary>
87  /// Invokes the manager at the end of the time step.
88  /// Samples and plots insight counts and population score.
89  /// </summary>
90  /// <param name="frontierTimeUtc">The current frontier time utc</param>
91  public void Step(DateTime frontierTimeUtc)
92  {
93  // Only add our charts to the algorithm when we actually have an insight
94  // We will still update our internal charts anyways, but this keeps Alpha charts out of
95  // algorithms that don't use the framework.
96  if (!_chartsAdded && _dailyInsightCount > 0)
97  {
98  _algorithm.AddChart(_insightScoreChart);
99  _algorithm.AddChart(_totalInsightCountChart);
100  _algorithm.AddChart(_totalInsightCountPerSymbolChart);
101 
102  _chartsAdded = true;
103  }
104 
105  // sample insight/symbol counts each utc day change
106  if (frontierTimeUtc.Date > _lastInsightCountSampleDateUtc)
107  {
108  _lastInsightCountSampleDateUtc = frontierTimeUtc.Date;
109 
110  // add sum of daily insight counts to the total insight count series
111  _totalInsightCountSeries.AddPoint(frontierTimeUtc.Date, _dailyInsightCount);
112 
113  // Create the pie chart every minute or so
114  PopulateChartWithSeriesPerSymbol(_totalInsightCountPerSymbol, _totalInsightCountPerSymbolChart, SeriesType.Treemap, frontierTimeUtc);
115 
116  // Resetting our storage
117  _dailyInsightCount = 0;
118  }
119 
120  // sample average population scores
121  if (frontierTimeUtc >= _nextChartSampleAlgorithmTimeUtc)
122  {
123  try
124  {
125  // verify these scores have been computed before taking the first sample
126  if (_statisticsManager.RollingAverageIsReady)
127  {
128  // sample the rolling averaged population scores
129  foreach (var scoreType in InsightManager.ScoreTypes)
130  {
131  var score = 100 * _statisticsManager.Statistics.RollingAveragedPopulationScore.GetScore(scoreType);
132  _insightScoreSeriesByScoreType[scoreType].AddPoint(frontierTimeUtc, score.SafeDecimalCast());
133  }
134  _nextChartSampleAlgorithmTimeUtc = frontierTimeUtc + SampleInterval;
135  }
136  }
137  catch (Exception err)
138  {
139  Log.Error(err);
140  }
141  }
142  }
143 
144  /// <summary>
145  /// Invoked after <see cref="IAlgorithm.Initialize"/> has been called.
146  /// Determines chart sample interval and initial sample times
147  /// </summary>
148  /// <remarks>
149  /// While the algorithm instance is provided, it's highly recommended to not maintain
150  /// a direct reference to it as there is no way to guarantee consistence reads.
151  /// </remarks>
152  /// <param name="algorithmStartDate">The start date of the algorithm</param>
153  /// <param name="algorithmEndDate">The end date of the algorithm</param>
154  /// <param name="algorithmUtcTime">The algorithm's current utc time</param>
155  public void InitializeForRange(DateTime algorithmStartDate, DateTime algorithmEndDate, DateTime algorithmUtcTime)
156  {
157  if (_liveMode)
158  {
159  // live mode we'll sample each minute
160  SampleInterval = Time.OneMinute;
161  }
162  else
163  {
164  // space out backtesting samples evenly
165  var backtestPeriod = algorithmEndDate - algorithmStartDate;
166  SampleInterval = TimeSpan.FromTicks(backtestPeriod.Ticks / BacktestChartSamples);
167  }
168 
169  _nextChartSampleAlgorithmTimeUtc = algorithmUtcTime + SampleInterval;
170  _lastInsightCountSampleDateUtc = algorithmUtcTime.RoundDown(Time.OneDay);
171  }
172 
173  /// <summary>
174  /// Handles the <see cref="IAlgorithm.InsightsGenerated"/> event.
175  /// Keep daily and total count of insights by symbol
176  /// </summary>
177  /// <param name="context">The newly generated insight analysis context</param>
178  public void OnInsightGenerated(InsightAnalysisContext context)
179  {
180  if (!_totalInsightCountPerSymbol.ContainsKey(context.Symbol))
181  {
182  _totalInsightCountPerSymbol[context.Symbol] = 1;
183  }
184  else
185  {
186  // track total count per symbol
187  _totalInsightCountPerSymbol[context.Symbol] += 1;
188  }
189 
190  _dailyInsightCount++;
191  }
192 
193  /// <summary>
194  /// NOP - Charting is more concerned with population vs individual insights
195  /// </summary>
196  /// <param name="context">Context whose insight has just completed analysis</param>
197  public void OnInsightClosed(InsightAnalysisContext context)
198  {
199  }
200 
201  /// <summary>
202  /// NOP - Charting is more concerned with population vs individual insights
203  /// </summary>
204  /// <param name="context">Context whose insight has just completed analysis</param>
205  public void OnInsightAnalysisCompleted(InsightAnalysisContext context)
206  {
207  }

209  /// <summary>
210  /// Creates series for each symbol and adds a value corresponding to the specified data
211  /// </summary>
212  private void PopulateChartWithSeriesPerSymbol(Dictionary<Symbol, int> data, Chart chart, SeriesType seriesType, DateTime frontierTimeUtc)
213  {
214  foreach (var kvp in data)
215  {
216  var symbol = kvp.Key;
217  var count = kvp.Value;
218 
219  Series series;
220  if (!chart.Series.TryGetValue(symbol.Value, out series))
221  {
222  series = new Series(symbol.Value, seriesType, null);
223  chart.Series.Add(series.Name, series);
224  }
225  series.AddPoint(frontierTimeUtc, count);
226  }
227  }
228  }
