16 using Python.Runtime;
17 using QuantConnect.Data;
18 using QuantConnect.Python;
19 using QuantConnect.Securities.Interfaces;
20 
21 namespace QuantConnect.Securities
22 {
23  /// <summary>
24  /// Python Wrapper for custom security data filters from Python
25  /// </summary>
26  public class SecurityDataFilterPythonWrapper : ISecurityDataFilter
27  {
28  private readonly dynamic _dataFilter;
29 
30  /// <summary>
31  /// Creates a new instance
32  /// </summary>
33  /// <param name="dataFilter">The Python class to wrapp</param>
34  public SecurityDataFilterPythonWrapper(PyObject dataFilter)
35  {
36  _dataFilter = dataFilter.ValidateImplementationOf<ISecurityDataFilter>();
37  }
38 
39  /// <summary>
40  /// Performs Filter method from Python instance returning true to accept, or false to fail/reject the data point.
41  /// </summary>
42  /// <param name="data">BasData data object we're filtering</param>
43  /// <param name="vehicle">Security vehicle for filter</param>
44  public bool Filter(Security vehicle, BaseData data)
45  {
46  using (Py.GIL())
47  {
48  return (_dataFilter.Filter(vehicle, data) as PyObject).GetAndDispose<bool>();
49  }
50  }
51  }
52 }
