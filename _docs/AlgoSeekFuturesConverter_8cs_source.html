63  /// <summary>
64  /// Give the reference date and source directory, convert the algoseek data into n-resolutions LEAN format.
65  /// </summary>
66  public void Convert()
67  {
68  Log.Trace("AlgoSeekFuturesConverter.Convert(): Copying remote raw data files locally.");
69  //Get the list of available raw files, copy from its remote location to a local folder and then for each file open a separate streamer.
70 
71  var files = GetFilesInRawFolder()
72  .Where(f => (f.Extension == ".gz" || f.Extension == ".bz2") && !f.Name.Contains("option"))
73  .Select(remote => remote.CopyTo(Path.Combine(Path.GetTempPath(), remote.Name), true))
74  .ToList();
75 
76  Log.Trace("AlgoSeekFuturesConverter.Convert(): Loading {0} AlgoSeekFuturesReader for {1} ", files.Count(), _referenceDate);
77 
78  //Initialize parameters
79  var totalLinesProcessed = 0L;
80  var totalFiles = files.Count();
81  var totalFilesProcessed = 0;
82  var start = DateTime.MinValue;
83 
84  var symbolMultipliers = LoadSymbolMultipliers();
85 
86  //Extract each file massively in parallel.
87  Parallel.ForEach(files, file =>
88  {
89  try
90  {
91  Log.Trace("Remote File :" + file);
92 
93  var csvFile = Path.Combine(_source.FullName, Path.GetFileNameWithoutExtension(file.Name));
94 
95  Log.Trace("Source File :" + csvFile);
96 
97  if (!File.Exists(csvFile))
98  {
99  // create the directory first or else 7z will fail
100  var csvFileInfo = new FileInfo(csvFile);
101  Directory.CreateDirectory(csvFileInfo.DirectoryName);
102 
103  Log.Trace("AlgoSeekFuturesConverter.Convert(): Extracting " + file);
104 
105  // Never time out extracting an archive; they can be pretty big
106  // and take a while to extract depending on the computer running this application
107  Compression.Extract7ZipArchive(file.FullName, _source.FullName, -1);
108  }
109 
110  // setting up local processors
111  var processors = new Processors();
112 
113  var reader = new AlgoSeekFuturesReader(csvFile, symbolMultipliers, _symbolFilter);
114  if (start == DateTime.MinValue)
115  {
116  start = DateTime.Now;
117  }
118 
119  if (reader.Current != null) // reader contains the data
120  {
121  do
122  {
123  var tick = reader.Current as Tick;
124 
125  if (tick.Symbol.ID.Symbol == "VX" && (
126  tick.BidPrice >= 998m || tick.AskPrice >= 998m))
127  {
128  // Invalid value for VX futures. Invalid prices in raw data are 998/999
129  continue;
130  }
131  //Add or create the consolidator-flush mechanism for symbol:
132  List<List<AlgoSeekFuturesProcessor>> symbolProcessors;
133  if (!processors.TryGetValue(tick.Symbol, out symbolProcessors))
134  {
135  symbolProcessors = new List<List<AlgoSeekFuturesProcessor>>(3)
136  {
137  { _resolutions.Select(x => new AlgoSeekFuturesProcessor(tick.Symbol, _referenceDate, TickType.Trade, x, _destination)).ToList() },
138  { _resolutions.Select(x => new AlgoSeekFuturesProcessor(tick.Symbol, _referenceDate, TickType.Quote, x, _destination)).ToList() },
139  { _resolutions.Select(x => new AlgoSeekFuturesProcessor(tick.Symbol, _referenceDate, TickType.OpenInterest, x, _destination)).ToList() }
140  };
141 
142  processors[tick.Symbol] = symbolProcessors;
143  }
144 
145  // Pass current tick into processor: enum 0 = trade; 1 = quote, 2 = oi
146  foreach (var processor in symbolProcessors[(int)tick.TickType])
147  {
148  processor.Process(tick);
149  }
150 
151  if (Interlocked.Increment(ref totalLinesProcessed) % 1000000m == 0)
152  {
153  var pro = (double)processors.Values.SelectMany(p => p.SelectMany(x => x)).Count();
154  var symbols = (double)processors.Keys.Count();
155  Log.Trace("AlgoSeekFuturesConverter.Convert(): Processed {0,3}M ticks( {1}k / sec); Memory in use: {2} MB; Total progress: {3}%, Processor per symbol {4}", Math.Round(totalLinesProcessed / 1000000m, 2), Math.Round(totalLinesProcessed / 1000L / (DateTime.Now - start).TotalSeconds), Process.GetCurrentProcess().WorkingSet64 / (1024 * 1024), 100 * totalFilesProcessed / totalFiles, pro / symbols);
156  }
157 
158  }
159  while (reader.MoveNext());
160 
161  Log.Trace("AlgoSeekFuturesConverter.Convert(): Performing final flush to disk... ");
162  Flush(processors, DateTime.MaxValue, true);
163  }
164 
165  processors = null;
166  GC.Collect();
167  GC.WaitForPendingFinalizers();
168 
169  Log.Trace("AlgoSeekFuturesConverter.Convert(): Finished processing file: " + file);
170  Interlocked.Increment(ref totalFilesProcessed);
171  }
172  catch(Exception err)
173  {
174  Log.Error("Exception caught! File: {0} Err: {1} Source {2} Stack {3}", file, err.Message, err.Source, err.StackTrace);
175  }
176  });
177 
178 
179  }

181  /// <summary>
182  /// Gets the files in raw folder.
183  /// </summary>
184  /// <returns>List of files in source folder</returns>
185  private IEnumerable<FileInfo> GetFilesInRawFolder()
186  {
187  var files = new List<FileInfo>();
188 
189  var command = OS.IsLinux ? "ls" : "cmd.exe";
190  var arguments = OS.IsWindows ? "/c dir /b /a-d" : string.Empty;
191 
192  var processStartInfo = new ProcessStartInfo(command, arguments)
193  {
194  CreateNoWindow = true,
195  WindowStyle = ProcessWindowStyle.Hidden,
196  UseShellExecute = false,
197  RedirectStandardOutput = true,
198  WorkingDirectory = _remote.FullName
199  };
200 
201  using (var process = new Process())
202  {
203 
204  process.StartInfo = processStartInfo;
205  process.Start();
206 
207  while (!process.StandardOutput.EndOfStream)
208  {
209  var line = process.StandardOutput.ReadLine();
210  if (line != null)
211  {
212  files.Add(new FileInfo(Path.Combine(_remote.FullName, line)));
213  }
214  }
215  process.WaitForExit();
216  }
217 
218  return files;
219 
220  }

222  /// <summary>
223  /// Private method loads symbol multipliers from algoseek csv file
224  /// </summary>
225  /// <returns></returns>
226  private Dictionary<string, decimal> LoadSymbolMultipliers()
227  {
228  const int columnUnderlying = 0;
229  const int columnMultipleFactor = 2;
230 
231  return File.ReadAllLines("AlgoSeekFuturesConverter/AlgoSeek.US.Futures.PriceMultipliers.1.1.csv")
232  .Select(line => line.ToCsvData())
233  // skipping empty fields
234  .Where(line => !string.IsNullOrEmpty(line[columnUnderlying]) &&
235  !string.IsNullOrEmpty(line[columnMultipleFactor]))
236  // skipping header
237  .Skip(1)
238  .ToDictionary(line => line[columnUnderlying],
239  line => line[columnMultipleFactor].ConvertInvariant<decimal>());
240  }

242  private void Flush(Processors processors, DateTime time, bool final)
243  {
244  foreach (var symbol in processors.Keys)
245  {
246  processors[symbol].ForEach(p => p.ForEach(x => x.FlushBuffer(time, final)));
247  }

250  /// <summary>
251  /// Compress the queue buffers directly to a zip file. Lightening fast as streaming ram-> compressed zip.
252  /// </summary>
253  public void Package(DateTime date)
254  {
255  var zipper = OS.IsWindows ? "C:/Program Files/7-Zip/7z.exe" : "7z";
256 
257  Log.Trace("AlgoSeekFuturesConverter.Package(): Zipping all files ...");
258 
259  var destination = Path.Combine(_destination, "future");
260  Directory.CreateDirectory(destination);
261  var dateMask = date.ToStringInvariant(DateFormat.EightCharacter);
262 
263  var files =
264  Directory.EnumerateFiles(destination, dateMask + "*.csv", SearchOption.AllDirectories)
265  .GroupBy(x => Directory.GetParent(x).FullName)
266  .ToList();
267 
268  // Zip each file massively in parallel
269  Parallel.ForEach(files, file =>
270  //foreach (var file in files)
271  {
272  try
273  {
274  var outputFileName = file.Key + ".zip";
275 
276  // Create and open a new ZIP file
277  var filesToCompress = Directory.GetFiles(file.Key, "*.csv", SearchOption.AllDirectories);
278  var zip = ZipFile.Open(outputFileName, ZipArchiveMode.Create);
279 
280  foreach (var fileToCompress in filesToCompress)
281  {
282  // Add the entry for each file
283  zip.CreateEntryFromFile(fileToCompress, Path.GetFileName(fileToCompress), CompressionLevel.Optimal);
284  }
285 
286  // Dispose of the object when we are done
287  zip.Dispose();
288 
289  try
290  {
291  Directory.Delete(file.Key, true);
292  }
293  catch (Exception err)
294  {
295  Log.Error("Directory.Delete returned error: " + err.Message);
296  }
297  }
298  catch (Exception err)
299  {
300  Log.Error("File: {0} Err: {1} Source {2} Stack {3}", file, err.Message, err.Source, err.StackTrace);
301  }
302  });
303  }
