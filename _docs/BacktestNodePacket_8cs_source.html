23 namespace QuantConnect.Packets

25  /// <summary>
26  /// Algorithm backtest task information packet.
27  /// </summary>
28  public class BacktestNodePacket : AlgorithmNodePacket
29  {
30  // default random id, static so its one per process
31  private static readonly string DefaultId
32  = Guid.NewGuid().ToString("N", CultureInfo.InvariantCulture);
33 
34  /// <summary>
35  /// Name of the backtest as randomly defined in the IDE.
36  /// </summary>
37  [JsonProperty(PropertyName = "sName")]
38  public string Name = "";
39 
40  /// <summary>
41  /// BacktestId / Algorithm Id for this task
42  /// </summary>
43  [JsonProperty(PropertyName = "sBacktestID")]
44  public string BacktestId = DefaultId;
45 
46  /// <summary>
47  /// Optimization Id for this task
48  /// </summary>
49  [JsonProperty(PropertyName = "sOptimizationID")]
50  public string OptimizationId;
51 
52  /// <summary>
53  /// Backtest start-date as defined in the Initialize() method.
54  /// </summary>
55  [JsonProperty(PropertyName = "dtPeriodStart")]
56  public DateTime? PeriodStart;
57 
58  /// <summary>
59  /// Backtest end date as defined in the Initialize() method.
60  /// </summary>
61  [JsonProperty(PropertyName = "dtPeriodFinish")]
62  public DateTime? PeriodFinish;
63 
64  /// <summary>
65  /// Backtest maximum end date
66  /// </summary>
67  [JsonProperty(PropertyName = "dtOutOfSampleMaxEndDate")]
68  public DateTime? OutOfSampleMaxEndDate;
69 
70  /// <summary>
71  /// The backtest out of sample day count
72  /// </summary>
73  [JsonProperty(PropertyName = "iOutOfSampleDays")]
74  public int OutOfSampleDays;
75 
76  /// <summary>
77  /// Estimated number of trading days in this backtest task based on the start-end dates.
78  /// </summary>
79  [JsonProperty(PropertyName = "iTradeableDates")]
80  public int TradeableDates = 0;
81 
82  /// <summary>
83  /// True, if this is a debugging backtest
84  /// </summary>
85  [JsonProperty(PropertyName = "bDebugging")]
86  public bool IsDebugging;
87 
88  /// <summary>
89  /// Optional initial cash amount if set
90  /// </summary>
91  public CashAmount? CashAmount;
92 
93  /// <summary>
94  /// Algorithm running mode.
95  /// </summary>
96  [JsonIgnore]
97  public override AlgorithmMode AlgorithmMode
98  {
99  get
100  {
101  return OptimizationId.IsNullOrEmpty() ? AlgorithmMode.Backtesting : AlgorithmMode.Optimization;
102  }
103  }
104 
105  /// <summary>
106  /// Default constructor for JSON
107  /// </summary>
108  public BacktestNodePacket()
109  : base(PacketType.BacktestNode)
110  {

112  {
113  MinuteLimit = 500,
114  SecondLimit = 100,
115  TickLimit = 30
116  };
117  }
118 
119  /// <summary>
120  /// Initialize the backtest task packet.
121  /// </summary>
122  public BacktestNodePacket(int userId, int projectId, string sessionId, byte[] algorithmData, decimal startingCapital, string name)
123  : this (userId, projectId, sessionId, algorithmData, name, new CashAmount(startingCapital, Currencies.USD))
124  {
125  }
126 
127  /// <summary>
128  /// Initialize the backtest task packet.
129  /// </summary>
130  public BacktestNodePacket(int userId, int projectId, string sessionId, byte[] algorithmData, string name, CashAmount? startingCapital = null)
131  : base(PacketType.BacktestNode)
132  {
133  UserId = userId;
134  Algorithm = algorithmData;
135  SessionId = sessionId;
136  ProjectId = projectId;
137  Name = name;
138  CashAmount = startingCapital;
139  Language = Language.CSharp;
140  Controls = new Controls
141  {
142  MinuteLimit = 500,
143  SecondLimit = 100,
144  TickLimit = 30
145  };
146  }
147  }
