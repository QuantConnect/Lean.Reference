16 using System.Collections.Generic;
17 using System.Linq;
18 using System.Text.RegularExpressions;
19 
20 namespace QuantConnect.Report.ReportElements
21 {
22  public class ParametersReportElement : ReportElement
23  {
24  private IReadOnlyDictionary<string, string> _parameters;
25  private readonly string _template;
26 
27  /// <summary>
28  /// Creates a two column table for the Algorithm's Parameters
29  /// </summary>
30  /// <param name="name">Name of the widget</param>
31  /// <param name="key">Location of injection</param>
32  /// <param name="backtestConfiguration">The configuration of the backtest algorithm</param>
33  /// <param name="liveConfiguration">The configuration of the live algorithm</param>
34  /// <param name="template">HTML template to use</param>
35  public ParametersReportElement(string name, string key, AlgorithmConfiguration backtestConfiguration, AlgorithmConfiguration liveConfiguration, string template)
36  {
37  Name = name;
38  Key = key;
39  _template = template;
40 
41  if (liveConfiguration != null)
42  {
43  _parameters = liveConfiguration.Parameters;
44  }
45  else if (backtestConfiguration != null)
46  {
47  _parameters = backtestConfiguration.Parameters;
48  }
49  else
50  {
51  // This case only happens for the unit tests, then we just create an empty dictionary
52  _parameters = new Dictionary<string, string>();
53  }
54  }
55 
56  /// <summary>
57  /// Generates a HTML two column table for the Algorithm's Parameters
58  /// </summary>
59  /// <returns>Returns a string representing a HTML two column table</returns>
60  public override string Render()
61  {
62  var items = new List<string>();
63  int parameterIndex;
64  var columns = (new Regex(@"{{\$KEY(\d+?)}}")).Matches(_template).Count;
65 
66  for (parameterIndex = 0; parameterIndex < _parameters.Count;)
67  {
68  var template = _template;
69  int column;
70  for (column = 0; column < columns; column++)
71  {
72  var currTemplateKey = "{{$KEY" + column.ToString() + "}}";
73  var currTemplateValue = "{{$VALUE" + column.ToString() + "}}";
74 
75  if (parameterIndex < _parameters.Count)
76  {
77  var parameter = _parameters.ElementAt(parameterIndex);
78  template = template.Replace(currTemplateKey, parameter.Key);
79  template = template.Replace(currTemplateValue, parameter.Value);
80  }
81  else
82  {
83  template = template.Replace(currTemplateKey, string.Empty);
84  template = template.Replace(currTemplateValue, string.Empty);
85  }
86 
87  parameterIndex++;
88  }
89 
90  if (column == 0)
91  {
92  parameterIndex++;
93  }
94 
95  items.Add(template);
96  }
97 
98  if (Key == ReportKey.ParametersPageStyle)
99  {
100  if (items.Count == 0)
101  {
102  return "display: none;";
103  }
104 
105  return string.Empty;
106  }
107 
108  var parameters= string.Join("\n", items);
109  return parameters;
110  }
111  }
112 }
