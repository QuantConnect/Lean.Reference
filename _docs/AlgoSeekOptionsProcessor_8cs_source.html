27  /// <summary>
28  /// Processor for caching and consolidating ticks;
29  /// then flushing the ticks in memory to disk when triggered.
30  /// </summary>
31  public class AlgoSeekOptionsProcessor
32  {
33  private string _zipPath;
34  private string _entryPath;
35  private Symbol _symbol;
36  private TickType _tickType;
37  private Resolution _resolution;
38  private Queue<IBaseData> _queue;
39  private string _dataDirectory;
40  private IDataConsolidator _consolidator;
41  private DateTime _referenceDate;
42  private static string[] _windowsRestrictedNames =
43  {
44  "con", "prn", "aux", "nul"
45  };

47  /// <summary>
48  /// Zip entry name for the option contract
49  /// </summary>
50  public string EntryPath
51  {
52  get
53  {
54  if (_entryPath == null)
55  {
56  _entryPath = SafeName(LeanData.GenerateZipEntryName(_symbol, _referenceDate, _resolution, _tickType));
57  }

59  }

64  /// Zip file path for the option contract collection
65  /// </summary>
66  public string ZipPath
67  {
68  get
69  {
70  if (_zipPath == null)
71  {
72  _zipPath = Path.Combine(_dataDirectory, SafeName(LeanData.GenerateRelativeZipFilePath(Safe(_symbol), _referenceDate, _resolution, _tickType).Replace(".zip", string.Empty))) + ".zip";

113  /// <summary>
114  /// Create a new AlgoSeekOptionsProcessor for enquing consolidated bars and flushing them to disk
115  /// </summary>
116  /// <param name="symbol">Symbol for the processor</param>
117  /// <param name="date">Reference date for the processor</param>
118  /// <param name="tickType">TradeBar or QuoteBar to generate</param>
119  /// <param name="resolution">Resolution to consolidate</param>
120  /// <param name="dataDirectory">Data directory for LEAN</param>
121  public AlgoSeekOptionsProcessor(Symbol symbol, DateTime date, TickType tickType, Resolution resolution, string dataDirectory)
122  {
123  _symbol = Safe(symbol);
124  _tickType = tickType;
125  _referenceDate = date;
126  _resolution = resolution;
127  _queue = new Queue<IBaseData>();
128  _dataDirectory = dataDirectory;
129 
130  // Setup the consolidator for the requested resolution
131  if (resolution == Resolution.Tick) throw new NotSupportedException();
132 
133  switch (tickType)
134  {
135  case TickType.Trade:
136  _consolidator = new TickConsolidator(resolution.ToTimeSpan());
137  break;
138  case TickType.Quote:
139  _consolidator = new TickQuoteBarConsolidator(resolution.ToTimeSpan());
140  break;
141  case TickType.OpenInterest:
142  _consolidator = new OpenInterestConsolidator(resolution.ToTimeSpan());
143  break;
144  }
145 
146  // On consolidating the bars put the bar into a queue in memory to be written to disk later.
147  _consolidator.DataConsolidated += (sender, consolidated) =>
148  {
149  _queue.Enqueue(consolidated);
150  };
151  }
152 
153  /// <summary>
154  /// Process the tick; add to the con
155  /// </summary>
156  /// <param name="data"></param>
157  public void Process(Tick data)
158  {
159  if (data.TickType != _tickType)
160  {
161  return;
162  }
163 
164  _consolidator.Update(data);
165  }
166 
167  /// <summary>
168  /// Write the in memory queues to the disk.
169  /// </summary>
170  /// <param name="frontierTime">Current foremost tick time</param>
171  /// <param name="finalFlush">Indicates is this is the final push to disk at the end of the data</param>
172  public void FlushBuffer(DateTime frontierTime, bool finalFlush)
173  {
174  //Force the consolidation if time has past the bar
175  _consolidator.Scan(frontierTime);
176 
177  // If this is the final packet dump it to the queue
178  if (finalFlush && _consolidator.WorkingData != null)
179  {
180  _queue.Enqueue(_consolidator.WorkingData);
181  }
182  }

184  /// <summary>
185  /// Add filtering to safe check the symbol for windows environments
186  /// </summary>
187  /// <param name="symbol">Symbol to rename if required</param>
188  /// <returns>Renamed symbol for reserved names</returns>
189  private static Symbol Safe(Symbol symbol)
190  {
191  if (OS.IsWindows)
192  {
193  if (_windowsRestrictedNames.Contains(symbol.Value.ToLowerInvariant()) ||
194  _windowsRestrictedNames.Contains(symbol.Underlying.Value.ToLowerInvariant()))
195  {
196  symbol = Symbol.CreateOption(SafeName(symbol.Underlying.Value), Market.USA, OptionStyle.American, symbol.ID.OptionRight, symbol.ID.StrikePrice, symbol.ID.Date);
197  }
198  }
199  return symbol;
200  }
201 
202  private static string SafeName(string fileName)
203  {
204  if (OS.IsWindows)
205  {
206  if (_windowsRestrictedNames.Contains(fileName.ToLowerInvariant()))
207  return "_" + fileName;
208  }
209  return fileName;
210  }
211  }
