16 using System;
17 using System.Collections.Generic;
18 using System.Linq;
19 using QuantConnect.Data.Fundamental;
20 using QuantConnect.Data.UniverseSelection;
21 using QuantConnect.Securities;
22 
23 namespace QuantConnect.Algorithm.Framework.Selection
24 {
25  /// <summary>
26  /// Provides a base class for defining equity coarse/fine fundamental selection models
27  /// </summary>
28  public abstract class FundamentalUniverseSelectionModel : UniverseSelectionModel
29  {
30  private readonly bool _filterFineData;
31  private readonly UniverseSettings _universeSettings;
32 
33  /// <summary>
34  /// Initializes a new instance of the <see cref="FundamentalUniverseSelectionModel"/> class
35  /// </summary>
36  /// <param name="filterFineData">True to also filter using fine fundamental data, false to only filter on coarse data</param>
37  protected FundamentalUniverseSelectionModel(bool filterFineData)
38  : this(filterFineData, null)
39  {
40  }
41 
42  /// <summary>
43  /// Initializes a new instance of the <see cref="FundamentalUniverseSelectionModel"/> class
44  /// </summary>
45  /// <param name="filterFineData">True to also filter using fine fundamental data, false to only filter on coarse data</param>
46  /// <param name="universeSettings">The settings used when adding symbols to the algorithm, specify null to use algorithm.UniverseSettings</param>
47  protected FundamentalUniverseSelectionModel(bool filterFineData, UniverseSettings universeSettings)
48  {
49  _filterFineData = filterFineData;
50  _universeSettings = universeSettings;
51  }
52 
53  /// <summary>
54  /// Creates a new fundamental universe using this class's selection functions
55  /// </summary>
56  /// <param name="algorithm">The algorithm instance to create universes for</param>
57  /// <returns>The universe defined by this model</returns>
58  public override IEnumerable<Universe> CreateUniverses(QCAlgorithm algorithm)
59  {
60  var universe = CreateCoarseFundamentalUniverse(algorithm);
61  if (_filterFineData)
62  {
63  universe = new FineFundamentalFilteredUniverse(universe, fine => SelectFine(algorithm, fine));
64  }
65  yield return universe;
66  }
67 
68  /// <summary>
69  /// Creates the coarse fundamental universe object.
70  /// This is provided to allow more flexibility when creating coarse universe.
71  /// </summary>
72  /// <param name="algorithm">The algorithm instance</param>
73  /// <returns>The coarse fundamental universe</returns>
74  public virtual Universe CreateCoarseFundamentalUniverse(QCAlgorithm algorithm)
75  {
76  var universeSettings = _universeSettings ?? algorithm.UniverseSettings;
77  return new CoarseFundamentalUniverse(universeSettings, coarse =>
78  {
79  // if we're using fine fundamental selection than exclude symbols without fine data
80  if (_filterFineData)
81  {
82  coarse = coarse.Where(c => c.HasFundamentalData);
83  }
84 
85  return SelectCoarse(algorithm, coarse);
86  });
87  }
88 
89  /// <summary>
90  /// Defines the coarse fundamental selection function.
91  /// </summary>
92  /// <param name="algorithm">The algorithm instance</param>
93  /// <param name="coarse">The coarse fundamental data used to perform filtering</param>
94  /// <returns>An enumerable of symbols passing the filter</returns>
95  public abstract IEnumerable<Symbol> SelectCoarse(QCAlgorithm algorithm, IEnumerable<CoarseFundamental> coarse);
96 
97  /// <summary>
98  /// Defines the fine fundamental selection function.
99  /// </summary>
100  /// <param name="algorithm">The algorithm instance</param>
101  /// <param name="fine">The fine fundamental data used to perform filtering</param>
102  /// <returns>An enumerable of symbols passing the filter</returns>
103  public virtual IEnumerable<Symbol> SelectFine(QCAlgorithm algorithm, IEnumerable<FineFundamental> fine)
104  {
105  // default impl performs no filtering of fine data
106  return fine.Select(f => f.Symbol);
107  }
108 
109  /// <summary>
110  /// Convenience method for creating a selection model that uses only coarse data
111  /// </summary>
112  /// <param name="coarseSelector">Selects symbols from the provided coarse data set</param>
113  /// <returns>A new universe selection model that will select US equities according to the selection function specified</returns>
114  public static IUniverseSelectionModel Coarse(Func<IEnumerable<CoarseFundamental>, IEnumerable<Symbol>> coarseSelector)
115  {
116  return new CoarseFundamentalUniverseSelectionModel(coarseSelector);
117  }
118 
119  /// <summary>
120  ///
121  /// </summary>
122  /// <param name="coarseSelector">Selects symbols from the provided coarse data set</param>
123  /// <param name="fineSelector">Selects symbols from the provided fine data set (this set has already been filtered according to the coarse selection)</param>
124  /// <returns>A new universe selection model that will select US equities according to the selection functions specified</returns>
125  public static IUniverseSelectionModel Fine(Func<IEnumerable<CoarseFundamental>, IEnumerable<Symbol>> coarseSelector, Func<IEnumerable<FineFundamental>, IEnumerable<Symbol>> fineSelector)
126  {
127  return new FineFundamentalUniverseSelectionModel(coarseSelector, fineSelector);
128  }
129  }
130 }
