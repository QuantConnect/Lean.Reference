29  /// <summary>
30  /// This is an implementation of <see cref="IDataQueueHandler"/> used to handle multiple live datafeeds
31  /// </summary>
32  public class DataQueueHandlerManager : IDataQueueHandler, IDataQueueUniverseProvider
33  {
34  private ITimeProvider _frontierTimeProvider;
35  private readonly Dictionary<SubscriptionDataConfig, Queue<IDataQueueHandler>> _dataConfigAndDataHandler = new();
36 
37  /// <summary>
38  /// Collection of data queue handles being used
39  /// </summary>
40  /// <remarks>Protected for testing purposes</remarks>
41  protected List<IDataQueueHandler> DataHandlers { get; } = new();
42 
43  /// <summary>
44  /// True if the composite queue handler has any <see cref="IDataQueueUniverseProvider"/> instance
45  /// </summary>
46  public bool HasUniverseProvider => DataHandlers.OfType<IDataQueueUniverseProvider>().Any();
47 
48  /// <summary>
49  /// Event triggered when an unsupported configuration is detected
50  /// </summary>
51  public event EventHandler<SubscriptionDataConfig> UnsupportedConfiguration;
52 
53  /// <summary>
54  /// Subscribe to the specified configuration
55  /// </summary>
56  /// <param name="dataConfig">defines the parameters to subscribe to a data feed</param>
57  /// <param name="newDataAvailableHandler">handler to be fired on new data available</param>
58  /// <returns>The new enumerator for this subscription request</returns>
59  public IEnumerator<BaseData> Subscribe(SubscriptionDataConfig dataConfig, EventHandler newDataAvailableHandler)
60  {
61  foreach (var dataHandler in DataHandlers)
62  {
63  // Emit ticks & custom data as soon as we get them, they don't need any kind of batching behavior applied to them
64  // only use the frontier time provider if we need to
65  var immediateEmission = dataConfig.Resolution == Resolution.Tick || dataConfig.IsCustomData || _frontierTimeProvider == null;
66  var exchangeTimeZone = dataConfig.ExchangeTimeZone;
67 
68  var enumerator = dataHandler.Subscribe(dataConfig, immediateEmission ? newDataAvailableHandler
69  : (sender, eventArgs) => {
70  // let's only wake up the main thread if the data point is allowed to be emitted, else we could fill forward previous bar and not let this one through
71  var dataAvailable = eventArgs as NewDataAvailableEventArgs;
72  if (dataAvailable == null || dataAvailable.DataPoint == null
73  || dataAvailable.DataPoint.EndTime.ConvertToUtc(exchangeTimeZone) <= _frontierTimeProvider.GetUtcNow())
74  {
75  newDataAvailableHandler?.Invoke(sender, eventArgs);
76  }
77  });
78 
79  // Check if the enumerator is not empty
80  if (enumerator != null)
81  {
82  if(!_dataConfigAndDataHandler.TryGetValue(dataConfig, out var dataQueueHandlers))
83  {
84  // we can get the same subscription request multiple times, the aggregator manager handles updating each enumerator
85  // but we need to keep track so we can call unsubscribe later to the target data queue handler
86  _dataConfigAndDataHandler[dataConfig] = dataQueueHandlers = new Queue<IDataQueueHandler>();
87  }
88  dataQueueHandlers.Enqueue(dataHandler);
89 
90  if (immediateEmission)
91  {
92  return enumerator;
93  }
94 
95  return new FrontierAwareEnumerator(enumerator, _frontierTimeProvider,
96  new TimeZoneOffsetProvider(exchangeTimeZone, _frontierTimeProvider.GetUtcNow(), Time.EndOfTime)
97  );
98  }
99  }
100 
101  // filter out warning for expected cases to reduce noise
102  if (!dataConfig.Symbol.Value.Contains("-UNIVERSE-", StringComparison.InvariantCultureIgnoreCase)
103  && dataConfig.Type != typeof(Delisting)
104  && !dataConfig.Symbol.IsCanonical())
105  {
106  UnsupportedConfiguration?.Invoke(this, dataConfig);
107  }
108  return null;
109  }
110 
111  /// <summary>
112  /// Removes the specified configuration
113  /// </summary>
114  /// <param name="dataConfig">Subscription config to be removed</param>
115  public virtual void Unsubscribe(SubscriptionDataConfig dataConfig)
116  {
117  if (_dataConfigAndDataHandler.TryGetValue(dataConfig, out var dataHandlers))
118  {
119  var dataHandler = dataHandlers.Dequeue();
120  dataHandler.Unsubscribe(dataConfig);
121 
122  if (dataHandlers.Count == 0)
123  {
124  // nothing left
125  _dataConfigAndDataHandler.Remove(dataConfig);
126  }
127  }
128  }
129 
130  /// <summary>
131  /// Sets the job we're subscribing for
132  /// </summary>
133  /// <param name="job">Job we're subscribing for</param>
134  public void SetJob(LiveNodePacket job)
135  {
136  var dataHandlersConfig = job.DataQueueHandler;
137  Log.Trace($"CompositeDataQueueHandler.SetJob(): will use {dataHandlersConfig}");
138  foreach (var dataHandlerName in dataHandlersConfig.DeserializeList())
139  {
140  var dataHandler = Composer.Instance.GetExportedValueByTypeName<IDataQueueHandler>(dataHandlerName);
141  dataHandler.SetJob(job);
142  DataHandlers.Add(dataHandler);
143  }
144 
145  InitializeFrontierTimeProvider();
146  }
147 
148  /// <summary>
149  /// Returns whether the data provider is connected
150  /// </summary>

152  public bool IsConnected => true;
153 
154  /// <summary>
155  /// Performs application-defined tasks associated with freeing, releasing, or resetting unmanaged resources.
156  /// </summary>
157  public void Dispose()
158  {
159  foreach (var dataHandler in DataHandlers)
160  {
161  dataHandler.Dispose();
162  }
163  }

165  /// <summary>
166  /// Method returns a collection of Symbols that are available at the data source.
167  /// </summary>
168  /// <param name="symbol">Symbol to lookup</param>
169  /// <param name="includeExpired">Include expired contracts</param>
170  /// <param name="securityCurrency">Expected security currency(if any)</param>
171  /// <returns>Enumerable of Symbols, that are associated with the provided Symbol</returns>
172  public IEnumerable<Symbol> LookupSymbols(Symbol symbol, bool includeExpired, string securityCurrency = null)
173  {
174  foreach (var dataHandler in GetUniverseProviders())
175  {
176  var result = dataHandler.LookupSymbols(symbol, includeExpired, securityCurrency).ToList();
177  if (result.Any())
178  {
179  return result;
180  }
181  }
182  return Enumerable.Empty<Symbol>();
183  }
184 
185  /// <summary>
186  /// Returns whether selection can take place or not.
187  /// </summary>
188  /// <remarks>This is useful to avoid a selection taking place during invalid times, for example IB reset times or when not connected,
189  /// because if allowed selection would fail since IB isn't running and would kill the algorithm</remarks>
190  /// <returns>True if selection can take place</returns>
191  public bool CanPerformSelection()
192  {
193  return GetUniverseProviders().Any(provider => provider.CanPerformSelection());
194  }
195 
196  /// <summary>
197  /// Creates the frontier time provider instance
198  /// </summary>
199  /// <remarks>Protected for testing purposes</remarks>
200  protected void InitializeFrontierTimeProvider()
201  {
202  var timeProviders = DataHandlers.OfType<ITimeProvider>().ToList();
203  if (timeProviders.Any())
204  {
205  Log.Trace($"DataQueueHandlerManager.InitializeFrontierTimeProvider(): will use the following IDQH frontier time providers: [{string.Join(",", timeProviders.Select(x => x.GetType()))}]");
206  _frontierTimeProvider = new CompositeTimeProvider(timeProviders);
207  }
208  }
209 
210  private IEnumerable<IDataQueueUniverseProvider> GetUniverseProviders()
211  {
212  var yielded = false;
213  foreach (var universeProvider in DataHandlers.OfType<IDataQueueUniverseProvider>())
214  {
215  yielded = true;
216  yield return universeProvider;
217  }
218 
219  if (!yielded)
220  {
221  throw new NotSupportedException("The DataQueueHandler does not support Options and Futures.");
222  }
223  }
224  }
