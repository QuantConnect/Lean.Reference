16 using System;
17 using Newtonsoft.Json;
18 using System.Collections.Generic;
19 
20 namespace QuantConnect.Orders
21 {
22  /// <summary>
23  /// Manager of a group of orders
24  /// </summary>
25  public class GroupOrderManager
26  {
27  /// <summary>
28  /// The unique order group Id
29  /// </summary>
30  public long Id { get; }
31 
32  /// <summary>
33  /// The group order quantity
34  /// </summary>
35  public decimal Quantity { get; internal set; }
36 
37  /// <summary>
38  /// The total order count associated with this order group
39  /// </summary>
40  public int Count { get; }
41 
42  /// <summary>
43  /// The limit price associated with this order group if any
44  /// </summary>
45  public decimal LimitPrice { get; set; }
46 
47  /// <summary>
48  /// The order Ids in this group
49  /// </summary>
50  /// <remarks>In live trading we process orders in a dedicated thread so we need to be thread safe</remarks>
51  public HashSet<int> OrderIds { get; }
52 
53  /// <summary>
54  /// Order Direction Property based off Quantity.
55  /// </summary>
56  public OrderDirection Direction
57  {
58  get
59  {
60  if (Quantity > 0)
61  {
62  return OrderDirection.Buy;
63  }
64  if (Quantity < 0)
65  {
66  return OrderDirection.Sell;
67  }
68  return OrderDirection.Hold;
69  }
70  }
71 
72  /// <summary>
73  /// Get the absolute quantity for this combo order
74  /// </summary>
75  [JsonIgnore]
76  public decimal AbsoluteQuantity => Math.Abs(Quantity);
77 
78  /// <summary>
79  /// Creates a new empty instance
80  /// </summary>
81  public GroupOrderManager()
82  {
83  }
84 
85  /// <summary>
86  /// Creates a new instance
87  /// </summary>
88  /// <param name="id">This order group unique Id</param>
89  /// <param name="legCount">The order leg count</param>
90  /// <param name="quantity">The group order quantity</param>
91  /// <param name="limitPrice">The limit price associated with this order group if any</param>
92  public GroupOrderManager(int id, int legCount, decimal quantity, decimal limitPrice = 0)
93  {
94  Id = id;
95  Count = legCount;
96  Quantity = quantity;
97  LimitPrice = limitPrice;
98  OrderIds = new (capacity: legCount);
99  }
100  }
101 }
