153  /// </summary>

168  Start = start;

178  /// <summary>
179  /// Updates this <see cref="RenkoBar"/> with the specified values and returns whether or not this bar is closed
180  /// </summary>
181  /// <param name="time">The current time</param>
182  /// <param name="currentValue">The current value</param>
183  /// <param name="volumeSinceLastUpdate">The volume since the last update called on this instance</param>
184  /// <returns>True if this bar <see cref="IsClosed"/></returns>
185  public bool Update(DateTime time, decimal currentValue, decimal volumeSinceLastUpdate)
186  {
187  if (Type == RenkoType.Wicked)
188  throw new InvalidOperationException("A \"Wicked\" RenkoBar cannot be updated!");
189 
190  // can't update a closed renko bar
191  if (IsClosed) return true;
192  if (Start == DateTime.MinValue) Start = time;
193  EndTime = time;
194 
195  // compute the min/max closes this renko bar can have
196  decimal lowClose = Open - BrickSize;
197  decimal highClose = Open + BrickSize;
198 
199  Close = Math.Min(highClose, Math.Max(lowClose, currentValue));
200  Volume += volumeSinceLastUpdate;
201 
202  // determine if this data caused the bar to close
203  if (currentValue <= lowClose || currentValue >= highClose)
204  {
205  IsClosed = true;
206  }
207 
208  if (Close > High) High = Close;
209  if (Close < Low) Low = Close;
210 
211  return IsClosed;
212  }
