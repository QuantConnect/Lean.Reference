17 using QuantConnect.Interfaces;

19 namespace QuantConnect.Data.Auxiliary
20 {
21  /// <summary>
22  /// Mapping extensions helper methods
23  /// </summary>
24  public static class MappingExtensions
25  {
26  /// <summary>
27  /// Helper method to resolve the mapping file to use.
28  /// </summary>
29  /// <remarks>This method is aware of the data type being added for <see cref="SecurityType.Base"/>
30  /// to the <see cref="SecurityIdentifier.Symbol"/> value</remarks>
31  /// <param name="mapFileProvider">The map file provider</param>
32  /// <param name="dataConfig">The configuration to fetch the map file for</param>
33  /// <returns>The mapping file to use</returns>
34  public static MapFile ResolveMapFile(this IMapFileProvider mapFileProvider, SubscriptionDataConfig dataConfig)
35  {
36  var resolver = MapFileResolver.Empty;
37  if(dataConfig.TickerShouldBeMapped())
38  {
39  resolver = mapFileProvider.Get(AuxiliaryDataKey.Create(dataConfig.Symbol));
40  }
41  return resolver.ResolveMapFile(dataConfig.Symbol , dataConfig.Type.Name, dataConfig.DataMappingMode);
42  }
43 
44  /// <summary>
45  /// Helper method to resolve the mapping file to use.
46  /// </summary>
47  /// <remarks>This method is aware of the data type being added for <see cref="SecurityType.Base"/>
48  /// to the <see cref="SecurityIdentifier.Symbol"/> value</remarks>
49  /// <param name="mapFileResolver">The map file resolver</param>
50  /// <param name="symbol">The symbol that we want to map</param>
51  /// <param name="dataType">The string data type name if any</param>
52  /// <param name="mappingMode">The mapping mode to use if any</param>
53  /// <returns>The mapping file to use</returns>
54  public static MapFile ResolveMapFile(this MapFileResolver mapFileResolver,
55  Symbol symbol,
56  string dataType = null,
57  DataMappingMode? mappingMode = null)
58  {
59  // Load the symbol and date to complete the mapFile checks in one statement
60  var symbolID = symbol.HasUnderlying ? symbol.Underlying.ID.Symbol : symbol.ID.Symbol;
61  if (dataType == null && symbol.SecurityType == SecurityType.Base)
62  {
63  symbol.ID.Symbol.TryGetCustomDataType(out dataType);
64  }
65  symbolID = symbol.SecurityType == SecurityType.Base && dataType != null ? symbolID.RemoveFromEnd($".{dataType}") : symbolID;
66 
67  MapFile result;
68  if (ReferenceEquals(mapFileResolver, MapFileResolver.Empty))
69  {
70  result = mapFileResolver.ResolveMapFile(symbol.Value, Time.BeginningOfTime);
71  }
72  else
73  {
74  var date = symbol.HasUnderlying ? symbol.Underlying.ID.Date : symbol.ID.Date;
75  result = mapFileResolver.ResolveMapFile(symbolID, date);
76  }
77  result.DataMappingMode = mappingMode;
78 
79  return result;
80  }
81  }
82 }
