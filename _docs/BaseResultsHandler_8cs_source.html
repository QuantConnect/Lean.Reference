276  /// <summary>
277  /// Gets the orders generated starting from the provided <see cref="ITransactionHandler.OrderEvents"/> position
278  /// </summary>
279  /// <returns>The delta orders</returns>
280  protected virtual Dictionary<int, Order> GetDeltaOrders(int orderEventsStartPosition, Func<int, bool> shouldStop)
281  {
282  var deltaOrders = new Dictionary<int, Order>();
283 
284  foreach (var orderId in TransactionHandler.OrderEvents.Skip(orderEventsStartPosition).Select(orderEvent => orderEvent.OrderId))
285  {
286  LastDeltaOrderPosition++;
287  if (deltaOrders.ContainsKey(orderId))
288  {
289  // we can have more than 1 order event per order id
290  continue;
291  }
292 
293  var order = Algorithm.Transactions.GetOrderById(orderId);
294  if (order == null)
295  {
296  // this shouldn't happen but just in case
297  continue;
298  }
299 
300  // for charting
301  order.Price = order.Price.SmartRounding();
302 
303  deltaOrders[orderId] = order;
304 
305  if (shouldStop(deltaOrders.Count))
306  {
307  break;
308  }
309  }
310 
311  return deltaOrders;
312  }

451  /// <summary>
452  /// Samples portfolio equity, benchmark, and daily performance
453  /// Called by scheduled event every night at midnight algorithm time
454  /// </summary>
455  /// <param name="time">Current UTC time in the AlgorithmManager loop</param>
456  public virtual void Sample(DateTime time)
457  {
458  var currentPortfolioValue = GetPortfolioValue();
459  var portfolioPerformance = DailyPortfolioValue == 0 ? 0 : Math.Round((currentPortfolioValue - DailyPortfolioValue) * 100 / DailyPortfolioValue, 10);
460 
461  // Update our max portfolio value
462  CumulativeMaxPortfolioValue = Math.Max(currentPortfolioValue, CumulativeMaxPortfolioValue);
463 
464  // Sample all our default charts
465  SampleEquity(time, currentPortfolioValue);
466  SampleBenchmark(time, GetBenchmarkValue(time));
467  SamplePerformance(time, portfolioPerformance);
468  SampleDrawdown(time, currentPortfolioValue);
469  SampleSalesVolume(time);
470  SampleExposure(time, currentPortfolioValue);
471  SampleCapacity(time);
472 
473  // Update daily portfolio value; works because we only call sample once a day
474  DailyPortfolioValue = currentPortfolioValue;
475  }

543  /// <summary>
544  /// Sample portfolio exposure long/short ratios by security type
545  /// </summary>
546  /// <param name="time">Time of the sample</param>
547  /// <param name="currentPortfolioValue">Current value of the portfolio</param>
548  protected virtual void SampleExposure(DateTime time, decimal currentPortfolioValue)
549  {
550  // Will throw in this case, just return without sampling
551  if (currentPortfolioValue == 0)
552  {
553  return;
554  }
555 
556  // Split up our holdings in one enumeration into long and shorts holding values
557  // only process those that we hold stock in.
558  var shortHoldings = new Dictionary<SecurityType, decimal>();
559  var longHoldings = new Dictionary<SecurityType, decimal>();
560  foreach (var holding in Algorithm.Portfolio.Values.Where(x => x.HoldStock))
561  {
562  // Ensure we have a value for this security type in both our dictionaries
563  if (!longHoldings.ContainsKey(holding.Symbol.SecurityType))
564  {
565  longHoldings.Add(holding.Symbol.SecurityType, 0);
566  shortHoldings.Add(holding.Symbol.SecurityType, 0);
567  }
568 
569  // Long Position
570  if (holding.HoldingsValue > 0)
571  {
572  longHoldings[holding.Symbol.SecurityType] += holding.HoldingsValue;
573  }
574  // Short Position
575  else
576  {
577  shortHoldings[holding.Symbol.SecurityType] += holding.HoldingsValue;
578  }
579  }
580 
581  // Sample our long and short positions
582  SampleExposureHelper(PositionSide.Long, time, currentPortfolioValue, longHoldings);
583  SampleExposureHelper(PositionSide.Short, time, currentPortfolioValue, shortHoldings);
584  }

632  /// <summary>
633  /// Gets the algorithm runtime statistics
634  /// </summary>
635  protected SortedDictionary<string, string> GetAlgorithmRuntimeStatistics(Dictionary<string, string> summary, CapacityEstimate capacityEstimate = null)
636  {
637  var runtimeStatistics = new SortedDictionary<string, string>();
638  lock (RuntimeStatistics)
639  {
640  foreach (var pair in RuntimeStatistics)
641  {
642  runtimeStatistics.Add(pair.Key, pair.Value);
643  }
644  }
645 
646  if (summary.ContainsKey("Probabilistic Sharpe Ratio"))
647  {
648  runtimeStatistics["Probabilistic Sharpe Ratio"] = summary["Probabilistic Sharpe Ratio"];
649  }
650  else
651  {
652  runtimeStatistics["Probabilistic Sharpe Ratio"] = "0%";
653  }
654 
655  runtimeStatistics["Unrealized"] = AlgorithmCurrencySymbol + Algorithm.Portfolio.TotalUnrealizedProfit.ToStringInvariant("N2");
656  runtimeStatistics["Fees"] = $"-{AlgorithmCurrencySymbol}{Algorithm.Portfolio.TotalFees.ToStringInvariant("N2")}";
657  runtimeStatistics["Net Profit"] = AlgorithmCurrencySymbol + Algorithm.Portfolio.TotalProfit.ToStringInvariant("N2");
658  runtimeStatistics["Return"] = GetNetReturn().ToStringInvariant("P");
659  runtimeStatistics["Equity"] = AlgorithmCurrencySymbol + Algorithm.Portfolio.TotalPortfolioValue.ToStringInvariant("N2");
660  runtimeStatistics["Holdings"] = AlgorithmCurrencySymbol + Algorithm.Portfolio.TotalHoldingsValue.ToStringInvariant("N2");
661  runtimeStatistics["Volume"] = AlgorithmCurrencySymbol + Algorithm.Portfolio.TotalSaleVolume.ToStringInvariant("N2");
662  if (capacityEstimate != null)
663  {
664  runtimeStatistics["Capacity"] = AlgorithmCurrencySymbol + capacityEstimate.Capacity.RoundToSignificantDigits(2).ToFinancialFigures();
665  }
666 
667  return runtimeStatistics;
668  }

670  /// <summary>
671  /// Will generate the statistics results and update the provided runtime statistics
672  /// </summary>
673  protected StatisticsResults GenerateStatisticsResults(Dictionary<string, Chart> charts,
674  SortedDictionary<DateTime, decimal> profitLoss = null, CapacityEstimate estimatedStrategyCapacity = null)
675  {
676  var statisticsResults = new StatisticsResults();
677  if (profitLoss == null)
678  {
679  profitLoss = new SortedDictionary<DateTime, decimal>();
680  }
681 
682  try
683  {
684  //Generates error when things don't exist (no charting logged, runtime errors in main algo execution)
685  const string strategyEquityKey = "Strategy Equity";
686  const string equityKey = "Equity";
687  const string dailyPerformanceKey = "Daily Performance";
688  const string benchmarkKey = "Benchmark";
689 
690  // make sure we've taken samples for these series before just blindly requesting them
691  if (charts.ContainsKey(strategyEquityKey) &&
692  charts[strategyEquityKey].Series.ContainsKey(equityKey) &&
693  charts[strategyEquityKey].Series.ContainsKey(dailyPerformanceKey) &&
694  charts.ContainsKey(benchmarkKey) &&
695  charts[benchmarkKey].Series.ContainsKey(benchmarkKey))
696  {
697  var equity = charts[strategyEquityKey].Series[equityKey].Values;
698  var performance = charts[strategyEquityKey].Series[dailyPerformanceKey].Values;
699  var totalTransactions = Algorithm.Transactions.GetOrders(x => x.Status.IsFill()).Count();
700  var benchmark = charts[benchmarkKey].Series[benchmarkKey].Values;
701 
702  var trades = Algorithm.TradeBuilder.ClosedTrades;
703

733  private void ProcessAlgorithmLogsImpl(ConcurrentQueue<string> concurrentQueue, PacketType packetType, int? messageQueueLimit = null)
734  {
735  if (concurrentQueue.IsEmpty)
736  {
737  return;
738  }
739 
740  var endTime = DateTime.UtcNow.AddMilliseconds(250).Ticks;
741  var currentMessageCount = -1;
742  while (DateTime.UtcNow.Ticks < endTime && concurrentQueue.TryDequeue(out var message))
743  {
744  if (messageQueueLimit.HasValue)
745  {
746  if (currentMessageCount == -1)
747  {
748  // this is expensive, so let's get it once
749  currentMessageCount = Messages.Count;
750  }
751  if (currentMessageCount > messageQueueLimit)
752  {
753  if (!_packetDroppedWarning)
754  {
755  _packetDroppedWarning = true;
756  // this shouldn't happen in most cases, queue limit is high and consumed often but just in case let's not silently drop packets without a warning
757  Messages.Enqueue(new HandledErrorPacket(AlgorithmId, "Your algorithm messaging has been rate limited to prevent browser flooding."));
758  }
759  //if too many in the queue already skip the logging and drop the messages
760  continue;
761  }
762  }
763 
764  if (packetType == PacketType.Debug)
765  {
766  Messages.Enqueue(new DebugPacket(ProjectId, AlgorithmId, CompileId, message));
767  }
768  else if (packetType == PacketType.Log)
769  {
770  Messages.Enqueue(new LogPacket(AlgorithmId, message));
771  }
772  else if (packetType == PacketType.HandledError)
773  {
774  Messages.Enqueue(new HandledErrorPacket(AlgorithmId, message));
775  }
776  AddToLogStore(message);
777 
778  // increase count after we add
779  currentMessageCount++;
780  }
781  }
782  }
