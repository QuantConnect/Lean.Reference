364  /// <summary>
365  /// Gets the orders generated starting from the provided <see cref="ITransactionHandler.OrderEvents"/> position
366  /// </summary>
367  /// <returns>The delta orders</returns>
368  protected virtual Dictionary<int, Order> GetDeltaOrders(int orderEventsStartPosition, Func<int, bool> shouldStop)
369  {
370  var deltaOrders = new Dictionary<int, Order>();
371 
372  foreach (var orderId in TransactionHandler.OrderEvents.Skip(orderEventsStartPosition).Select(orderEvent => orderEvent.OrderId))
373  {
374  LastDeltaOrderPosition++;
375  if (deltaOrders.ContainsKey(orderId))
376  {
377  // we can have more than 1 order event per order id
378  continue;
379  }
380 
381  var order = Algorithm.Transactions.GetOrderById(orderId);
382  if (order == null)
383  {
384  // this shouldn't happen but just in case
385  continue;
386  }
387 
388  // for charting
389  order.Price = order.Price.SmartRounding();
390 
391  deltaOrders[orderId] = order;
392 
393  if (shouldStop(deltaOrders.Count))
394  {
395  break;
396  }
397  }
398 
399  return deltaOrders;
400  }

529  /// <summary>
530  /// Samples portfolio equity, benchmark, and daily performance
531  /// Called by scheduled event every night at midnight algorithm time
532  /// </summary>
533  /// <param name="time">Current UTC time in the AlgorithmManager loop</param>
534  public virtual void Sample(DateTime time)
535  {
536  var currentPortfolioValue = GetPortfolioValue();
537  var portfolioPerformance = DailyPortfolioValue == 0 ? 0 : Math.Round((currentPortfolioValue - DailyPortfolioValue) * 100 / DailyPortfolioValue, 10);
538 
539  // Update our max portfolio value
540  CumulativeMaxPortfolioValue = Math.Max(currentPortfolioValue, CumulativeMaxPortfolioValue);
541 
542  // Sample all our default charts
543  UpdateAlgorithmEquity();
544  SampleEquity(time);
545  SampleBenchmark(time, GetBenchmarkValue(time));
546  SamplePerformance(time, portfolioPerformance);
547  SampleDrawdown(time, currentPortfolioValue);
548  SampleSalesVolume(time);
549  SampleExposure(time, currentPortfolioValue);
550  SampleCapacity(time);
551  SamplePortfolioTurnover(time, currentPortfolioValue);
552  SamplePortfolioMargin(time, currentPortfolioValue);
553 
554  // Update daily portfolio value; works because we only call sample once a day
555  DailyPortfolioValue = currentPortfolioValue;
556  }

627  /// <summary>
628  /// Sample portfolio turn over of the strategy
629  /// </summary>
630  /// <param name="time">Time of the sample</param>
631  /// <param name="currentPortfolioValue">Current equity value</param>
632  protected virtual void SamplePortfolioTurnover(DateTime time, decimal currentPortfolioValue)
633  {
634  if(currentPortfolioValue != 0)
635  {
636  if(Algorithm.StartDate == time.ConvertFromUtc(Algorithm.TimeZone))
637  {
638  // the first sample in backtesting is at start, we only want to sample after a full algorithm execution date
639  return;
640  }
641  var currentTotalSaleVolume = Algorithm.Portfolio.TotalSaleVolume;
642 
643  decimal todayPortfolioTurnOver;
644  if (_previousPortfolioTurnoverSample == time)
645  {
646  // we are sampling the same time twice, this can happen if we sample at the start of the portfolio loop
647  // and the algorithm happen to end at the same time and we trigger the final sample to take into account that last loop
648  // this new sample will overwrite the previous, so we resample using T-2 sales volume
649  todayPortfolioTurnOver = (currentTotalSaleVolume - _previousSalesVolume[1]) / currentPortfolioValue;
650  }
651  else
652  {
653  todayPortfolioTurnOver = (currentTotalSaleVolume - _previousSalesVolume[0]) / currentPortfolioValue;
654  }
655 
656  _previousSalesVolume.Add(currentTotalSaleVolume);
657  _previousPortfolioTurnoverSample = time;
658 
659  Sample(PortfolioTurnoverKey, PortfolioTurnoverKey, 0, SeriesType.Line, new ChartPoint(time, todayPortfolioTurnOver), "%");
660  }
661  }

678  /// <summary>
679  /// Sample portfolio exposure long/short ratios by security type
680  /// </summary>
681  /// <param name="time">Time of the sample</param>
682  /// <param name="currentPortfolioValue">Current value of the portfolio</param>
683  protected virtual void SampleExposure(DateTime time, decimal currentPortfolioValue)
684  {
685  // Will throw in this case, just return without sampling
686  if (currentPortfolioValue == 0)
687  {
688  return;
689  }
690 
691  // Split up our holdings in one enumeration into long and shorts holding values
692  // only process those that we hold stock in.
693  var shortHoldings = new Dictionary<SecurityType, decimal>();
694  var longHoldings = new Dictionary<SecurityType, decimal>();
695  foreach (var holding in Algorithm.Portfolio.Values)
696  {
697  // Ensure we have a value for this security type in both our dictionaries
698  if (!longHoldings.ContainsKey(holding.Symbol.SecurityType))
699  {
700  longHoldings.Add(holding.Symbol.SecurityType, 0);
701  shortHoldings.Add(holding.Symbol.SecurityType, 0);
702  }
703 
704  var holdingsValue = holding.HoldingsValue;
705  if (holdingsValue == 0)
706  {
707  continue;
708  }
709 
710  // Long Position
711  if (holdingsValue > 0)
712  {
713  longHoldings[holding.Symbol.SecurityType] += holdingsValue;
714  }
715  // Short Position
716  else
717  {
718  shortHoldings[holding.Symbol.SecurityType] += holdingsValue;
719  }
720  }

936  /// <summary>
937  /// Processes algorithm logs.
938  /// Logs of the same type are batched together one per line and are sent out
939  /// </summary>
940  protected void ProcessAlgorithmLogs(int? messageQueueLimit = null)
941  {
942  ProcessAlgorithmLogsImpl(Algorithm.DebugMessages, PacketType.Debug, messageQueueLimit);
943  ProcessAlgorithmLogsImpl(Algorithm.ErrorMessages, PacketType.HandledError, messageQueueLimit);
944  ProcessAlgorithmLogsImpl(Algorithm.LogMessages, PacketType.Log, messageQueueLimit);

946 
947  private void ProcessAlgorithmLogsImpl(ConcurrentQueue<string> concurrentQueue, PacketType packetType, int? messageQueueLimit = null)
948  {
949  if (concurrentQueue.IsEmpty)
950  {
951  return;
952  }
953 
954  var endTime = DateTime.UtcNow.AddMilliseconds(250).Ticks;
955  var currentMessageCount = -1;
956  while (DateTime.UtcNow.Ticks < endTime && concurrentQueue.TryDequeue(out var message))
957  {
958  if (messageQueueLimit.HasValue)
959  {
960  if (currentMessageCount == -1)
961  {
962  // this is expensive, so let's get it once
963  currentMessageCount = Messages.Count;
964  }
965  if (currentMessageCount > messageQueueLimit)
966  {
967  if (!_packetDroppedWarning)
968  {
969  _packetDroppedWarning = true;
970  // this shouldn't happen in most cases, queue limit is high and consumed often but just in case let's not silently drop packets without a warning
971  Messages.Enqueue(new HandledErrorPacket(AlgorithmId, "Your algorithm messaging has been rate limited to prevent browser flooding."));
972  }
973  //if too many in the queue already skip the logging and drop the messages
974  continue;
975  }
976  }
977 
978  if (packetType == PacketType.Debug)
979  {
980  Messages.Enqueue(new DebugPacket(ProjectId, AlgorithmId, CompileId, message));
981  }
982  else if (packetType == PacketType.Log)
983  {
984  Messages.Enqueue(new LogPacket(AlgorithmId, message));
985  }
986  else if (packetType == PacketType.HandledError)
987  {
988  Messages.Enqueue(new HandledErrorPacket(AlgorithmId, message));
989  }
990  AddToLogStore(message);
991 
992  // increase count after we add
993  currentMessageCount++;
994  }
995  }
