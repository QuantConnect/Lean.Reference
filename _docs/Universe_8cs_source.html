144  /// <summary>
145  /// Determines whether or not the specified security can be removed from
146  /// this universe. This is useful to prevent securities from being taken
147  /// out of a universe before the algorithm has had enough time to make
148  /// decisions on the security
149  /// </summary>
150  /// <param name="utcTime">The current utc time</param>
151  /// <param name="security">The security to check if its ok to remove</param>
152  /// <returns>True if we can remove the security, false otherwise</returns>
153  public virtual bool CanRemoveMember(DateTime utcTime, Security security)
154  {
155  // can always remove securities after dispose requested
156  if (DisposeRequested)
157  {
158  return true;
159  }
160 
161  // can always remove delisted securities from the universe
162  if (security.IsDelisted)
163  {
164  return true;
165  }
166 
167  Member member;
168  if (Securities.TryGetValue(security.Symbol, out member))
169  {
170  if (_minimumTimeInUniverseRoundingInterval == null)
171  {
172  // lets set _minimumTimeInUniverseRoundingInterval once
173  _minimumTimeInUniverseRoundingInterval = UniverseSettings.MinimumTimeInUniverse;
174  AdjustMinimumTimeInUniverseRoundingInterval();
175  }
176 
177  var timeInUniverse = utcTime - member.Added;
178  if (timeInUniverse.Round(_minimumTimeInUniverseRoundingInterval.Value)
179  >= UniverseSettings.MinimumTimeInUniverse)
180  {
181  return true;
182  }
183  }
184  return false;
185  }

187  /// <summary>
188  /// Performs universe selection using the data specified
189  /// </summary>
190  /// <param name="utcTime">The current utc time</param>
191  /// <param name="data">The symbols to remain in the universe</param>
192  /// <returns>The data that passes the filter</returns>
193  public IEnumerable<Symbol> PerformSelection(DateTime utcTime, BaseDataCollection data)
194  {
195  // select empty set of symbols after dispose requested
196  if (DisposeRequested)
197  {
198  OnSelectionChanged();
199  return Enumerable.Empty<Symbol>();
200  }
201 
202  var selections = data.FilteredContracts;
203  if (selections == null)
204  {
205  // only trigger selection if it hasn't already been run
206  var result = SelectSymbols(utcTime, data);
207  if (ReferenceEquals(result, Unchanged))
208  {
209  data.FilteredContracts = _previousSelections;
210  return Unchanged;
211  }
212 
213  selections = result.ToHashSet();
214  data.FilteredContracts = selections;
215  }
216 
217  var hasDiffs = _previousSelections.AreDifferent(selections);
218  _previousSelections = selections;
219  if (!hasDiffs)
220  {
221  return Unchanged;
222  }
223 
224  OnSelectionChanged(selections);
225  return selections;
226  }

306  /// <summary>
307  /// Adds the specified security to this universe
308  /// </summary>
309  /// <param name="utcTime">The current utc date time</param>
310  /// <param name="security">The security to be added</param>
311  /// <param name="isInternal">True if internal member</param>
312  /// <returns>True if the security was successfully added,
313  /// false if the security was already in the universe</returns>
314  internal virtual bool AddMember(DateTime utcTime, Security security, bool isInternal)
315  {
316  // never add members to disposed universes
317  if (DisposeRequested)
318  {
319  return false;
320  }
321 
322  if (security.IsDelisted)
323  {
324  return false;
325  }
326 
327  return Securities.TryAdd(security.Symbol, new Member(utcTime, security, isInternal));
328  }

330  /// <summary>
331  /// Tries to remove the specified security from the universe. This
332  /// will first check to verify that we can remove the security by
333  /// calling the <see cref="CanRemoveMember"/> function.
334  /// </summary>
335  /// <param name="utcTime">The current utc time</param>
336  /// <param name="security">The security to be removed</param>
337  /// <returns>True if the security was successfully removed, false if
338  /// we're not allowed to remove or if the security didn't exist</returns>
339  internal virtual bool RemoveMember(DateTime utcTime, Security security)
340  {
341  if (CanRemoveMember(utcTime, security))
342  {
343  Member member;
344  return Securities.TryRemove(security.Symbol, out member);
345  }
346  return false;
347  }

382  /// <summary>
383  /// Will adjust the <see cref="_minimumTimeInUniverseRoundingInterval"/>
384  /// so rounding is performed as expected
385  /// </summary>
386  private void AdjustMinimumTimeInUniverseRoundingInterval()
387  {
388  if (_minimumTimeInUniverseRoundingInterval >= Time.OneDay)
389  {
390  _minimumTimeInUniverseRoundingInterval = Time.OneDay;
391  }
392  else if (_minimumTimeInUniverseRoundingInterval >= Time.OneHour)
393  {
394  _minimumTimeInUniverseRoundingInterval = Time.OneHour;
395  }
396  else if (_minimumTimeInUniverseRoundingInterval >= Time.OneMinute)
397  {
398  _minimumTimeInUniverseRoundingInterval = Time.OneMinute;
399  }
400  else if (_minimumTimeInUniverseRoundingInterval >= Time.OneSecond)
401  {
402  _minimumTimeInUniverseRoundingInterval = Time.OneSecond;
403  }
404  }
