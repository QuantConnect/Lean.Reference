16 using Deedle;
17 using QuantConnect.Packets;
18 using System;
19 using System.Collections.Generic;
20 using System.Globalization;
21 using System.Linq;
22 
23 namespace QuantConnect.Report.ReportElements
24 {
25  public class SharpeRatioReportElement : ReportElement
26  {
27  /// <summary>
28  /// The number of trading days per year to get better result of statistics
29  /// </summary>
30  private double _tradingDaysPerYear;
31 
32  /// <summary>
33  /// Live result object
34  /// </summary>
35  protected LiveResult LiveResult { get; }
36 
37  /// <summary>
38  /// Backtest result object
39  /// </summary>
40  protected BacktestResult BacktestResult { get; }
41 
42  /// <summary>
43  /// Sharpe Ratio from a backtest
44  /// </summary>
45  public virtual decimal? BacktestResultValue => BacktestResult?.TotalPerformance?.PortfolioStatistics?.SharpeRatio;
46 
47  /// <summary>
48  /// Estimate the sharpe ratio of the strategy.
49  /// </summary>
50  /// <param name="name">Name of the widget</param>
51  /// <param name="key">Location of injection</param>
52  /// <param name="backtest">Backtest result object</param>
53  /// <param name="live">Live result object</param>
54  /// <param name="tradingDaysPerYear">The number of trading days per year to get better result of statistics</param>
55  public SharpeRatioReportElement(string name, string key, BacktestResult backtest, LiveResult live, int tradingDaysPerYear)
56  {
57  LiveResult = live;
58  BacktestResult = backtest;
59  Name = name;
60  Key = key;
61  _tradingDaysPerYear = Convert.ToDouble(tradingDaysPerYear, CultureInfo.InvariantCulture);
62  }
63 
64  /// <summary>
65  /// The generated output string to be injected
66  /// </summary>
67  public override string Render()
68  {
69  if (LiveResult == null)
70  {
71  Result = BacktestResultValue;
72  return BacktestResultValue?.ToString("F1") ?? "-";
73  }
74 
75  var equityPoints = ResultsUtil.EquityPoints(LiveResult);
76  var performance = DeedleUtil.PercentChange(new Series<DateTime, double>(equityPoints).ResampleEquivalence(date => date.Date, s => s.LastValue()));
77  if (performance.ValueCount == 0)
78  {
79  return "-";
80  }
81 
82  var sixMonthsAgo = performance.LastKey().AddDays(-180);
83  var trailingPerformance = performance.Where(series => series.Key >= sixMonthsAgo && series.Key.DayOfWeek != DayOfWeek.Saturday && series.Key.DayOfWeek != DayOfWeek.Sunday)
84  .Values
85  .ToList();
86 
87  var annualStandardDeviation = trailingPerformance.Count < 7 ? 0 : GetAnnualStandardDeviation(trailingPerformance, _tradingDaysPerYear);
88  if (annualStandardDeviation <= 0)
89  {
90  return "-";
91  }
92 
93  var annualPerformance = Statistics.Statistics.AnnualPerformance(trailingPerformance, _tradingDaysPerYear);
94  var liveResultValue = Statistics.Statistics.SharpeRatio(annualPerformance, annualStandardDeviation, 0.0);
95  Result = liveResultValue;
96  return liveResultValue.ToString("F2");
97  }
98 
99  /// <summary>
100  /// Get annual standard deviation
101  /// </summary>
102  /// <param name="trailingPerformance">The performance for the last period</param>
103  /// <param name="tradingDaysPerYear">The number of trading days per year to get better result of statistics</param>
104  /// <returns>Annual standard deviation.</returns>
105  public virtual double GetAnnualStandardDeviation(List<double> trailingPerformance, double tradingDaysPerYear)
106  {
107  return Statistics.Statistics.AnnualStandardDeviation(trailingPerformance, tradingDaysPerYear);
108  }
109  }
110 }
