26  /// <summary>

28  /// </summary>

31  /// <summary>

36  {
37  }
38 
39  /// <summary>
40  /// True if this subscription request should be streamed
41  /// </summary>
42  public virtual bool ShouldStreamSubscription(SubscriptionDataConfig config)
43  {
44  return IsStreamingType(config) || !config.IsCustomData && config.Type != typeof(CoarseFundamental) && config.Type != typeof(Fundamental) && config.Type != typeof(MarginInterestRate);
45  }
46 
47  /// <summary>
48  /// Returns true if the data type for the given subscription configuration supports streaming
49  /// </summary>
50  protected static bool IsStreamingType(SubscriptionDataConfig configuration)
51  {
52  var dataTypeInstance = configuration.Type.GetBaseDataInstance();
53  return dataTypeInstance.GetSource(configuration, DateTime.UtcNow, true)
54  .TransportMedium == SubscriptionTransportMedium.Streaming;
55  }
56  }
