95  /// <summary>
96  /// The main processing method steps through the messaging queue and processes the messages one by one.
97  /// </summary>
98  protected override void Run()
99  {
100  try
101  {
102  while (!(ExitTriggered && Messages.IsEmpty))
103  {
104  //While there's no work to do, go back to the algorithm:
105  if (Messages.IsEmpty)
106  {
107  ExitEvent.WaitOne(50);
108  }
109  else
110  {
111  //1. Process Simple Messages in Queue
112  Packet packet;
113  if (Messages.TryDequeue(out packet))
114  {
115  MessagingHandler.Send(packet);
116  }
117  }
118 
119  //2. Update the packet scanner:
120  Update();
121 
122  } // While !End.
123  }
124  catch (Exception err)
125  {
126  // unexpected error, we need to close down shop
127  Algorithm.SetRuntimeError(err, "ResultHandler");
128  }
129 
130  Log.Trace("BacktestingResultHandler.Run(): Ending Thread...");
131  } // End Run();

133  /// <summary>
134  /// Send a backtest update to the browser taking a latest snapshot of the charting data.
135  /// </summary>
136  private void Update()
137  {
138  try
139  {
140  //Sometimes don't run the update, if not ready or we're ending.
141  if (Algorithm?.Transactions == null || ExitTriggered || !Algorithm.GetLocked())
142  {
143  return;
144  }
145 
146  var utcNow = DateTime.UtcNow;
147  if (utcNow <= _nextUpdate || _progressMonitor.ProcessedDays < _daysProcessedFrontier) return;
148 
149  var deltaOrders = GetDeltaOrders(LastDeltaOrderPosition, shouldStop: orderCount => orderCount >= 50);
150  // Deliberately skip to the end of order event collection to prevent overloading backtesting UX
151  LastDeltaOrderPosition = TransactionHandler.OrderEvents.Count();
152 
153  //Reset loop variables:
154  try
155  {
156  _daysProcessedFrontier = _progressMonitor.ProcessedDays + 1;
157  _nextUpdate = utcNow.AddSeconds(3);
158  }
159  catch (Exception err)
160  {
161  Log.Error(err, "Can't update variables");
162  }
163 
164  var deltaCharts = new Dictionary<string, Chart>();
165  var serverStatistics = GetServerStatistics(utcNow);
166  var performanceCharts = new Dictionary<string, Chart>();
167 
168  // Process our charts updates
169  lock (ChartLock)
170  {
171  foreach (var kvp in Charts)
172  {
173  var chart = kvp.Value;
174 
175  // Get a copy of this chart with updates only since last request
176  var updates = chart.GetUpdates();
177  if (!updates.IsEmpty())
178  {
179  deltaCharts.Add(chart.Name, updates);
180  }
181 
182  // Update our algorithm performance charts
183  if (AlgorithmPerformanceCharts.Contains(kvp.Key))
184  {
185  performanceCharts[kvp.Key] = chart.Clone();
186  }
187 
188  if (updates.Name == PortfolioMarginKey)
189  {
190  PortfolioMarginChart.RemoveSinglePointSeries(updates);
191  }
192  }
193  }
194 
195  //Get the runtime statistics from the user algorithm:
196  var summary = GenerateStatisticsResults(performanceCharts, estimatedStrategyCapacity: _capacityEstimate).Summary;
197  var runtimeStatistics = GetAlgorithmRuntimeStatistics(summary, _capacityEstimate);
198 
199  var progress = _progressMonitor.Progress;
200 
201  //1. Cloud Upload -> Upload the whole packet to S3 Immediately:
202  if (utcNow > _nextS3Update)
203  {
204  // For intermediate backtesting results, we truncate the order list to include only the last 100 orders
205  // The final packet will contain the full list of orders.
206  const int maxOrders = 100;
207  var orderCount = TransactionHandler.Orders.Count;
208 
209  var completeResult = new BacktestResult(new BacktestResultParameters(
210  Charts,
211  orderCount > maxOrders ? TransactionHandler.Orders.Skip(orderCount - maxOrders).ToDictionary() : TransactionHandler.Orders.ToDictionary(),
212  Algorithm.Transactions.TransactionRecord,
213  new Dictionary<string, string>(),
214  runtimeStatistics,
215  new Dictionary<string, AlgorithmPerformance>(),
216  // we store the last 100 order events, the final packet will contain the full list
217  TransactionHandler.OrderEvents.Reverse().Take(100).ToList(), state: GetAlgorithmState()));
218 
219  StoreResult(new BacktestResultPacket(_job, completeResult, Algorithm.EndDate, Algorithm.StartDate, progress));
220 
221  _nextS3Update = DateTime.UtcNow.AddSeconds(30);
222  }
223 
224  //2. Backtest Update -> Send the truncated packet to the backtester:
225  var splitPackets = SplitPackets(deltaCharts, deltaOrders, runtimeStatistics, progress, serverStatistics);
226 
227  foreach (var backtestingPacket in splitPackets)
228  {
229  MessagingHandler.Send(backtestingPacket);
230  }
231 
232  // let's re update this value after we finish just in case, so we don't re enter in the next loop
233  _nextUpdate = DateTime.UtcNow.Add(MainUpdateInterval);
234  }
235  catch (Exception err)
236  {
237  Log.Error(err);
238  }
239  }

241  /// <summary>
242  /// Run over all the data and break it into smaller packets to ensure they all arrive at the terminal
243  /// </summary>
244  public virtual IEnumerable<BacktestResultPacket> SplitPackets(Dictionary<string, Chart> deltaCharts, Dictionary<int, Order> deltaOrders, SortedDictionary<string, string> runtimeStatistics, decimal progress, Dictionary<string, string> serverStatistics)
245  {
246  // break the charts into groups
247  var splitPackets = new List<BacktestResultPacket>();
248  foreach (var chart in deltaCharts.Values)
249  {
250  splitPackets.Add(new BacktestResultPacket(_job, new BacktestResult
251  {
252  Charts = new Dictionary<string, Chart>
253  {
254  {chart.Name, chart}
255  }
256  }, Algorithm.EndDate, Algorithm.StartDate, progress));
257  }
258 
259  // only send orders if there is actually any update
260  if (deltaOrders.Count > 0)
261  {
262  // Add the orders into the charting packet:
263  splitPackets.Add(new BacktestResultPacket(_job, new BacktestResult { Orders = deltaOrders }, Algorithm.EndDate, Algorithm.StartDate, progress));
264  }
265 
266  //Add any user runtime statistics into the backtest.
267  splitPackets.Add(new BacktestResultPacket(_job, new BacktestResult { ServerStatistics = serverStatistics, RuntimeStatistics = runtimeStatistics }, Algorithm.EndDate, Algorithm.StartDate, progress));
268 
269 
270  return splitPackets;
271  }

273  /// <summary>
274  /// Save the snapshot of the total results to storage.
275  /// </summary>
276  /// <param name="packet">Packet to store.</param>
277  protected override void StoreResult(Packet packet)
278  {
279  try
280  {
281  // Make sure this is the right type of packet:
282  if (packet.Type != PacketType.BacktestResult) return;
283 
284  // Port to packet format:
285  var result = packet as BacktestResultPacket;
286 
287  if (result != null)
288  {
289  // Get Storage Location:
290  var key = $"{AlgorithmId}.json";
291 
292  BacktestResult results;
293  lock (ChartLock)
294  {
295  results = new BacktestResult(new BacktestResultParameters(
296  result.Results.Charts.ToDictionary(x => x.Key, x => x.Value.Clone()),
297  result.Results.Orders,
298  result.Results.ProfitLoss,
299  result.Results.Statistics,
300  result.Results.RuntimeStatistics,
301  result.Results.RollingWindow,
302  null, // null order events, we store them separately
303  result.Results.TotalPerformance,
304  result.Results.AlgorithmConfiguration,
305  result.Results.State));
306 
307  if (result.Results.Charts.TryGetValue(PortfolioMarginKey, out var marginChart))
308  {
309  PortfolioMarginChart.RemoveSinglePointSeries(marginChart);
310  }
311  }
312  // Save results
313  SaveResults(key, results);
314 
315  // Store Order Events in a separate file
316  StoreOrderEvents(Algorithm?.UtcTime ?? DateTime.UtcNow, result.Results.OrderEvents);
317  }
318  else
319  {
320  Log.Error("BacktestingResultHandler.StoreResult(): Result Null.");
321  }
322  }
323  catch (Exception err)
324  {
325  Log.Error(err);
326  }

329  /// <summary>
330  /// Send a final analysis result back to the IDE.
331  /// </summary>
332  protected void SendFinalResult()
333  {
334  try
335  {
336  var endTime = DateTime.UtcNow;
337  BacktestResultPacket result;
338  // could happen if algorithm failed to init
339  if (Algorithm != null)
340  {
341  //Convert local dictionary:
342  var charts = new Dictionary<string, Chart>(Charts);
343  var orders = new Dictionary<int, Order>(TransactionHandler.Orders);
344  var profitLoss = new SortedDictionary<DateTime, decimal>(Algorithm.Transactions.TransactionRecord);
345  var statisticsResults = GenerateStatisticsResults(charts, profitLoss, _capacityEstimate);
346  var runtime = GetAlgorithmRuntimeStatistics(statisticsResults.Summary, capacityEstimate: _capacityEstimate);
347 
348  FinalStatistics = statisticsResults.Summary;
349 
350  // clear the trades collection before placing inside the backtest result
351  foreach (var ap in statisticsResults.RollingPerformances.Values)
352  {
353  ap.ClosedTrades.Clear();
354  }
355  var orderEvents = TransactionHandler.OrderEvents.ToList();
356  //Create a result packet to send to the browser.
357  result = new BacktestResultPacket(_job,
358  new BacktestResult(new BacktestResultParameters(charts, orders, profitLoss, statisticsResults.Summary, runtime,
359  statisticsResults.RollingPerformances, orderEvents, statisticsResults.TotalPerformance,
360  AlgorithmConfiguration.Create(Algorithm, _job), GetAlgorithmState(endTime))),
361  Algorithm.EndDate, Algorithm.StartDate);
362  }
363  else
364  {
365  result = BacktestResultPacket.CreateEmpty(_job);
366  result.Results.State = GetAlgorithmState(endTime);
367  }
368 
369  result.ProcessingTime = (endTime - StartTime).TotalSeconds;
370  result.DateFinished = DateTime.Now;
371  result.Progress = 1;
372 
373  StoreInsights();
374 
375  //Place result into storage.
376  StoreResult(result);
377 
378  result.Results.ServerStatistics = GetServerStatistics(endTime);
379  //Second, send the truncated packet:
380  MessagingHandler.Send(result);
381 
382  Log.Trace("BacktestingResultHandler.SendAnalysisResult(): Processed final packet");
383  }
384  catch (Exception err)
385  {
386  Log.Error(err);
387  }
388  }

391  /// Set the Algorithm instance for ths result.
392  /// </summary>

394  /// <param name="startingPortfolioValue">Algorithm starting capital for statistics calculations</param>
395  /// <remarks>While setting the algorithm the backtest result handler.</remarks>
396  public virtual void SetAlgorithm(IAlgorithm algorithm, decimal startingPortfolioValue)
397  {
398  Algorithm = algorithm;
399  Algorithm.SetStatisticsService(this);
400  State["Name"] = Algorithm.Name;
401  StartingPortfolioValue = startingPortfolioValue;
402  DailyPortfolioValue = StartingPortfolioValue;
403  CumulativeMaxPortfolioValue = StartingPortfolioValue;
404  AlgorithmCurrencySymbol = Currencies.GetCurrencySymbol(Algorithm.AccountCurrency);
405  _capacityEstimate = new CapacityEstimate(Algorithm);
406  _progressMonitor = new BacktestProgressMonitor(Algorithm.TimeKeeper, Algorithm.EndDate);
407 
408  //Get the resample period:
409  var totalMinutes = (algorithm.EndDate - algorithm.StartDate).TotalMinutes;
410  var resampleMinutes = totalMinutes < MinimumSamplePeriod * Samples ? MinimumSamplePeriod : totalMinutes / Samples; // Space out the sampling every
411  ResamplePeriod = TimeSpan.FromMinutes(resampleMinutes);
412  Log.Trace("BacktestingResultHandler(): Sample Period Set: " + resampleMinutes.ToStringInvariant("00.00"));
413 
414  //Set the security / market types.
415  var types = new List<SecurityType>();
416  foreach (var kvp in Algorithm.Securities)
417  {
418  var security = kvp.Value;
419 
420  if (!types.Contains(security.Type)) types.Add(security.Type);
421  }
422  SecurityType(types);
423 
424  ConfigureConsoleTextWriter(algorithm);
425 
426  // Wire algorithm name and tags updates
427  algorithm.NameUpdated += (sender, name) => AlgorithmNameUpdated(name);
428  algorithm.TagsUpdated += (sender, tags) => AlgorithmTagsUpdated(tags);
429  }

539  /// <summary>
540  /// Add a sample to the chart specified by the chartName, and seriesName.
541  /// </summary>
542  /// <param name="chartName">String chart name to place the sample.</param>
543  /// <param name="seriesIndex">Type of chart we should create if it doesn't already exist.</param>
544  /// <param name="seriesName">Series name for the chart.</param>
545  /// <param name="seriesType">Series type for the chart.</param>
546  /// <param name="value">Value for the chart sample.</param>
547  /// <param name="unit">Unit of the sample</param>
548  protected override void Sample(string chartName, string seriesName, int seriesIndex, SeriesType seriesType, ISeriesPoint value,
549  string unit = "$")
550  {
551  // Sampling during warming up period skews statistics
552  if (Algorithm.IsWarmingUp)
553  {
554  return;
555  }
556 
557  lock (ChartLock)
558  {
559  //Add a copy locally:
560  Chart chart;
561  if (!Charts.TryGetValue(chartName, out chart))
562  {
563  chart = new Chart(chartName);
564  Charts.AddOrUpdate(chartName, chart);
565  }
566 
567  //Add the sample to our chart:
568  BaseSeries series;
569  if (!chart.Series.TryGetValue(seriesName, out series))
570  {
571  series = BaseSeries.Create(seriesType, seriesName, seriesIndex, unit);
572  chart.Series.Add(seriesName, series);
573  }
574 
575  //Add our value:
576  if (series.Values.Count == 0 || value.Time > series.Values[series.Values.Count - 1].Time
577  // always sample portfolio turnover and use latest value
578  || chartName == PortfolioTurnoverKey)
579  {
580  series.AddPoint(value);
581  }
582  }
583  }

596  /// <summary>
597  /// Add a range of samples from the users algorithms to the end of our current list.
598  /// </summary>
599  /// <param name="updates">Chart updates since the last request.</param>
600  protected void SampleRange(IEnumerable<Chart> updates)
601  {
602  lock (ChartLock)
603  {
604  foreach (var update in updates)
605  {
606  //Create the chart if it doesn't exist already:
607  Chart chart;
608  if (!Charts.TryGetValue(update.Name, out chart))
609  {
610  chart = new Chart(update.Name);
611  Charts.AddOrUpdate(update.Name, chart);
612  }
613 
614  //Add these samples to this chart.
615  foreach (var series in update.Series.Values)
616  {
617  // let's assert we are within series count limit
618  if (_chartSeriesCount.Count < _job.Controls.MaximumChartSeries)
619  {
620  _chartSeriesCount.Add(series.Name);
621  }
622  else if (!_chartSeriesCount.Contains(series.Name))
623  {
624  // above the limit and this is a new series
625  if(!_chartSeriesCountExceededError)
626  {
627  _chartSeriesCountExceededError = true;
628  DebugMessage($"Exceeded maximum chart series count for organization tier, new series will be ignored. Limit is currently set at {_job.Controls.MaximumChartSeries}. https://qnt.co/docs-charting-quotas");
629  }
630  continue;
631  }
632 
633  if (series.Values.Count > 0)
634  {
635  var thisSeries = chart.TryAddAndGetSeries(series.Name, series, forceAddNew: false);
636  if (series.SeriesType == SeriesType.Pie)
637  {
638  var dataPoint = series.ConsolidateChartPoints();
639  if (dataPoint != null)
640  {
641  thisSeries.AddPoint(dataPoint);
642  }
643  }
644  else
645  {
646  var values = thisSeries.Values;
647  if ((values.Count + series.Values.Count) <= _job.Controls.MaximumDataPointsPerChartSeries) // check chart data point limit first
648  {
649  //We already have this record, so just the new samples to the end:
650  values.AddRange(series.Values);
651  }
652  else if (!_chartSeriesExceededDataPoints.Contains(chart.Name + series.Name))
653  {
654  _chartSeriesExceededDataPoints.Add(chart.Name + series.Name);
655  DebugMessage($"Exceeded maximum data points per series for organization tier, chart update skipped. Chart Name {update.Name}. Series name {series.Name}. https://qnt.co/docs-charting-quotas" +
656  $"Limit is currently set at {_job.Controls.MaximumDataPointsPerChartSeries}");
657  }
658  }
659  }
660  }
661  }
662  }

769  /// <summary>
770  /// Configures the <see cref="Console.Out"/> and <see cref="Console.Error"/> <see cref="TextWriter"/>
771  /// instances. By default, we forward <see cref="Console.WriteLine(string)"/> to <see cref="IAlgorithm.Debug"/>.
772  /// This is perfect for running in the cloud, but since they're processed asynchronously, the ordering of these
773  /// messages with respect to <see cref="Log"/> messages is broken. This can lead to differences in regression
774  /// test logs based solely on the ordering of messages. To disable this forwarding, set <code>"forward-console-messages"</code>
775  /// to <code>false</code> in the configuration.
776  /// </summary>
777  protected virtual void ConfigureConsoleTextWriter(IAlgorithm algorithm)
778  {
779  if (Config.GetBool("forward-console-messages", true))
780  {
781  // we need to forward Console.Write messages to the algorithm's Debug function
782  Console.SetOut(new FuncTextWriter(algorithm.Debug));
783  Console.SetError(new FuncTextWriter(algorithm.Error));
784  }
785  else
786  {
787  // we need to forward Console.Write messages to the standard Log functions
788  Console.SetOut(new FuncTextWriter(msg => Log.Trace(msg)));
789  Console.SetError(new FuncTextWriter(msg => Log.Error(msg)));
790  }
791  }
