17 using System;
18 using System.IO;
19 using System.Linq;
20 using QuantConnect.Data;
21 using QuantConnect.Util;
22 using System.Diagnostics;
23 using QuantConnect.Logging;
24 using System.Threading.Tasks;
25 using System.Collections.Generic;
26 using QuantConnect.ToolBox.CoinApi;

28 namespace QuantConnect.ToolBox.CoinApiDataConverter
29 {
30  /// <summary>
31  /// Console application for converting CoinApi raw data into Lean data format for high resolutions (tick, second and minute)
32  /// </summary>
33  public class CoinApiDataConverter
34  {
35  /// <summary>
36  /// List of supported exchanges
37  /// </summary>
38  private static readonly HashSet<string> SupportedMarkets = new[]
39  {
40  Market.GDAX,
41  Market.Bitfinex,
42  Market.Binance,
43  Market.FTX,
44  Market.FTXUS,
45  Market.Kraken,
46  Market.BinanceUS
47  }.ToHashSet();
48 
49  private readonly DirectoryInfo _rawDataFolder;
50  private readonly DirectoryInfo _destinationFolder;
51  private readonly DateTime _processingDate;
52  private readonly string _market;
53 
54  /// <summary>
55  /// CoinAPI data converter.
56  /// </summary>
57  /// <param name="date">the processing date.</param>
58  /// <param name="rawDataFolder">path to the raw data folder.</param>
59  /// <param name="destinationFolder">destination of the newly generated files.</param>
60  /// <param name="market">The market to process (optional). Defaults to processing all markets in parallel.</param>
61  public CoinApiDataConverter(DateTime date, string rawDataFolder, string destinationFolder, string market = null)
62  {
63  _market = string.IsNullOrWhiteSpace(market)
64  ? null
65  : market.ToLowerInvariant();
66 
67  _processingDate = date;
68  _rawDataFolder = new DirectoryInfo(Path.Combine(rawDataFolder, SecurityType.Crypto.ToLower(), "coinapi"));
69  if (!_rawDataFolder.Exists)
70  {
71  throw new ArgumentException($"CoinApiDataConverter(): Source folder not found: {_rawDataFolder.FullName}");
72  }
73 
74  _destinationFolder = new DirectoryInfo(destinationFolder);
75  _destinationFolder.Create();
76  }
77 
78  /// <summary>
79  /// Runs this instance.
80  /// </summary>
81  /// <returns></returns>
82  public bool Run()
83  {
84  var stopwatch = Stopwatch.StartNew();
85 
86  var symbolMapper = new CoinApiSymbolMapper();
87  var success = true;
88 
89  // There were cases of files with with an extra suffix, following pattern:
90  // <TickType>-<ID>-<Exchange>_SPOT_<BaseCurrency>_<QuoteCurrency>_<ExtraSuffix>.csv.gz
91  // Those cases should be ignored for SPOT prices.
92  var tradesFolder = new DirectoryInfo(
93  Path.Combine(
94  _rawDataFolder.FullName,
95  "trades",
96  _processingDate.ToStringInvariant(DateFormat.EightCharacter)));
97 
98  var quotesFolder = new DirectoryInfo(
99  Path.Combine(
100  _rawDataFolder.FullName,
101  "quotes",
102  _processingDate.ToStringInvariant(DateFormat.EightCharacter)));
103 
104  var rawMarket = _market != null &&
105  CoinApiSymbolMapper.MapMarketsToExchangeIds.TryGetValue(_market, out var rawMarketValue)
106  ? rawMarketValue
107  : null;
108 
109  // Distinct by tick type and first two parts of the raw file name, separated by '-'.
110  // This prevents us from double processing the same ticker twice, in case we're given
111  // two raw data files for the same symbol. Related: https://github.com/QuantConnect/Lean/pull/3262
112  var apiDataReader = new CoinApiDataReader(symbolMapper);
113  var filesToProcessCandidates = tradesFolder.EnumerateFiles("*.gz")
114  .Concat(quotesFolder.EnumerateFiles("*.gz"))
115  .Where(f => f.Name.Contains("SPOT") && (rawMarket == null || f.Name.Contains(rawMarket)))
116  .Where(f => f.Name.Split('_').Length == 4)
117  .ToList();
118 
119  var filesToProcessKeys = new HashSet<string>();
120  var filesToProcess = new List<FileInfo>();
121 
122  foreach (var candidate in filesToProcessCandidates)
123  {
124  try
125  {
126  var entryData = apiDataReader.GetCoinApiEntryData(candidate, _processingDate);
127  CurrencyPairUtil.DecomposeCurrencyPair(entryData.Symbol, out var baseCurrency,
128  out var quoteCurrency);
129 
130  if (!candidate.FullName.Contains(baseCurrency) && !candidate.FullName.Contains(quoteCurrency))
131  {
132  throw new Exception($"Skipping {candidate.FullName} we have the wrong symbol {entryData.Symbol}!");
133  }
134 
135  var key = candidate.Directory.Parent.Name + entryData.Symbol.ID;
136  if (filesToProcessKeys.Add(key))
137  {
138  // Separate list from HashSet to preserve ordering of viable candidates
139  filesToProcess.Add(candidate);
140  }
141  }
142  catch (Exception err)
143  {
144  // Most likely the exchange isn't supported. Log exception message to avoid excessive stack trace spamming in console output
145  Log.Error(err.Message);
146  }
147  }
148 
149  Parallel.ForEach(filesToProcess, (file, loopState) =>
150  {
151  Log.Trace($"CoinApiDataConverter(): Starting data conversion from source file: {file.Name}...");
152  try
153  {
154  ProcessEntry(apiDataReader, file);
155  }
156  catch (Exception e)
157  {
158  Log.Error(e, $"CoinApiDataConverter(): Error processing entry: {file.Name}");
159  success = false;
160  loopState.Break();
161  }
162  }
163  );
164 
165  Log.Trace($"CoinApiDataConverter(): Finished in {stopwatch.Elapsed}");
166  return success;
167  }
168 
169  /// <summary>
170  /// Processes the entry.
171  /// </summary>
172  /// <param name="coinapiDataReader">The coinapi data reader.</param>
173  /// <param name="file">The file.</param>
174  private void ProcessEntry(CoinApiDataReader coinapiDataReader, FileInfo file)
175  {
176  var entryData = coinapiDataReader.GetCoinApiEntryData(file, _processingDate);
177 
178  if (!SupportedMarkets.Contains(entryData.Symbol.ID.Market))
179  {
180  // only convert data for supported exchanges
181  return;
182  }
183 
184  var tickData = coinapiDataReader.ProcessCoinApiEntry(entryData, file);
185 
186  // in some cases the first data points from '_processingDate' get's included in the previous date file
187  // so we will ready previous date data and drop most of it just to save these midnight ticks
188  var yesterdayDate = _processingDate.AddDays(-1);
189  var yesterdaysFile = new FileInfo(file.FullName.Replace(
190  _processingDate.ToStringInvariant(DateFormat.EightCharacter),
191  yesterdayDate.ToStringInvariant(DateFormat.EightCharacter)));
192  if (yesterdaysFile.Exists)
193  {
194  var yesterdaysEntryData = coinapiDataReader.GetCoinApiEntryData(yesterdaysFile, yesterdayDate);
195  tickData = tickData.Concat(coinapiDataReader.ProcessCoinApiEntry(yesterdaysEntryData, yesterdaysFile));
196  }
197  else
198  {
199  Log.Error($"CoinApiDataConverter(): yesterdays data file not found '{yesterdaysFile.FullName}'");
200  }
201 
202  // materialize the enumerable into a list, since we need to enumerate over it twice
203  var ticks = tickData.Where(tick => tick.Time.Date == _processingDate)
204  .OrderBy(t => t.Time)
205  .ToList();
206 
207  var writer = new LeanDataWriter(Resolution.Tick, entryData.Symbol, _destinationFolder.FullName, entryData.TickType);
208  writer.Write(ticks);
209 
210  Log.Trace($"CoinApiDataConverter(): Starting consolidation for {entryData.Symbol.Value} {entryData.TickType}");
211  var consolidators = new List<TickAggregator>();
212 
213  if (entryData.TickType == TickType.Trade)
214  {
215  consolidators.AddRange(new[]
216  {
217  new TradeTickAggregator(Resolution.Second),
218  new TradeTickAggregator(Resolution.Minute)
219  });
220  }
221  else
222  {
223  consolidators.AddRange(new[]
224  {
225  new QuoteTickAggregator(Resolution.Second),
226  new QuoteTickAggregator(Resolution.Minute)
227  });
228  }
229 
230  foreach (var tick in ticks)
231  {
232  if (tick.Suspicious)
233  {
234  // When CoinAPI loses connectivity to the exchange, they indicate
235  // it in the data by providing a value of `-1` for bid/ask price.
236  // We will keep it in tick data, but will remove it from consolidated data.
237  continue;
238  }
239 
240  foreach (var consolidator in consolidators)
241  {
242  consolidator.Update(tick);
243  }
244  }
245 
246  foreach (var consolidator in consolidators)
247  {
248  writer = new LeanDataWriter(consolidator.Resolution, entryData.Symbol, _destinationFolder.FullName, entryData.TickType);
249  writer.Write(consolidator.Flush());
250  }
251  }
252  }
