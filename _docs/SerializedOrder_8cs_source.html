222  /// <summary>
223  /// Creates a new serialized order instance based on the provided order
224  /// </summary>
225  public SerializedOrder(Order order, string algorithmId)
226  {
227  AlgorithmId = algorithmId;
228 
229  OrderId = order.Id;
230  ContingentId = order.ContingentId;
231  BrokerId = order.BrokerId;
232  Symbol = order.Symbol.ID.ToString();
233  Price = order.Price;
234  PriceCurrency = order.PriceCurrency;
235  Quantity = order.Quantity;
236  Type = order.Type;
237  Status = order.Status;
238  Tag = order.Tag;
239  Direction = order.Direction;
240  PriceAdjustmentMode = order.PriceAdjustmentMode;
241 
242  CreatedTime = Time.DateTimeToUnixTimeStamp(order.CreatedTime);
243  if (order.LastFillTime.HasValue)
244  {
245  LastFillTime = Time.DateTimeToUnixTimeStamp(order.LastFillTime.Value);
246  }
247  if (order.LastUpdateTime.HasValue)
248  {
249  LastUpdateTime = Time.DateTimeToUnixTimeStamp(order.LastUpdateTime.Value);
250  }
251  if (order.CanceledTime.HasValue)
252  {
253  CanceledTime = Time.DateTimeToUnixTimeStamp(order.CanceledTime.Value);
254  }
255  if (order.OrderSubmissionData != null)
256  {
257  SubmissionAskPrice = order.OrderSubmissionData.AskPrice;
258  SubmissionBidPrice = order.OrderSubmissionData.BidPrice;
259  SubmissionLastPrice = order.OrderSubmissionData.LastPrice;
260  }
261 
262  var timeInForceType = order.Properties.TimeInForce.GetType().Name;
263  // camelcase the type name, lowering the first char
264  TimeInForceType = char.ToLowerInvariant(timeInForceType[0]) + timeInForceType.Substring(1);
265  if (order.Properties.TimeInForce is GoodTilDateTimeInForce)
266  {
267  var expiry = (order.Properties.TimeInForce as GoodTilDateTimeInForce).Expiry;
268  TimeInForceExpiry = Time.DateTimeToUnixTimeStamp(expiry);
269  }
270 
271  if (order.Type == OrderType.Limit)
272  {
273  var limit = order as LimitOrder;
274  LimitPrice = limit.LimitPrice;
275  }
276  else if (order.Type == OrderType.StopLimit)
277  {
278  var stopLimit = order as StopLimitOrder;
279  LimitPrice = stopLimit.LimitPrice;
280  StopPrice = stopLimit.StopPrice;
281  StopTriggered = stopLimit.StopTriggered;
282  }
283  else if (order.Type == OrderType.StopMarket)
284  {
285  var stopMarket = order as StopMarketOrder;
286  StopPrice = stopMarket.StopPrice;
287  }
288  else if (order.Type == OrderType.TrailingStop)
289  {
290  var trailingStop = order as TrailingStopOrder;
291  StopPrice = trailingStop.StopPrice;
292  TrailingAmount = trailingStop.TrailingAmount;
293  TrailingAsPercentage = trailingStop.TrailingAsPercentage;
294  }
295  else if (order.Type == OrderType.LimitIfTouched)
296  {
297  var limitIfTouched = order as LimitIfTouchedOrder;
298  LimitPrice = limitIfTouched.LimitPrice;
299  TriggerPrice = limitIfTouched.TriggerPrice;
300  TriggerTouched = limitIfTouched.TriggerTouched;
301  }
302 
303  GroupOrderManager = order.GroupOrderManager;
304  }
305  }
