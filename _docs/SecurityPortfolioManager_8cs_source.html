394  /// <summary>
395  /// Total portfolio value if we sold all holdings at current market rates.
396  /// </summary>
397  /// <remarks>Cash + TotalUnrealisedProfit + TotalUnleveredAbsoluteHoldingsCost</remarks>
398  /// <seealso cref="Cash"/>
399  /// <seealso cref="TotalUnrealizedProfit"/>
400  /// <seealso cref="TotalUnleveredAbsoluteHoldingsCost"/>
401  public decimal TotalPortfolioValue
402  {
403  get
404  {
405  if (!_isTotalPortfolioValueValid)
406  {
407  decimal totalHoldingsValueWithoutForexCryptoFutureCfd = 0;
408  decimal totalFuturesAndCfdHoldingsValue = 0;
409  foreach (var kvp in Securities.Where((pair, i) => pair.Value.Holdings.Quantity != 0))
410  {
411  var position = kvp.Value;
412  var securityType = position.Type;
413  // We can't include forex in this calculation since we would be double accounting with respect to the cash book
414  // We also exclude futures and CFD as they are calculated separately because they do not impact the account's cash.
415  // We include futures options as part of this calculation because IB chooses to change our account's cash balance
416  // when we buy or sell a futures options contract.
417  if (securityType != SecurityType.Forex && securityType != SecurityType.Crypto &&
418  securityType != SecurityType.Future && securityType != SecurityType.Cfd)
419  {
420  totalHoldingsValueWithoutForexCryptoFutureCfd += position.Holdings.HoldingsValue;
421  }
422 
423  // Futures and CFDs don't impact account cash, so they must be calculated
424  // by applying the unrealized P&L to the cash balance.
425  if (securityType == SecurityType.Future || securityType == SecurityType.Cfd)
426  {
427  totalFuturesAndCfdHoldingsValue += position.Holdings.UnrealizedProfit;
428  }
429  }
430 
431  _totalPortfolioValue = CashBook.TotalValueInAccountCurrency +
432  UnsettledCashBook.TotalValueInAccountCurrency +
433  totalHoldingsValueWithoutForexCryptoFutureCfd +
434  totalFuturesAndCfdHoldingsValue;
435 
436  _isTotalPortfolioValueValid = true;
437  }
438 
439  return _totalPortfolioValue;
440  }
441  }

548  {
549  accountCurrency = accountCurrency.LazyToUpper();
550 
551  // only allow setting account currency once
552  // we could try to set it twice when backtesting and the job packet specifies the initial CashAmount to use
553  if (_setAccountCurrencyWasCalled)
554  {
555  if (accountCurrency != CashBook.AccountCurrency)
556  {
557  Log.Trace("SecurityPortfolioManager.SetAccountCurrency():" +
558  $" account currency has already been set to {CashBook.AccountCurrency}." +

560  }
561  return;
562  }

564 
565  if (Securities.Count > 0)
566  {

568  "Cannot change AccountCurrency after adding a Security. " +

573  {
574  throw new InvalidOperationException("SecurityPortfolioManager.SetAccountCurrency(): " +
575  "Cannot change AccountCurrency after setting cash. " +
576  "Please move SetAccountCurrency() before SetCash().");
577  }
578 
579  Log.Trace("SecurityPortfolioManager.SetAccountCurrency():" +
580  $" setting account currency to {accountCurrency}");
581 
582  UnsettledCashBook.AccountCurrency = accountCurrency;
583  CashBook.AccountCurrency = accountCurrency;

662  /// <summary>
663  /// Applies a dividend to the portfolio
664  /// </summary>
665  /// <param name="dividend">The dividend to be applied</param>
666  /// <param name="liveMode">True if live mode, false for backtest</param>
667  /// <param name="mode">The <see cref="DataNormalizationMode"/> for this security</param>
668  public void ApplyDividend(Dividend dividend, bool liveMode, DataNormalizationMode mode)
669  {
670  // we currently don't properly model dividend payable dates, so in
671  // live mode it's more accurate to rely on the brokerage cash sync
672  if (liveMode)
673  {
674  return;
675  }
676 
677  var security = Securities[dividend.Symbol];
678 
679  // only apply dividends when we're in raw mode or split adjusted mode
680  if (mode == DataNormalizationMode.Raw || mode == DataNormalizationMode.SplitAdjusted)
681  {

690  /// <summary>
691  /// Applies a split to the portfolio
692  /// </summary>
693  /// <param name="split">The split to be applied</param>
694  /// <param name="liveMode">True if live mode, false for backtest</param>
695  /// <param name="mode">The <see cref="DataNormalizationMode"/> for this security</param>
696  public void ApplySplit(Split split, bool liveMode, DataNormalizationMode mode)
697  {
698  var security = Securities[split.Symbol];
699 
700  // only apply splits to equities
701  if (security.Type != SecurityType.Equity)
702  {
703  return;
704  }
705 
706  // only apply splits in live or raw data mode
707  if (!liveMode && mode != DataNormalizationMode.Raw)
708  {
709  return;
710  }
711 
712  // we need to modify our holdings in lght of the split factor
713  var quantity = security.Holdings.Quantity / split.SplitFactor;
714  var avgPrice = security.Holdings.AveragePrice * split.SplitFactor;
715 
716  // we'll model this as a cash adjustment
717  var leftOver = quantity - (int)quantity;
718 
719  security.Holdings.SetHoldings(avgPrice, (int)quantity);
720 
721  // build a 'next' value to update the market prices in light of the split factor
722  var next = security.GetLastData();
723  if (next == null)
724  {
725  // sometimes we can get splits before we receive data which
726  // will cause this to return null, in this case we can't possibly
727  // have any holdings or price to set since we haven't received
728  // data yet, so just do nothing
729  _baseCurrencyCash.AddAmount(leftOver * split.ReferencePrice * split.SplitFactor);
730  return;
731  }
732  next.Value *= split.SplitFactor;
733 
734  // make sure to modify open/high/low as well for tradebar data types
735  var tradeBar = next as TradeBar;
736  if (tradeBar != null)
737  {
738  tradeBar.Open *= split.SplitFactor;
739  tradeBar.High *= split.SplitFactor;
740  tradeBar.Low *= split.SplitFactor;
741  }
742 
743  // make sure to modify bid/ask as well for tradebar data types
744  var tick = next as Tick;
745  if (tick != null)
746  {
747  tick.AskPrice *= split.SplitFactor;
748  tick.BidPrice *= split.SplitFactor;
749  }
750 
751  security.SetMarketPrice(next);
752  _baseCurrencyCash.AddAmount(leftOver * next.Price);
753 
754  // security price updated
755  InvalidateTotalPortfolioValue();
756  }
