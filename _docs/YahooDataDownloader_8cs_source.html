16 using System.Collections.Generic;

21 namespace QuantConnect.ToolBox.YahooDownloader

23  /// <summary>
24  /// Yahoo Data Downloader class
25  /// </summary>
26  public class YahooDataDownloader : IDataDownloader
27  {
28  /// <summary>
29  /// Get historical data enumerable for a single symbol, type and resolution given this start and end time (in UTC).
30  /// </summary>
31  /// <param name="dataDownloaderGetParameters">model class for passing in parameters for historical data</param>
32  /// <returns>Enumerable of base data for this symbol</returns>
33  public IEnumerable<BaseData> Get(DataDownloaderGetParameters dataDownloaderGetParameters)
34  {
35  var symbol = dataDownloaderGetParameters.Symbol;
36  var resolution = dataDownloaderGetParameters.Resolution;
37  var startUtc = dataDownloaderGetParameters.StartUtc;
38  var endUtc = dataDownloaderGetParameters.EndUtc;
39  var tickType = dataDownloaderGetParameters.TickType;
40 
41  if (tickType != TickType.Trade)
42  {
43  return Enumerable.Empty<BaseData>();
44  }
45 
46  if (resolution != Resolution.Daily)
47  {
48  throw new ArgumentException("The YahooDataDownloader can only download daily data.");
49  }
50 
51  if (symbol.ID.SecurityType != SecurityType.Equity)
52  {
53  throw new NotSupportedException("SecurityType not available: " + symbol.ID.SecurityType);
54  }
55 
56  if (endUtc < startUtc)
57  {
58  throw new ArgumentException("The end date must be greater or equal than the start date.");
59  }
60 
61  return GetEnumerator(symbol, startUtc, endUtc);
62  }
63 
64  private static IEnumerable<BaseData> GetEnumerator(Symbol symbol, DateTime startDateTime, DateTime endDateTime)
65  {
66  var data = Historical.Get(symbol.Value, startDateTime, endDateTime, "history");
67 
68  foreach (var item in data)
69  {
70  yield return new TradeBar(item.Date, symbol, item.Open, item.High, item.Low, item.Close, (long)item.Volume, TimeSpan.FromDays(1));
71  }
72  }
73 
74 
75  /// <summary>
76  /// Download Dividend and Split data from Yahoo
77  /// </summary>
78  /// <param name="symbol">Symbol of the data to download</param>
79  /// <param name="startUtc">Get data after this time</param>
80  /// <param name="endUtc">Get data before this time</param>
81  /// <returns></returns>
82  public Queue<BaseData> DownloadSplitAndDividendData(Symbol symbol, DateTime startUtc, DateTime endUtc)
83  {
84  var split = Historical.GetRaw(symbol.Value, startUtc, endUtc, "split");
85  var dividend = Historical.GetRaw(symbol.Value, startUtc, endUtc, "dividend");
86  var parsed = new List<BaseData>();
87 
88  foreach (var data in new[] { split, dividend })
89  {
90 
91  bool isSplit = false;
92 
93  foreach (var item in data.Split('\n'))
94  {
95  if (item == "Date,Stock Splits")
96  {
97  isSplit = true;
98  continue;
99  }
100  if (item == "Date,Dividends")
101  {
102  continue;
103  }
104  if (item == "")
105  {
106  break;
107  }
108 
109  string[] values = item.Split(',');
110 
111  if (isSplit)
112  {
113  parsed.Add(new Split
114  {
115  Time = Parse.DateTimeExact(values[0].Replace("-", string.Empty), DateFormat.EightCharacter),
116  Value = ParseAmount(values[1])
117  });
118  }
119  else
120  {
121  parsed.Add(new Dividend
122  {
123  Time = Parse.DateTimeExact(values[0].Replace("-", string.Empty), DateFormat.EightCharacter),
124  Value = Parse.Decimal(values[1], System.Globalization.NumberStyles.Float)
125  });
126  }
127  }
128  }
129 
130  return new Queue<BaseData>(parsed.OrderByDescending(x => x.Time));
131  }
132 
133  /// <summary>
134  /// Put the split ratio into a decimal format
135  /// </summary>
136  /// <param name="splitFactor">Split ratio</param>
137  /// <returns>Decimal representing the split ratio</returns>
138  private decimal ParseAmount(string splitFactor)
139  {
140  var factors = splitFactor.Split(':');
141  return Parse.Decimal(factors[0]) / Parse.Decimal(factors[1]);
142  }
143 
144  }
