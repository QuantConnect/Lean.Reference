173  Cash = _cashBook?.ToDictionary(pair => pair.Key, pair => pair.Value);

175  AccountCurrencySymbol = AccountCurrency != null ? Currencies.GetCurrencySymbol(AccountCurrency) : null;
176  }
177  }
