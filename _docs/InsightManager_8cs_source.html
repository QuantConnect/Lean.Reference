26  /// <summary>
27  /// Encapsulates the storage and on-line scoring of insights.
28  /// </summary>
29  /// <remarks>
30  /// This type assumes a forward progression of time, and as such, methods invoked will
31  /// return data that is current as of the last update time.
32  /// This type is designed to be invoked from two separate threads. That does not mean this type
33  /// is thread-safe in the general sense, but given a particular invocation pattern. The goal is
34  /// to allow the algorithm thread to continue while scoring of the insights happens on a separate
35  /// thread. Insights are added on the algorithm thread via AddInsights and scoring updates are pushed
36  /// on the insight thrad via UpdateScores. This means that the various collections (open, closed, updated)
37  /// are potentially at different frontiers. In fact, it is the common case where the openInsightContexts
38  /// collection is ahead of everything else.
39  /// </remarks>
40  public class InsightManager : IInsightManager, IDisposable
41  {
42  /// <summary>
43  /// Gets all insight score types
44  /// </summary>
45  public static readonly IReadOnlyCollection<InsightScoreType> ScoreTypes = Enum.GetValues(typeof(InsightScoreType)).Cast<InsightScoreType>().ToArray();
46 
47  private readonly double _extraAnalysisPeriodRatio;
48  private readonly List<IInsightManagerExtension> _extensions;
49  private readonly IInsightScoreFunctionProvider _scoreFunctionProvider;
50 
51  private readonly object _lock;
52  private readonly HashSet<InsightAnalysisContext> _updatedInsightContexts;
53  private readonly HashSet<InsightAnalysisContext> _openInsightContexts;
54  private readonly ConcurrentDictionary<Guid, InsightAnalysisContext> _closedInsightContexts;
55 
56  /// <summary>
57  /// Enumerable of insights still under analysis
58  /// </summary>
59  public IEnumerable<Insight> OpenInsights
60  {
61  get
62  {
63  lock (_lock)
64  {
65  return _openInsightContexts.Select(context => context.Insight).ToList();
66  }
67  }
68  }
69 
70  /// <summary>
71  /// Enumerable of insights who's analysis has been completed
72  /// </summary>
73  public IEnumerable<Insight> ClosedInsights => _closedInsightContexts.Select(kvp => kvp.Value.Insight);
74 
75  /// <summary>
76  /// Enumerable of all internally maintained insights
77  /// </summary>
78  public IEnumerable<Insight> AllInsights => OpenInsights.Concat(ClosedInsights);

80  /// <summary>
81  /// Gets the unique set of symbols from analysis contexts that will
82  /// </summary>
83  /// <param name="frontierTimeUtc"></param>
84  /// <returns></returns>
85  public IEnumerable<InsightAnalysisContext> ContextsOpenAt(DateTime frontierTimeUtc)
86  {
87  lock (_lock)
88  {
89  return _openInsightContexts.Where(context => context.AnalysisEndTimeUtc <= frontierTimeUtc).ToList();
90  }
91  }

93  /// <summary>
94  /// Initializes a new instance of the <see cref="InsightManager"/> class
95  /// </summary>
96  /// <param name="scoreFunctionProvider">Provides scoring functions by insight type/score type</param>
97  /// <param name="extraAnalysisPeriodRatio">Ratio of the insight period to keep the analysis open</param>
98  /// <param name="extensions">Extensions used to perform tasks at certain events</param>
99  public InsightManager(IInsightScoreFunctionProvider scoreFunctionProvider, double extraAnalysisPeriodRatio, params IInsightManagerExtension[] extensions)
100  {
101  if (extraAnalysisPeriodRatio < 0)
102  {
103  throw new ArgumentOutOfRangeException(nameof(extraAnalysisPeriodRatio), "extraAnalysisPeriodRatio must be greater than or equal to zero.");
104  }
105 
106  _scoreFunctionProvider = scoreFunctionProvider;
107  _extraAnalysisPeriodRatio = extraAnalysisPeriodRatio;
108  _extensions = extensions?.ToList() ?? new List<IInsightManagerExtension>();
109 
110  _lock = new object();
111  _openInsightContexts = new HashSet<InsightAnalysisContext>();
112  _updatedInsightContexts = new HashSet<InsightAnalysisContext>();
113  _closedInsightContexts = new ConcurrentDictionary<Guid, InsightAnalysisContext>();
114  }
115 
116  /// <summary>
117  /// Add an extension to this manager
118  /// </summary>
119  /// <param name="extension">The extension to be added</param>
120  public void AddExtension(IInsightManagerExtension extension)
121  {
122  _extensions.Add(extension);
123  }
124 
125  /// <summary>
126  /// Initializes any extensions for the specified backtesting range
127  /// </summary>
128  /// <param name="start">The start date of the backtest (current time in live mode)</param>
129  /// <param name="end">The end date of the backtest (<see cref="Time.EndOfTime"/> in live mode)</param>
130  /// <param name="current">The algorithm's current utc time</param>
131  public void InitializeExtensionsForRange(DateTime start, DateTime end, DateTime current)
132  {
133  foreach (var extension in _extensions)
134  {
135  extension.InitializeForRange(start, end, current);
136  }
137  }

139  /// <summary>
140  /// Steps the manager forward in time, accepting new state information and potentialy newly generated insights
141  /// </summary>
142  /// <param name="frontierTimeUtc">The frontier time of the insight analysis</param>
143  /// <param name="securityValuesCollection">Snap shot of the securities at the frontier time</param>
144  /// <param name="generatedInsights">Any insight generated by the algorithm at the frontier time</param>
145  public void Step(DateTime frontierTimeUtc, ReadOnlySecurityValuesCollection securityValuesCollection, GeneratedInsightsCollection generatedInsights)
146  {
147  lock (_lock)
148  {
149  if (generatedInsights != null && generatedInsights.Insights.Count > 0)
150  {
151  foreach (var insight in generatedInsights.Insights)
152  {
153  // save initial security values and deterine analysis period
154  var initialValues = securityValuesCollection[insight.Symbol];
155  var analysisPeriod = insight.Period + TimeSpan.FromTicks((long)(_extraAnalysisPeriodRatio * insight.Period.Ticks));
156 
157  // set this as an open analysis context
158  var context = new InsightAnalysisContext(insight, initialValues, analysisPeriod);
159  _openInsightContexts.Add(context);
160 
161  if (context.InitialValues.Price == 0)
162  {
163  Log.Error(Invariant($"InsightManager.Step(): Warning {frontierTimeUtc} UTC: insight {insight} initial price value is 0"));
164  }
165 
166  // let everyone know we've received an insight
167  _extensions.ForEach(e => e.OnInsightGenerated(context));
168  }
169  }
170 
171  UpdateScores(securityValuesCollection);
172 
173  foreach (var extension in _extensions)
174  {
175  extension.Step(frontierTimeUtc);
176  }
177  }
178  }

180  /// <summary>
181  /// Removes insights from the manager with the specified ids
182  /// </summary>
183  /// <param name="insightIds">The insights ids to be removed</param>
184  public void RemoveInsights(IEnumerable<Guid> insightIds)
185  {
186  foreach (var id in insightIds)
187  {
188  InsightAnalysisContext context;
189  _closedInsightContexts.TryRemove(id, out context);
190  }
191  }
192 
193  /// <summary>
194  /// Gets all insight analysis contexts that have been updated since this method's last invocation.
195  /// Contexts are marked as not updated during the enumeration, so in order to remove a context from
196  /// the updated set, the enumerable must be enumerated.
197  /// </summary>
198  /// <returns></returns>
199  public IEnumerable<InsightAnalysisContext> GetUpdatedContexts()
200  {
201  lock (_lock)
202  {
203  var copy = _updatedInsightContexts.ToList();
204  _updatedInsightContexts.Clear();
205  return copy;
206  }
207  }

209  /// <summary>
210  /// Updates all open insight scores
211  /// </summary>
212  private void UpdateScores(ReadOnlySecurityValuesCollection securityValuesCollection)
213  {
214  // for performance be lazy to initialize collection
215  List<InsightAnalysisContext> removals = null;
216 
217  foreach (var context in _openInsightContexts)
218  {
219  // was this insight period closed before we update the times?
220  var previouslyClosed = context.InsightPeriodClosed;
221 
222  // update the security values: price/volatility
223  context.SetCurrentValues(securityValuesCollection[context.Symbol]);
224 
225  // update scores for each score type
226  var currentTimeUtc = context.CurrentValues.TimeUtc;
227  foreach (var scoreType in ScoreTypes)
228  {
229  if (!context.ShouldAnalyze(scoreType))
230  {
231  // not all insights can receive every score type, for example, insight.Magnitude==null, not point in doing magnitude scoring
232  continue;
233  }
234 
235  // resolve and evaluate the scoring function, storing the result in the context
236  var function = _scoreFunctionProvider.GetScoreFunction(context.Insight.Type, scoreType);
237  var score = function.Evaluate(context, scoreType);
238  context.Score.SetScore(scoreType, score, currentTimeUtc);
239  }
240 
241  // it wasn't closed and now it is closed, fire the event.
242  if (!previouslyClosed && context.InsightPeriodClosed)
243  {
244  _extensions.ForEach(e => e.OnInsightClosed(context));
245  }
246 
247  // if this score has been finalized, remove it from the open set
248  if (currentTimeUtc >= context.AnalysisEndTimeUtc)
249  {
250  context.Score.Finalize(currentTimeUtc);
251 
252  // set the last value used for scoring
253  context.Insight.ReferenceValueFinal = context.CurrentValues.Get(context.Insight.Type);
254 
255  _extensions.ForEach(e => e.OnInsightAnalysisCompleted(context));
256 
257  var id = context.Insight.Id;
258  _closedInsightContexts[id] = context;
259 
260  if (removals == null)
261  {
262  removals = new List<InsightAnalysisContext>();
263  }
264  removals.Add(context);
265  }
266 
267  // mark the context as having been updated
268  _updatedInsightContexts.Add(context);
269  }
270 
271  if (removals != null)
272  {
273  _openInsightContexts.RemoveWhere(removals.Contains);
274  }
275  }

277  /// <summary>Performs application-defined tasks associated with freeing, releasing, or resetting unmanaged resources.</summary>
278  /// <filterpriority>2</filterpriority>
279  public void Dispose()
280  {
281  foreach (var ext in _extensions)
282  {
283  (ext as IDisposable)?.DisposeSafely();
284  }
285  }
