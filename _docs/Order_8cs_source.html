178  /// <summary>
179  /// Returns true if the order is a marketable order.
180  /// </summary>
181  public bool IsMarketable
182  {
183  get
184  {
185  if (Type == OrderType.Limit)
186  {
187  // check if marketable limit order using bid/ask prices
188  var limitOrder = (LimitOrder)this;
189  return OrderSubmissionData != null &&
190  (Direction == OrderDirection.Buy && limitOrder.LimitPrice >= OrderSubmissionData.AskPrice ||
191  Direction == OrderDirection.Sell && limitOrder.LimitPrice <= OrderSubmissionData.BidPrice);
192  }
193 
194  return Type == OrderType.Market || Type == OrderType.ComboMarket;
195  }
196  }

378  /// <summary>
379  /// Creates a new Order instance from a SerializedOrder instance
380  /// </summary>
381  /// <remarks>Used by the <see cref="SerializedOrderJsonConverter"/></remarks>
382  public static Order FromSerialized(SerializedOrder serializedOrder)
383  {
384  var sid = SecurityIdentifier.Parse(serializedOrder.Symbol);
385  var symbol = new Symbol(sid, sid.Symbol);
386 
387  TimeInForce timeInForce = null;
388  var type = System.Type.GetType($"QuantConnect.Orders.TimeInForces.{serializedOrder.TimeInForceType}", throwOnError: false, ignoreCase: true);
389  if (type != null)
390  {
391  timeInForce = (TimeInForce) Activator.CreateInstance(type, true);
392  if (timeInForce is GoodTilDateTimeInForce)
393  {
394  var expiry = QuantConnect.Time.UnixTimeStampToDateTime(serializedOrder.TimeInForceExpiry.Value);
395  timeInForce = new GoodTilDateTimeInForce(expiry);
396  }
397  }
398 
399  var createdTime = QuantConnect.Time.UnixTimeStampToDateTime(serializedOrder.CreatedTime);
400 
401  var order = CreateOrder(serializedOrder.OrderId, serializedOrder.Type, symbol, serializedOrder.Quantity,
402  DateTime.SpecifyKind(createdTime, DateTimeKind.Utc),
403  serializedOrder.Tag,
404  new OrderProperties { TimeInForce = timeInForce },
405  serializedOrder.LimitPrice ?? 0,
406  serializedOrder.StopPrice ?? 0,
407  serializedOrder.TriggerPrice ?? 0,
408  serializedOrder.TrailingAmount ?? 0,
409  serializedOrder.TrailingAsPercentage ?? false,
410  serializedOrder.GroupOrderManager);
411 
412  order.OrderSubmissionData = new OrderSubmissionData(serializedOrder.SubmissionBidPrice,
413  serializedOrder.SubmissionAskPrice,
414  serializedOrder.SubmissionLastPrice);
415 
416  order.BrokerId = serializedOrder.BrokerId;
417  order.ContingentId = serializedOrder.ContingentId;
418  order.Price = serializedOrder.Price;
419  order.PriceCurrency = serializedOrder.PriceCurrency;
420  order.Status = serializedOrder.Status;
421  order.PriceAdjustmentMode = serializedOrder.PriceAdjustmentMode;
422 
423  if (serializedOrder.LastFillTime.HasValue)
424  {
425  var time = QuantConnect.Time.UnixTimeStampToDateTime(serializedOrder.LastFillTime.Value);
426  order.LastFillTime = DateTime.SpecifyKind(time, DateTimeKind.Utc);
427  }
428  if (serializedOrder.LastUpdateTime.HasValue)
429  {
430  var time = QuantConnect.Time.UnixTimeStampToDateTime(serializedOrder.LastUpdateTime.Value);
431  order.LastUpdateTime = DateTime.SpecifyKind(time, DateTimeKind.Utc);
432  }
433  if (serializedOrder.CanceledTime.HasValue)
434  {
435  var time = QuantConnect.Time.UnixTimeStampToDateTime(serializedOrder.CanceledTime.Value);
436  order.CanceledTime = DateTime.SpecifyKind(time, DateTimeKind.Utc);
437  }
438 
439  return order;
440  }

454  private static Order CreateOrder(int orderId, OrderType type, Symbol symbol, decimal quantity, DateTime time,
455  string tag, IOrderProperties properties, decimal limitPrice, decimal stopPrice, decimal triggerPrice, decimal trailingAmount,
456  bool trailingAsPercentage, GroupOrderManager groupOrderManager)
457  {
458  Order order;
459  switch (type)
460  {
461  case OrderType.Market:
462  order = new MarketOrder(symbol, quantity, time, tag, properties);
463  break;
464 
465  case OrderType.Limit:
466  order = new LimitOrder(symbol, quantity, limitPrice, time, tag, properties);
467  break;
468 
469  case OrderType.StopMarket:
470  order = new StopMarketOrder(symbol, quantity, stopPrice, time, tag, properties);
471  break;
472 
473  case OrderType.StopLimit:
474  order = new StopLimitOrder(symbol, quantity, stopPrice, limitPrice, time, tag, properties);
475  break;
476 
477  case OrderType.TrailingStop:
478  order = new TrailingStopOrder(symbol, quantity, stopPrice, trailingAmount, trailingAsPercentage, time, tag, properties);
479  break;
480 
481  case OrderType.LimitIfTouched:
482  order = new LimitIfTouchedOrder(symbol, quantity, triggerPrice, limitPrice, time, tag, properties);
483  break;
484 
485  case OrderType.MarketOnOpen:
486  order = new MarketOnOpenOrder(symbol, quantity, time, tag, properties);
487  break;
488 
489  case OrderType.MarketOnClose:
490  order = new MarketOnCloseOrder(symbol, quantity, time, tag, properties);
491  break;
492 
493  case OrderType.OptionExercise:
494  order = new OptionExerciseOrder(symbol, quantity, time, tag, properties);
495  break;
496 
497  case OrderType.ComboLimit:
498  order = new ComboLimitOrder(symbol, quantity, limitPrice, time, groupOrderManager, tag, properties);
499  break;
500 
501  case OrderType.ComboLegLimit:
502  order = new ComboLegLimitOrder(symbol, quantity, limitPrice, time, groupOrderManager, tag, properties);
503  break;
504 
505  case OrderType.ComboMarket:
506  order = new ComboMarketOrder(symbol, quantity, time, groupOrderManager, tag, properties);
507  break;
508 
509  default:
510  throw new ArgumentOutOfRangeException();
511  }
512  order.Status = OrderStatus.New;
513  order.Id = orderId;
514  if (groupOrderManager != null)
515  {
516  lock (groupOrderManager.OrderIds)
517  {
518  groupOrderManager.OrderIds.Add(orderId);
519  }
520  }
521  return order;
522  }
523  }
