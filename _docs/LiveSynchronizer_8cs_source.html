26 namespace QuantConnect.Lean.Engine.DataFeeds

28  /// <summary>
29  /// Implementation of the <see cref="ISynchronizer"/> interface which provides the mechanism to stream live data to the algorithm
30  /// </summary>
31  public class LiveSynchronizer : Synchronizer
32  {
33  private ITimeProvider _timeProvider;
34  private LiveTimeProvider _frontierTimeProvider;
35  private RealTimeScheduleEventService _realTimeScheduleEventService;
36  private readonly int _batchingDelay = Config.GetInt("consumer-batching-timeout-ms");
37  private readonly ManualResetEventSlim _newLiveDataEmitted = new ManualResetEventSlim(false);
38 
39  /// <summary>
40  /// Continuous UTC time provider
41  /// </summary>
42  public override ITimeProvider TimeProvider => _timeProvider;
43 
44  /// <summary>
45  /// Initializes the instance of the Synchronizer class
46  /// </summary>
47  public override void Initialize(
48  IAlgorithm algorithm,
49  IDataFeedSubscriptionManager dataFeedSubscriptionManager)
50  {
51  base.Initialize(algorithm, dataFeedSubscriptionManager);
52 
53  // the time provider, is the real time provider
54  _timeProvider = GetTimeProvider();
55  _frontierTimeProvider = new LiveTimeProvider(realTime: TimeProvider);
56  // the synchronizer will use our '_frontierTimeProvider' which initially during warmup will be using
57  // the base time provider which is the subscription based time provider (like backtesting)
58  // once wawrmup finishes it will start using the realtime provider
59  SubscriptionSynchronizer.SetTimeProvider(_frontierTimeProvider);
60 
61  // attach event handlers to subscriptions
62  dataFeedSubscriptionManager.SubscriptionAdded += (sender, subscription) =>
63  {
64  subscription.NewDataAvailable += OnSubscriptionNewDataAvailable;
65  };
66 
67  dataFeedSubscriptionManager.SubscriptionRemoved += (sender, subscription) =>
68  {
69  subscription.NewDataAvailable -= OnSubscriptionNewDataAvailable;
70  };
71 
72  _realTimeScheduleEventService = new RealTimeScheduleEventService(new RealTimeProvider());
73  // this schedule event will be our time pulse
74  _realTimeScheduleEventService.NewEvent += (sender, args) => _newLiveDataEmitted.Set();
75  }
76 
77  /// <summary>
78  /// Returns an enumerable which provides the data to stream to the algorithm
79  /// </summary>
80  public override IEnumerable<TimeSlice> StreamData(CancellationToken cancellationToken)
81  {
82  PostInitialize();
83 
84  var shouldSendExtraEmptyPacket = false;
85  var nextEmit = DateTime.MinValue;
86  var lastLoopStart = DateTime.UtcNow;
87 
88  var enumerator = SubscriptionSynchronizer
89  .Sync(SubscriptionManager.DataFeedSubscriptions, cancellationToken)
90  .GetEnumerator();
91 
92  var previousWasTimePulse = false;
93  while (!cancellationToken.IsCancellationRequested)
94  {
95  var now = DateTime.UtcNow;
96  if (!previousWasTimePulse)
97  {
98  if (!_newLiveDataEmitted.IsSet
99  // we warmup as fast as we can even if no new data point is available
100  && !Algorithm.IsWarmingUp)
101  {
102  // if we just crossed into the next second let's loop again, we will flush any consolidator bar
103  // else we will wait to be notified by the subscriptions or our scheduled event service every second
104  if (lastLoopStart.Second == now.Second)
105  {
106  _realTimeScheduleEventService.ScheduleEvent(TimeSpan.FromMilliseconds(GetPulseDueTime(now)), now);
107  _newLiveDataEmitted.Wait();
108  }
109  }
110  _newLiveDataEmitted.Reset();
111  }
112 
113  lastLoopStart = now;
114 
115  TimeSlice timeSlice;
116  try
117  {
118  if (!enumerator.MoveNext())
119  {
120  // the enumerator ended
121  break;
122  }
123 
124  timeSlice = enumerator.Current;
125  }
126  catch (Exception err)
127  {
128  // notify the algorithm about the error, so it can be reported to the user
129  Algorithm.SetRuntimeError(err, "LiveSynchronizer");
130  shouldSendExtraEmptyPacket = true;
131  break;
132  }
133 
134  // check for cancellation
135  if (timeSlice == null || cancellationToken.IsCancellationRequested) break;
136 
137  var frontierUtc = FrontierTimeProvider.GetUtcNow();
138  // emit on data or if we've elapsed a full second since last emit or there are security changes
139  if (timeSlice.SecurityChanges != SecurityChanges.None
140  || timeSlice.IsTimePulse
141  || timeSlice.Data.Count != 0
142  || frontierUtc >= nextEmit)
143  {
144  previousWasTimePulse = timeSlice.IsTimePulse;
145  yield return timeSlice;
146 
147  // ignore if time pulse because we will emit a slice with the same time just after this one
148  if (!timeSlice.IsTimePulse)
149  {
150  // force emitting every second since the data feed is
151  // the heartbeat of the application
152  nextEmit = frontierUtc.RoundDown(Time.OneSecond).Add(Time.OneSecond);
153  }
154  }
155  }
156 
157  if (shouldSendExtraEmptyPacket)
158  {
159  // send last empty packet list before terminating,
160  // so the algorithm manager has a chance to detect the runtime error
161  // and exit showing the correct error instead of a timeout
162  nextEmit = FrontierTimeProvider.GetUtcNow().RoundDown(Time.OneSecond);
163  if (!cancellationToken.IsCancellationRequested)
164  {
165  var timeSlice = TimeSliceFactory.Create(
166  nextEmit,
167  new List<DataFeedPacket>(),
168  SecurityChanges.None,
169  new Dictionary<Universe, BaseDataCollection>());
170  yield return timeSlice;
171  }
172  }
173 
174  enumerator.DisposeSafely();
175  Log.Trace("LiveSynchronizer.GetEnumerator(): Exited thread.");
176  }
177 
178  /// <summary>
179  /// Free resources
180  /// </summary>
181  public override void Dispose()
182  {
183  _newLiveDataEmitted.Set();
184  _newLiveDataEmitted?.DisposeSafely();
185  _realTimeScheduleEventService?.DisposeSafely();
186  }
187 
188  /// <summary>
189  /// Gets the <see cref="ITimeProvider"/> to use. By default this will load the
190  /// <see cref="RealTimeProvider"/> for live mode, else <see cref="SubscriptionFrontierTimeProvider"/>
191  /// </summary>
192  /// <returns>The <see cref="ITimeProvider"/> to use</returns>
193  protected override ITimeProvider GetTimeProvider()
194  {
195  return RealTimeProvider.Instance;
196  }
197 
198  /// <summary>
199  /// Performs additional initialization steps after algorithm initialization
200  /// </summary>
201  protected override void PostInitialize()
202  {
203  base.PostInitialize();
204  _frontierTimeProvider.Initialize(base.GetTimeProvider());
205  }
206 
207  /// <summary>
208  /// Will return the amount of milliseconds that are missing for the next time pulse
209  /// </summary>
210  protected virtual int GetPulseDueTime(DateTime now)
211  {
212  // let's wait until the next second starts
213  return 1000 - now.Millisecond + _batchingDelay;
214  }
215 
216  /// <summary>
217  /// Trigger new data event
218  /// </summary>
219  /// <param name="sender">Sender of the event</param>
220  /// <param name="args">Event information</param>
221  protected virtual void OnSubscriptionNewDataAvailable(object sender, EventArgs args)
222  {
223  _newLiveDataEmitted.Set();
224  }
225  }
