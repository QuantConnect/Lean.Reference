16 using QuantConnect.Data;
17 using QuantConnect.Data.Consolidators;
18 using QuantConnect.Data.Market;
19 using QuantConnect.Lean.Engine.DataFeeds.Enumerators;
20 using QuantConnect.Logging;
21 using System;
22 using System.Collections.Concurrent;
23 using System.Collections.Generic;
24 using System.Linq;

26 namespace QuantConnect.Lean.Engine.DataFeeds
27 {
28  /// <summary>
29  /// Aggregates ticks and bars based on given subscriptions.
30  /// Current implementation is based on <see cref="IDataConsolidator"/> that consolidates ticks and put them into enumerator.
31  /// </summary>
32  public class AggregationManager : IDataAggregator
33  {
34  private readonly ConcurrentDictionary<SecurityIdentifier, List<KeyValuePair<SubscriptionDataConfig, ScannableEnumerator<BaseData>>>> _enumerators
35  = new ConcurrentDictionary<SecurityIdentifier, List<KeyValuePair<SubscriptionDataConfig, ScannableEnumerator<BaseData>>>>();
36 
37  /// <summary>
38  /// Continuous UTC time provider
39  /// </summary>
40  protected ITimeProvider TimeProvider { get; set; } = RealTimeProvider.Instance;
41 
42  /// <summary>
43  /// Add new subscription to current <see cref="IDataAggregator"/> instance
44  /// </summary>
45  /// <param name="dataConfig">defines the parameters to subscribe to a data feed</param>
46  /// <param name="newDataAvailableHandler">handler to be fired on new data available</param>
47  /// <returns>The new enumerator for this subscription request</returns>
48  public IEnumerator<BaseData> Add(SubscriptionDataConfig dataConfig, EventHandler newDataAvailableHandler)
49  {
50  var consolidator = GetConsolidator(dataConfig);
51  var isPeriodBased = (dataConfig.Type.Name == nameof(QuoteBar) ||
52  dataConfig.Type.Name == nameof(TradeBar) ||
53  dataConfig.Type.Name == nameof(OpenInterest)) &&
54  dataConfig.Resolution != Resolution.Tick;
55  var enumerator = new ScannableEnumerator<BaseData>(consolidator, dataConfig.ExchangeTimeZone, TimeProvider, newDataAvailableHandler, isPeriodBased);
56 
57  _enumerators.AddOrUpdate(
58  dataConfig.Symbol.ID,
59  new List<KeyValuePair<SubscriptionDataConfig, ScannableEnumerator<BaseData>>> { new KeyValuePair<SubscriptionDataConfig, ScannableEnumerator<BaseData>>(dataConfig, enumerator) },
60  (k, v) => { return v.Concat(new[] { new KeyValuePair<SubscriptionDataConfig, ScannableEnumerator<BaseData>>(dataConfig, enumerator) }).ToList(); });
61 
62  return enumerator;
63  }
64 
65  /// <summary>
66  /// Removes the handler with the specified identifier
67  /// </summary>
68  /// <param name="dataConfig">Subscription data configuration to be removed</param>
69  public bool Remove(SubscriptionDataConfig dataConfig)
70  {
71  List<KeyValuePair<SubscriptionDataConfig, ScannableEnumerator<BaseData>>> enumerators;
72  if (_enumerators.TryGetValue(dataConfig.Symbol.ID, out enumerators))
73  {
74  if (enumerators.Count == 1)
75  {
76  List<KeyValuePair<SubscriptionDataConfig, ScannableEnumerator<BaseData>>> output;
77  return _enumerators.TryRemove(dataConfig.Symbol.ID, out output);
78  }
79  else
80  {
81  _enumerators[dataConfig.Symbol.ID] = enumerators.Where(pair => pair.Key != dataConfig).ToList();
82  return true;
83  }
84  }
85  else
86  {
87  Log.Debug($"AggregationManager.Update(): IDataConsolidator for symbol ({dataConfig.Symbol.Value}) was not found.");
88  return false;
89  }
90  }
91 
92  /// <summary>
93  /// Add new data to aggregator
94  /// </summary>
95  /// <param name="input">The new data</param>
96  public void Update(BaseData input)
97  {
98  try
99  {
100  List<KeyValuePair<SubscriptionDataConfig, ScannableEnumerator<BaseData>>> enumerators;
101  if (_enumerators.TryGetValue(input.Symbol.ID, out enumerators))
102  {
103  for (var i = 0; i < enumerators.Count; i++)
104  {
105  var kvp = enumerators[i];
106 
107  // for non tick resolution subscriptions drop suspicious ticks
108  if (kvp.Key.Resolution != Resolution.Tick)
109  {
110  var tick = input as Tick;
111  if (tick != null && tick.Suspicious)
112  {
113  continue;
114  }
115  }
116 
117  kvp.Value.Update(input);
118  }
119  }
120  }
121  catch (Exception exception)
122  {
123  Log.Error(exception);
124  }
125  }
126 
127  /// <summary>
128  /// Dispose of the aggregation manager.
129  /// </summary>
130  public void Dispose() { }
131 
132  /// <summary>
133  /// Gets the consolidator to aggregate data for the given config
134  /// </summary>
135  protected virtual IDataConsolidator GetConsolidator(SubscriptionDataConfig config)
136  {
137  var period = config.Resolution.ToTimeSpan();
138  if (config.Type == typeof(QuoteBar))
139  {
140  return new TickQuoteBarConsolidator(period);
141  }
142  if (config.Type == typeof(TradeBar))
143  {
144  return new TickConsolidator(period);
145  }
146  if (config.Type == typeof(OpenInterest))
147  {
148  return new OpenInterestConsolidator(period);
149  }
150  if (config.Type == typeof(Tick))
151  {
152  return FilteredIdentityDataConsolidator.ForTickType(config.TickType);
153  }
154  if (config.Type == typeof(Split))
155  {
156  return new IdentityDataConsolidator<Split>();
157  }
158  if (config.Type == typeof(Dividend))
159  {
160  return new IdentityDataConsolidator<Dividend>();
161  }
162 
163  // streaming custom data subscriptions can pass right through
164  return new FilteredIdentityDataConsolidator<BaseData>(data => data.GetType() == config.Type);
165  }
166  }
167 }
