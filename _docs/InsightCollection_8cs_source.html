16 using System;
17 using System.Collections;
18 using System.Collections.Concurrent;
19 using System.Collections.Generic;
20 using System.Linq;
21 
22 namespace QuantConnect.Algorithm.Framework.Alphas
23 {
24  /// <summary>
25  /// Provides a collection for managing insights. This type provides collection access semantics
26  /// as well as dictionary access semantics through TryGetValue, ContainsKey, and this[symbol]
27  /// </summary>
28  public class InsightCollection : ICollection<Insight>
29  {
30  // for performance lets keep the next insight expiration time
31  private DateTime? _nextExpiryTime;
32  private readonly ConcurrentDictionary<Symbol, List<Insight>> _insights = new ConcurrentDictionary<Symbol, List<Insight>>();
33 
34  /// <summary>Gets the number of elements contained in the <see cref="T:System.Collections.Generic.ICollection`1" />.</summary>
35  /// <returns>The number of elements contained in the <see cref="T:System.Collections.Generic.ICollection`1" />.</returns>
36  public int Count => _insights.Aggregate(0, (i, kvp) => i + kvp.Value.Count);
37 
38  /// <summary>Gets a value indicating whether the <see cref="T:System.Collections.Generic.ICollection`1" /> is read-only.</summary>
39  /// <returns>true if the <see cref="T:System.Collections.Generic.ICollection`1" /> is read-only; otherwise, false.</returns>
40  public bool IsReadOnly => false;
41 
42  /// <summary>Adds an item to the <see cref="T:System.Collections.Generic.ICollection`1" />.</summary>
43  /// <param name="item">The object to add to the <see cref="T:System.Collections.Generic.ICollection`1" />.</param>
44  /// <exception cref="T:System.NotSupportedException">The <see cref="T:System.Collections.Generic.ICollection`1" /> is read-only.</exception>
45  public void Add(Insight item)
46  {
47  _nextExpiryTime = null;
48  _insights.AddOrUpdate(item.Symbol, s => new List<Insight> {item}, (s, list) =>
49  {
50  list.Add(item);
51  return list;
52  });
53  }
54 
55  /// <summary>
56  /// Adds each item in the specified enumerable of insights to this collection
57  /// </summary>
58  /// <param name="insights">The insights to add to this collection</param>
59  public void AddRange(IEnumerable<Insight> insights)
60  {
61  foreach (var insight in insights)
62  {
63  Add(insight);
64  }
65  }
66 
67  /// <summary>Removes all items from the <see cref="T:System.Collections.Generic.ICollection`1" />.</summary>
68  /// <exception cref="T:System.NotSupportedException">The <see cref="T:System.Collections.Generic.ICollection`1" /> is read-only. </exception>
69  public void Clear()
70  {
71  _nextExpiryTime = null;
72  _insights.Clear();
73  }
74 
75  /// <summary>Determines whether the <see cref="T:System.Collections.Generic.ICollection`1" /> contains a specific value.</summary>
76  /// <returns>true if <paramref name="item" /> is found in the <see cref="T:System.Collections.Generic.ICollection`1" />; otherwise, false.</returns>
77  /// <param name="item">The object to locate in the <see cref="T:System.Collections.Generic.ICollection`1" />.</param>
78  public bool Contains(Insight item)
79  {
80  List<Insight> symbolInsights;
81  if (_insights.TryGetValue(item.Symbol, out symbolInsights))
82  {
83  return symbolInsights.Contains(item);
84  }
85 
86  return false;
87  }
88 
89  /// <summary>
90  /// Determines whether insights exist in this collection for the specified symbol
91  /// </summary>
92  /// <param name="symbol">The symbol key</param>
93  /// <returns>True if there are insights for the symbol in this collection</returns>
94  public bool ContainsKey(Symbol symbol)
95  {
96  List<Insight> insights;
97  return _insights.TryGetValue(symbol, out insights) && insights.Count > 0;
98  }
99 
100  /// <summary>Copies the elements of the <see cref="T:System.Collections.Generic.ICollection`1" /> to an <see cref="T:System.Array" />, starting at a particular <see cref="T:System.Array" /> index.</summary>
101  /// <param name="array">The one-dimensional <see cref="T:System.Array" /> that is the destination of the elements copied from <see cref="T:System.Collections.Generic.ICollection`1" />. The <see cref="T:System.Array" /> must have zero-based indexing.</param>
102  /// <param name="arrayIndex">The zero-based index in <paramref name="array" /> at which copying begins.</param>
103  /// <exception cref="T:System.ArgumentNullException">
104  /// <paramref name="array" /> is null.</exception>
105  /// <exception cref="T:System.ArgumentOutOfRangeException">
106  /// <paramref name="arrayIndex" /> is less than 0.</exception>
107  /// <exception cref="T:System.ArgumentException">The number of elements in the source <see cref="T:System.Collections.Generic.ICollection`1" /> is greater than the available space from <paramref name="arrayIndex" /> to the end of the destination <paramref name="array" />.</exception>
108  public void CopyTo(Insight[] array, int arrayIndex)
109  {
110  // Avoid calling `ToList` on insights to avoid potential infinite loop (issue #3168)
111  Array.Copy(_insights.SelectMany(kvp => kvp.Value).ToArray(), 0, array, arrayIndex, Count);
112  }
113 
114  /// <summary>Removes the first occurrence of a specific object from the <see cref="T:System.Collections.Generic.ICollection`1" />.</summary>
115  /// <returns>true if <paramref name="item" /> was successfully removed from the <see cref="T:System.Collections.Generic.ICollection`1" />; otherwise, false. This method also returns false if <paramref name="item" /> is not found in the original <see cref="T:System.Collections.Generic.ICollection`1" />.</returns>
116  /// <param name="item">The object to remove from the <see cref="T:System.Collections.Generic.ICollection`1" />.</param>
117  /// <exception cref="T:System.NotSupportedException">The <see cref="T:System.Collections.Generic.ICollection`1" /> is read-only.</exception>
118  public bool Remove(Insight item)
119  {
120  List<Insight> symbolInsights;
121  if (_insights.TryGetValue(item.Symbol, out symbolInsights))
122  {
123  _nextExpiryTime = null;
124 
125  if (symbolInsights.Remove(item))
126  {
127  // remove empty list from dictionary
128  if (symbolInsights.Count == 0)
129  {
130  _insights.TryRemove(item.Symbol, out symbolInsights);
131  }
132 
133  return true;
134  }
135  }
136 
137  return false;
138  }
139 
140  /// <summary>
141  /// Dictionary accessor returns a list of insights for the specified symbol
142  /// </summary>
143  /// <param name="symbol">The symbol key</param>
144  /// <returns>List of insights for the symbol</returns>
145  public List<Insight> this[Symbol symbol]
146  {
147  get { return _insights[symbol]; }
148  set { _insights[symbol] = value; }
149  }
150 
151  /// <summary>
152  /// Attempts to get the list of insights with the specified symbol key
153  /// </summary>
154  /// <param name="symbol">The symbol key</param>
155  /// <param name="insights">The insights for the specified symbol, or null if not found</param>
156  /// <returns>True if insights for the specified symbol were found, false otherwise</returns>
157  public bool TryGetValue(Symbol symbol, out List<Insight> insights)
158  {
159  return _insights.TryGetValue(symbol, out insights);
160  }
161 
162  /// <summary>Returns an enumerator that iterates through the collection.</summary>
163  /// <returns>A <see cref="T:System.Collections.Generic.IEnumerator`1" /> that can be used to iterate through the collection.</returns>
164  /// <filterpriority>1</filterpriority>
165  public IEnumerator<Insight> GetEnumerator()
166  {
167  return _insights.SelectMany(kvp => kvp.Value).GetEnumerator();
168  }
169 
170  /// <summary>Returns an enumerator that iterates through a collection.</summary>
171  /// <returns>An <see cref="T:System.Collections.IEnumerator" /> object that can be used to iterate through the collection.</returns>
172  /// <filterpriority>2</filterpriority>
173  IEnumerator IEnumerable.GetEnumerator()
174  {
175  return GetEnumerator();
176  }

178  /// <summary>
179  /// Removes the symbol and its insights
180  /// </summary>
181  /// <param name="symbols">List of symbols that will be removed</param>
182  public void Clear(Symbol[] symbols)
183  {
184  _nextExpiryTime = null;
185  foreach (var symbol in symbols)
186  {
187  List<Insight> insights;
188  _insights.TryRemove(symbol, out insights);
189  }
190  }
191 
192  /// <summary>
193  /// Gets the next expiry time UTC
194  /// </summary>
195  public DateTime? GetNextExpiryTime()
196  {
197  if (Count == 0)
198  {
199  return null;
200  }
201 
202  if (_nextExpiryTime != null)
203  {
204  return _nextExpiryTime;
205  }
206 
207  _nextExpiryTime = _insights.Min(x => x.Value.Min(i => i.CloseTimeUtc));
208  return _nextExpiryTime;
209  }

211  /// <summary>
212  /// Gets the last generated active insight
213  /// </summary>
214  /// <returns>Collection of insights that are active</returns>
215  public ICollection<Insight> GetActiveInsights(DateTime utcTime)
216  {
217  var activeInsights = new List<Insight>();
218  foreach (var kvp in _insights)
219  {
220  foreach (var insight in kvp.Value)
221  {
222  if (insight.IsActive(utcTime))
223  {
224  activeInsights.Add(insight);
225  }
226  }
227  }
228  return activeInsights;
229  }
230 
231  /// <summary>
232  /// Returns true if there are active insights for a given symbol and time
233  /// </summary>
234  /// <param name="symbol">The symbol key</param>
235  /// <param name="utcTime">Time that determines whether the insight has expired</param>
236  /// <returns></returns>
237  public bool HasActiveInsights(Symbol symbol, DateTime utcTime)
238  {
239  List<Insight> insights;
240  if (TryGetValue(symbol, out insights))
241  {
242  return insights.Any(i => i.IsActive(utcTime));
243  }
244  return false;
245  }
246 
247  /// <summary>
248  /// Remove all expired insights from the collection and retuns them
249  /// </summary>
250  /// <param name="utcTime">Time that determines whether the insight has expired</param>
251  /// <returns>Expired insights that were removed</returns>
252  public ICollection<Insight> RemoveExpiredInsights(DateTime utcTime)
253  {
254  var removedInsights = new List<Insight>();
255  foreach (var kvp in _insights)
256  {
257  foreach (var insight in kvp.Value)
258  {
259  if (insight.IsExpired(utcTime))
260  {
261  removedInsights.Add(insight);
262  }
263  }
264  }
265  foreach (var insight in removedInsights)
266  {
267  Remove(insight);
268  }
269  return removedInsights;
270  }
271  }
