16 using System;
17 using QuantConnect.Data;
18 using QuantConnect.Interfaces;
19 using System.Collections.Generic;
20 
21 namespace QuantConnect.Lean.Engine.DataFeeds
22 {
23  /// <summary>
24  /// Provides an implementation of <see cref="ISubscriptionDataSourceReader"/> that reads zip entry names
25  /// </summary>
26  public class ZipEntryNameSubscriptionDataSourceReader : ISubscriptionDataSourceReader
27  {
28  private readonly IDataCacheProvider _dataProvider;
29  private readonly SubscriptionDataConfig _config;
30  private readonly DateTime _date;
31  private readonly bool _isLiveMode;
32  private readonly BaseData _factory;
33 
34  /// <summary>
35  /// Event fired when the specified source is considered invalid, this may
36  /// be from a missing file or failure to download a remote source
37  /// </summary>
38  public event EventHandler<InvalidSourceEventArgs> InvalidSource;
39 
40  /// <summary>
41  /// Initializes a new instance of the <see cref="ZipEntryNameSubscriptionDataSourceReader"/> class
42  /// </summary>
43  /// <param name="dataProvider">Used to fetch data</param>
44  /// <param name="config">The subscription's configuration</param>
45  /// <param name="date">The date this factory was produced to read data for</param>
46  /// <param name="isLiveMode">True if we're in live mode, false for backtesting</param>
47  public ZipEntryNameSubscriptionDataSourceReader(IDataCacheProvider dataProvider, SubscriptionDataConfig config, DateTime date, bool isLiveMode)
48  {
49  _date = date;
50  _config = config;
51  _isLiveMode = isLiveMode;
52  _dataProvider = dataProvider;
53  _factory = config.GetBaseDataInstance();
54  }
55 
56  /// <summary>
57  /// Reads the specified <paramref name="source"/>
58  /// </summary>
59  /// <param name="source">The source to be read</param>
60  /// <returns>An <see cref="IEnumerable{BaseData}"/> that contains the data in the source</returns>
61  public IEnumerable<BaseData> Read(SubscriptionDataSource source)
62  {
63  List<string> entryNames;
64  try
65  {
66  entryNames = _dataProvider.GetZipEntries(source.Source);
67  }
68  catch (Exception err)
69  {
70  OnInvalidSource(source, err);
71  yield break;
72  }
73 
74  foreach (var entryFileName in entryNames)
75  {
76  var instance = _factory.Reader(_config, entryFileName, _date, _isLiveMode);
77  if (instance != null && instance.EndTime != default(DateTime))
78  {
79  yield return instance;
80  }
81  }
82  }
83 
84  /// <summary>
85  /// Event invocator for the <see cref="InvalidSource"/> event
86  /// </summary>
87  /// <param name="source">The <see cref="SubscriptionDataSource"/> that was invalid</param>
88  /// <param name="exception">The exception if one was raised, otherwise null</param>
89  private void OnInvalidSource(SubscriptionDataSource source, Exception exception)
90  {
91  InvalidSource?.Invoke(this, new InvalidSourceEventArgs(source, exception));
92  }
93  }
94 }
