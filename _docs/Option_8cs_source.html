276  /// <summary>
277  /// Returns the directional quantity of underlying shares that are going to change hands on exercise/assignment of all
278  /// contracts held by this account, taking into account the contract's <see cref="Right"/> as well as the contract's current
279  /// <see cref="ContractUnitOfTrade"/>, which may have recently changed due to a split/reverse split in the underlying security.
280  /// </summary>
281  /// <remarks>
282  /// Long option positions result in exercise while short option positions result in assignment. This function uses the term
283  /// exercise loosely to refer to both situations.
284  /// </remarks>
285  public decimal GetExerciseQuantity()
286  {
287  // negate Holdings.Quantity to match an equivalent order
288  return GetExerciseQuantity(-Holdings.Quantity);
289  }

291  /// <summary>
292  /// Returns the directional quantity of underlying shares that are going to change hands on exercise/assignment of the
293  /// specified <paramref name="exerciseOrderQuantity"/>, taking into account the contract's <see cref="Right"/> as well
294  /// as the contract's current <see cref="ContractUnitOfTrade"/>, which may have recently changed due to a split/reverse
295  /// split in the underlying security.
296  /// </summary>
297  /// <remarks>
298  /// Long option positions result in exercise while short option positions result in assignment. This function uses the term
299  /// exercise loosely to refer to both situations.
300  /// </remarks>
301  /// <paramref name="exerciseOrderQuantity">The quantity of contracts being exercised as provided by the <see cref="OptionExerciseOrder"/>.
302  /// A negative value indicates exercise (we are long and the order quantity is negative to bring us (closer) to zero.
303  /// A positive value indicates assignment (we are short and the order quantity is positive to bring us (closer) to zero.</paramref>
304  public decimal GetExerciseQuantity(decimal exerciseOrderQuantity)
305  {
306  // when exerciseOrderQuantity > 0 [ we are short ]
307  // && right == call => we sell to contract holder => negative
308  // && right == put => we buy from contract holder => positive
309 
310  // when exerciseOrderQuantity < 0 [ we are long ]
311  // && right == call => we buy from contract holder => positive
312  // && right == put => we sell to contract holder => negative
313 
314  var sign = Right == OptionRight.Call ? -1 : 1;
315  return sign * exerciseOrderQuantity * ContractUnitOfTrade;
316  }

326  /// <summary>

333  /// <summary>
334  /// Option payoff function at expiration time
335  /// </summary>
336  /// <param name="underlyingPrice">The price of the underlying</param>
337  /// <returns></returns>
338  public decimal GetPayOff(decimal underlyingPrice)
339  {
340  return Right == OptionRight.Call ? underlyingPrice - StrikePrice : StrikePrice - underlyingPrice;

499  /// <summary>
500  /// Sets the <see cref="ContractFilter"/> to a new universe selection function
501  /// </summary>
502  /// <param name="universeFunc">new universe selection function</param>
503  public void SetFilter(Func<OptionFilterUniverse, OptionFilterUniverse> universeFunc)
504  {
505  ContractFilter = new FuncSecurityDerivativeFilter(universe =>

509  return result.ApplyTypesFilter();

513  /// <summary>
514  /// Sets the <see cref="ContractFilter"/> to a new universe selection function
515  /// </summary>
516  /// <param name="universeFunc">new universe selection function</param>
517  public void SetFilter(PyObject universeFunc)
518  {
519  ContractFilter = new FuncSecurityDerivativeFilter(universe =>
520  {
521  var optionUniverse = universe as OptionFilterUniverse;
522  using (Py.GIL())
523  {
524  PyObject result = (universeFunc as dynamic)(optionUniverse);
525 
526  //Try to convert it to the possible outcomes and process it
527  //Must try filter first, if it is a filter and you try and convert it to
528  //list, TryConvert() with catch an exception. Later Python algo will break on
529  //this exception because we are using Py.GIL() and it will see the error set
530  OptionFilterUniverse filter;
531  List<Symbol> list;
532 
533  if ((result).TryConvert(out filter))
534  {
535  optionUniverse = filter;
536  }
537  else if ((result).TryConvert(out list))
538  {
539  optionUniverse = optionUniverse.WhereContains(list);
540  }
541  else
542  {
543  throw new ArgumentException($"QCAlgorithm.SetFilter: result type {result.GetPythonType()} from " +
544  $"filter function is not a valid argument, please return either a OptionFilterUniverse or a list of symbols");
545  }
546  }
547  return optionUniverse.ApplyTypesFilter();
548  });
549  }
