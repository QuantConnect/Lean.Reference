37  /// <summary>
38  /// Runs LEAN to calculate the portfolio at a given time from <see cref="Order"/> objects.

165  /// <summary>
166  /// Internal method to get the history for the given securities
167  /// </summary>
168  /// <param name="algorithm">Algorithm</param>
169  /// <param name="securities">Securities to get history for</param>
170  /// <param name="resolution">Resolution to retrieve data in</param>
171  /// <returns>History of the given securities</returns>
172  /// <remarks>Method is static because we want to use it from the constructor as well</remarks>
173  private static IEnumerable<Slice> GetHistory(IAlgorithm algorithm, List<Security> securities, Resolution resolution)
174  {
175  var historyRequests = new List<Data.HistoryRequest>();
176  var historyRequestFactory = new HistoryRequestFactory(algorithm);
177 
178  // Create the history requests
179  foreach (var security in securities)
180  {
181  var configs = algorithm.SubscriptionManager
182  .SubscriptionDataConfigService
183  .GetSubscriptionDataConfigs(security.Symbol, includeInternalConfigs: true);
184 
185  // we need to order and select a specific configuration type
186  // so the conversion rate is deterministic
187  var configToUse = configs.OrderBy(x => x.TickType).First();
188 
189  var startTime = historyRequestFactory.GetStartTimeAlgoTz(
190  security.Symbol,
191  1,
192  resolution,
193  security.Exchange.Hours,
194  configToUse.DataTimeZone);
195  var endTime = algorithm.EndDate;
196 
197  historyRequests.Add(historyRequestFactory.CreateHistoryRequest(
198  configToUse,
199  startTime,
200  endTime,
201  security.Exchange.Hours,
202  resolution
203  ));
204  }
205 
206  return algorithm.HistoryProvider.GetHistory(historyRequests, algorithm.TimeZone).ToList();
207  }

209  /// <summary>
210  /// Gets the history for the given symbols from the <paramref name="start"/> to the <paramref name="end"/>
211  /// </summary>
212  /// <param name="symbols">Symbols to request history for</param>
213  /// <param name="start">Start date of history request</param>
214  /// <param name="end">End date of history request</param>
215  /// <param name="resolution">Resolution of history request</param>
216  /// <returns>Enumerable of slices</returns>
217  public static IEnumerable<Slice> GetHistory(List<Symbol> symbols, DateTime start, DateTime end, Resolution resolution)
218  {
219  // Handles the conversion of Symbol to Security for us.
220  var looper = new PortfolioLooper(0, new List<Order>(), resolution);
221  var securities = new List<Security>();
222 
223  looper.Algorithm.SetStartDate(start);
224  looper.Algorithm.SetEndDate(end);
225 
226  foreach (var symbol in symbols)
227  {
228  var configs = looper.Algorithm.SubscriptionManager.SubscriptionDataConfigService.Add(symbol, resolution, false, false);
229  securities.Add(looper.Algorithm.Securities.CreateSecurity(symbol, configs));
230  }
231 
232  return GetHistory(looper.Algorithm, securities, resolution);
233  }

235  /// <summary>
236  /// Gets the point in time portfolio over multiple deployments
237  /// </summary>
238  /// <param name="equityCurve">Equity curve series</param>
239  /// <param name="orders">Orders</param>
240  /// <param name="algorithmConfiguration">Optional parameter to override default algorithm configuration</param>
241  /// <param name="liveSeries">Equity curve series originates from LiveResult</param>
242  /// <returns>Enumerable of <see cref="PointInTimePortfolio"/></returns>
243  public static IEnumerable<PointInTimePortfolio> FromOrders(Series<DateTime, double> equityCurve, IEnumerable<Order> orders,
244  AlgorithmConfiguration algorithmConfiguration = null, bool liveSeries = false)
245  {
246  // Don't do anything if we have no orders or equity curve to process
247  if (!orders.Any() || equityCurve.IsEmpty)
248  {
249  yield break;
250  }
251 
252  // Chunk different deployments into separate Lists for separate processing
253  var portfolioDeployments = new List<List<Order>>();
254 
255  // Orders are guaranteed to start counting from 1. This ensures that we have
256  // no collision at all with the start of a deployment
257  var previousOrderId = 0;
258  var currentDeployment = new List<Order>();
259 
260  // Make use of reference semantics to add new deployments to the list
261  portfolioDeployments.Add(currentDeployment);
262 
263  foreach (var order in orders)
264  {
265  // In case we have two different deployments with only a single
266  // order in the deployments, <= was chosen because it covers duplicate values
267  if (order.Id <= previousOrderId)
268  {
269  currentDeployment = new List<Order>();
270  portfolioDeployments.Add(currentDeployment);
271  }
272 
273  currentDeployment.Add(order);
274  previousOrderId = order.Id;
275  }
276 
277  PortfolioLooper looper = null;
278  PointInTimePortfolio prev = null;
279  foreach (var deploymentOrders in portfolioDeployments)
280  {
281  if (deploymentOrders.Count == 0)
282  {
283  Log.Trace($"PortfolioLooper.FromOrders(): Deployment contains no orders");
284  continue;
285  }
286  var startTime = deploymentOrders.First().Time;
287  var deployment = equityCurve.Where(kvp => kvp.Key <= startTime);
288  if (deployment.IsEmpty)
289  {
290  Log.Trace($"PortfolioLooper.FromOrders(): Equity series is empty after filtering with upper bound: {startTime}");
291  continue;
292  }
293 
294  // Skip any deployments that haven't been ran long enough to be generated in live mode
295  if (liveSeries && deploymentOrders.First().Time.Date == deploymentOrders.Last().Time.Date)
296  {
297  Log.Trace("PortfolioLooper.FromOrders(): Filtering deployment because it has not been deployed for more than one day");
298  continue;
299  }
300 
301  // For every deployment, we want to start fresh.
302  looper = new PortfolioLooper(deployment.LastValue(), deploymentOrders, algorithmConfiguration: algorithmConfiguration);
303 
304  foreach (var portfolio in looper.ProcessOrders(deploymentOrders))
305  {
306  prev = portfolio;
307  yield return portfolio;
308  }
309  }
310 
311  if (prev != null)
312  {
313  yield return new PointInTimePortfolio(prev, equityCurve.LastKey());
314  }
315 
316  looper.DisposeSafely();
317  }

319  /// <summary>
320  /// Process the orders
321  /// </summary>
322  /// <param name="orders">orders</param>
323  /// <returns>PointInTimePortfolio</returns>
324  private IEnumerable<PointInTimePortfolio> ProcessOrders(IEnumerable<Order> orders)
325  {
326  // Portfolio.ProcessFill(...) does not filter out invalid orders. We must do so ourselves
327  foreach (var order in orders)
328  {
329  Algorithm.SetDateTime(order.Time);
330 
331  var orderSecurity = Algorithm.Securities[order.Symbol];
332  DateTime lastFillTime;
333 
334  if ((order.Type == OrderType.MarketOnOpen || order.Type == OrderType.MarketOnClose) &&
335  (order.Status == OrderStatus.Filled || order.Status == OrderStatus.PartiallyFilled) && order.LastFillTime == null)
336  {
337  lastFillTime = order.Time;
338  }
339  else if (order.LastFillTime == null)
340  {
341  Log.Trace($"Order with ID: {order.Id} has been skipped because of null LastFillTime");
342  continue;
343  }
344  else
345  {
346  lastFillTime = order.LastFillTime.Value;
347  }
348 
349  var tick = new Tick { Quantity = order.Quantity, AskPrice = order.Price, BidPrice = order.Price, Value = order.Price, EndTime = lastFillTime };
350  var tradeBar = new TradeBar
351  {
352  Open = order.Price,
353  High = order.Price,
354  Low = order.Price,
355  Close = order.Price,
356  Volume = order.Quantity,
357 
358  DataType = MarketDataType.TradeBar,
359  Period = TimeSpan.Zero,
360  Symbol = order.Symbol,
361  Time = lastFillTime,
362  };
363 
364  // Required for crypto so that the Cache Price is updated accordingly,
365  // since its `Security.Price` implementation explicitly requests TradeBars.
366  // For most asset types this might be enough as well, but there is the
367  // possibility that some trades might get filtered, so we cover that
368  // case by setting the market price via Tick as well.
369  orderSecurity.SetMarketPrice(tradeBar);
370  orderSecurity.SetMarketPrice(tick);
371 
372  // Check if we have a base currency (i.e. forex or crypto that requires currency conversion)
373  // to ensure the proper conversion rate is set for them
374  var baseCurrency = orderSecurity as IBaseCurrencySymbol;
375 
376  if (baseCurrency != null)
377  {
378  // We want slices that apply to either this point in time, or the last most recent point in time
379  var updateSlices = _conversionSlices.Where(x => x.Time <= order.Time).ToList();
380 
381  // This is put here because there can potentially be no slices
382  if (updateSlices.Count != 0)
383  {
384  var updateSlice = updateSlices.Last();
385 
386  foreach (var quoteBar in updateSlice.QuoteBars.Values)
387  {
388  Algorithm.Securities[quoteBar.Symbol].SetMarketPrice(quoteBar);
389  }
390  }
391  }
392 
393  // Update our cash holdings before we invalidate the portfolio value
394  // to calculate the proper cash value of other assets the algo owns
395  foreach (var cash in Algorithm.Portfolio.CashBook.Values.Where(x => x.CurrencyConversion != null))
396  {
397  cash.Update();
398  }
399 
400  // Securities prices might have been updated, so we need to recalculate how much
401  // money we have in our portfolio, otherwise we risk being out of date and
402  // calculate on stale data.
403  Algorithm.Portfolio.InvalidateTotalPortfolioValue();
404 
405  var orderEvent = new OrderEvent(order, order.Time, Orders.Fees.OrderFee.Zero) { FillPrice = order.Price, FillQuantity = order.Quantity };
406 
407  // Process the order
408  Algorithm.Portfolio.ProcessFills(new List<OrderEvent> { orderEvent });
409 
410  // Create portfolio statistics and return back to the user
411  yield return new PointInTimePortfolio(order, Algorithm.Portfolio);
412  }
413  }
414  }
