16 using System;
17 using System.Collections.Generic;
18 using System.Dynamic;
19 using System.Linq.Expressions;
20 using System.Reflection;
21 using QuantConnect.Util;

23 namespace QuantConnect.Data
24 {
25  /// <summary>
26  /// Dynamic Data Class: Accept flexible data, adapting to the columns provided by source.
27  /// </summary>
28  /// <remarks>Intended for use with Quandl class.</remarks>
29  public abstract class DynamicData : BaseData, IDynamicMetaObjectProvider
30  {
31  private static readonly MethodInfo SetPropertyMethodInfo = typeof(DynamicData).GetMethod("SetProperty");
32  private static readonly MethodInfo GetPropertyMethodInfo = typeof(DynamicData).GetMethod("GetProperty");
33 
34  private readonly IDictionary<string, object> _storage = new Dictionary<string, object>();
35 
36  /// <summary>
37  /// Get the metaObject required for Dynamism.
38  /// </summary>
39  public DynamicMetaObject GetMetaObject(Expression parameter)
40  {
41  return new GetSetPropertyDynamicMetaObject(parameter, this, SetPropertyMethodInfo, GetPropertyMethodInfo);
42  }
43 
44  /// <summary>
45  /// Sets the property with the specified name to the value. This is a case-insensitve search.
46  /// </summary>
47  /// <param name="name">The property name to set</param>
48  /// <param name="value">The new property value</param>
49  /// <returns>Returns the input value back to the caller</returns>
50  public object SetProperty(string name, object value)
51  {
52  name = name.ToLowerInvariant();
53 
54  if (name == "time")
55  {
56  Time = (DateTime)value;
57  }
58  if (name == "endtime")
59  {
60  EndTime = (DateTime)value;
61  }
62  if (name == "value")
63  {
64  Value = (decimal)value;
65  }
66  if (name == "symbol")
67  {
68  if (value is string)
69  {
70  Symbol = SymbolCache.GetSymbol((string) value);
71  }
72  else
73  {
74  Symbol = (Symbol) value;
75  }
76  }
77  // reaodnly
78  //if (name == "Price")
79  //{
80  // return Price = (decimal) value;
81  //}
82  _storage[name] = value;
83  return value;
84  }
85 
86  /// <summary>
87  /// Gets the property's value with the specified name. This is a case-insensitve search.
88  /// </summary>
89  /// <param name="name">The property name to access</param>
90  /// <returns>object value of BaseData</returns>
91  public object GetProperty(string name)
92  {
93  name = name.ToLowerInvariant();
94 
95  // redirect these calls to the base types properties
96  if (name == "time")
97  {
98  return Time;
99  }
100  if (name == "endtime")
101  {
102  return EndTime;
103  }
104  if (name == "value")
105  {
106  return Value;
107  }
108  if (name == "symbol")
109  {
110  return Symbol;
111  }
112  if (name == "price")
113  {
114  return Price;
115  }
116 
117  object value;
118  if (!_storage.TryGetValue(name, out value))
119  {
120  // let the user know the property name that we couldn't find
121  throw new KeyNotFoundException(
122  $"Property with name \'{name}\' does not exist. Properties: Time, Symbol, Value {string.Join(", ", _storage.Keys)}"
123  );
124  }
125 
126  return value;
127  }
128 
129  /// <summary>
130  /// Gets whether or not this dynamic data instance has a property with the specified name.
131  /// This is a case-insensitve search.
132  /// </summary>
133  /// <param name="name">The property name to check for</param>
134  /// <returns>True if the property exists, false otherwise</returns>
135  public bool HasProperty(string name)
136  {
137  return _storage.ContainsKey(name.ToLowerInvariant());
138  }
139 
140  /// <summary>
141  /// Gets the storage dictionary
142  /// Python algorithms need this information since DynamicMetaObject does not work
143  /// </summary>
144  /// <returns>Dictionary that stores the paramenters names and values</returns>
145  public IDictionary<string, object> GetStorageDictionary()
146  {
147  return _storage;
148  }
149 
150  /// <summary>
151  /// Return a new instance clone of this object, used in fill forward
152  /// </summary>
153  /// <remarks>
154  /// This base implementation uses reflection to copy all public fields and properties
155  /// </remarks>
156  /// <returns>A clone of the current object</returns>
157  public override BaseData Clone()
158  {
159  var clone = ObjectActivator.Clone(this);
160  foreach (var kvp in _storage)
161  {
162  // don't forget to add the dynamic members!
163  clone._storage.Add(kvp);
164  }
165  return clone;
166  }
167  }
168 }
