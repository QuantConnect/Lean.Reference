17 using System;
18 using System.Collections.Generic;
19 using NodaTime;
20 using QuantConnect.Data;
21 using QuantConnect.Securities;
22 using QuantConnect.Util;
23 
24 namespace QuantConnect.Lean.Engine.DataFeeds.Enumerators
25 {
26  /// <summary>
27  /// An implementation of the <see cref="FillForwardEnumerator"/> that uses an <see cref="ITimeProvider"/>
28  /// to determine if a fill forward bar needs to be emitted
29  /// </summary>
30  public class LiveFillForwardEnumerator : FillForwardEnumerator
31  {
32  private readonly ITimeProvider _timeProvider;
33 
34  /// <summary>
35  /// Initializes a new instance of the <see cref="LiveFillForwardEnumerator"/> class that accepts
36  /// a reference to the fill forward resolution, useful if the fill forward resolution is dynamic
37  /// and changing as the enumeration progresses
38  /// </summary>
39  /// <param name="timeProvider">The source of time used to gauage when this enumerator should emit extra bars when
40  /// null data is returned from the source enumerator</param>
41  /// <param name="enumerator">The source enumerator to be filled forward</param>
42  /// <param name="exchange">The exchange used to determine when to insert fill forward data</param>
43  /// <param name="fillForwardResolution">The resolution we'd like to receive data on</param>
44  /// <param name="isExtendedMarketHours">True to use the exchange's extended market hours, false to use the regular market hours</param>
45  /// <param name="subscriptionEndTime">The end time of the subscription, once passing this date the enumerator will stop</param>
46  /// <param name="dataResolution">The source enumerator's data resolution</param>
47  /// <param name="dataTimeZone">Time zone of the underlying source data</param>
48  public LiveFillForwardEnumerator(ITimeProvider timeProvider, IEnumerator<BaseData> enumerator, SecurityExchange exchange, IReadOnlyRef<TimeSpan> fillForwardResolution, bool isExtendedMarketHours, DateTime subscriptionEndTime, TimeSpan dataResolution, DateTimeZone dataTimeZone)
49  : base(enumerator, exchange, fillForwardResolution, isExtendedMarketHours, subscriptionEndTime, dataResolution, dataTimeZone)
50  {
51  _timeProvider = timeProvider;
52  }
53 
54  /// <summary>
55  /// Determines whether or not fill forward is required, and if true, will produce the new fill forward data
56  /// </summary>
57  /// <param name="fillForwardResolution"></param>
58  /// <param name="previous">The last piece of data emitted by this enumerator</param>
59  /// <param name="next">The next piece of data on the source enumerator, this may be null</param>
60  /// <param name="fillForward">When this function returns true, this will have a non-null value, null when the function returns false</param>
61  /// <returns>True when a new fill forward piece of data was produced and should be emitted by this enumerator</returns>
62  protected override bool RequiresFillForwardData(TimeSpan fillForwardResolution, BaseData previous, BaseData next, out BaseData fillForward)
63  {
64  // convert times to UTC for accurate comparisons and differences across DST changes
65  fillForward = null;
66  // Add a delay to the time we expect a data point if we've configured a delay for batching
67  var nextExpectedDataPointTimeUtc = previous.EndTime.ConvertToUtc(Exchange.TimeZone) + fillForwardResolution;
68  if (next != null)
69  {
70  // if not future data, just return the 'next'
71  if (next.EndTime.ConvertToUtc(Exchange.TimeZone) <= nextExpectedDataPointTimeUtc)
72  {
73  return false;
74  }
75  // next is future data, fill forward in between
76  var clone = previous.Clone(true);
77  clone.Time = previous.Time + fillForwardResolution;
78  fillForward = clone;
79  return true;
80  }
81 
82  // the underlying enumerator returned null, check to see if time has passed for fill forwarding
83  if (nextExpectedDataPointTimeUtc <= _timeProvider.GetUtcNow())
84  {
85  var clone = previous.Clone(true);
86  clone.Time = previous.Time + fillForwardResolution;
87  fillForward = clone;
88  return true;
89  }
90 
91  return false;
92  }
93  }
94 }
