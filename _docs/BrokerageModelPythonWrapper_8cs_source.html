77  /// <summary>
78  /// Gets a map of the default markets to be used for each security type
79  /// </summary>
80  public IReadOnlyDictionary<SecurityType, string> DefaultMarkets
81  {
82  get
83  {
84  using (Py.GIL())
85  {
86  var markets = _model.DefaultMarkets;
87  if ((markets as PyObject).TryConvert(out IReadOnlyDictionary<SecurityType, string> csharpDic))
88  {
89  return csharpDic;
90  }
91 
92  var dic = new Dictionary<SecurityType, string>();
93  foreach (var item in markets)
94  {
95  using var pyItem = item as PyObject;
96  var market = pyItem.As<SecurityType>();
97  dic[market] = markets[item];
98  }
99 
100  (markets as PyObject).Dispose();
101  return dic;
102  }
103  }

119  /// <summary>

137  /// <summary>
138  /// Returns true if the brokerage could accept this order. This takes into account
139  /// order type, security type, and order size limits.
140  /// </summary>
141  /// <remarks>
142  /// For example, a brokerage may have no connectivity at certain times, or an order rate/size limit
143  /// </remarks>
144  /// <param name="security">The security being ordered</param>
145  /// <param name="order">The order to be processed</param>
146  /// <param name="message">If this function returns false, a brokerage message detailing why the order may not be submitted</param>
147  /// <returns>True if the brokerage could process the order, false otherwise</returns>
148  public bool CanSubmitOrder(Security security, Order order, out BrokerageMessageEvent message)
149  {
150  using (Py.GIL())
151  {
152  using var result = _model.CanSubmitOrder(security, order, out message) as PyObject;
153  // Since pythonnet does not support out parameters, the methods return
154  // a tuple where the out parameter comes after the other returned values
155  if (!PyTuple.IsTupleType(result))
156  {
157  throw new ArgumentException($@"{_model.__class__.__name__}.CanSubmitOrder(): Must return a tuple value where the first value is a bool and the second a BrokerageMessageEvent");
158  }
159 
160  message = result[1].As<BrokerageMessageEvent>();
161  return result[0].As<bool>();
162  }
163  }
