367  /// <summary>
368  /// Construct a new security vehicle based on the user options.
369  /// </summary>
370  protected Security(Symbol symbol,
371  Cash quoteCurrency,
372  SymbolProperties symbolProperties,
373  SecurityExchange exchange,
374  SecurityCache cache,
375  ISecurityPortfolioModel portfolioModel,
376  IFillModel fillModel,
377  IFeeModel feeModel,
378  ISlippageModel slippageModel,
379  ISettlementModel settlementModel,
380  IVolatilityModel volatilityModel,
381  IBuyingPowerModel buyingPowerModel,
382  ISecurityDataFilter dataFilter,
383  IPriceVariationModel priceVariationModel,
384  ICurrencyConverter currencyConverter,
385  IRegisteredSecurityDataTypesProvider registeredTypesProvider
386  )
387  {
388  if (symbolProperties == null)
389  {
390  throw new ArgumentNullException(nameof(symbolProperties), "Security requires a valid SymbolProperties instance.");
391  }
392 
393  if (symbolProperties.QuoteCurrency != quoteCurrency.Symbol)
394  {
395  throw new ArgumentException("symbolProperties.QuoteCurrency must match the quoteCurrency.Symbol");
396  }
397 
398  Symbol = symbol;
399  _subscriptionsBag = new ();
400  QuoteCurrency = quoteCurrency;
401  SymbolProperties = symbolProperties;
402  IsTradable = true;
403  Cache = cache;
404  Exchange = exchange;
405  DataFilter = dataFilter;
406  PriceVariationModel = priceVariationModel;
407  PortfolioModel = portfolioModel;
408  BuyingPowerModel = buyingPowerModel;
409  FillModel = fillModel;
410  FeeModel = feeModel;
411  SlippageModel = slippageModel;
412  SettlementModel = settlementModel;
413  VolatilityModel = volatilityModel;
414  Holdings = new SecurityHolding(this, currencyConverter);
415  Data = new DynamicSecurityData(registeredTypesProvider, Cache);
416 
417  UpdateSubscriptionProperties();
418  }
