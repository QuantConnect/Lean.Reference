16 using System;
17 using System.Linq;
18 using QuantConnect.Orders;
19 using System.Collections.Generic;
20 using System.Collections.Specialized;

22 namespace QuantConnect.Securities.Positions
23 {
24  /// <summary>
25  /// Responsible for managing the resolution of position groups for an algorithm
26  /// </summary>
27  public class PositionManager
28  {
29  private bool _requiresGroupResolution;
30 
31  private PositionGroupCollection _groups;
32  private readonly SecurityManager _securities;
33  private readonly IPositionGroupResolver _resolver;
34  private readonly IPositionGroupBuyingPowerModel _defaultModel;
35 
36  /// <summary>
37  /// Gets the set of currently resolved position groups
38  /// </summary>
39  public PositionGroupCollection Groups
40  {
41  get
42  {
43  ResolvePositionGroups();
44  return _groups;
45  }
46  private set
47  {
48  _groups = value;
49  }
50  }
51 
52  /// <summary>
53  /// Gets whether or not the algorithm is using only default position groups
54  /// </summary>
55  public bool IsOnlyDefaultGroups => Groups.IsOnlyDefaultGroups;
56 
57  /// <summary>
58  /// Initializes a new instance of the <see cref="PositionManager"/> class
59  /// </summary>
60  /// <param name="securities">The algorithm's security manager</param>
61  public PositionManager(SecurityManager securities)
62  {
63  _securities = securities;
64  Groups = PositionGroupCollection.Empty;
65  _defaultModel = new SecurityPositionGroupBuyingPowerModel();
66  _resolver = new CompositePositionGroupResolver(new OptionStrategyPositionGroupResolver(securities),
67  new SecurityPositionGroupResolver(_defaultModel));
68 
69  // we must be notified each time our holdings change, so each time a security is added, we
70  // want to bind to its SecurityHolding.QuantityChanged event so we can trigger the resolver
71 
72  securities.CollectionChanged += (sender, args) =>
73  {
74  var items = args.NewItems ?? new List<object>();
75  if (args.OldItems != null)
76  {
77  foreach (var item in args.OldItems)
78  {
79  items.Add(item);
80  }
81  }
82 
83  foreach (Security security in items)
84  {
85  if (args.Action == NotifyCollectionChangedAction.Add)
86  {
87  security.Holdings.QuantityChanged += HoldingsOnQuantityChanged;
88  if (security.Invested)
89  {
90  // if this security has holdings then we'll need to resolve position groups
91  _requiresGroupResolution = true;
92  }
93  }
94  else if (args.Action == NotifyCollectionChangedAction.Remove)
95  {
96  security.Holdings.QuantityChanged -= HoldingsOnQuantityChanged;
97  if (security.Invested)
98  {
99  // only trigger group resolution if we had holdings in the removed security
100  _requiresGroupResolution = true;
101  }
102  }
103  }
104  };
105  }
106 
107  /// <summary>
108  /// Gets the <see cref="IPositionGroup"/> matching the specified <paramref name="key"/>. If one is not found,
109  /// then a new empty position group is returned.
110  /// </summary>
111  public IPositionGroup this[PositionGroupKey key] => Groups[key];
112 
113  /// <summary>
114  /// Creates a position group for the specified order, pulling
115  /// </summary>
116  /// <param name="order">The order</param>
117  /// <returns>A new position group matching the provided order</returns>
118  public IPositionGroup CreatePositionGroup(Order order)
119  {
120  IPositionGroup group;
121  var newPositions = order.CreatePositions(_securities).ToList();
122 
123  // We send new and current positions to try resolve any strategy being executed by multiple orders
124  // else the PositionGroup we will get out here will just be the default in those cases
125  if (!_resolver.TryGroup(newPositions, Groups, out group))
126  {
127  throw new InvalidOperationException($"Unable to create group for order: {order.Id}");
128  }
129 
130  return group;
131  }
132 
133  /// <summary>
134  /// Resolves position groups using the specified collection of positions
135  /// </summary>
136  /// <param name="positions">The positions to be grouped</param>
137  /// <returns>A collection of position groups containing all of the provided positions</returns>
138  public PositionGroupCollection ResolvePositionGroups(PositionCollection positions)
139  {
140  return _resolver.Resolve(positions);
141  }
142 
143  /// <summary>
144  /// Determines which position groups could be impacted by changes in the specified positions
145  /// </summary>
146  /// <param name="positions">The positions to be changed</param>
147  /// <returns>All position groups that need to be re-evaluated due to changes in the positions</returns>
148  public IEnumerable<IPositionGroup> GetImpactedGroups(IReadOnlyCollection<IPosition> positions)
149  {
150  return _resolver.GetImpactedGroups(Groups, positions);
151  }
152 
153  /// <summary>
154  /// Creates a <see cref="PositionGroupKey"/> for the security's default position group
155  /// </summary>
156  public PositionGroupKey CreateDefaultKey(Security security)
157  {
158  return new PositionGroupKey(_defaultModel, security);
159  }
160 
161  /// <summary>
162  /// Gets or creates the default position group for the specified <paramref name="security"/>
163  /// </summary>
164  /// <remarks>
165  /// TODO: position group used here is the default, is this what callers want?
166  /// </remarks>
167  public IPositionGroup GetOrCreateDefaultGroup(Security security)
168  {
169  var key = CreateDefaultKey(security);
170  return Groups[key];
171  }
172 
173  private void HoldingsOnQuantityChanged(object sender, SecurityHoldingQuantityChangedEventArgs e)
174  {
175  _requiresGroupResolution = true;
176  }
177 
178  /// <summary>
179  /// Resolves the algorithm's position groups from all of its holdings
180  /// </summary>
181  private void ResolvePositionGroups()
182  {
183  if (_requiresGroupResolution)
184  {
185  _requiresGroupResolution = false;
186  // TODO : Replace w/ special IPosition impl to always equal security.Quantity and we'll
187  // use them explicitly for resolution collection so we don't do this each time
188  var investedPositions = _securities.Where(kvp => kvp.Value.Invested).Select(kvp => (IPosition)new Position(kvp.Value));
189  var positionsCollection = new PositionCollection(investedPositions);
190  Groups = ResolvePositionGroups(positionsCollection);
191  }
192  }
193  }
194 }
