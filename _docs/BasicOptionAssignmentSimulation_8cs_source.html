28  /// <summary>
29  /// This market conditions simulator emulates exercising of short option positions in the portfolio.
30  /// Simulator implements basic no-arb argument: when time value of the option contract is close to zero
31  /// it assigns short legs getting profit close to expiration dates in deep ITM positions. User algorithm then receives
32  /// assignment event from LEAN. Simulator randomly scans for arbitrage opportunities every two hours or so.
33  /// </summary>
34  public class BasicOptionAssignmentSimulation : IBacktestingMarketSimulation
35  {
36  // we start simulating assignments 4 days prior to expiration
37  private readonly TimeSpan _priorExpiration = new TimeSpan(4,0,0,0);
38 
39  // we focus only on deep ITM calls and puts (at least 5% away from price)
40  private const decimal _deepITM = 0.05m;
41 
42  // we rescan portfolio for new contracts and expirations every month
43  private readonly TimeSpan _securitiesRescanPeriod = new TimeSpan(30, 0, 0, 0);
44 
45  // we try to generate new assignments every 2 hours
46  private readonly TimeSpan _assignmentScanPeriod = new TimeSpan(0, 2, 0, 0);
47 
48  // last update time
49  private DateTime _lastUpdate = DateTime.MinValue;
50  private Queue<DateTime> _assignmentScans;
51  private readonly Random _rand = new Random(12345);
52 
53  /// <summary>
54  /// We generate a list of time points when we would like to run our simulation. we then return true if the time is in the list.
55  /// </summary>
56  /// <returns></returns>
57  public bool IsReadyToSimulate(IAlgorithm algorithm)
58  {
59  if (_lastUpdate == DateTime.MinValue ||
60  algorithm.UtcTime - _lastUpdate > _securitiesRescanPeriod)
61  {
62  var expirations = algorithm.Securities.Select(x => x.Key)
63  .Where(x => x.ID.SecurityType.IsOption() &&
64  x.ID.Date > algorithm.Time &&
65  x.ID.Date - algorithm.Time <= _securitiesRescanPeriod &&
66  (x.ID.OptionStyle == OptionStyle.American ||
67  (x.ID.OptionStyle == OptionStyle.European && x.ID.Date.Date == algorithm.Time.Date)))
68  .Select(x => x.ID.Date)
69  .OrderBy(x => x)
70  .ToList();
71 
72  var scansCount = _priorExpiration.TotalMinutes / _assignmentScanPeriod.TotalMinutes;
73 
74  // we generate a list of random dates when we plan to search for opportunities to assign short positions.
75  var scans = new List<DateTime>();
76 
77  foreach (var expirationDate in expirations)
78  {
79  var startDate = expirationDate - _priorExpiration;
80 
81  foreach (var count in Enumerable.Range(0, (int)scansCount))
82  {
83  scans.Add(startDate.AddMinutes(count * _assignmentScanPeriod.TotalMinutes));
84  }
85  }
86  var randomizedScans = scans
87  .DistinctBy(x => new DateTime(x.Year, x.Month, x.Day, x.Hour, 0, 0)) // DistinctBy hour
88  .OrderBy(x => x)
89  .Select(x => x.AddMinutes(_rand.NextDouble() * _assignmentScanPeriod.TotalMinutes));
90 
91  _assignmentScans = new Queue<DateTime>(randomizedScans);
92 
93  _lastUpdate = algorithm.UtcTime;
94  }
95 
96  if (_assignmentScans.Count > 0)
97  {
98  // we check if new simulation date has arrived. It may happen that several of them had.. due to exchange hours, weekends, etc.
99  // we fast forward through unused items
100  if (algorithm.UtcTime >= _assignmentScans.Peek())
101  {
102  while (_assignmentScans.Count > 0 &&
103  algorithm.UtcTime >= _assignmentScans.Peek())
104  {
105  _assignmentScans.Dequeue();
106  }
107 
108  return true;
109  }
110 
111  return false;
112  }
113 
114  return false;
115  }

117  /// <summary>
118  /// We simulate activity of market makers on expiration. Trying to get profit close to expiration dates in deep ITM positions.
119  /// This version of the simulator exercises short positions in full.
120  /// </summary>
121  public void SimulateMarketConditions(IBrokerage brokerage, IAlgorithm algorithm)
122  {
123  if (!IsReadyToSimulate(algorithm)) return;
124 
125  var backtestingBrokerage = (BacktestingBrokerage)brokerage;
126 
127  Func<Symbol, bool> deepITM = symbol =>
128  {
129  var underlyingPrice = algorithm.Securities[symbol.Underlying].Close;
130 
131  var result =
132  symbol.ID.OptionRight == OptionRight.Call
133  ? (underlyingPrice - symbol.ID.StrikePrice) / underlyingPrice > _deepITM
134  : (symbol.ID.StrikePrice - underlyingPrice) / underlyingPrice > _deepITM;
135 
136  return result;
137  };
138 
139  foreach(var pair in algorithm.Securities
140  // we take only options that expire soon
141  .Where(x => x.Key.ID.SecurityType.IsOption() &&
142  ((x.Key.ID.OptionStyle == OptionStyle.American && x.Key.ID.Date - algorithm.UtcTime <= _priorExpiration) ||
143  (x.Key.ID.OptionStyle == OptionStyle.European && x.Key.ID.Date.Date == algorithm.UtcTime.Date)))
144  // we look into short positions only (short for user means long for us)
145  .Where(x => x.Value.Holdings.IsShort)
146  // we take only deep ITM strikes
147  .Where(x => deepITM(x.Key))
148  // we estimate P/L
149  .Where(x => EstimateArbitragePnL((Option)x.Value,
150  (OptionHolding)x.Value.Holdings,
151  algorithm.Securities[x.Value.Symbol.Underlying],
152  algorithm.Portfolio.CashBook) > 0.0m)
153  .OrderBy(pair => pair.Key))
154  {
155  // we exercise options with positive expected P/L (over basic sale of option)
156  backtestingBrokerage.ActivateOptionAssignment((Option) pair.Value, (int) ((OptionHolding) pair.Value.Holdings).AbsoluteQuantity);
157  }
158 
159  }

161  private decimal EstimateArbitragePnL(Option option,
162  OptionHolding holding,
163  Security underlying,
164  ICurrencyConverter currencyConverter)
165  {
166  // no-arb argument:
167  // if our long deep ITM position has a large B/A spread and almost no time value, it may be interesting for us
168  // to exercise the option and close the resulting position in underlying instrument, if we want to exit now.
169 
170  // User's short option position is our long one.
171  // In order to sell ITM position we take option bid price as an input
172  var optionPrice = option.BidPrice;
173 
174  // we are interested in underlying bid price if we exercise calls and want to sell the underlying immediately.
175  // we are interested in underlying ask price if we exercise puts
176  var underlyingPrice = option.Symbol.ID.OptionRight == OptionRight.Call
177  ? underlying.BidPrice
178  : underlying.AskPrice;
179 
180  // quantity is normally negative algo's holdings, but since we're modeling the contract holder (counter-party)
181  // it's negative THEIR holdings. holding.Quantity is negative, so if counter-party exercises, they would reduce holdings
182  var underlyingQuantity = option.GetExerciseQuantity(holding.Quantity);
183 
184  // Scenario 1 (base): we just close option position
185  var marketOrder1 = new MarketOrder(option.Symbol, -holding.Quantity, option.LocalTime.ConvertToUtc(option.Exchange.TimeZone));
186  var orderFee1 = currencyConverter.ConvertToAccountCurrency(option.FeeModel.GetOrderFee(
187  new OrderFeeParameters(option, marketOrder1)).Value);
188 
189  var basePnL = (optionPrice - holding.AveragePrice) * -holding.Quantity
190  * option.QuoteCurrency.ConversionRate
191  * option.SymbolProperties.ContractMultiplier
192  - orderFee1.Amount;
193 
194  // Scenario 2 (alternative): we exercise option and then close underlying position
195  var optionExerciseOrder2 = new OptionExerciseOrder(option.Symbol, (int)holding.AbsoluteQuantity, option.LocalTime.ConvertToUtc(option.Exchange.TimeZone));
196  var optionOrderFee2 = currencyConverter.ConvertToAccountCurrency(option.FeeModel.GetOrderFee(
197  new OrderFeeParameters(option, optionExerciseOrder2)).Value);
198 
199  var underlyingOrderFee2Amount = 0m;
200 
201  // Cash settlements do not open a position for the underlying.
202  // For Physical Delivery, we calculate the order fee since we have to close the position
203  if (option.ExerciseSettlement == SettlementType.PhysicalDelivery)
204  {
205  var underlyingMarketOrder2 = new MarketOrder(underlying.Symbol, -underlyingQuantity,
206  underlying.LocalTime.ConvertToUtc(underlying.Exchange.TimeZone));
207  var underlyingOrderFee2 = currencyConverter.ConvertToAccountCurrency(underlying.FeeModel.GetOrderFee(
208  new OrderFeeParameters(underlying, underlyingMarketOrder2)).Value);
209  underlyingOrderFee2Amount = underlyingOrderFee2.Amount;
210  }
211 
212  // calculating P/L of the two transactions (exercise option and then close underlying position)
213  var altPnL = (underlyingPrice - option.StrikePrice) * underlyingQuantity * underlying.QuoteCurrency.ConversionRate * option.ContractUnitOfTrade
214  - underlyingOrderFee2Amount
215  - holding.AveragePrice * holding.AbsoluteQuantity * option.SymbolProperties.ContractMultiplier * option.QuoteCurrency.ConversionRate
216  - optionOrderFee2.Amount;
217 
218  return altPnL - basePnL;
219  }
220  }
