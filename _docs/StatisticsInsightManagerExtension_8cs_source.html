17 using System;
18 using QuantConnect.Algorithm.Framework.Alphas;
19 using QuantConnect.Algorithm.Framework.Alphas.Analysis;
20 using QuantConnect.Interfaces;
21 using QuantConnect.Statistics;

23 namespace QuantConnect.Lean.Engine.Alphas
24 {
25  /// <summary>
26  /// Manages alpha statistics responsbilities
27  /// </summary>
28  public class StatisticsInsightManagerExtension : IInsightManagerExtension
29  {
30  private readonly double _smoothingFactor;
31  private readonly int _rollingAverageIsReadyCount;
32  private readonly bool _requireRollingAverageWarmup;
33  private readonly decimal _tradablePercentOfVolume;
34  private readonly KellyCriterionManager _kellyCriterionManager;
35  private DateTime _lastKellyCriterionUpdate;
36 
37  /// <summary>
38  /// Gets the current statistics. The values are current as of the time specified
39  /// in <see cref="AlphaRuntimeStatistics.MeanPopulationScore"/> and <see cref="AlphaRuntimeStatistics.RollingAveragedPopulationScore"/>
40  /// </summary>
41  public AlphaRuntimeStatistics Statistics { get; }
42 
43  /// <summary>
44  /// Gets whether or not the rolling average statistics is ready
45  /// </summary>
46  public bool RollingAverageIsReady => !_requireRollingAverageWarmup || Statistics.TotalInsightsAnalysisCompleted >= _rollingAverageIsReadyCount;
47 
48  /// <summary>
49  /// Initializes a new instance of the <see cref="StatisticsInsightManagerExtension"/> class
50  /// </summary>
51  /// <param name="accountCurrencyProvider">The account currency provider</param>
52  /// <param name="tradablePercentOfVolume">Percent of volume of first bar used to estimate the maximum number of tradable shares. Defaults to 1%</param>
53  /// <param name="period">The period used for exponential smoothing of scores - this is a number of insights. Defaults to 100 insight predictions.</param>
54  /// <param name="requireRollingAverageWarmup">Specify true to force the population average scoring to warmup before plotting.</param>
55  public StatisticsInsightManagerExtension(
56  IAccountCurrencyProvider accountCurrencyProvider,
57  decimal tradablePercentOfVolume = 0.01m,
58  int period = 100,
59  bool requireRollingAverageWarmup = false)
60  {
61  Statistics = new AlphaRuntimeStatistics(accountCurrencyProvider);
62  _tradablePercentOfVolume = tradablePercentOfVolume;
63  _smoothingFactor = 2.0 / (period + 1.0);
64 
65  // use normal ema warmup period
66  _rollingAverageIsReadyCount = period;
67  _requireRollingAverageWarmup = requireRollingAverageWarmup;
68 
69  _kellyCriterionManager = new KellyCriterionManager();
70  }
71 
72  /// <summary>
73  /// Handles the <see cref="IAlgorithm.InsightsGenerated"/> event
74  /// Increments total, long and short counters. Updates long/short ratio
75  /// </summary>
76  /// <param name="context">The newly generated insight context</param>
77  public void OnInsightGenerated(InsightAnalysisContext context)
78  {
79  // incremement total insight counter
80  Statistics.TotalInsightsGenerated++;
81 
82  // update long/short ratio statistics
83  if (context.Insight.Direction == InsightDirection.Up)
84  {
85  Statistics.LongCount++;
86  }
87  else if (context.Insight.Direction == InsightDirection.Down)
88  {
89  Statistics.ShortCount++;
90  }
91  }
92 
93  /// <summary>
94  /// Computes an estimated value for the insight. This is intended to be invoked at the end of the
95  /// insight period, i.e, when now == insight.GeneratedTimeUtc + insight.Period;
96  /// </summary>
97  /// <param name="context">Context whose insight has just closed</param>
98  public void OnInsightClosed(InsightAnalysisContext context)
99  {
100  // increment closed insight counter
101  Statistics.TotalInsightsClosed += 1;
102 
103  // tradable volume (purposefully includes fractional shares)
104  var volume = _tradablePercentOfVolume * context.InitialValues.Volume;
105 
106  // value of the entering the trade in the account currency
107  var enterValue = context.InitialValues.Price * context.InitialValues.QuoteCurrencyConversionRate;
108 
109  // value of exiting the trade in the account currency
110  var exitValue = context.CurrentValues.Price * context.CurrentValues.QuoteCurrencyConversionRate;
111 
112  // total value delta between enter and exit values
113  var insightValue = (int)context.Insight.Direction * (exitValue - enterValue);
114 
115  var insightValueFactoredByTradableVolume = insightValue * volume;
116 
117  context.Insight.EstimatedValue = insightValueFactoredByTradableVolume;
118  Statistics.TotalAccumulatedEstimatedAlphaValue += insightValueFactoredByTradableVolume;
119 
120  // just in case..
121  if (enterValue != 0)
122  {
123  _kellyCriterionManager.AddNewValue(
124  (int)context.Insight.Direction * (exitValue / enterValue - 1),
125  context.Insight.GeneratedTimeUtc);
126  }
127  }
128 
129  /// <summary>
130  /// Updates the specified statistics with the new scores
131  /// </summary>
132  /// <param name="context">Context whose insight has just completed analysis</param>
133  public void OnInsightAnalysisCompleted(InsightAnalysisContext context)
134  {
135  // increment analysis completed counter
136  Statistics.TotalInsightsAnalysisCompleted += 1;
137 
138  foreach (var scoreType in InsightManager.ScoreTypes)
139  {
140  if (!context.ShouldAnalyze(scoreType))
141  {
142  continue;
143  }
144  var score = context.Score.GetScore(scoreType);
145  var currentTime = context.CurrentValues.TimeUtc;
146 
147  // online population average
148  var mean = Statistics.MeanPopulationScore.GetScore(scoreType);
149  var newMean = mean + (score - mean) / Statistics.TotalInsightsAnalysisCompleted;
150  Statistics.MeanPopulationScore.SetScore(scoreType, newMean, currentTime);
151 
152  var newEma = newMean;
153  if (Statistics.TotalInsightsAnalysisCompleted > 4)
154  {
155  // compute the traditional ema
156  var ema = Statistics.RollingAveragedPopulationScore.GetScore(scoreType);
157  newEma = score * _smoothingFactor + ema * (1 - _smoothingFactor);
158  }
159  Statistics.RollingAveragedPopulationScore.SetScore(scoreType, newEma, currentTime);
160  }
161  }
162 
163  /// <summary>
164  /// Invokes the manager at the end of the time step.
165  /// </summary>
166  /// <param name="frontierTimeUtc">The current frontier time utc</param>
167  public void Step(DateTime frontierTimeUtc)
168  {
169  Statistics.SetDate(frontierTimeUtc);
170 
171  if (_lastKellyCriterionUpdate.Date != frontierTimeUtc)
172  {
173  _lastKellyCriterionUpdate = frontierTimeUtc;
174 
175  _kellyCriterionManager.UpdateScores();
176 
177  Statistics.KellyCriterionEstimate = _kellyCriterionManager.KellyCriterionEstimate;
178  Statistics.KellyCriterionProbabilityValue = _kellyCriterionManager.KellyCriterionProbabilityValue;
179  }
180  }
181 
182  /// <summary>
183  /// Allows the extension to initialize itself over the expected range
184  /// </summary>
185  /// <param name="algorithmStartDate">The start date of the algorithm</param>
186  /// <param name="algorithmEndDate">The end date of the algorithm</param>
187  /// <param name="algorithmUtcTime">The algorithm's current utc time</param>
188  public void InitializeForRange(DateTime algorithmStartDate, DateTime algorithmEndDate, DateTime algorithmUtcTime)
189  {
190  Statistics.SetStartDate(algorithmStartDate);
191  }
192  }
