17 using System;
18 using System.Collections.Generic;
19 using System.Linq;
20 using System.Text;

22 namespace QuantConnect.ToolBox.IQFeed
23 {
24  public class Time
25  {
26  public Time(int hour = 0, int minute = 0, int second = 0)
27  {
28  _hour = hour;
29  _minute = minute;
30  _second = second;
31  }
32  public Time(string textTime)
33  {
34  if (textTime.Length < 8) { _hour = 0; _minute = 0; _second = 0; return; }
35  if (!int.TryParse(textTime.Substring(0, 2), out _hour)) _hour = 0;
36  if (!int.TryParse(textTime.Substring(3, 2), out _minute)) _minute = 0;
37  if (!int.TryParse(textTime.Substring(6, 2), out _second)) _second = 0;
38  }
39  public int Hour { get { return _hour; } set { _hour = value; } }
40  public int Minute { get { return _minute; } set { _minute = value; } }
41  public int Second { get { return _second; } set { _second = value; } }
42  public string IQFeedFormat =>
43  $"{_hour.ToStringInvariant("00")}{_minute.ToStringInvariant("00")}{_second.ToStringInvariant("00")}";
44 
45  #region private
46  private int _hour;
47  private int _minute;
48  private int _second;
49  #endregion
50  }

52  public enum PeriodType { Second, Minute, Hour }
53  public class Interval
54  {
55  public Interval(PeriodType periodType, int periods)
56  {
57  _periodType = periodType;
58  _periods = periods;
59  }
60  public PeriodType PeriodType { get { return _periodType; } }
61  public int Periods { get { return _periods; } }
62  public int Seconds
63  {
64  get
65  {
66  switch (_periodType)
67  {
68  case PeriodType.Second: return _periods;
69  case PeriodType.Minute: return _periods * 60;
70  case PeriodType.Hour: return _periods * 3600;
71  }
72  throw new Exception("May not get seconds for " + _periodType.ToString());
73  }
74  }
75  #region private
76  private PeriodType _periodType;
77  private int _periods;
78  #endregion
79  }
