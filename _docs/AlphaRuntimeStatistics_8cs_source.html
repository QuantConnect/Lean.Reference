200  /// <summary>
201  /// Suggested Value of the Alpha On A Monthly Basis For Licensing
202  /// </summary>
203  [JsonProperty(DefaultValueHandling = DefaultValueHandling.Ignore), JsonConverter(typeof(StringDecimalJsonConverter), true)]
204  public decimal EstimatedMonthlyAlphaValue
205  {
206  get
207  {
208  if (_daysCompleted == 0)
209  {
210  return _overrideEstimatedMonthlyAlphaValue;
211  }
212  return (TotalAccumulatedEstimatedAlphaValue / (decimal) _daysCompleted) * 30;
213  }
214  private set { _overrideEstimatedMonthlyAlphaValue = value; }
215  }

241  /// <summary>
242  /// Creates a dictionary containing the statistics
243  /// </summary>
244  public Dictionary<string, string> ToDictionary()
245  {
246  var accountCurrencySymbol = Currencies.GetCurrencySymbol(_accountCurrencyProvider?.AccountCurrency ?? Currencies.USD);
247 
248  return new Dictionary<string, string>
249  {
250  {"Fitness Score", $"{Invariant(FitnessScore)}"},
251  {"Kelly Criterion Estimate", $"{Invariant(KellyCriterionEstimate)}"},
252  {"Kelly Criterion Probability Value", $"{Invariant(KellyCriterionProbabilityValue)}"},
253  {"Sortino Ratio", $"{Invariant(SortinoRatio)}"},
254  {"Return Over Maximum Drawdown", $"{Invariant(ReturnOverMaxDrawdown)}"},
255  {"Portfolio Turnover", $"{Invariant(PortfolioTurnover)}"},
256  {"Total Insights Generated", $"{Invariant(TotalInsightsGenerated)}"},
257  {"Total Insights Closed", $"{Invariant(TotalInsightsClosed)}"},
258  {"Total Insights Analysis Completed", $"{Invariant(TotalInsightsAnalysisCompleted)}"},
259  {"Long Insight Count", $"{Invariant(LongCount)}"},
260  {"Short Insight Count", $"{Invariant(ShortCount)}"},
261  {"Long/Short Ratio", $"{Invariant(Math.Round(100*LongShortRatio, 2))}%"},
262  {"Estimated Monthly Alpha Value", $"{accountCurrencySymbol}{Invariant(EstimatedMonthlyAlphaValue.SmartRounding())}"},
263  {"Total Accumulated Estimated Alpha Value", $"{accountCurrencySymbol}{Invariant(TotalAccumulatedEstimatedAlphaValue.SmartRounding())}"},
264  {"Mean Population Estimated Insight Value", $"{accountCurrencySymbol}{Invariant(MeanPopulationEstimatedInsightValue.SmartRounding())}"},
265  {"Mean Population Direction", $"{Invariant(Math.Round(100 * MeanPopulationScore.Direction, 4))}%"},
266  {"Mean Population Magnitude", $"{Invariant(Math.Round(100 * MeanPopulationScore.Magnitude, 4))}%"},
267  {"Rolling Averaged Population Direction", $"{Invariant(Math.Round(100 * RollingAveragedPopulationScore.Direction, 4))}%"},
268  {"Rolling Averaged Population Magnitude", $"{Invariant(Math.Round(100 * RollingAveragedPopulationScore.Magnitude, 4))}%"},
269  };
270  }
