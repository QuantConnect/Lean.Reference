16 using System;
17 using System.Collections.Generic;
18 using System.Linq;
19 using Newtonsoft.Json;
20 
21 namespace QuantConnect.Data.Fundamental
22 {
23  /// <summary>
24  /// Abstract base class for multi-period fields
25  /// </summary>
26  public abstract class MultiPeriodField
27  {
28  /// <summary>
29  /// Helper, and internal struct use to hold the values for a period.
30  /// </summary>
31  /// <remarks>For performance using a struct versus a class, this allows us to save
32  /// on references</remarks>
33  protected struct PeriodField
34  {
35  /// <summary>
36  /// Creates a new period field instance
37  /// </summary>
38  public PeriodField(byte period, decimal value)
39  {
40  Period = period;
41  Value = value;
42  }
43 
44  /// <summary>
45  /// The period associated with this value <see cref="PeriodAsByte"/>
46  /// </summary>
47  public byte Period { get; set; }
48 
49  /// <summary>
50  /// The value for this period
51  /// </summary>
52  public decimal Value { get; set; }
53  }
54 
55  private bool StoreIsEmpty => Store == null || Store.Length == 0;
56 
57  /// <summary>
58  /// The dictionary store containing all values for the multi-period field
59  /// </summary>
60  /// <remarks>For performance using and array versus any other collection
61  /// this allows us to save in memory footprint and speed. We expect few amount of
62  /// elements in each collection but multiple collections <see cref="FineFundamental"/></remarks>
63  protected PeriodField[] Store;
64 
65  /// <summary>
66  /// Gets the default period for the field
67  /// </summary>
68  protected virtual byte DefaultPeriod => StoreIsEmpty ? PeriodAsByte.NoPeriod : Store.FirstOrDefault().Period;
69 
70  /// <summary>
71  /// Creates a new instance
72  /// </summary>
73  protected MultiPeriodField(IDictionary<string, decimal> store = null)
74  {
75  if (store != null)
76  {
77  Store = store.Select(kvp => new PeriodField(PeriodAsByte.Convert(kvp.Key), kvp.Value)).ToArray();
78  }
79  }
80 
81  /// <summary>
82  /// Gets the value of the field for the requested period
83  /// </summary>
84  /// <param name="period">The requested period</param>
85  /// <returns>The value for the period</returns>
86  public decimal GetPeriodValue(string period)
87  {
88  return GetPeriodValue(PeriodAsByte.Convert(period));
89  }
90 
91  /// <summary>
92  /// Internal implementation which gets the value of the field for the requested period
93  /// </summary>
94  protected decimal GetPeriodValue(byte period)
95  {
96  return StoreIsEmpty ? 0 : Store.FirstOrDefault(field => field.Period == period).Value;
97  }
98 
99  /// <summary>
100  /// Returns true if the field contains a value for the requested period
101  /// </summary>
102  /// <param name="period">The requested period</param>
103  public bool HasPeriodValue(string period) => !StoreIsEmpty && Store.Any(field => field.Period == PeriodAsByte.Convert(period));
104 
105  /// <summary>
106  /// Returns true if the field contains a value for the default period
107  /// </summary>
108  public bool HasValue => HasPeriodValue(PeriodAsByte.Convert(DefaultPeriod));
109 
110  /// <summary>
111  /// Gets the list of available period names for the field
112  /// </summary>
113  /// <returns>The list of periods</returns>
114  public IEnumerable<string> GetPeriodNames()
115  {
116  return StoreIsEmpty
117  ? Enumerable.Empty<string>() : Store.Select(field => PeriodAsByte.Convert(field.Period));
118  }
119 
120  /// <summary>
121  /// Gets a dictionary of period names and values for the field
122  /// </summary>
123  /// <returns>The dictionary of period names and values</returns>
124  public IReadOnlyDictionary<string, decimal> GetPeriodValues()
125  {
126  return StoreIsEmpty
127  ? new Dictionary<string, decimal>()
128  : Store.ToDictionary(field => PeriodAsByte.Convert(field.Period), field => field.Value);
129  }
130 
131  /// <summary>
132  /// Returns the default value for the field
133  /// </summary>
134  [JsonIgnore]
135  public decimal Value
136  {
137  get
138  {
139  if (StoreIsEmpty)
140  {
141  return 0;
142  }
143 
144  return HasValue ? GetPeriodValue(DefaultPeriod) : Store.First().Value;
145  }
146  }
147 
148  /// <summary>
149  /// Returns the default value for the field
150  /// </summary>
151  /// <param name="field"></param>
152  public static implicit operator decimal(MultiPeriodField field)
153  {
154  return field.Value;
155  }
156 
157  /// <summary>
158  /// Sets the value of the field for the specified period
159  /// </summary>
160  /// <param name="period">The period</param>
161  /// <param name="value">The value to be set</param>
162  public void SetPeriodValue(string period, decimal value)
163  {
164  SetPeriodValue(PeriodAsByte.Convert(period), value);
165  }
166 
167  /// <summary>
168  /// Internal implementation which sets the value of the field for the specified period
169  /// </summary>
170  protected void SetPeriodValue(byte period, decimal value)
171  {
172  if (!StoreIsEmpty)
173  {
174  for (var i = 0; i < Store.Length; i++)
175  {
176  if (Store[i].Period == period)
177  {
178  Store[i].Value = value;
179  return;
180  }
181  }
182  }
183 
184  // if we are here it means the array does not have the value
185  // we have to create a new array and add the value
186  var index = 0;
187  if (Store == null)
188  {
189  Store = new PeriodField[1];
190  }
191  else
192  {
193  var newSize = Store.Length + 1;
194  Array.Resize(ref Store, newSize);
195  index = newSize - 1;
196  }
197  Store[index] = new PeriodField(period, value);
198  }
199 
200  /// <summary>
201  /// Returns true if the field has at least one value for one period
202  /// </summary>
203  public bool HasValues()
204  {
205  return !StoreIsEmpty;
206  }
207 
208  /// <summary>
209  /// Applies updated values from <paramref name="update"/> to this instance
210  /// </summary>
211  /// <remarks>Used to apply data updates to the current instance. This WILL overwrite existing values.</remarks>
212  /// <param name="update">The next data update for this instance</param>
213  public void UpdateValues(MultiPeriodField update)
214  {
215  if (update?.Store == null)
216  return;
217 
218  foreach (var kvp in update.Store)
219  {
220  SetPeriodValue(kvp.Period, kvp.Value);
221  }
222  }
223 
224  /// <summary>
225  /// Returns a string that represents the current object.
226  /// </summary>
227  /// <returns>
228  /// A string that represents the current object.
229  /// </returns>
230  /// <filterpriority>2</filterpriority>
231  public override string ToString()
232  {
233  return StoreIsEmpty
234  ? "" : string.Join(";", Store.Select(x => $"{PeriodAsByte.Convert(x.Period)}:{x.Value}"));
235  }
236  }
237 }
