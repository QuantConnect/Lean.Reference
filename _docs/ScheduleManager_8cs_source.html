26  /// <summary>
27  /// Provides access to the real time handler's event scheduling feature
28  /// </summary>

47  /// <summary>
48  /// Initializes a new instance of the <see cref="ScheduleManager"/> class
49  /// </summary>
50  /// <param name="securities">Securities manager containing the algorithm's securities</param>
51  /// <param name="timeZone">The algorithm's time zone</param>
52  public ScheduleManager(SecurityManager securities, DateTimeZone timeZone)
53  {
54  _securities = securities;
55  DateRules = new DateRules(securities, timeZone);
56  TimeRules = new TimeRules(securities, timeZone);
57 
58  // used for storing any events before the event schedule is set
59  _preInitializedEvents = new List<ScheduledEvent>();
60  }
61 
62  /// <summary>
63  /// Sets the <see cref="IEventSchedule"/> implementation
64  /// </summary>
65  /// <param name="eventSchedule">The event schedule implementation to be used. This is the IRealTimeHandler</param>
66  internal void SetEventSchedule(IEventSchedule eventSchedule)
67  {
68  if (eventSchedule == null)
69  {
70  throw new ArgumentNullException(nameof(eventSchedule));
71  }
72 
73  lock (_eventScheduleLock)
74  {
75  _eventSchedule = eventSchedule;
76 
77  // load up any events that were added before we were ready to send them to the scheduler
78  foreach (var scheduledEvent in _preInitializedEvents)
79  {
80  _eventSchedule.Add(scheduledEvent);
81  }
82  }
83  }
84 
85  /// <summary>
86  /// Adds the specified event to the schedule
87  /// </summary>
88  /// <param name="scheduledEvent">The event to be scheduled, including the date/times the event fires and the callback</param>
89  public void Add(ScheduledEvent scheduledEvent)
90  {
91  lock (_eventScheduleLock)
92  {
93  if (_eventSchedule != null)
94  {
95  _eventSchedule.Add(scheduledEvent);
96  }
97  else
98  {
99  _preInitializedEvents.Add(scheduledEvent);
100  }
101  }
102  }
103 
104  /// <summary>
105  /// Removes the specified event from the schedule
106  /// </summary>
107  /// <param name="scheduledEvent">The event to be removed</param>
108  public void Remove(ScheduledEvent scheduledEvent)
109  {
110  lock (_eventScheduleLock)
111  {
112  if (_eventSchedule != null)
113  {
114  _eventSchedule.Remove(scheduledEvent);
115  }
116  else
117  {
118  _preInitializedEvents.RemoveAll(se => Equals(se, scheduledEvent));
119  }
120  }
121  }

169  /// <summary>
170  /// Schedules the callback to run using the specified date and time rules
171  /// </summary>
172  /// <param name="name">The event's unique name</param>
173  /// <param name="dateRule">Specifies what dates the event should run</param>
174  /// <param name="timeRule">Specifies the times on those dates the event should run</param>
175  /// <param name="callback">The callback to be invoked</param>
176  public ScheduledEvent On(string name, IDateRule dateRule, ITimeRule timeRule, PyObject callback)
177  {
178  return On(name, dateRule, timeRule, (n, d) => { using (Py.GIL()) callback.Invoke(); });
179  }
180 
181  /// <summary>
182  /// Schedules the callback to run using the specified date and time rules
183  /// </summary>
184  /// <param name="name">The event's unique name</param>
185  /// <param name="dateRule">Specifies what dates the event should run</param>
186  /// <param name="timeRule">Specifies the times on those dates the event should run</param>
187  /// <param name="callback">The callback to be invoked</param>
188  public ScheduledEvent On(string name, IDateRule dateRule, ITimeRule timeRule, Action<string, DateTime> callback)
189  {
190  // back the date up to ensure we get all events, the event scheduler will skip past events that whose time has passed
191  var dates = GetDatesDeferred(dateRule, _securities);
192  var eventTimes = timeRule.CreateUtcEventTimes(dates);
193  var scheduledEvent = new ScheduledEvent(name, eventTimes, callback);
194  Add(scheduledEvent);
195 
196  Log.Trace($"Event Name \"{scheduledEvent.Name}\", scheduled to run.");
197  return scheduledEvent;
198  }

277  /// <summary>
278  /// Helper methods to defer the evaluation of the current time until the dates are enumerated for the first time.
279  /// This allows for correct support for warmup period
280  /// </summary>
281  internal static IEnumerable<DateTime> GetDatesDeferred(IDateRule dateRule, SecurityManager securities)
282  {
283  foreach (var item in dateRule.GetDates(securities.UtcTime.Date.AddDays(-1), Time.EndOfTime))
284  {
285  yield return item;
286  }
287  }
