108  /// <summary>
109  /// Creates a new subscription to provide data for the specified security.
110  /// </summary>
111  /// <param name="request">Defines the subscription to be added, including start/end times the universe and security</param>
112  /// <returns>The created <see cref="Subscription"/> if successful, null otherwise</returns>
113  public override Subscription CreateSubscription(SubscriptionRequest request)
114  {
115  Subscription subscription = null;
116  try
117  {
118  // create and add the subscription to our collection
119  subscription = request.IsUniverseSubscription
120  ? CreateUniverseSubscription(request)
121  : CreateDataSubscription(request);
122  }
123  catch (Exception err)
124  {
125  Log.Error(err, $"CreateSubscription(): Failed configuration: '{request.Configuration}'");
126  // kill the algorithm, this shouldn't happen
127  _algorithm.SetRuntimeError(err, $"Failed to subscribe to {request.Configuration.Symbol}");
128  }
129 
130  return subscription;
131  }
132 
133  /// <summary>
134  /// Removes the subscription from the data feed, if it exists
135  /// </summary>
136  /// <param name="subscription">The subscription to remove</param>
137  public override void RemoveSubscription(Subscription subscription)
138  {
139  var symbol = subscription.Configuration.Symbol;
140 
141  // remove the subscriptions
142  if (!_channelProvider.ShouldStreamSubscription(subscription.Configuration))
143  {
144  _customExchange.RemoveEnumerator(symbol);
145  }
146  else
147  {
148  _dataQueueHandler.UnsubscribeWithMapping(subscription.Configuration);
149  }
150  }

193  /// <summary>
194  /// Creates a new subscription for the specified security
195  /// </summary>
196  /// <param name="request">The subscription request</param>
197  /// <returns>A new subscription instance of the specified security</returns>
198  private Subscription CreateDataSubscription(SubscriptionRequest request)
199  {
200  Subscription subscription = null;
201 
202  var localEndTime = request.EndTimeUtc.ConvertFromUtc(request.Security.Exchange.TimeZone);
203  var timeZoneOffsetProvider = new TimeZoneOffsetProvider(request.Configuration.ExchangeTimeZone, request.StartTimeUtc, request.EndTimeUtc);
204 
205  IEnumerator<BaseData> enumerator = null;
206  if (!_channelProvider.ShouldStreamSubscription(request.Configuration))
207  {
208  if (!Tiingo.IsAuthCodeSet)
209  {
210  // we're not using the SubscriptionDataReader, so be sure to set the auth token here
211  Tiingo.SetAuthCode(Config.Get("tiingo-auth-token"));
212  }
213 
214  var factory = new LiveCustomDataSubscriptionEnumeratorFactory(_timeProvider, _algorithm.ObjectStore);
215  var enumeratorStack = factory.CreateEnumerator(request, _dataProvider);
216 
217  var enqueable = new EnqueueableEnumerator<BaseData>();
218  _customExchange.AddEnumerator(request.Configuration.Symbol, enumeratorStack, handleData: data =>
219  {
220  enqueable.Enqueue(data);
221 
222  subscription?.OnNewDataAvailable();
223  });
224 
225  enumerator = enqueable;
226  }
227  else
228  {
229  var auxEnumerators = new List<IEnumerator<BaseData>>();
230 
231  if (LiveAuxiliaryDataEnumerator.TryCreate(request.Configuration, _timeProvider, request.Security.Cache, _mapFileProvider,
232  _factorFileProvider, request.StartTimeLocal, out var auxDataEnumator))
233  {
234  auxEnumerators.Add(auxDataEnumator);
235  }
236 
237  EventHandler handler = (_, _) => subscription?.OnNewDataAvailable();
238  enumerator = Subscribe(request.Configuration, handler, IsExpired);
239 
240  if (auxEnumerators.Count > 0)
241  {
242  enumerator = new LiveAuxiliaryDataSynchronizingEnumerator(_timeProvider, request.Configuration.ExchangeTimeZone, enumerator, auxEnumerators);
243  }
244  }
245 
246  // scale prices before 'SubscriptionFilterEnumerator' since it updates securities realtime price
247  // and before fill forwarding so we don't happen to apply twice the factor
248  if (request.Configuration.PricesShouldBeScaled(liveMode: true))
249  {
250  enumerator = new PriceScaleFactorEnumerator(
251  enumerator,
252  request.Configuration,
253  _factorFileProvider,
254  liveMode: true);
255  }
256 
257  if (request.Configuration.FillDataForward)
258  {
259  var fillForwardResolution = _subscriptions.UpdateAndGetFillForwardResolution(request.Configuration);
260 
261  enumerator = new LiveFillForwardEnumerator(_frontierTimeProvider, enumerator, request.Security.Exchange, fillForwardResolution, request.Configuration.ExtendedMarketHours, localEndTime, request.Configuration.Increment, request.Configuration.DataTimeZone);
262  }
263 
264  // make our subscriptions aware of the frontier of the data feed, prevents future data from spewing into the feed
265  enumerator = new FrontierAwareEnumerator(enumerator, _frontierTimeProvider, timeZoneOffsetProvider);
266 
267  // define market hours and user filters to incoming data after the frontier enumerator so during warmup we avoid any realtime data making it's way into the securities
268  if (request.Configuration.IsFilteredSubscription)
269  {
270  enumerator = new SubscriptionFilterEnumerator(enumerator, request.Security, localEndTime, request.Configuration.ExtendedMarketHours, true, request.ExchangeHours);
271  }
272 
273  enumerator = GetWarmupEnumerator(request, enumerator);
274 
275  var subscriptionDataEnumerator = new SubscriptionDataEnumerator(request.Configuration, request.Security.Exchange.Hours, timeZoneOffsetProvider, enumerator, request.IsUniverseSubscription);
276  subscription = new Subscription(request, subscriptionDataEnumerator, timeZoneOffsetProvider);
277 
278  return subscription;
279  }

281  /// <summary>
282  /// Helper method to determine if the symbol associated with the requested configuration is expired or not
283  /// </summary>
284  /// <remarks>This is useful during warmup where we can be requested to add some already expired asset. We want to skip sending it
285  /// to our live <see cref="_dataQueueHandler"/> instance to avoid explosions. But we do want to add warmup enumerators</remarks>
286  private bool IsExpired(SubscriptionDataConfig dataConfig)
287  {
288  var mapFile = _mapFileProvider.ResolveMapFile(dataConfig);
289  var delistingDate = dataConfig.Symbol.GetDelistingDate(mapFile);
290  return _timeProvider.GetUtcNow().Date > delistingDate.ConvertToUtc(dataConfig.ExchangeTimeZone);
291  }

298  /// <summary>
299  /// Creates a new subscription for universe selection
300  /// </summary>
301  /// <param name="request">The subscription request</param>
302  private Subscription CreateUniverseSubscription(SubscriptionRequest request)
303  {
304  Subscription subscription = null;
305 
306  // TODO : Consider moving the creating of universe subscriptions to a separate, testable class
307 
308  // grab the relevant exchange hours
309  var config = request.Universe.Configuration;
310  var localEndTime = request.EndTimeUtc.ConvertFromUtc(request.Security.Exchange.TimeZone);
311  var tzOffsetProvider = new TimeZoneOffsetProvider(request.Configuration.ExchangeTimeZone, request.StartTimeUtc, request.EndTimeUtc);
312 
313  IEnumerator<BaseData> enumerator = null;
314 
315  var timeTriggered = request.Universe as ITimeTriggeredUniverse;
316  if (timeTriggered != null)
317  {
318  Log.Trace($"LiveTradingDataFeed.CreateUniverseSubscription(): Creating user defined universe: {config.Symbol.ID}");
319 
320  // spoof a tick on the requested interval to trigger the universe selection function
321  var enumeratorFactory = new TimeTriggeredUniverseSubscriptionEnumeratorFactory(timeTriggered, MarketHoursDatabase.FromDataFolder(), _frontierTimeProvider);
322  enumerator = enumeratorFactory.CreateEnumerator(request, _dataProvider);
323 
324  enumerator = new FrontierAwareEnumerator(enumerator, _timeProvider, tzOffsetProvider);
325 
326  var enqueueable = new EnqueueableEnumerator<BaseData>();
327  _customExchange.AddEnumerator(new EnumeratorHandler(config.Symbol, enumerator, enqueueable));
328  enumerator = enqueueable;
329  }
330  else if (config.Type.IsAssignableTo(typeof(ETFConstituentUniverse)) || config.Type.IsAssignableTo(typeof(FundamentalUniverse)))
331  {
332  Log.Trace($"LiveTradingDataFeed.CreateUniverseSubscription(): Creating {config.Type.Name} universe: {config.Symbol.ID}");
333 
334  // Will try to pull data from the data folder every 10min, file with yesterdays date.
335  // If lean is started today it will trigger initial coarse universe selection
336  var factory = new LiveCustomDataSubscriptionEnumeratorFactory(_timeProvider,
337  _algorithm.ObjectStore,
338  // we adjust time to the previous tradable date
339  time => Time.GetStartTimeForTradeBars(request.Security.Exchange.Hours, time, Time.OneDay, 1, false, config.DataTimeZone),
340  TimeSpan.FromMinutes(10)
341  );
342  var enumeratorStack = factory.CreateEnumerator(request, _dataProvider);
343 
344  // aggregates each coarse data point into a single BaseDataCollection
345  var aggregator = new BaseDataCollectionAggregatorEnumerator(enumeratorStack, config.Symbol, true);
346  var enqueable = new EnqueueableEnumerator<BaseData>();
347  _customExchange.AddEnumerator(config.Symbol, aggregator, handleData: data =>
348  {
349  enqueable.Enqueue(data);
350  subscription?.OnNewDataAvailable();
351  });
352 
353  enumerator = GetConfiguredFrontierAwareEnumerator(enqueable, tzOffsetProvider,
354  // advance time if before 23pm or after 5am and not on Saturdays
355  time => time.Hour < 23 && time.Hour > 5 && time.DayOfWeek != DayOfWeek.Saturday);
356  }
357  else if (request.Universe is OptionChainUniverse)
358  {
359  Log.Trace("LiveTradingDataFeed.CreateUniverseSubscription(): Creating option chain universe: " + config.Symbol.ID);
360 
361  Func<SubscriptionRequest, IEnumerator<BaseData>> configure = (subRequest) =>
362  {
363  var fillForwardResolution = _subscriptions.UpdateAndGetFillForwardResolution(subRequest.Configuration);
364  var input = Subscribe(subRequest.Configuration, (sender, args) => subscription?.OnNewDataAvailable(), (_) => false);
365  return new LiveFillForwardEnumerator(_frontierTimeProvider, input, subRequest.Security.Exchange, fillForwardResolution, subRequest.Configuration.ExtendedMarketHours, localEndTime, subRequest.Configuration.Increment, subRequest.Configuration.DataTimeZone);
366  };
367 
368  var symbolUniverse = GetUniverseProvider(SecurityType.Option);
369 
370  var enumeratorFactory = new OptionChainUniverseSubscriptionEnumeratorFactory(configure, symbolUniverse, _timeProvider);
371  enumerator = enumeratorFactory.CreateEnumerator(request, _dataProvider);
372 
373  enumerator = new FrontierAwareEnumerator(enumerator, _frontierTimeProvider, tzOffsetProvider);
374  }
375  else if (request.Universe is FuturesChainUniverse)
376  {
377  Log.Trace("LiveTradingDataFeed.CreateUniverseSubscription(): Creating futures chain universe: " + config.Symbol.ID);
378 
379  var symbolUniverse = GetUniverseProvider(SecurityType.Future);
380 
381  enumerator = new DataQueueFuturesChainUniverseDataCollectionEnumerator(request, symbolUniverse, _timeProvider);
382  enumerator = new FrontierAwareEnumerator(enumerator, _frontierTimeProvider, tzOffsetProvider);
383  }
384  else
385  {
386  Log.Trace("LiveTradingDataFeed.CreateUniverseSubscription(): Creating custom universe: " + config.Symbol.ID);
387 
388  var factory = new LiveCustomDataSubscriptionEnumeratorFactory(_timeProvider, _algorithm.ObjectStore);
389  var enumeratorStack = factory.CreateEnumerator(request, _dataProvider);
390  enumerator = new BaseDataCollectionAggregatorEnumerator(enumeratorStack, config.Symbol, liveMode: true);
391 
392  var enqueueable = new EnqueueableEnumerator<BaseData>();
393  _customExchange.AddEnumerator(new EnumeratorHandler(config.Symbol, enumerator, enqueueable));
394  enumerator = enqueueable;
395  }
396 
397  enumerator = GetWarmupEnumerator(request, enumerator);
398 
399  // create the subscription
400  var subscriptionDataEnumerator = new SubscriptionDataEnumerator(request.Configuration, request.Security.Exchange.Hours, tzOffsetProvider, enumerator, request.IsUniverseSubscription);
401  subscription = new Subscription(request, subscriptionDataEnumerator, tzOffsetProvider);
402 
403  // send the subscription for the new symbol through to the data queuehandler
404  if (_channelProvider.ShouldStreamSubscription(subscription.Configuration))
405  {
406  Subscribe(request.Configuration, (sender, args) => subscription?.OnNewDataAvailable(), (_) => false);
407  }
408 
409  return subscription;
410  }

412  /// <summary>
413  /// Build and apply the warmup enumerators when required
414  /// </summary>
415  private IEnumerator<BaseData> GetWarmupEnumerator(SubscriptionRequest request, IEnumerator<BaseData> liveEnumerator)
416  {
417  if (_algorithm.IsWarmingUp)
418  {
419  var warmupRequest = new SubscriptionRequest(request, endTimeUtc: _timeProvider.GetUtcNow(),
420  // we will not fill forward each warmup enumerators separately but concatenated bellow
421  configuration: new SubscriptionDataConfig(request.Configuration, fillForward: false,
422  resolution: _algorithm.Settings.WarmupResolution));
423  if (warmupRequest.TradableDaysInDataTimeZone.Any()
424  // make sure there is at least room for a single bar of the requested resolution, else can cause issues with some history providers
425  // this could happen when we create some internal subscription whose start time is 'Now', which we don't really want to warmup
426  && warmupRequest.EndTimeUtc - warmupRequest.StartTimeUtc >= warmupRequest.Configuration.Resolution.ToTimeSpan()
427  // since we change the resolution, let's validate it's still valid configuration (example daily equity quotes are not!)
428  && LeanData.IsValidConfiguration(warmupRequest.Configuration.SecurityType, warmupRequest.Configuration.Resolution, warmupRequest.Configuration.TickType))
429  {
430  // since we will source data locally and from the history provider, let's limit the history request size
431  // by setting a start date respecting the 'MaximumWarmupHistoryDaysLookBack'
432  var historyWarmup = warmupRequest;
433  var warmupHistoryStartDate = warmupRequest.EndTimeUtc.AddDays(-MaximumWarmupHistoryDaysLookBack);
434  if (warmupHistoryStartDate > warmupRequest.StartTimeUtc)
435  {
436  historyWarmup = new SubscriptionRequest(warmupRequest, startTimeUtc: warmupHistoryStartDate);
437  }
438 
439  // let's keep track of the last point we got from the file based enumerator and start our history enumeration from this point
440  // this is much more efficient since these duplicated points will be dropped by the filter righ away causing memory usage spikes
441  var lastPointTracker = new LastPointTracker();
442 
443  var synchronizedWarmupEnumerator = TryAddFillForwardEnumerator(warmupRequest,
444  // we concatenate the file based and history based warmup enumerators, dropping duplicate time stamps
445  new ConcatEnumerator(true, GetFileBasedWarmupEnumerator(warmupRequest, lastPointTracker), GetHistoryWarmupEnumerator(historyWarmup, lastPointTracker)) { CanEmitNull = false },
446  // if required by the original request, we will fill forward the Synced warmup data
447  request.Configuration.FillDataForward,
448  _algorithm.Settings.WarmupResolution);
449 
450  // don't let future data past. We let null pass because that's letting the next enumerator know we've ended because we always return true in live
451  synchronizedWarmupEnumerator = new FilterEnumerator<BaseData>(synchronizedWarmupEnumerator, data => data == null || data.EndTime <= warmupRequest.EndTimeLocal);
452 
453  // the order here is important, concat enumerator will keep the last enumerator given and dispose of the rest
454  liveEnumerator = new ConcatEnumerator(true, synchronizedWarmupEnumerator, liveEnumerator);
455  }
456  }
457  return liveEnumerator;
458  }

460  /// <summary>
461  /// File based warmup enumerator
462  /// </summary>
463  private IEnumerator<BaseData> GetFileBasedWarmupEnumerator(SubscriptionRequest warmup, LastPointTracker lastPointTracker)
464  {
465  IEnumerator<BaseData> result = null;
466  try
467  {
468  result = new FilterEnumerator<BaseData>(CreateEnumerator(warmup),
469  data =>
470  {
471  // don't let future data past, nor fill forward, that will be handled after merging with the history request response
472  if (data == null || data.EndTime < warmup.EndTimeLocal && !data.IsFillForward)
473  {
474  if (data != null)
475  {
476  lastPointTracker.LastDataPoint = data;
477  }
478  return true;
479  }
480  return false;
481  });
482  }
483  catch (Exception e)
484  {
485  Log.Error(e, $"File based warmup: {warmup.Configuration}");
486  }
487  return result;
488  }

490  /// <summary>
491  /// History based warmup enumerator
492  /// </summary>
493  private IEnumerator<BaseData> GetHistoryWarmupEnumerator(SubscriptionRequest warmup, LastPointTracker lastPointTracker)
494  {
495  IEnumerator<BaseData> result;
496  if (warmup.IsUniverseSubscription)
497  {
498  // we ignore the fill forward time span argument because we will fill forwared the concatenated file and history based enumerators next in the stack
499  result = CreateUniverseEnumerator(warmup, createUnderlyingEnumerator: (req, _) => GetHistoryWarmupEnumerator(req, lastPointTracker));
500  }
501  else
502  {
503  // we create an enumerable of which we get the enumerator to defer the creation of the history request until the file based enumeration ended
504  // and potentially the 'lastPointTracker' is available to adjust our start time
505  result = new[] { warmup }.SelectMany(_ =>
506  {
507  var startTimeUtc = warmup.StartTimeUtc;
508  if (lastPointTracker != null && lastPointTracker.LastDataPoint != null)
509  {
510  var utcLastPointTime = lastPointTracker.LastDataPoint.Time.ConvertToUtc(warmup.ExchangeHours.TimeZone);
511  if (utcLastPointTime > startTimeUtc)
512  {
513  if (Log.DebuggingEnabled)
514  {
515  Log.Debug($"LiveTradingDataFeed.GetHistoryWarmupEnumerator(): Adjusting history warmup start time to {utcLastPointTime} from {startTimeUtc} for {warmup.Configuration}");
516  }
517  startTimeUtc = utcLastPointTime;
518  }
519  }
520  var historyRequest = new Data.HistoryRequest(warmup.Configuration, warmup.ExchangeHours, startTimeUtc, warmup.EndTimeUtc);
521  try
522  {
523  return _algorithm.HistoryProvider.GetHistory(new[] { historyRequest }, _algorithm.TimeZone).Select(slice =>
524  {
525  try
526  {
527  var data = slice.Get(historyRequest.DataType);
528  return (BaseData)data[warmup.Configuration.Symbol];
529  }
530  catch (Exception e)
531  {
532  Log.Error(e, $"History warmup: {warmup.Configuration}");
533  }
534  return null;
535  });
536  }
537  catch
538  {
539  // some history providers could throw if they do not support a type
540  }
541  return Enumerable.Empty<BaseData>();
542  }).GetEnumerator();
543  }
544 
545  return new FilterEnumerator<BaseData>(result,
546  // don't let future data past, nor fill forward, that will be handled after merging with the file based enumerator
547  data => data == null || data.EndTime < warmup.EndTimeLocal && !data.IsFillForward);
548  }

550  /// <summary>
551  /// Will wrap the provided enumerator with a <see cref="FrontierAwareEnumerator"/>
552  /// using a <see cref="PredicateTimeProvider"/> that will advance time based on the provided
553  /// function
554  /// </summary>
555  /// <remarks>Won't advance time if now.Hour is bigger or equal than 23pm, less or equal than 5am or Saturday.
556  /// This is done to prevent universe selection occurring in those hours so that the subscription changes
557  /// are handled correctly.</remarks>
558  private IEnumerator<BaseData> GetConfiguredFrontierAwareEnumerator(
559  IEnumerator<BaseData> enumerator,
560  TimeZoneOffsetProvider tzOffsetProvider,
561  Func<DateTime, bool> customStepEvaluator)
562  {
563  var stepTimeProvider = new PredicateTimeProvider(_frontierTimeProvider, customStepEvaluator);
564 
565  return new FrontierAwareEnumerator(enumerator, stepTimeProvider, tzOffsetProvider);
566  }
