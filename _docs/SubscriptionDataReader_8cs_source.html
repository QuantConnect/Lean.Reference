127  /// <summary>
128  /// Subscription data reader takes a subscription request, loads the type, accepts the data source and enumerate on the results.
129  /// </summary>
130  /// <param name="config">Subscription configuration object</param>
131  /// <param name="dataRequest">The data request</param>
132  /// <param name="mapFileProvider">Used for resolving the correct map files</param>
133  /// <param name="factorFileProvider">Used for getting factor files</param>
134  /// <param name="dataCacheProvider">Used for caching files</param>
135  /// <param name="dataProvider">The data provider to use</param>
136  public SubscriptionDataReader(SubscriptionDataConfig config,
137  BaseDataRequest dataRequest,
138  IMapFileProvider mapFileProvider,
139  IFactorFileProvider factorFileProvider,
140  IDataCacheProvider dataCacheProvider,
141  IDataProvider dataProvider,
142  IObjectStore objectStore)
143  {
144  //Save configuration of data-subscription:
145  _config = config;
146 
147  //Save Start and End Dates:
148  _periodStart = dataRequest.StartTimeLocal;
149  _periodFinish = dataRequest.EndTimeLocal;
150  _mapFileProvider = mapFileProvider;
151  _factorFileProvider = factorFileProvider;
152  _dataCacheProvider = dataCacheProvider;
153 
154  //Save access to securities
155  _tradeableDates = dataRequest.TradableDaysInDataTimeZone.GetEnumerator();
156  _dataProvider = dataProvider;
157  _objectStore = objectStore;

160  /// <summary>
161  /// Initializes the <see cref="SubscriptionDataReader"/> instance
162  /// </summary>
163  /// <remarks>Should be called after all consumers of <see cref="NewTradableDate"/> event are set,
164  /// since it will produce events.</remarks>
165  public void Initialize()
166  {
167  if (_initialized)
168  {
169  return;
170  }
171 
172  //Save the type of data we'll be getting from the source.
173  try
174  {
175  _dataFactory = _config.GetBaseDataInstance();
176  }
177  catch (ArgumentException exception)
178  {
179  OnInvalidConfigurationDetected(new InvalidConfigurationDetectedEventArgs(_config.Symbol, exception.Message));
180  _endOfStream = true;
181  return;
182  }
183 
184  // If Tiingo data, set the access token in data factory
185  var tiingo = _dataFactory as TiingoPrice;
186  if (tiingo != null)
187  {
188  if (!Tiingo.IsAuthCodeSet)
189  {
190  Tiingo.SetAuthCode(Config.Get("tiingo-auth-token"));
191  }
192  }
193 
194  // load up the map files for equities, options, and custom data if it supports it.
195  // Only load up factor files for equities
196  if (_dataFactory.RequiresMapping())
197  {
198  try
199  {
200  var mapFile = _mapFileProvider.ResolveMapFile(_config);
201 
202  // only take the resolved map file if it has data, otherwise we'll use the empty one we defined above
203  if (mapFile.Any()) _mapFile = mapFile;
204 
205  if (_config.PricesShouldBeScaled())
206  {
207  var factorFile = _factorFileProvider.Get(_config.Symbol);
208  _hasScaleFactors = factorFile != null;
209  if (_hasScaleFactors)
210  {
211  _factorFile = factorFile;
212 
213  // if factor file has minimum date, update start period if before minimum date
214  if (_factorFile != null && _factorFile.FactorFileMinimumDate.HasValue)
215  {
216  if (_periodStart < _factorFile.FactorFileMinimumDate.Value)
217  {
218  _periodStart = _factorFile.FactorFileMinimumDate.Value;
219 
220  OnNumericalPrecisionLimited(
221  new NumericalPrecisionLimitedEventArgs(_config.Symbol,
222  $"[{_config.Symbol.Value}, {_factorFile.FactorFileMinimumDate.Value.ToShortDateString()}]"));
223  }
224  }
225  }
226 
227  if (_periodStart < mapFile.FirstDate)
228  {
229  _periodStart = mapFile.FirstDate;
230 
231  OnStartDateLimited(
232  new StartDateLimitedEventArgs(_config.Symbol,
233  $"[{_config.Symbol.Value}," +
234  $" {mapFile.FirstDate.ToString("yyyy-MM-dd", CultureInfo.InvariantCulture)}]"));
235  }
236  }
237  }
238  catch (Exception err)
239  {
240  Log.Error(err, "Fetching Price/Map Factors: " + _config.Symbol.ID + ": ");
241  }
242  }
243 
244  _factorFile ??= _config.Symbol.GetEmptyFactorFile();
245  _mapFile ??= new MapFile(_config.Symbol.Value, Enumerable.Empty<MapFileRow>());
246 
247  _delistingDate = _config.Symbol.GetDelistingDate(_mapFile);
248 
249  // adding a day so we stop at EOD
250  _delistingDate = _delistingDate.AddDays(1);
251  UpdateDataEnumerator(true);
252 
253  _initialized = true;
254  }

256  /// <summary>
257  /// Advances the enumerator to the next element of the collection.
258  /// </summary>
259  /// <returns>
260  /// true if the enumerator was successfully advanced to the next element; false if the enumerator has passed the end of the collection.
261  /// </returns>
262  /// <exception cref="T:System.InvalidOperationException">The collection was modified after the enumerator was created. </exception><filterpriority>2</filterpriority>
263  public bool MoveNext()
264  {
265  if (!_initialized)
266  {
267  // Late initialization so it is performed in the data feed stack
268  // and not in the algorithm thread
269  Initialize();
270  }
271 
272  if (_endOfStream)
273  {
274  return false;
275  }
276 
277  if (Current != null)
278  {
279  // only save previous price data
280  _previous = Current;
281  }
282 
283  if (_subscriptionFactoryEnumerator == null)
284  {
285  _endOfStream = true;
286  return false;
287  }
288 
289  do
290  {
291  if (_pastDelistedDate)
292  {
293  break;
294  }
295  // keep enumerating until we find something that is within our time frame
296  while (_subscriptionFactoryEnumerator.MoveNext())
297  {
298  var instance = _subscriptionFactoryEnumerator.Current;
299  if (instance == null)
300  {
301  // keep reading until we get valid data
302  continue;
303  }
304 
305  // prevent emitting past data, this can happen when switching symbols on daily data
306  if (_previous != null && _config.Resolution != Resolution.Tick)
307  {
308  if (_config.IsCustomData)
309  {
310  // Skip the point if time went backwards for custom data?
311  // TODO: Should this be the case for all datapoints?
312  if (instance.EndTime < _previous.EndTime) continue;
313  }
314  else
315  {
316  // all other resolutions don't allow duplicate end times
317  if (instance.EndTime <= _previous.EndTime) continue;
318  }
319  }
320 
321  if (instance.EndTime < _periodStart)
322  {
323  // keep reading until we get a value on or after the start
324  _previous = instance;
325  continue;
326  }
327 
328  // if we move past our current 'date' then we need to do daily things, such
329  // as updating factors and symbol mapping
330  var shouldSkip = false;
331 
332  while (instance.Time.ConvertTo(_config.ExchangeTimeZone, _config.DataTimeZone).Date > _tradeableDates.Current)
333  {
334  var currentTradeableDate = _tradeableDates.Current;
335  if (UpdateDataEnumerator(false))
336  {
337  shouldSkip = true;
338  if (_subscriptionFactoryEnumerator == null)
339  {
340  // if null enumerator we have not been mapped into something new, we just ended,
341  // let's double check this data point should be skipped or not based on current tradeable date
342  shouldSkip = instance.Time.ConvertTo(_config.ExchangeTimeZone, _config.DataTimeZone).Date > _tradeableDates.Current;
343  if (shouldSkip)
344  {
345  // the end, no new enumerator and current instance is beyond current date
346  _endOfStream = true;
347  return false;
348  }
349  }
350  break;
351  }
352 
353  if (currentTradeableDate == _tradeableDates.Current)
354  {
355  // if tradeable dates did not advanced let's not check again
356  break;
357  }
358  }
359  if(shouldSkip)
360  {
361  // Skip current 'instance' if its start time is beyond the current date, fixes GH issue 3912
362  continue;
363  }
364 
365  // We have to perform this check after refreshing the enumerator, if appropriate
366  // 'instance' could be a data point far in the future due to remapping (GH issue 5232) in which case it will be dropped
367  if (instance.Time > _periodFinish)
368  {
369  // stop reading when we get a value after the end
370  _endOfStream = true;
371  return false;
372  }
373 
374  // we've made it past all of our filters, we're withing the requested start/end of the subscription,
375  // we've satisfied user and market hour filters, so this data is good to go as current
376  Current = instance;
377 
378  // we keep the last raw price registered before we return so we are not affected by anyone (price scale) modifying our current
379  _lastRawPrice = Current.Price;
380  return true;
381  }
382 
383  // we've ended the enumerator, time to refresh
384  UpdateDataEnumerator(true);
385  }
386  while (_subscriptionFactoryEnumerator != null);
387 
388  _endOfStream = true;
389  return false;
390  }
391 
392  /// <summary>
393  /// Resolves the next enumerator to be used in <see cref="MoveNext"/> and updates
394  /// <see cref="_subscriptionFactoryEnumerator"/>
395  /// </summary>
396  /// <returns>True, if the enumerator has been updated (even if updated to null)</returns>
397  private bool UpdateDataEnumerator(bool endOfEnumerator)
398  {
399  do
400  {
401  // always advance the date enumerator, this function is intended to be
402  // called on date changes, never return null for live mode, we'll always
403  // just keep trying to refresh the subscription
404  DateTime date;
405  if (!TryGetNextDate(out date))
406  {
407  _subscriptionFactoryEnumerator = null;
408  // if we run out of dates then we're finished with this subscription
409  return true;
410  }
411 
412  // fetch the new source, using the data time zone for the date
413  var newSource = _dataFactory.GetSource(_config, date, false);
414 
415  // check if we should create a new subscription factory
416  var sourceChanged = _source != newSource && newSource.Source != "";
417  if (sourceChanged)
418  {
419  // dispose of the current enumerator before creating a new one
420  Dispose();
421 
422  // save off for comparison next time
423  _source = newSource;
424  var subscriptionFactory = CreateSubscriptionFactory(newSource, _dataFactory, _dataProvider);
425  _subscriptionFactoryEnumerator = subscriptionFactory.Read(newSource).GetEnumerator();
426  return true;
427  }
428 
429  // if there's still more in the enumerator and we received the same source from the GetSource call
430  // above, then just keep using the same enumerator as we were before
431  if (!endOfEnumerator) // && !sourceChanged is always true here
432  {
433  return false;
434  }
435 
436  // keep churning until we find a new source or run out of tradeable dates
437  // in live mode tradeable dates won't advance beyond today's date, but
438  // TryGetNextDate will return false if it's already at today
439  }
440  while (true);
441  }

443  private ISubscriptionDataSourceReader CreateSubscriptionFactory(SubscriptionDataSource source, BaseData baseDataInstance, IDataProvider dataProvider)
444  {
445  var factory = SubscriptionDataSourceReader.ForSource(source, _dataCacheProvider, _config, _tradeableDates.Current, false, baseDataInstance, dataProvider, _objectStore);
446  AttachEventHandlers(factory, source);
447  return factory;
448  }

450  private void AttachEventHandlers(ISubscriptionDataSourceReader dataSourceReader, SubscriptionDataSource source)
451  {
452  dataSourceReader.InvalidSource += (sender, args) =>
453  {
454  if (_config.IsCustomData && !_config.Type.GetBaseDataInstance().IsSparseData())
455  {
456  OnDownloadFailed(
457  new DownloadFailedEventArgs(_config.Symbol,
458  "We could not fetch the requested data. " +
459  "This may not be valid data, or a failed download of custom data. " +
460  $"Skipping source ({args.Source.Source})."));
461  return;
462  }
463 
464  switch (args.Source.TransportMedium)
465  {
466  case SubscriptionTransportMedium.LocalFile:
467  // the local uri doesn't exist, write an error and return null so we we don't try to get data for today
468  // Log.Trace(string.Format("SubscriptionDataReader.GetReader(): Could not find QC Data, skipped: {0}", source));
469  break;
470 
471  case SubscriptionTransportMedium.RemoteFile:
472  OnDownloadFailed(
473  new DownloadFailedEventArgs(_config.Symbol,
474  $"Error downloading custom data source file, skipped: {source} " +
475  $"Error: {args.Exception.Message}", args.Exception.StackTrace));
476  break;
477 
478  case SubscriptionTransportMedium.Rest:
479  break;
480 
481  case SubscriptionTransportMedium.ObjectStore:
482  break;
483 
484  default:
485  throw new ArgumentOutOfRangeException();
486  }
487  };
488 
489  if (dataSourceReader is TextSubscriptionDataSourceReader)
490  {
491  // handle empty files/instantiation errors
492  var textSubscriptionFactory = (TextSubscriptionDataSourceReader)dataSourceReader;
493  // handle parser errors
494  textSubscriptionFactory.ReaderError += (sender, args) =>
495  {
496  OnReaderErrorDetected(
497  new ReaderErrorDetectedEventArgs(_config.Symbol,
498  $"Error invoking {_config.Symbol} data reader. " +
499  $"Line: {args.Line} Error: {args.Exception.Message}",
500  args.Exception.StackTrace));
501  };
502  }
503  }

505  /// <summary>
506  /// Iterates the tradeable dates enumerator
507  /// </summary>
508  /// <param name="date">The next tradeable date</param>
509  /// <returns>True if we got a new date from the enumerator, false if it's exhausted, or in live mode if we're already at today</returns>
510  private bool TryGetNextDate(out DateTime date)
511  {
512  while (_tradeableDates.MoveNext())
513  {
514  date = _tradeableDates.Current;
515 
516  OnNewTradableDate(new NewTradableDateEventArgs(date, _previous, _config.Symbol, _lastRawPrice));
517 
518  if (_pastDelistedDate || date > _delistingDate)
519  {
520  // if we already passed our delisting date we stop
521  _pastDelistedDate = true;
522  break;
523  }
524 
525  if (!_mapFile.HasData(date))
526  {
527  continue;
528  }
529 
530  // don't do other checks if we haven't gotten data for this date yet
531  if (_previous != null && _previous.EndTime.ConvertTo(_config.ExchangeTimeZone, _config.DataTimeZone) > _tradeableDates.Current)
532  {
533  continue;
534  }
535 
536  // we've passed initial checks,now go get data for this date!
537  return true;
538  }
539 
540  // no more tradeable dates, we've exhausted the enumerator
541  date = DateTime.MaxValue.Date;
542  return false;
543  }
