74  /// <summary>

128  /// <summary>
129  /// Initializes a new instance of the <see cref="ClassicRenkoConsolidator" /> class.
130  /// </summary>
131  /// <param name="barSize">The size of each bar in units of the value produced by <paramref name="selector"/></param>
132  /// <param name="selector">Extracts the value from a data instance to be formed into a <see cref="RenkoBar"/>. The default
133  /// value is (x => x.Value) the <see cref="IBaseData.Value"/> property on <see cref="IBaseData"/></param>
134  /// <param name="volumeSelector">Extracts the volume from a data instance. The default value is null which does
135  /// not aggregate volume per bar.</param>
136  /// <param name="evenBars">When true bar open/close will be a multiple of the barSize</param>
137  public ClassicRenkoConsolidator(decimal barSize,
138  PyObject selector,
139  PyObject volumeSelector = null,
140  bool evenBars = true)
141  : this(barSize, evenBars)
142  {
143  using (Py.GIL())
144  {
145  if (selector != null && !selector.IsNone())
146  {
147  if (!selector.TryConvertToDelegate(out _selector))
148  {
149  throw new ArgumentException(
150  "Unable to convert parameter 'selector' to delegate type Func<IBaseData, decimal>");
151  }
152  }
153  else
154  {
155  _selector = x => x.Value;
156  }
157 
158  if (volumeSelector != null && !volumeSelector.IsNone())
159  {
160  if (!volumeSelector.TryConvertToDelegate(out _volumeSelector))
161  {
162  throw new ArgumentException(
163  "Unable to convert parameter 'volumeSelector' to delegate type Func<IBaseData, decimal>");
164  }
165  }
166  else
167  {
168  _volumeSelector = x => 0;
169  }
170  }
171  }

173  /// <summary>
174  /// Updates this consolidator with the specified data
175  /// </summary>
176  /// <param name="data">The new data for the consolidator</param>
177  public void Update(IBaseData data)
178  {
179  var currentValue = _selector(data);
180  var volume = _volumeSelector(data);
181 
182  decimal? close = null;
183 
184  // if we're already in a bar then update it
185  if (_currentBar != null)
186  {
187  _currentBar.Update(data.Time, currentValue, volume);
188 
189  // if the update caused this bar to close, fire the event and reset the bar
190  if (_currentBar.IsClosed)
191  {
192  close = _currentBar.Close;
193  OnDataConsolidated(_currentBar);
194  _currentBar = null;
195  }
196  }
197 
198  if (_currentBar == null)
199  {
200  var open = close ?? currentValue;
201  if (_evenBars && !close.HasValue)
202  {
203  open = Math.Ceiling(open / _barSize) * _barSize;
204  }
205 
206  _currentBar = new RenkoBar(data.Symbol, data.Time, _barSize, open, volume);
207  }
208  }

226  private static void EpsilonCheck(decimal barSize)
227  {
228  if (barSize < Extensions.GetDecimalEpsilon())
229  {
230  throw new ArgumentOutOfRangeException(nameof(barSize),
231  "RenkoConsolidator bar size must be positve and greater than 1e-28");
232  }

236  /// Event invocator for the DataConsolidated event. This should be invoked
237  /// by derived classes when they have consolidated a new piece of data.
238  /// </summary>
239  /// <param name="consolidated">The newly consolidated data</param>
240  private void OnDataConsolidated(RenkoBar consolidated)
241  {
242  DataConsolidated?.Invoke(this, consolidated);
243 
244  _dataConsolidatedHandler?.Invoke(this, consolidated);
