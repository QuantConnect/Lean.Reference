28 namespace QuantConnect.ToolBox.CoinApiDataConverter

30  /// <summary>
31  /// Reader class for CoinAPI crypto raw data.
32  /// </summary>
33  public class CoinApiDataReader
34  {
35  private readonly ISymbolMapper _symbolMapper;
36 
37  /// <summary>
38  /// Creates a new instance of the <see cref="CoinApiDataReader"/> class
39  /// </summary>
40  /// <param name="symbolMapper">The symbol mapper</param>
41  public CoinApiDataReader(ISymbolMapper symbolMapper)
42  {
43  _symbolMapper = symbolMapper;
44  }
45 
46  /// <summary>
47  /// Gets the coin API entry data.
48  /// </summary>
49  /// <param name="file">The source file.</param>
50  /// <param name="processingDate">The processing date.</param>
51  /// <returns></returns>
52  public CoinApiEntryData GetCoinApiEntryData(FileInfo file, DateTime processingDate)
53  {
54  // crypto/<market>/<date>/<ticktype>-563-BITFINEX_SPOT_BTC_USD.csv.gz
55  var tickType = file.FullName.Contains("trades") ? TickType.Trade : TickType.Quote;
56 
57  var symbolId = Path.GetFileNameWithoutExtension(Path.GetFileNameWithoutExtension(file.Name)).Split('-').Last();
58 
59  var symbol = _symbolMapper.GetLeanSymbol(symbolId, SecurityType.Crypto, null);
60 
61  return new CoinApiEntryData
62  {
63  Name = file.Name,
64  Symbol = symbol,
65  TickType = tickType,
66  Date = processingDate
67  };
68  }
69 
70  /// <summary>
71  /// Gets an list of ticks for a given CoinAPI source file.
72  /// </summary>
73  /// <param name="entryData">The entry data.</param>
74  /// <param name="file">The file.</param>
75  /// <returns></returns>
76  /// <exception cref="Exception">CoinApiDataReader.ProcessCoinApiEntry(): CSV header not found for entry name: {entryData.Name}</exception>
77  public IEnumerable<Tick> ProcessCoinApiEntry(CoinApiEntryData entryData, FileInfo file)
78  {
79  Log.Trace("CoinApiDataReader.ProcessTarEntry(): Processing " +
80  $"{entryData.Symbol.ID.Market}-{entryData.Symbol.Value}-{entryData.TickType} " +
81  $"for {entryData.Date:yyyy-MM-dd}");
82 
83 
84  using (var stream = new GZipStream(file.OpenRead(), CompressionMode.Decompress))
85  using (var reader = new StreamReader(stream))
86  {
87  var headerLine = reader.ReadLine();
88  if (headerLine == null)
89  {
90  throw new Exception($"CoinApiDataReader.ProcessCoinApiEntry(): CSV header not found for entry name: {entryData.Name}");
91  }
92 
93  var headerParts = headerLine.Split(';').ToList();
94 
95  var tickList = entryData.TickType == TickType.Trade
96  ? ParseTradeData(entryData.Symbol, reader, headerParts)
97  : ParseQuoteData(entryData.Symbol, reader, headerParts);
98 
99  foreach (var tick in tickList)
100  {
101  yield return tick;
102  }
103  }
104  }
105 
106  /// <summary>
107  /// Parses CoinAPI trade data.
108  /// </summary>
109  /// <param name="symbol">The symbol.</param>
110  /// <param name="reader">The reader.</param>
111  /// <param name="headerParts">The header parts.</param>
112  /// <returns></returns>
113  private IEnumerable<Tick> ParseTradeData(Symbol symbol, StreamReader reader, List<string> headerParts)
114  {
115  var columnTime = headerParts.FindIndex(x => x == "time_exchange");
116  var columnPrice = headerParts.FindIndex(x => x == "price");
117  var columnQuantity = headerParts.FindIndex(x => x == "base_amount");
118 
119  string line;
120  while ((line = reader.ReadLine()) != null)
121  {
122  var lineParts = line.Split(';');
123 
124  var time = DateTime.Parse(lineParts[columnTime], CultureInfo.InvariantCulture);
125  var price = lineParts[columnPrice].ToDecimal();
126  var quantity = lineParts[columnQuantity].ToDecimal();
127 
128  yield return new Tick
129  {
130  Symbol = symbol,
131  Time = time,
132  Value = price,
133  Quantity = quantity,
134  TickType = TickType.Trade,
135  Suspicious = price == -1
136  };
137  }
138  }
139 
140  /// <summary>
141  /// Parses CoinAPI quote data.
142  /// </summary>
143  /// <param name="symbol">The symbol.</param>
144  /// <param name="reader">The reader.</param>
145  /// <param name="headerParts">The header parts.</param>
146  /// <returns></returns>
147  private IEnumerable<Tick> ParseQuoteData(Symbol symbol, StreamReader reader, List<string> headerParts)
148  {
149  var columnTime = headerParts.FindIndex(x => x == "time_exchange");
150  var columnAskPrice = headerParts.FindIndex(x => x == "ask_px");
151  var columnAskSize = headerParts.FindIndex(x => x == "ask_sx");
152  var columnBidPrice = headerParts.FindIndex(x => x == "bid_px");
153  var columnBidSize = headerParts.FindIndex(x => x == "bid_sx");
154 
155  var previousAskPrice = 0m;
156  var previousBidPrice = 0m;
157 
158  string line;
159  while ((line = reader.ReadLine()) != null)
160  {
161  var lineParts = line.Split(';');
162 
163  var time = DateTime.Parse(lineParts[columnTime], CultureInfo.InvariantCulture);
164  var askPrice = lineParts[columnAskPrice].ToDecimal();
165  var askSize = lineParts[columnAskSize].ToDecimal();
166  var bidPrice = lineParts[columnBidPrice].ToDecimal();
167  var bidSize = lineParts[columnBidSize].ToDecimal();
168 
169  if (askPrice == previousAskPrice && bidPrice == previousBidPrice)
170  {
171  // only save quote if bid price or ask price changed
172  continue;
173  }
174 
175  previousAskPrice = askPrice;
176  previousBidPrice = bidPrice;
177 
178  yield return new Tick
179  {
180  Symbol = symbol,
181  Time = time,
182  AskPrice = askPrice,
183  AskSize = askSize,
184  BidPrice = bidPrice,
185  BidSize = bidSize,
186  TickType = TickType.Quote,
187  Suspicious = bidPrice == -1m || askPrice == -1m
188  };
189  }
190  }
191  }
