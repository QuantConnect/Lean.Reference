167  /// <summary>
168  /// Compose result data packet - with tradable dates from the backtest job task and the partial result packet.
169  /// </summary>
170  /// <param name="job">Job that started this request</param>
171  /// <param name="results">Results class for the Backtest job</param>
172  /// <param name="endDate">The algorithms backtest end date</param>
173  /// <param name="startDate">The algorithms backtest start date</param>
174  /// <param name="progress">Progress of the packet. For the packet we assume progess of 100%.</param>
175  public BacktestResultPacket(BacktestNodePacket job, BacktestResult results, DateTime endDate, DateTime startDate, decimal progress = 1m)
176  : this()
177  {
178  try
179  {
180  Progress = Math.Round(progress, 3);
181  SessionId = job.SessionId;
182  PeriodFinish = endDate;
183  PeriodStart = startDate;
184  CompileId = job.CompileId;
185  Channel = job.Channel;
186  BacktestId = job.BacktestId;
187  OptimizationId = job.OptimizationId;
188  Results = results;
189  Name = job.Name;
190  UserId = job.UserId;
191  ProjectId = job.ProjectId;
192  SessionId = job.SessionId;
193  TradeableDates = job.TradeableDates;
194  }
195  catch (Exception err) {
196  Log.Error(err);
197  }
198  }
