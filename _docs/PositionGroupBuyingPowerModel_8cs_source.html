65  /// <summary>
66  /// Computes the impact on the portfolio's buying power from adding the position group to the portfolio. This is
67  /// a 'what if' analysis to determine what the state of the portfolio would be if these changes were applied. The
68  /// delta (before - after) is the margin requirement for adding the positions and if the margin used after the changes
69  /// are applied is less than the total portfolio value, this indicates sufficient capital.
70  /// </summary>
71  /// <param name="parameters">An object containing the portfolio and a position group containing the contemplated
72  /// changes to the portfolio</param>
73  /// <returns>Returns the portfolio's total portfolio value and margin used before and after the position changes are applied</returns>
74  public virtual ReservedBuyingPowerImpact GetReservedBuyingPowerImpact(ReservedBuyingPowerImpactParameters parameters)
75  {
76  // This process aims to avoid having to compute buying power on the entire portfolio and instead determines
77  // the set of groups that can be impacted by the changes being contemplated. The only real way to determine
78  // the change in maintenance margin is to determine what groups we'll have after the changes and compute the
79  // margin based on that.
80  // 1. Determine impacted groups (depends on IPositionGroupResolver.GetImpactedGroups)
81  // 2. Compute the currently reserved buying power of impacted groups
82  // 3. Create position collection using impacted groups and apply contemplated changes
83  // 4. Resolve new position groups using position collection with applied contemplated changes
84  // 5. Compute the contemplated reserved buying power on these newly resolved groups
85 
86  // 1. Determine impacted groups
87  var positionManager = parameters.Portfolio.Positions;
88 
89  // 2. Compute current reserved buying power
90  var current = 0m;
91  var impactedGroups = new List<IPositionGroup>();
92 
93  // 3. Determine set of impacted positions to be grouped
94  var positions = parameters.Order.CreatePositions(parameters.Portfolio.Securities).ToList();
95 
96  var impactedPositions = positions.ToDictionary(p => p.Symbol);
97 
98  foreach (var impactedGroup in positionManager.GetImpactedGroups(positions))
99  {
100  impactedGroups.Add(impactedGroup);
101  current += impactedGroup.BuyingPowerModel.GetReservedBuyingPowerForPositionGroup(
102  parameters.Portfolio, impactedGroup
103  );
104 
105  foreach (var position in impactedGroup)
106  {
107  IPosition existing;
108  if (impactedPositions.TryGetValue(position.Symbol, out existing))
109  {
110  // if it already exists then combine it with the existing
111  impactedPositions[position.Symbol] = existing.Combine(position);
112  }
113  else
114  {
115  impactedPositions[position.Symbol] = position;
116  }
117  }
118  }
119 
120  // 4. Resolve new position groups
121  var contemplatedGroups = positionManager.ResolvePositionGroups(new PositionCollection(impactedPositions.Values));
122 
123  // 5. Compute contemplated reserved buying power
124  var contemplated = 0m;
125  foreach (var contemplatedGroup in contemplatedGroups)
126  {
127  contemplated += contemplatedGroup.BuyingPowerModel.GetMaintenanceMargin(
128  parameters.Portfolio, contemplatedGroup
129  );
130  }
131 
132  return new ReservedBuyingPowerImpact(
133  current, contemplated, impactedGroups, parameters.ContemplatedChanges, contemplatedGroups
134  );
135  }

137  /// <summary>
138  /// Check if there is sufficient buying power for the position group to execute this order.
139  /// </summary>
140  /// <param name="parameters">An object containing the portfolio, the position group and the order</param>
141  /// <returns>Returns buying power information for an order against a position group</returns>
142  public virtual HasSufficientBuyingPowerForOrderResult HasSufficientBuyingPowerForOrder(
143  HasSufficientPositionGroupBuyingPowerForOrderParameters parameters
144  )
145  {
146  // The addition of position groups requires that we not only check initial margin requirements, but also
147  // that we confirm that after the changes have been applied and the new groups resolved our maintenance
148  // margin is still in a valid range (less than TPV). For this model, we use the security's sufficient buying
149  // power impl to confirm initial margin requirements and lean heavily on GetReservedBuyingPowerImpact for
150  // help with confirming that our expected maintenance margin is still less than TPV.
151  // 1. Confirm we have sufficient buying power to execute the trade using security's BP model
152  // 2. Confirm we pass position group specific checks
153  // 3. Confirm we haven't exceeded maintenance margin limits via GetReservedBuyingPowerImpact's delta
154 
155  // 1. Confirm we meet initial margin requirements, accounting for buffer
156  var availableBuyingPower = this.GetPositionGroupBuyingPower(
157  parameters.Portfolio, parameters.PositionGroup, parameters.Order.Direction
158  );
159 
160  // 2. Confirm we pass position group specific checks
161  var result = PassesPositionGroupSpecificBuyingPowerForOrderChecks(parameters, availableBuyingPower);
162  if (result?.IsSufficient == false)
163  {
164  return result;
165  }
166 
167  // 3. Confirm that the new groupings arising from the change doesn't make maintenance margin exceed TPV
168  var args = new ReservedBuyingPowerImpactParameters(parameters.Portfolio, parameters.PositionGroup, parameters.Order);
169  var deltaBuyingPower = GetReservedBuyingPowerImpact(args).Delta;
170  if (deltaBuyingPower <= availableBuyingPower)
171  {
172  return parameters.Sufficient();
173  }
174 
175  return parameters.Insufficient(Invariant(
176  $"Id: {parameters.Order.Id}, Maintenance Margin Delta: {deltaBuyingPower.Normalize()}, Free Margin: {availableBuyingPower.Value.Normalize()}"
177  ));
178  }

204  /// <summary>
205  /// Get the maximum position group order quantity to obtain a position with a given buying power
206  /// percentage. Will not take into account free buying power.
207  /// </summary>
208  /// <param name="parameters">An object containing the portfolio, the position group and the target
209  /// signed buying power percentage</param>
210  /// <returns>Returns the maximum allowed market order quantity and if zero, also the reason</returns>
211  public virtual GetMaximumLotsResult GetMaximumLotsForTargetBuyingPower(
212  GetMaximumLotsForTargetBuyingPowerParameters parameters
213  )
214  {
215  // In order to determine maximum order quantity for a particular amount of buying power, we must resolve
216  // the group's 'unit' as this will be the quantity step size. If we don't step according to these units
217  // then we could be left with a different type of group with vastly different margin requirements, so we
218  // must keep the ratios between all of the position quantities the same. First we'll determine the target
219  // buying power, taking into account RequiredFreeBuyingPowerPercent to ensure a buffer. Then we'll evaluate
220  // the initial margin requirement using the provided position group position quantities. From this value,
221  // we can determine if we need to add more quantity or remove quantity by looking at the delta from the target
222  // to the computed initial margin requirement. We can also compute, assuming linearity, the change in initial
223  // margin requirements for each 'unit' of the position group added. The final value we need before starting to
224  // iterate to solve for quantity is the minimum quantities. This is the 'unit' of the position group, and any
225  // quantities less than the unit's quantity would yield an entirely different group w/ different margin calcs.
226  // Now that we've resolved our target, our group unit and the unit's initial margin requirement, we can iterate
227  // increasing/decreasing quantities in multiples of the unit's quantities until we're within a unit's amount of
228  // initial margin to the target buying power.
229  // NOTE: The first estimate MUST be greater than the target and iteration will successively decrease quantity estimates.
230  // 1. Determine current holdings of position group
231  // 2. If targeting zero, we can short circuit and return the negative of existing position quantities
232  // 3. Determine target buying power, taking into account RequiredFreeBuyingPowerPercent
233  // 4. Determine current used margin [we're using initial here to match BuyingPowerModel]
234  // 5. Determine if we need to buy or sell to reach the target and convert to absolutes
235  // 6. Resolve the group's 'unit' quantities, this is our step size
236  // 7. Compute the initial margin requirement for a single unit
237  // 7a. Compute and add order fees into the unit initial margin requirement
238  // 8. Verify the target is greater than 1 unit's initial margin, otherwise exit w/ zero
239  // 8a. Define minimum bounds on the target as target - (unit order margin + fees)
240  // 9. Assuming linearity, compute estimate of absolute order quantity to reach target
241  // 10. Begin iterating
242  // 11. For each quantity estimate, compute initial margin requirement
243  // 12. Compute order fees and add to the initial margin requirement
244  // 13. Check to see if current estimate yields is w/in one unit's margin of target (must be less than)
245  // 14. Compute a new quantity estimate
246  // 15. After 13 results in ending iteration, return result w/ direction from #5
247 
248  var portfolio = parameters.Portfolio;
249 
250  // 1. Determine current holdings of position group
251  var currentPositionGroup = portfolio.Positions[parameters.PositionGroup.Key];
252 
253  // 2. If targeting zero, short circuit and return the negative of existing quantities
254  if (parameters.TargetBuyingPower == 0m)
255  {
256  return parameters.Result(currentPositionGroup.Quantity);
257  }
258 
259  // 3. Determine target buying power, taking into account RequiredFreeBuyingPowerPercent
260  var bufferFactor = 1 - RequiredFreeBuyingPowerPercent;
261  var targetBufferFactor = bufferFactor * parameters.TargetBuyingPower;
262 
263  var totalPortfolioValue = portfolio.TotalPortfolioValue;
264 
265  // 4. Determine initial margin requirement for current holdings
266  var currentSignedUsedMargin = 0m;
267  if (currentPositionGroup.Quantity != 0)
268  {
269  currentSignedUsedMargin = this.GetInitialMarginRequirement(portfolio, currentPositionGroup);
270  }
271 
272  // 5. Determine if we need to buy or sell to reach target, we'll work in the land of absolutes after this
273  var signedTarget = targetBufferFactor * totalPortfolioValue - currentSignedUsedMargin;
274  var globalTarget = Math.Abs(signedTarget);
275  var direction = Math.Sign(signedTarget);
276 
277  // 6. Resolve 'unit' -- this defines our step size
278  var groupUnit = parameters.PositionGroup.Key.CreateUnitGroup();
279 
280  // 7. Compute initial margin requirement for a single unit
281  var absUnitMargin = this.GetInitialMarginRequirement(portfolio, groupUnit);
282  if (absUnitMargin == 0m)
283  {
284  // likely due to missing price data
285  var zeroPricedPosition = parameters.PositionGroup.FirstOrDefault(
286  p => portfolio.Securities.GetValueOrDefault(p.Symbol)?.Price == 0m
287  );
288  return parameters.Error(zeroPricedPosition?.Symbol.GetZeroPriceMessage()
289  ?? $"Computed zero initial margin requirement for {parameters.PositionGroup.GetUserFriendlyName()}."
290  );
291  }
292 
293  // 7a. Compute fees for a single unit - if fees and price are linear, we'll be able to solve exactly w/out iteration
294  var contemplatedOrderFees = GetOrderFeeInAccountCurrency(portfolio, groupUnit);
295  absUnitMargin += contemplatedOrderFees;
296 
297  // 8. Verify target is more that the unit margin -- for groups, minimum is same as unit margin
298  if (absUnitMargin > globalTarget)
299  {
300  return parameters.SilenceNonErrorReasons
301  ? parameters.Zero()
302  : parameters.Zero(
303  $"The target order margin {globalTarget} is less than the minimum initial margin: {absUnitMargin}"
304  );
305  }
306 
307  // 9. Compute initial position group quantity estimate -- group quantities are whole numbers [number of lots/unit quantities]
308  var positionGroupQuantity = Math.Floor(globalTarget / absUnitMargin);
309  var positionGroup = groupUnit.WithQuantity(positionGroupQuantity);
310 
311  // 9a. Compute fees for initial estimate and adjust targets. We take fees out of the target to decouple it from the quantity root finding
312 
313  // 10. Begin iterating until order quantity is within target absFinalOrderMargin bounds (coming from above)
314  var loopCount = 0;
315  const int maxLoopCount = 5;
316  contemplatedOrderFees = GetOrderFeeInAccountCurrency(portfolio, positionGroup);
317  var orderMarginWithoutFees = this.GetInitialMarginRequirement(portfolio, positionGroup);
318  var orderMargin = orderMarginWithoutFees + contemplatedOrderFees;
319 
320  // 8a. Define lower bounds on target, we seek an order that is >= targetMinimum and <= target
321  var target = Math.Abs(targetBufferFactor * (totalPortfolioValue - contemplatedOrderFees) - currentSignedUsedMargin);
322  var targetMinimum = target - absUnitMargin;
323 
324  var lastOrderQuantity = 0m;
325  while (orderMarginWithoutFees > target || orderMarginWithoutFees < targetMinimum)
326  {
327  // Evaluate delta target and compute new quantity estimate
328  var deltaTarget = target - orderMarginWithoutFees;
329  var marginPerUnit = orderMarginWithoutFees / positionGroupQuantity;
330  var deltaQuantity = (int) Math.Floor(deltaTarget / marginPerUnit);
331  if (deltaQuantity == 0)
332  {
333  deltaQuantity = orderMarginWithoutFees > target ? -1 : 1;
334  }
335 
336  positionGroupQuantity += deltaQuantity;
337 
338  if (positionGroupQuantity <= 0)
339  {
340  return parameters.Zero(
341  $"The target order margin {target} is less than the minimum {absUnitMargin}"
342  );
343  }
344 
345  ArgumentException error;
346  if (UnableToConverge(lastOrderQuantity, positionGroupQuantity, groupUnit, portfolio, target, orderMargin, absUnitMargin, contemplatedOrderFees, out error))
347  {
348  throw error;
349  }
350 
351  if (loopCount >= maxLoopCount)
352  {
353  break;
354  }
355 
356  // 12. Update order margin with new quantity estimate
357  positionGroup = positionGroup.WithQuantity(positionGroupQuantity);
358  contemplatedOrderFees = GetOrderFeeInAccountCurrency(portfolio, positionGroup);
359  orderMarginWithoutFees = this.GetInitialMarginRequirement(portfolio, positionGroup);
360  orderMargin = orderMarginWithoutFees + contemplatedOrderFees;
361 
362  loopCount++;
363  lastOrderQuantity = positionGroupQuantity;
364 
365  // Update margin per unit so we can refine our targets
366  marginPerUnit = orderMarginWithoutFees / positionGroupQuantity;
367 
368  // if we're iterating, it's possible that we have an exotic/non-linear fee or price structure
369  // to remove a potential source of non-linearity, we deduct the fees from our target to focus on quantity root finding
370  target = Math.Abs(targetBufferFactor * (totalPortfolioValue - contemplatedOrderFees) - currentSignedUsedMargin);
371  targetMinimum = target - marginPerUnit;
372  }
373 
374  // 15. Incorporate direction back into the result
375  return parameters.Result(direction * positionGroupQuantity);
376  }

378  /// <summary>
379  /// Get the maximum market position group order quantity to obtain a delta in the buying power used by a position group.
380  /// The deltas sign defines the position side to apply it to, positive long, negative short.
381  /// </summary>
382  /// <param name="parameters">An object containing the portfolio, the position group and the delta buying power</param>
383  /// <returns>Returns the maximum allowed market order quantity and if zero, also the reason</returns>
384  /// <remarks>Used by the margin call model to reduce the position by a delta percent.</remarks>
385  public virtual GetMaximumLotsResult GetMaximumLotsForDeltaBuyingPower(
386  GetMaximumLotsForDeltaBuyingPowerParameters parameters
387  )
388  {
389  // we convert this delta request into a target buying power request through projection
390  // by determining the currently used (reserved) buying power and adding the delta to
391  // arrive at a target buying power percentage
392 
393  var currentPositionGroup = parameters.Portfolio.Positions[parameters.PositionGroup.Key];
394  var usedBuyingPower = parameters.PositionGroup.BuyingPowerModel.GetReservedBuyingPowerForPositionGroup(
395  parameters.Portfolio, currentPositionGroup
396  );
397 
398  var signedUsedBuyingPower = Math.Sign(currentPositionGroup.Quantity) * usedBuyingPower;
399  var targetBuyingPower = signedUsedBuyingPower + parameters.DeltaBuyingPower;
400 
401  var targetBuyingPowerPercent = parameters.Portfolio.TotalPortfolioValue != 0
402  ? targetBuyingPower / parameters.Portfolio.TotalPortfolioValue
403  : 0;
404 
405  return GetMaximumLotsForTargetBuyingPower(new GetMaximumLotsForTargetBuyingPowerParameters(
406  parameters.Portfolio, parameters.PositionGroup, targetBuyingPowerPercent, parameters.MinimumOrderMarginPortfolioPercentage
407  ));
408  }

410  /// <summary>
411  /// Gets the buying power available for a position group trade
412  /// </summary>
413  /// <param name="parameters">A parameters object containing the algorithm's portfolio, security, and order direction</param>
414  /// <returns>The buying power available for the trade</returns>
415  public PositionGroupBuyingPower GetPositionGroupBuyingPower(PositionGroupBuyingPowerParameters parameters)
416  {
417  // SecurityPositionGroupBuyingPowerModel models buying power the same as non-grouped, so we can simply delegate
418  // to the security's model. For posterity, however, I'll lay out the process for computing the available buying
419  // power for a position group trade. There's two separate cases, one where we're increasing the position and one
420  // where we're decreasing the position and potentially crossing over zero. When decreasing the position we have
421  // to account for the reserved buying power that the position currently holds and add that to any free buying power
422  // in the portfolio.
423  // 1. Get portfolio's MarginRemaining (free buying power)
424  // 2. Determine if closing position
425  // 2a. Add reserved buying power freed up by closing the position
426  // 2b. Rebate initial buying power required for current position [to match current behavior, might not be possible]
427 
428  // 1. Get MarginRemaining
429  var buyingPower = parameters.Portfolio.MarginRemaining;
430 
431  // 2. Determine if closing position
432  IPositionGroup existing;
433  if (parameters.Portfolio.Positions.Groups.TryGetGroup(parameters.PositionGroup.Key, out existing) &&
434  parameters.Direction.Closes(existing.GetPositionSide()))
435  {
436  // 2a. Add reserved buying power of current position
437  buyingPower += GetReservedBuyingPowerForPositionGroup(parameters);
438 
439  // 2b. Rebate the initial margin equivalent of current position
440  // this interface doesn't have a concept of initial margin as it's an impl detail of the BuyingPowerModel base class
441  buyingPower += this.GetInitialMarginRequirement(parameters.Portfolio, existing);
442  }
443 
444  return buyingPower;
445  }

476  /// <summary>
477  /// Checks if <paramref name="lastOrderQuantity"/> equals <paramref name="positionGroupQuantity"/> indicating we got the same result on this iteration
478  /// meaning we're unable to converge through iteration. This function was split out to support derived types using the same error message as well
479  /// as removing the added noise of the check and message creation.
480  /// </summary>
481  protected static bool UnableToConverge(decimal lastOrderQuantity, decimal positionGroupQuantity, IPositionGroup groupUnit, SecurityPortfolioManager portfolio, decimal target, decimal orderMargin, decimal absUnitMargin, decimal orderFees, out ArgumentException error)
482  {
483  // determine if we're unable to converge by seeing if quantity estimate hasn't changed
484  if (lastOrderQuantity == positionGroupQuantity)
485  {
486  string message;
487  if (groupUnit.Count == 1)
488  {
489  // single security group
490  var security = portfolio.Securities[groupUnit.Single().Symbol];
491  message = "GetMaximumPositionGroupOrderQuantityForTargetBuyingPower failed to converge to target order margin " +
492  Invariant($"{target}. Current order margin is {orderMargin}. Order quantity {positionGroupQuantity}. ") +
493  Invariant($"Lot size is {security.SymbolProperties.LotSize}. Order fees {orderFees}. Security symbol ") +
494  $"{security.Symbol}. Margin unit {absUnitMargin}.";
495  }
496  else
497  {
498  message = "GetMaximumPositionGroupOrderQuantityForTargetBuyingPower failed to converge to target order margin " +
499  Invariant($"{target}. Current order margin is {orderMargin}. Order quantity {positionGroupQuantity}. ") +
500  Invariant($"Position Group Unit is {groupUnit.Key}. Order fees {orderFees}. Position Group Name ") +
501  $"{groupUnit.GetUserFriendlyName()}. Margin unit {absUnitMargin}.";
502  }
503 
504  error = new ArgumentException(message);
505  return true;
506  }
507 
508  error = null;
509  return false;
510  }
