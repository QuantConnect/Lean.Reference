189  /// <summary>
190  /// Initializes a new instance of the <see cref="PortfolioStatistics"/> class
191  /// </summary>
192  /// <param name="profitLoss">Trade record of profits and losses</param>
193  /// <param name="equity">The list of daily equity values</param>
194  /// <param name="portfolioTurnover">The algorithm portfolio turnover</param>
195  /// <param name="listPerformance">The list of algorithm performance values</param>
196  /// <param name="listBenchmark">The list of benchmark values</param>
197  /// <param name="startingCapital">The algorithm starting capital</param>
198  /// <param name="riskFreeInterestRateModel">The risk free interest rate model to use</param>
199  /// <param name="tradingDaysPerYear">The number of trading days per year</param>
200  /// <param name="winCount">
201  /// The number of wins, including ITM options with profitLoss less than 0.
202  /// If this and <paramref name="lossCount"/> are null, they will be calculated from <paramref name="profitLoss"/>
203  /// </param>
204  /// <param name="lossCount">The number of losses</param>
205  public PortfolioStatistics(
206  SortedDictionary<DateTime, decimal> profitLoss,
207  SortedDictionary<DateTime, decimal> equity,
208  SortedDictionary<DateTime, decimal> portfolioTurnover,
209  List<double> listPerformance,
210  List<double> listBenchmark,
211  decimal startingCapital,
212  IRiskFreeInterestRateModel riskFreeInterestRateModel,
213  int tradingDaysPerYear,
214  int? winCount = null,
215  int? lossCount = null)
216  {
217  StartEquity = startingCapital;
218  EndEquity = equity.LastOrDefault().Value;
219 
220  if (portfolioTurnover.Count > 0)
221  {
222  PortfolioTurnover = portfolioTurnover.Select(kvp => kvp.Value).Average();
223  }
224 
225  if (startingCapital == 0
226  // minimum amount of samples to calculate variance
227  || listBenchmark.Count < 2
228  || listPerformance.Count < 2)
229  {
230  return;
231  }
232 
233  var runningCapital = startingCapital;
234  var totalProfit = 0m;
235  var totalLoss = 0m;
236  var totalWins = 0;
237  var totalLosses = 0;
238  foreach (var pair in profitLoss)
239  {
240  var tradeProfitLoss = pair.Value;
241 
242  if (tradeProfitLoss > 0)
243  {
244  totalProfit += tradeProfitLoss / runningCapital;
245  totalWins++;
246  }
247  else
248  {
249  totalLoss += tradeProfitLoss / runningCapital;
250  totalLosses++;
251  }
252 
253  runningCapital += tradeProfitLoss;
254  }
255 
256  AverageWinRate = totalWins == 0 ? 0 : totalProfit / totalWins;
257  AverageLossRate = totalLosses == 0 ? 0 : totalLoss / totalLosses;
258  ProfitLossRatio = AverageLossRate == 0 ? 0 : AverageWinRate / Math.Abs(AverageLossRate);
259 
260  // Set the actual total wins and losses count.
261  // Some options assignments (ITM) count as wins even though they are losses.
262  if (winCount.HasValue && lossCount.HasValue)
263  {
264  totalWins = winCount.Value;
265  totalLosses = lossCount.Value;
266  }
267 
268  var totalTrades = totalWins + totalLosses;
269  WinRate = totalTrades == 0 ? 0 : (decimal) totalWins / totalTrades;
270  LossRate = totalTrades == 0 ? 0 : (decimal) totalLosses / totalTrades;
271  Expectancy = WinRate * ProfitLossRatio - LossRate;
272 
273  if (startingCapital != 0)
274  {
275  TotalNetProfit = equity.Values.LastOrDefault() / startingCapital - 1;
276  }
277 
278  var fractionOfYears = (decimal) (equity.Keys.LastOrDefault() - equity.Keys.FirstOrDefault()).TotalDays / 365;
279  CompoundingAnnualReturn = Statistics.CompoundingAnnualPerformance(startingCapital, equity.Values.LastOrDefault(), fractionOfYears);
280 
281  Drawdown = DrawdownPercent(equity, 3);
282 
283  AnnualVariance = Statistics.AnnualVariance(listPerformance, tradingDaysPerYear).SafeDecimalCast();
284  AnnualStandardDeviation = (decimal) Math.Sqrt((double) AnnualVariance);
285 
286  var benchmarkAnnualPerformance = GetAnnualPerformance(listBenchmark, tradingDaysPerYear);
287  var annualPerformance = GetAnnualPerformance(listPerformance, tradingDaysPerYear);
288 
289  var riskFreeRate = riskFreeInterestRateModel.GetAverageRiskFreeRate(equity.Select(x => x.Key));
290  SharpeRatio = AnnualStandardDeviation == 0 ? 0 : Statistics.SharpeRatio(annualPerformance, AnnualStandardDeviation, riskFreeRate);
291 
292  var annualDownsideDeviation = Statistics.AnnualDownsideStandardDeviation(listPerformance, tradingDaysPerYear).SafeDecimalCast();
293  SortinoRatio = annualDownsideDeviation == 0 ? 0 : Statistics.SharpeRatio(annualPerformance, annualDownsideDeviation, riskFreeRate);
294 
295  var benchmarkVariance = listBenchmark.Variance();
296  Beta = benchmarkVariance.IsNaNOrZero() ? 0 : (decimal) (listPerformance.Covariance(listBenchmark) / benchmarkVariance);
297 
298  Alpha = Beta == 0 ? 0 : annualPerformance - (riskFreeRate + Beta * (benchmarkAnnualPerformance - riskFreeRate));
299 
300  TrackingError = (decimal)Statistics.TrackingError(listPerformance, listBenchmark, (double)tradingDaysPerYear);
301 
302  InformationRatio = TrackingError == 0 ? 0 : (annualPerformance - benchmarkAnnualPerformance) / TrackingError;
303 
304  TreynorRatio = Beta == 0 ? 0 : (annualPerformance - riskFreeRate) / Beta;
305 
306  // deannualize a 1 sharpe ratio
307  var benchmarkSharpeRatio = 1.0d / Math.Sqrt(tradingDaysPerYear);
308  ProbabilisticSharpeRatio = Statistics.ProbabilisticSharpeRatio(listPerformance, benchmarkSharpeRatio).SafeDecimalCast();
309 
310  ValueAtRisk99 = GetValueAtRisk(listPerformance, tradingDaysPerYear, 0.99d);
311  ValueAtRisk95 = GetValueAtRisk(listPerformance, tradingDaysPerYear, 0.95d);
312  }

321  /// <summary>
322  /// Drawdown maximum percentage.
323  /// </summary>
324  /// <param name="equityOverTime">The list of daily equity values</param>
325  /// <param name="rounding">The number of decimal places to round the result</param>
326  /// <returns>The drawdown percentage</returns>
327  private static decimal DrawdownPercent(SortedDictionary<DateTime, decimal> equityOverTime, int rounding = 2)
328  {
329  var prices = equityOverTime.Values.ToList();
330  if (prices.Count == 0) return 0;
331 
332  var drawdowns = new List<decimal>();
333  var high = prices[0];
334  foreach (var price in prices)
335  {
336  if (price > high) high = price;
337  if (high > 0) drawdowns.Add(price / high - 1);
338  }
339 
340  return Math.Round(Math.Abs(drawdowns.Min()), rounding);
341  }

343  /// <summary>
344  /// Annualized return statistic calculated as an average of daily trading performance multiplied by the number of trading days per year.
345  /// </summary>
346  /// <param name="performance">Dictionary collection of double performance values</param>
347  /// <param name="tradingDaysPerYear">Trading days per year for the assets in portfolio</param>
348  /// <remarks>May be inaccurate for forex algorithms with more trading days in a year</remarks>
349  /// <returns>Double annual performance percentage</returns>
350  private static decimal GetAnnualPerformance(List<double> performance, int tradingDaysPerYear)
351  {
352  return Statistics.AnnualPerformance(performance, tradingDaysPerYear).SafeDecimalCast();
353  }

355  private static decimal GetValueAtRisk(
356  List<double> performance,
357  int lookbackPeriodDays,
358  double confidenceLevel,
359  int rounding = 3)
360  {
361  var periodPerformance = performance.TakeLast(lookbackPeriodDays);
362  var mean = periodPerformance.Mean();
363  var standardDeviation = periodPerformance.StandardDeviation();
364  var valueAtRisk = (decimal)Normal.InvCDF(mean, standardDeviation, 1 - confidenceLevel);
365  return Math.Round(valueAtRisk, rounding);
366  }
