117  /// <summary>
118  /// Add a list of market data points to the local security cache for the current market price.
119  /// </summary>
120  /// <remarks>Internally uses <see cref="AddData"/> using the last data point of the provided list
121  /// and it stores by type the non fill forward points using <see cref="StoreData"/></remarks>
122  public void AddDataList(IReadOnlyList<BaseData> data, Type dataType, bool? containsFillForwardData = null)
123  {
124  var nonFillForwardData = data;
125  // maintaining regression requires us to NOT cache FF data
126  if (containsFillForwardData != false)
127  {
128  var dataFiltered = new List<BaseData>(data.Count);
129  for (var i = 0; i < data.Count; i++)
130  {
131  var dataPoint = data[i];
132  if (!dataPoint.IsFillForward)
133  {
134  dataFiltered.Add(dataPoint);
135  }
136  }
137  nonFillForwardData = dataFiltered;
138  }
139  if (nonFillForwardData.Count != 0)
140  {
141  StoreData(nonFillForwardData, dataType);
142  }
143  else if (dataType == typeof(OpenInterest))
144  {
145  StoreData(data, typeof(OpenInterest));
146  }
147 
148  var last = data[data.Count - 1];
149 
150  ProcessDataPoint(last, cacheByType: false);
151  }
152 
153  /// <summary>
154  /// Add a new market data point to the local security cache for the current market price.
155  /// Rules:
156  /// Don't cache fill forward data.
157  /// Always return the last observation.
158  /// If two consecutive data has the same time stamp and one is Quotebars and the other Tradebar, prioritize the Quotebar.
159  /// </summary>
160  public void AddData(BaseData data)
161  {
162  ProcessDataPoint(data, cacheByType: true);
163  }

165  /// <summary>
166  /// Will consume the given data point updating the cache state and it's properties
167  /// </summary>
168  /// <param name="data">The data point to process</param>
169  /// <param name="cacheByType">True if this data point should be cached by type</param>
170  protected virtual void ProcessDataPoint(BaseData data, bool cacheByType)
171  {
172  var tick = data as Tick;
173  if (tick?.TickType == TickType.OpenInterest)
174  {
175  if (cacheByType)
176  {
177  StoreDataPoint(data);
178  }
179  OpenInterest = (long)tick.Value;
180  return;
181  }
182 
183  // Only cache non fill-forward data and non auxiliary
184  if (data.IsFillForward) return;
185 
186  if (cacheByType)
187  {
188  StoreDataPoint(data);
189  }
190 
191  // we store auxiliary data by type but we don't use it to set 'lastData' nor price information
192  if (data.DataType == MarketDataType.Auxiliary) return;
193 
194  var isDefaultDataType = SubscriptionManager.IsDefaultDataType(data);
195 
196  // don't set _lastData if receive quotebar then tradebar w/ same end time. this
197  // was implemented to grant preference towards using quote data in the fill
198  // models and provide a level of determinism on the values exposed via the cache.
199  if ((_lastData == null
200  || _lastQuoteBarUpdate != data.EndTime
201  || data.DataType != MarketDataType.TradeBar)
202  // we will only set the default data type to preserve determinism and backwards compatibility
203  && isDefaultDataType)
204  {
205  _lastData = data;
206  }
207 
208  if (tick != null)
209  {
210  if (tick.Value != 0) Price = tick.Value;
211 
212  switch (tick.TickType)
213  {
214  case TickType.Trade:
215  if (tick.Quantity != 0) Volume = tick.Quantity;
216  break;
217 
218  case TickType.Quote:
219  if (tick.BidPrice != 0) BidPrice = tick.BidPrice;
220  if (tick.BidSize != 0) BidSize = tick.BidSize;
221 
222  if (tick.AskPrice != 0) AskPrice = tick.AskPrice;
223  if (tick.AskSize != 0) AskSize = tick.AskSize;
224  break;
225  }
226  return;
227  }
228 
229  var bar = data as IBar;
230  if (bar != null)
231  {
232  // we will only set OHLC values using the default data type to preserve determinism and backwards compatibility.
233  // Gives priority to QuoteBar over TradeBar, to be removed when default data type completely addressed GH issue 4196
234  if ((_lastQuoteBarUpdate != data.EndTime || _lastOHLCUpdate != data.EndTime) && isDefaultDataType)
235  {
236  _lastOHLCUpdate = data.EndTime;
237  if (bar.Open != 0) Open = bar.Open;
238  if (bar.High != 0) High = bar.High;
239  if (bar.Low != 0) Low = bar.Low;
240  if (bar.Close != 0)
241  {
242  Price = bar.Close;
243  Close = bar.Close;
244  }
245  }
246 
247  var tradeBar = bar as TradeBar;
248  if (tradeBar != null)
249  {
250  if (tradeBar.Volume != 0) Volume = tradeBar.Volume;
251  }
252 
253  var quoteBar = bar as QuoteBar;
254  if (quoteBar != null)
255  {
256  _lastQuoteBarUpdate = quoteBar.EndTime;
257  if (quoteBar.Ask != null && quoteBar.Ask.Close != 0) AskPrice = quoteBar.Ask.Close;
258  if (quoteBar.Bid != null && quoteBar.Bid.Close != 0) BidPrice = quoteBar.Bid.Close;
259  if (quoteBar.LastBidSize != 0) BidSize = quoteBar.LastBidSize;
260  if (quoteBar.LastAskSize != 0) AskSize = quoteBar.LastAskSize;
261  }
262  }
263  else if (data.DataType != MarketDataType.Auxiliary)
264  {
265  if(data.DataType != MarketDataType.Base || data.Price != 0)
266  {
267  Price = data.Price;
268  }
269  }
270  }

272  /// <summary>
273  /// Stores the specified data list in the cache WITHOUT updating any of the cache properties, such as Price
274  /// </summary>
275  /// <param name="data">The collection of data to store in this cache</param>
276  /// <param name="dataType">The data type</param>
277  public void StoreData(IReadOnlyList<BaseData> data, Type dataType)
278  {
279  if (dataType == typeof(Tick))
280  {
281  var tick = data[data.Count - 1] as Tick;
282  switch (tick?.TickType)
283  {
284  case TickType.Trade:
285  _lastTickTrades = data;
286  return;
287  case TickType.Quote:
288  _lastTickQuotes = data;
289  return;
290  }
291  }
292 
293  lock (_locker)
294  {
295  _dataByType ??= new();
296  _dataByType[dataType] = data;
297  }
298  }

314  public T GetData<T>()

316  {

318  if (!TryGetValue(typeof(T), out list) || list.Count == 0)

322  return list[list.Count - 1] as T;

380  /// <summary>
381  /// Gets whether or not this dynamic data instance has data stored for the specified type
382  /// </summary>
383  public bool TryGetValue(Type type, out IReadOnlyList<BaseData> data)
384  {
385  if (type == typeof(Fundamentals))
386  {
387  // for backwards compatibility
388  type = typeof(FundamentalUniverse);
389  }
390  else if (type == typeof(ETFConstituentData))
391  {
392  // for backwards compatibility
393  type = typeof(ETFConstituentUniverse);
394  }
395  else if (type == typeof(Tick))
396  {
397  var quote = _lastTickQuotes.LastOrDefault();
398  var trade = _lastTickTrades.LastOrDefault();
399  var isQuoteDefaultDataType = quote != null && SubscriptionManager.IsDefaultDataType(quote);
400  var isTradeDefaultDataType = trade != null && SubscriptionManager.IsDefaultDataType(trade);
401 
402  // Currently, IsDefaultDataType returns true for both cases,
403  // So we will return the list with the tick with the most recent timestamp
404  if (isQuoteDefaultDataType && isTradeDefaultDataType)
405  {
406  data = quote.EndTime > trade.EndTime ? _lastTickQuotes : _lastTickTrades;
407  return true;
408  }
409 
410  data = isQuoteDefaultDataType ? _lastTickQuotes : _lastTickTrades;
411  return data?.Count > 0;
412  }
413 
414  data = default;
415  return _dataByType != null && _dataByType.TryGetValue(type, out data);
416  }

418  [MethodImpl(MethodImplOptions.AggressiveInlining)]
419  private void StoreDataPoint(BaseData data)
420  {
421  if (data.GetType() == typeof(Tick))
422  {
423  var tick = data as Tick;
424  switch (tick?.TickType)
425  {
426  case TickType.Trade:
427  _lastTickTrades = new List<BaseData> { tick };
428  break;
429  case TickType.Quote:
430  _lastTickQuotes = new List<BaseData> { tick };
431  break;
432  }
433  }
434  else
435  {
436  lock (_locker)
437  {
438  _dataByType ??= new();
439  // Always keep track of the last observation
440  IReadOnlyList<BaseData> list;
441  if (!_dataByType.TryGetValue(data.GetType(), out list))
442  {
443  list = new List<BaseData> { data };
444  _dataByType[data.GetType()] = list;
445  }
446  else
447  {
448  // we KNOW this one is actually a list, so this is safe
449  // we overwrite the zero entry so we're not constantly newing up lists
450  ((List<BaseData>)list)[0] = data;
451  }
452  }
453  }

456  /// <summary>
457  /// Helper method that modifies the target security cache instance to use the
458  /// type cache of the source
459  /// </summary>
460  /// <remarks>Will set in the source cache any data already present in the target cache</remarks>
461  /// <remarks>This is useful for custom data securities which also have an underlying security,
462  /// will allow both securities to access the same data by type</remarks>
463  /// <param name="sourceToShare">The source cache to use</param>
464  /// <param name="targetToModify">The target security cache that will be modified</param>
465  public static void ShareTypeCacheInstance(SecurityCache sourceToShare, SecurityCache targetToModify)
466  {
467  sourceToShare._dataByType ??= new();
468  if (targetToModify._dataByType != null)
469  {
470  lock (targetToModify._locker)
471  {
472  lock (sourceToShare._locker)
473  {
474  foreach (var kvp in targetToModify._dataByType)
475  {
476  sourceToShare._dataByType.TryAdd(kvp.Key, kvp.Value);
477  }
478  }
479  }
480  }
481  targetToModify._dataByType = sourceToShare._dataByType;
482  targetToModify._lastTickTrades = sourceToShare._lastTickTrades;
483  targetToModify._lastTickQuotes = sourceToShare._lastTickQuotes;
484  }
