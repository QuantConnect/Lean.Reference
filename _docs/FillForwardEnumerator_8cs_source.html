30  /// <summary>
31  /// The FillForwardEnumerator wraps an existing base data enumerator and inserts extra 'base data' instances
32  /// on a specified fill forward resolution
33  /// </summary>

37  private BaseData _previous;
38  private bool _ended;

42  private readonly DateTimeZone _dataTimeZone;

47  private readonly IReadOnlyRef<TimeSpan> _fillForwardResolution;

54  /// <summary>
55  /// Initializes a new instance of the <see cref="FillForwardEnumerator"/> class that accepts
56  /// a reference to the fill forward resolution, useful if the fill forward resolution is dynamic
57  /// and changing as the enumeration progresses
58  /// </summary>
59  /// <param name="enumerator">The source enumerator to be filled forward</param>
60  /// <param name="exchange">The exchange used to determine when to insert fill forward data</param>
61  /// <param name="fillForwardResolution">The resolution we'd like to receive data on</param>
62  /// <param name="isExtendedMarketHours">True to use the exchange's extended market hours, false to use the regular market hours</param>
63  /// <param name="subscriptionEndTime">The end time of the subscrition, once passing this date the enumerator will stop</param>
64  /// <param name="dataResolution">The source enumerator's data resolution</param>
65  /// <param name="dataTimeZone">The time zone of the underlying source data. This is used for rounding calculations and
66  /// is NOT the time zone on the BaseData instances (unless of course data time zone equals the exchange time zone)</param>
67  public FillForwardEnumerator(IEnumerator<BaseData> enumerator,
68  SecurityExchange exchange,
69  IReadOnlyRef<TimeSpan> fillForwardResolution,
70  bool isExtendedMarketHours,
71  DateTime subscriptionEndTime,
72  TimeSpan dataResolution,
73  DateTimeZone dataTimeZone

77  Exchange = exchange;
78  _enumerator = enumerator;
79  _dataResolution = dataResolution;
80  _dataTimeZone = dataTimeZone;
81  _fillForwardResolution = fillForwardResolution;
82  _isExtendedMarketHours = isExtendedMarketHours;
83  // '_dataResolution' and '_subscriptionEndTime' are readonly they won't change, so lets calculate this once here since it's expensive
84  _subscriptionEndTimeRoundDownByDataResolution = RoundDown(_subscriptionEndTime, _dataResolution);
85  }

99  /// <summary>
100  /// Gets the current element in the collection.
101  /// </summary>
102  /// <returns>
103  /// The current element in the collection.
104  /// </returns>
105  /// <filterpriority>2</filterpriority>
106  object IEnumerator.Current => Current;

108  /// <summary>
109  /// Advances the enumerator to the next element of the collection.
110  /// </summary>
111  /// <returns>
112  /// true if the enumerator was successfully advanced to the next element; false if the enumerator has passed the end of the collection.
113  /// </returns>
114  /// <exception cref="T:System.InvalidOperationException">The collection was modified after the enumerator was created. </exception><filterpriority>2</filterpriority>
115  public bool MoveNext()
116  {
117  if (_delistedTime.HasValue)
118  {
119  // don't fill forward after data after the delisted date
120  if (_previous == null || _previous.EndTime >= _delistedTime.Value)
121  {
122  return false;
123  }
124  }
125 
126  if (Current != null && Current.DataType != MarketDataType.Auxiliary)
127  {
128  // only set the _previous if the last item we emitted was NOT auxilliary data,
129  // since _previous is used for fill forward behavior
130  _previous = Current;
131  }
132 
133  BaseData fillForward;
134 
135  if (!_isFillingForward)
136  {
137  // if we're filling forward we don't need to move next since we haven't emitted _enumerator.Current yet
138  if (!_enumerator.MoveNext())
139  {
140  _ended = true;
141  if (_delistedTime.HasValue)
142  {
143  // don't fill forward delisted data
144  return false;
145  }
146 
147  // check to see if we ran out of data before the end of the subscription
148  if (_previous == null || _previous.EndTime >= _subscriptionEndTime)
149  {
150  // we passed the end of subscription, we're finished
151  return false;
152  }
153 
154  // we can fill forward the rest of this subscription if required
155  var endOfSubscription = (Current ?? _previous).Clone(true);
156  endOfSubscription.Time = _subscriptionEndTimeRoundDownByDataResolution;
157  endOfSubscription.EndTime = endOfSubscription.Time + _dataResolution;
158  if (RequiresFillForwardData(_fillForwardResolution.Value, _previous, endOfSubscription, out fillForward))
159  {
160  // don't mark as filling forward so we come back into this block, subscription is done
161  //_isFillingForward = true;
162  Current = fillForward;
163  return true;
164  }
165 
166  // don't emit the last bar if the market isn't considered open!
167  if (!Exchange.IsOpenDuringBar(endOfSubscription.Time, endOfSubscription.EndTime, _isExtendedMarketHours))
168  {
169  return false;
170  }
171 
172  Current = endOfSubscription;
173  return true;
174  }
175  }
176  // If we are filling forward and the underlying is null, let's MoveNext() as long as it didn't end.
177  // This only applies for live trading, so that the LiveFillForwardEnumerator does not stall whenever
178  // we generate a fill-forward bar. The underlying enumerator is advanced so that we don't get stuck
179  // in a cycle of generating infinite fill-forward bars.
180  else if (_enumerator.Current == null && !_ended)
181  {
182  _ended = _enumerator.MoveNext();
183  }
184 
185  var underlyingCurrent = _enumerator.Current;
186  if (underlyingCurrent != null && underlyingCurrent.DataType == MarketDataType.Auxiliary)
187  {
188  var delisting = underlyingCurrent as Delisting;
189  if (delisting?.Type == DelistingType.Delisted)
190  {
191  _delistedTime = delisting.EndTime;
192  }
193  }
194 
195  if (_previous == null)
196  {
197  // first data point we dutifully emit without modification
198  Current = underlyingCurrent;
199  return true;
200  }
201 
202  if (RequiresFillForwardData(_fillForwardResolution.Value, _previous, underlyingCurrent, out fillForward))
203  {
204  if (_previous.EndTime >= _subscriptionEndTime)
205  {
206  // we passed the end of subscription, we're finished
207  return false;
208  }
209  // we require fill forward data because the _enumerator.Current is too far in future
210  _isFillingForward = true;
211  Current = fillForward;
212  return true;
213  }
214 
215  _isFillingForward = false;
216  Current = underlyingCurrent;
217  return true;
218  }

238  /// <summary>
239  /// Determines whether or not fill forward is required, and if true, will produce the new fill forward data
240  /// </summary>
241  /// <param name="fillForwardResolution"></param>
242  /// <param name="previous">The last piece of data emitted by this enumerator</param>
243  /// <param name="next">The next piece of data on the source enumerator</param>
244  /// <param name="fillForward">When this function returns true, this will have a non-null value, null when the function returns false</param>
245  /// <returns>True when a new fill forward piece of data was produced and should be emitted by this enumerator</returns>
246  protected virtual bool RequiresFillForwardData(TimeSpan fillForwardResolution, BaseData previous, BaseData next, out BaseData fillForward)
247  {
248  // convert times to UTC for accurate comparisons and differences across DST changes
249  var previousTimeUtc = previous.Time.ConvertToUtc(Exchange.TimeZone);
250  var nextTimeUtc = next.Time.ConvertToUtc(Exchange.TimeZone);
251  var nextEndTimeUtc = next.EndTime.ConvertToUtc(Exchange.TimeZone);
252 
253  if (nextEndTimeUtc < previousTimeUtc)
254  {
255  Log.Error("FillForwardEnumerator received data out of order. Symbol: " + previous.Symbol.ID);
256  fillForward = null;
257  return false;
258  }
259 
260  // check to see if the gap between previous and next warrants fill forward behavior
261  var nextPreviousTimeUtcDelta = nextTimeUtc - previousTimeUtc;
262  if (nextPreviousTimeUtcDelta <= fillForwardResolution && nextPreviousTimeUtcDelta <= _dataResolution)
263  {
264  fillForward = null;
265  return false;
266  }
267 
268  // every bar emitted MUST be of the data resolution.
269 
270  // compute end times of the four potential fill forward scenarios
271  // 1. the next fill forward bar. 09:00-10:00 followed by 10:00-11:00 where 01:00 is the fill forward resolution
272  // 2. the next data resolution bar, same as above but with the data resolution instead
273  // 3. the next fill forward bar following the next market open, 15:00-16:00 followed by 09:00-10:00 the following open market day
274  // 4. the next data resolution bar following the next market open, same as above but with the data resolution instead
275 
276  // the precedence for validation is based on the order of the end times, obviously if a potential match
277  // is before a later match, the earliest match should win.
278 
279  foreach (var item in GetSortedReferenceDateIntervals(previous, fillForwardResolution, _dataResolution))
280  {
281  // issue GH 4925 , more description https://github.com/QuantConnect/Lean/pull/4941
282  // To build Time/EndTime we always use '+'/'-' dataResolution
283  // DataTime TZ = UTC -5; Exchange TZ = America/New York (-5/-4)
284  // Standard TimeZone 00:00:00 + 1 day = 1.00:00:00
285  // Daylight Time 01:00:00 + 1 day = 1.01:00:00
286 
287  // daylight saving time starts/end at 2 a.m. on Sunday
288  // Having this information we find that the specific bar of Sunday
289  // Starts in one TZ (Standard TZ), but finishes in another (Daylight TZ) (consider winter => summer)
290  // During simple arithmetic operations like +/- we shift the time, but not the time zone
291  // which is sensitive for specific dates (daylight movement) if we are in Exchange TimeZone, for example
292  // We have 00:00:00 + 1 day = 1.00:00:00, so both are in Standard TZ, but we expect endTime in Daylight, i.e. 1.01:00:00
293 
294  // futher down double Convert (Exchange TZ => data TZ => Exchange TZ)
295  // allows us to calculate Time using it's own TZ (aka reapply)
296  // and don't rely on TZ of bar start/end time
297  // i.e. 00:00:00 + 1 day = 1.01:00:00, both start and end are in their own TZ
298  // it's interesting that NodaTime consider next
299  // if time great or equal than 01:00 AM it's considered as "moved" (Standard, not Daylight)
300  // when time less than 01:00 AM it's considered as previous TZ (Standard, not Daylight)
301  // it's easy to fix this behavior by substract 1 tick before first convert, and then return it back.
302  // so we work with 0:59:59.. AM instead.
303  // but now follow native behavior
304 
305  // all above means, that all Time values, calculated using simple +/- operations
306  // sticks to original Time Zone, swallowing its own TZ and movement i.e.
307  // EndTime = Time + resolution, both Time and EndTime in the TZ of Time (Standard/Daylight)
308  // Time = EndTime - resolution, both Time and EndTime in the TZ of EndTime (Standard/Daylight)
309 
310  // next.EndTime sticks to Time TZ,
311  // potentialBarEndTime should be calculated in the same way as bar.EndTime, i.e. Time + resolution
312  var potentialBarEndTime = RoundDown(item.ReferenceDateTime, item.Interval).ConvertToUtc(Exchange.TimeZone) + item.Interval;
313 
314  var period = _dataResolution;
315  if (next.Time == next.EndTime)
316  {
317  // we merge corporate event data points (mapping, delisting, splits, dividend) which do not have
318  // a period or resolution
319  period = TimeSpan.Zero;
320  }
321 
322  // to avoid duality it's necessary to compare potentialBarEndTime with
323  // next.EndTime calculated as Time + resolution,
324  // and both should be based on the same TZ (for example UTC)
325  var nextEndTimeUTC = next.Time.ConvertToUtc(Exchange.TimeZone) + period;
326  if (potentialBarEndTime < nextEndTimeUTC
327  // let's fill forward based on previous (which isn't auxiliary) if next is auxiliary and they share the end time
328  // we do allow emitting both an auxiliary data point and a Filled Forwared data for the same end time
329  || next.DataType == MarketDataType.Auxiliary && potentialBarEndTime == nextEndTimeUTC)
330  {
331  // to check open hours we need to convert potential
332  // bar EndTime into exchange time zone
333  var potentialBarEndTimeInExchangeTZ =
334  potentialBarEndTime.ConvertFromUtc(Exchange.TimeZone);
335  var nextFillForwardBarStartTime = potentialBarEndTimeInExchangeTZ - item.Interval;
336 
337  if (Exchange.IsOpenDuringBar(nextFillForwardBarStartTime, potentialBarEndTimeInExchangeTZ, _isExtendedMarketHours))
338  {
339  fillForward = previous.Clone(true);
340 
341  // bar are ALWAYS of the data resolution
342  fillForward.Time = (potentialBarEndTime - _dataResolution).ConvertFromUtc(Exchange.TimeZone);
343  fillForward.EndTime = potentialBarEndTimeInExchangeTZ;
344 
345  return true;
346  }
347  }
348  else
349  {
350  break;
351  }
352  }
353 
354  // the next is before the next fill forward time, so do nothing
355  fillForward = null;
356  return false;
357  }

359  private IEnumerable<ReferenceDateInterval> GetSortedReferenceDateIntervals(BaseData previous, TimeSpan fillForwardResolution, TimeSpan dataResolution)
360  {
361  if (fillForwardResolution < dataResolution)
362  {
363  return GetReferenceDateIntervals(previous.EndTime, fillForwardResolution, dataResolution);
364  }
365 
366  if (fillForwardResolution > dataResolution)
367  {
368  return GetReferenceDateIntervals(previous.EndTime, dataResolution, fillForwardResolution);
369  }
370 
371  return GetReferenceDateIntervals(previous.EndTime, fillForwardResolution);
372  }

374  /// <summary>
375  /// Get potential next fill forward bars.
376  /// </summary>
377  /// <remarks>Special case where fill forward resolution and data resolution are equal</remarks>
378  private IEnumerable<ReferenceDateInterval> GetReferenceDateIntervals(DateTime previousEndTime, TimeSpan resolution)
379  {
380  if (Exchange.IsOpenDuringBar(previousEndTime, previousEndTime + resolution, _isExtendedMarketHours))
381  {
382  // if next in market us it
383  yield return new ReferenceDateInterval(previousEndTime, resolution);
384  }
385 
386  // now we can try the bar after next market open
387  var marketOpen = Exchange.Hours.GetNextMarketOpen(previousEndTime, _isExtendedMarketHours);
388  yield return new ReferenceDateInterval(marketOpen, resolution);
389  }

391  /// <summary>
392  /// Get potential next fill forward bars.
393  /// </summary>
394  private IEnumerable<ReferenceDateInterval> GetReferenceDateIntervals(DateTime previousEndTime, TimeSpan smallerResolution, TimeSpan largerResolution)
395  {
396  if (Exchange.IsOpenDuringBar(previousEndTime, previousEndTime + smallerResolution, _isExtendedMarketHours))
397  {
398  yield return new ReferenceDateInterval(previousEndTime, smallerResolution);
399  }
400 
401  var result = new List<ReferenceDateInterval>(3);
402  // we need to round down because previous end time could be of the smaller resolution, in data TZ!
403  var start = RoundDown(previousEndTime, largerResolution);
404  if (Exchange.IsOpenDuringBar(start, start + largerResolution, _isExtendedMarketHours))
405  {
406  result.Add(new ReferenceDateInterval(start, largerResolution));
407  }
408 
409  // this is typically daily data being filled forward on a higher resolution
410  // since the previous bar was not in market hours then we can just fast forward
411  // to the next market open
412  var marketOpen = Exchange.Hours.GetNextMarketOpen(previousEndTime, _isExtendedMarketHours);
413  result.Add(new ReferenceDateInterval(marketOpen, smallerResolution));
414  result.Add(new ReferenceDateInterval(marketOpen, largerResolution));
415 
416  // we need to order them because they might not be in an incremental order and consumer expects them to be
417  foreach (var referenceDateInterval in result.OrderBy(interval => interval.ReferenceDateTime + interval.Interval))
418  {
419  yield return referenceDateInterval;
420  }
421  }

423  /// <summary>
424  /// We need to round down in data timezone.
425  /// For example GH issue 4392: Forex daily data, exchange tz time is 8PM, but time in data tz is 12AM
426  /// so rounding down on exchange tz will crop it, while rounding on data tz will return the same data point time.
427  /// Why are we even doing this? being able to determine the next valid data point for a resolution from a data point that might be in another resolution
428  /// </summary>
429  private DateTime RoundDown(DateTime value, TimeSpan interval)
430  {
431  return value.RoundDownInTimeZone(interval, Exchange.TimeZone, _dataTimeZone);
432  }
