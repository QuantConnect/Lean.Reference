123  /// <summary>
124  /// Method returns a collection of Symbols that are available at IQFeed.
125  /// </summary>
126  /// <param name="lookupName">String representing the name to lookup</param>
127  /// <param name="securityType">Expected security type of the returned symbols (if any)</param>
128  /// <param name="includeExpired">Include expired contracts</param>
129  /// <param name="securityCurrency">Expected security currency(if any)</param>
130  /// <param name="securityExchange">Expected security exchange name(if any)</param>
131  /// <returns></returns>
132  public IEnumerable<Symbol> LookupSymbols(string lookupName, SecurityType securityType, bool includeExpired, string securityCurrency = null, string securityExchange = null)
133  {
134  Func<Symbol, string> lookupFunc;
135 
136  switch (securityType)
137  {
138  case SecurityType.Option:
139  // for option, futures contract we search the underlying
140  lookupFunc = symbol => symbol.HasUnderlying ? symbol.Underlying.Value : string.Empty;
141  break;
142  case SecurityType.Future:
143  lookupFunc = symbol => symbol.ID.Symbol;
144  break;
145  default:
146  lookupFunc = symbol => symbol.Value;
147  break;
148  }
149 
150  var result = _symbolUniverse.Where(x => lookupFunc(x.Symbol) == lookupName &&
151  x.Symbol.ID.SecurityType == securityType &&
152  (securityCurrency == null || x.SecurityCurrency == securityCurrency) &&
153  (securityExchange == null || x.SecurityExchange == securityExchange))
154  .ToList();
155 
156  bool onDemandRequests = result.All(symbolData => !symbolData.IsDataLoaded());
157 
158  if (onDemandRequests)
159  {
160  var exchanges = securityType == SecurityType.Future ?
161  FuturesExchanges.Values.Reverse().ToArray() :
162  new string[] { };
163 
164  // sorting list of available contracts by exchange priority, taking the top 1
165  var symbolData =
166  result
167  .OrderByDescending(e => Array.IndexOf(exchanges, e))
168  .First();
169 
170  // we check if the result contains the data that needs to be loaded on demand (e.g. options, futures)
171  var loadedData = LoadSymbolOnDemand(symbolData);
172 
173  // Replace placeholder item in _symbolUniverse with the data loaded on demand
174  UpdateCollectionsOnDemand(symbolData, loadedData);
175 
176  // if we found some data that was loaded on demand, then we have to re-run the query to include that data into method output
177 
178  result = _symbolUniverse.Where(x => lookupFunc(x.Symbol) == lookupName &&
179  x.Symbol.ID.SecurityType == securityType &&
180  (securityCurrency == null || x.SecurityCurrency == securityCurrency) &&
181  (securityExchange == null || x.SecurityExchange == securityExchange))
182  .ToList();
183  }
184 
185  return result.Select(x => x.Symbol);
186  }

242  /// <summary>
243  /// Private method performs initial loading of data from IQFeed universe:
244  /// - method loads FX,equities, indices as is into memory
245  /// - method prepares ondemand data for options and futures and stores it to disk
246  /// - method updates futures mapping files if required
247  /// </summary>
248  /// <returns></returns>
249  private IEnumerable<SymbolData> LoadSymbols()
250  {
251  // default URI
252  const string uri = "http://www.dtniq.com/product/mktsymbols_v2.zip";
253 
254  if (!Directory.Exists(Globals.Cache)) Directory.CreateDirectory(Globals.Cache);
255 
256  // we try to check if we already downloaded the file and it is in cache. If yes, we use it. Otherwise, download new file.
257  IStreamReader reader;
258 
259  // we update the files every week
260  var dayOfWeek = DateTimeFormatInfo.CurrentInfo.Calendar.GetWeekOfYear(DateTime.Today, CalendarWeekRule.FirstDay, DayOfWeek.Monday);
261  var thisYearWeek = $"{DateTime.Today.ToStringInvariant("yyyy")}-{dayOfWeek.ToStringInvariant()}";
262 
263  var todayZipFileName = "IQFeed-symbol-universe-" + thisYearWeek + ".zip";
264  var todayFullZipName = Path.Combine(Globals.Cache, todayZipFileName);
265 
266  var todayCsvFileName = "mktsymbols_v2.txt";
267  var todayFullCsvName = Path.Combine(Globals.Cache, todayCsvFileName);
268 
269  var iqfeedNameMapFileName = "IQFeed-symbol-map.json";
270  var iqfeedNameMapFullName = Path.Combine("IQFeed", iqfeedNameMapFileName);
271 
272  var mapExists = File.Exists(iqfeedNameMapFullName);
273  var universeExists = File.Exists(todayFullZipName);
274 
275  if (mapExists)
276  {
277  Log.Trace("Loading IQFeed futures symbol map file...");
278  _iqFeedNameMap = JsonConvert.DeserializeObject<Dictionary<string, string>>(File.ReadAllText(iqfeedNameMapFullName));
279  }
280 
281  if (!universeExists)
282  {
283  Log.Trace("Loading and unzipping IQFeed symbol universe file ({0})...", uri);
284 
285  using (var client = new WebClient())
286  {
287  client.Proxy = WebRequest.GetSystemWebProxy();
288  client.DownloadFile(uri, todayFullZipName);
289  }
290 
291  Compression.Unzip(todayFullZipName, Globals.Cache, true);
292  }
293  else
294  {
295  Log.Trace("Found up-to-date IQFeed symbol universe file in local cache. Loading it...");
296  }
297 
298  var symbolCache = new Dictionary<Symbol, SymbolData>();
299  var symbolUniverse = new List<SymbolData>();
300 
301  long currentPosition = 0;
302  long prevPosition = 0;
303  long lineCounter = 0;
304  reader = new LocalFileSubscriptionStreamReader(_dataCacheProvider, todayFullCsvName);
305 
306  while (!reader.EndOfStream)
307  {
308  lineCounter++;
309 
310  if (lineCounter % 100000 == 0)
311  {
312  Log.Trace($"{lineCounter} lines read.");
313  }
314 
315  prevPosition = currentPosition;
316 
317  var line = reader.ReadLine();
318 
319  currentPosition += line.Length + NewLine.Length; // file position 'estimator' for ASCII file of IQFeed universe
320 
321  var columns = line.Split(Tabulation);
322 
323  if (columns[columnSymbol] == "TST$Y") continue; // skip test symbol
324 
325  if (columns.Length != totalColumns)
326  {
327  Log.Trace("Discrepancy found while parsing IQFeed symbol universe file. Expected 8 columns, but arrived {0}. Line: {1}", columns.Length, line);
328  continue;
329  }
330 
331  switch (columns[columnSecurityType])
332  {
333  case "INDEX":
334  case "EQUITY":
335 
336  // we load equities/indices in memory
337  symbolUniverse.Add(new SymbolData
338  {
339  Symbol = Symbol.Create(columns[columnSymbol], SecurityType.Equity, Market.USA),
340  SecurityCurrency = Currencies.USD,
341  SecurityExchange = Market.USA,
342  Ticker = columns[columnSymbol]
343  });
344  break;
345 
346  case "IEOPTION":
347 
348  var ticker = columns[columnSymbol];
349  var result = SymbolRepresentation.ParseOptionTickerIQFeed(ticker);
350  var optionUnderlying = result.Underlying;
351  var canonicalSymbol = Symbol.Create(optionUnderlying, SecurityType.Option, Market.USA);
352 
353  if (!symbolCache.ContainsKey(canonicalSymbol))
354  {
355  var placeholderSymbolData = new SymbolData
356  {
357  Symbol = canonicalSymbol,
358  SecurityCurrency = Currencies.USD,
359  SecurityExchange = Market.USA,
360  StartPosition = prevPosition,
361  EndPosition = currentPosition
362  };
363 
364  symbolCache.Add(canonicalSymbol, placeholderSymbolData);
365  }
366  else
367  {
368  symbolCache[canonicalSymbol].EndPosition = currentPosition;
369  }
370 
371  break;
372 
373  case "FOREX":
374 
375  // we use FXCM symbols only
376  if (columns[columnSymbol].EndsWith(".FXCM"))
377  {
378  var symbol = columns[columnSymbol].Replace(".FXCM", string.Empty);
379 
380  symbolUniverse.Add(new SymbolData
381  {
382  Symbol = Symbol.Create(symbol, SecurityType.Forex, Market.FXCM),
383  SecurityCurrency = Currencies.USD,
384  SecurityExchange = Market.FXCM,
385  Ticker = columns[columnSymbol]
386  });
387  }
388  break;
389 
390  case "FUTURE":
391 
392  // we are not interested in designated continuous contracts
393  if (columns[columnSymbol].EndsWith("#") || columns[columnSymbol].EndsWith("#C") || columns[columnSymbol].EndsWith("$$"))
394  continue;
395 
396  var futuresTicker = columns[columnSymbol].TrimStart(new[] { '@' });
397 
398  var parsed = SymbolRepresentation.ParseFutureTicker(futuresTicker);
399  var underlyingString = parsed.Underlying;
400 
401  if (_iqFeedNameMap.ContainsKey(underlyingString))
402  underlyingString = _iqFeedNameMap[underlyingString];
403  else
404  {
405  if (!mapExists)
406  {
407  if (!_iqFeedNameMap.ContainsKey(underlyingString))
408  {
409  // if map is not created yet, we request this information from IQFeed
410  var exchangeSymbol = _symbolFundamentalData.Request(columns[columnSymbol]).Item2;
411  if (!string.IsNullOrEmpty(exchangeSymbol))
412  {
413  _iqFeedNameMap[underlyingString] = exchangeSymbol;
414  underlyingString = exchangeSymbol;
415  }
416  }
417  }
418  }
419 
420  var market = GetFutureMarket(underlyingString, columns[columnExchange]);
421  canonicalSymbol = Symbol.Create(underlyingString, SecurityType.Future, market);
422 
423  if (!symbolCache.ContainsKey(canonicalSymbol))
424  {
425  var placeholderSymbolData = new SymbolData
426  {
427  Symbol = canonicalSymbol,
428  SecurityCurrency = Currencies.USD,
429  SecurityExchange = market,
430  StartPosition = prevPosition,
431  EndPosition = currentPosition
432  };
433 
434  symbolCache.Add(canonicalSymbol, placeholderSymbolData);
435  }
436  else
437  {
438  symbolCache[canonicalSymbol].EndPosition = currentPosition;
439  }
440 
441  break;
442 
443  default:
444 
445  continue;
446  }
447  }
448 
449  if (!mapExists)
450  {
451  Log.Trace("Saving IQFeed futures symbol map file...");
452  File.WriteAllText(iqfeedNameMapFullName, JsonConvert.SerializeObject(_iqFeedNameMap));
453  }
454 
455  symbolUniverse.AddRange(symbolCache.Values);
456 
457  Log.Trace("Finished loading IQFeed symbol universe file.");
458 
459  return symbolUniverse;
460  }

462  /// <summary>
463  /// Private method loads all option or future contracts for a particular underlying
464  /// symbol (placeholder) on demand by walking through the IQFeed universe file
465  /// </summary>
466  /// <param name="placeholder">Underlying symbol</param>
467  /// <returns></returns>
468  private IEnumerable<SymbolData> LoadSymbolOnDemand(SymbolData placeholder)
469  {
470  var dayOfWeek = DateTimeFormatInfo.CurrentInfo.Calendar.GetWeekOfYear(DateTime.Today, CalendarWeekRule.FirstDay, DayOfWeek.Monday);
471  var thisYearWeek = $"{DateTime.Today.ToStringInvariant("yyyy")}-{dayOfWeek.ToStringInvariant()}";
472 
473  var todayCsvFileName = "mktsymbols_v2.txt";
474  var todayFullCsvName = Path.Combine(Globals.Cache, todayCsvFileName);
475 
476  var reader = new LocalFileSubscriptionStreamReader(_dataCacheProvider, todayFullCsvName, placeholder.StartPosition);
477 
478  Log.Trace("Loading data on demand for {0}...", placeholder.Symbol.ID);
479 
480  var symbolUniverse = new List<SymbolData>();
481 
482  long currentPosition = placeholder.StartPosition;
483 
484  while (!reader.EndOfStream && currentPosition <= placeholder.EndPosition)
485  {
486  var line = reader.ReadLine();
487 
488  currentPosition += line.Length + NewLine.Length;
489 
490  var columns = line.Split(Tabulation);
491 
492  if (columns.Length != totalColumns)
493  {
494  continue;
495  }
496 
497  switch (columns[columnSecurityType])
498  {
499  case "IEOPTION":
500 
501  var ticker = columns[columnSymbol];
502  var result = SymbolRepresentation.ParseOptionTickerIQFeed(ticker);
503 
504  symbolUniverse.Add(new SymbolData
505  {
506  Symbol = Symbol.CreateOption(result.Underlying,
507  Market.USA,
508  OptionStyle.American,
509  result.OptionRight,
510  result.OptionStrike,
511  result.ExpirationDate),
512  SecurityCurrency = Currencies.USD,
513  SecurityExchange = Market.USA,
514  Ticker = columns[columnSymbol]
515  });
516 
517  break;
518 
519  case "FUTURE":
520  // we are not interested in designated continuous contracts
521  if ( columns[columnSymbol].EndsWith( "#" ) || columns[columnSymbol].EndsWith( "#C" ) )
522  {
523  continue;
524  }
525 
526  var futuresTicker = columns[columnSymbol].TrimStart(new[] { '@' });
527 
528  var parsed = SymbolRepresentation.ParseFutureTicker(futuresTicker);
529  var underlyingString = parsed.Underlying;
530 
531  if (_iqFeedNameMap.ContainsKey(underlyingString))
532  underlyingString = _iqFeedNameMap[underlyingString];
533 
534  if (underlyingString != placeholder.Symbol.Value.TrimStart('/'))
535  {
536  continue;
537  }
538 
539  var market = GetFutureMarket(underlyingString, columns[columnExchange]);
540  // Futures contracts have different idiosyncratic expiration dates that IQFeed symbol universe file doesn't contain
541  // We request IQFeed explicitly for the exact expiration data of each contract
542 
543  var expirationDate = _symbolFundamentalData.Request(columns[columnSymbol]).Item1;
544 
545  if (expirationDate == DateTime.MinValue)
546  {
547  // contract is outdated
548  continue;
549  }
550 
551  symbolUniverse.Add(new SymbolData
552  {
553  Symbol = Symbol.CreateFuture(underlyingString,
554  market,
555  expirationDate),
556  SecurityCurrency = Currencies.USD,
557  SecurityExchange = market,
558  Ticker = columns[columnSymbol]
559  });
560 
561  break;
562 
563  default:
564 
565  continue;
566  }
567  }
568 
569  return symbolUniverse;
570  }

645  /// <summary>
646  /// Private class that helps requesting IQFeed fundamental data
647  /// </summary>
648  public class SymbolFundamentalData : IQLevel1Client
649  {
650  public SymbolFundamentalData() : base(IQFeedDefault.BufferSize)
651  {
652  }
653 
654  /// <summary>
655  /// Method returns two fields of the fundamental data that we need: expiration date (tuple field 1),
656  /// and exchange root symbol (tuple field 2)
657  /// </summary>
658  public Tuple<DateTime, string> Request(string ticker)
659  {
660  const int timeout = 180; // sec
661  var manualResetEvent = new ManualResetEvent(false);
662 
663  var expiry = DateTime.MinValue;
664  var rootSymbol = string.Empty;
665 
666  EventHandler<Level1FundamentalEventArgs> dataEventHandler = (sender, e) =>
667  {
668  if (e.Symbol == ticker)
669  {
670  expiry = e.ExpirationDate;
671  rootSymbol = e.ExchangeRoot;
672 
673  manualResetEvent.Set();
674  }
675  };
676  EventHandler<Level1SummaryUpdateEventArgs> noDataEventHandler = (sender, e) =>
677  {
678  if (e.Symbol == ticker && e.NotFound)
679  {
680  manualResetEvent.Set();
681  }
682  };
683 
684  Level1FundamentalEvent += dataEventHandler;
685  Level1SummaryUpdateEvent += noDataEventHandler;
686 
687  Subscribe(ticker);
688 
689  if (!manualResetEvent.WaitOne(timeout * 1000))
690  {
691  Log.Error("SymbolFundamentalData.Request() failed to receive response from IQFeed within {0} seconds", timeout);
692  }
693 
694  Unsubscribe(ticker);
695 
696  Level1SummaryUpdateEvent -= noDataEventHandler;
697 
698  Level1FundamentalEvent -= dataEventHandler;
699 
700  return Tuple.Create(expiry, rootSymbol);
701  }
702  }
