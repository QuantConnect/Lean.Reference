157  /// <summary>
158  /// Class for generating a SKU for a node with a given configuration
159  /// Every SKU is made up of 3 variables:
160  /// - Target environment (L for live, B for Backtest, R for Research)
161  /// - CPU core count
162  /// - Dedicated RAM (GB)
163  /// </summary>
164  public class SKU
165  {
166  /// <summary>
167  /// The number of CPU cores in the node
168  /// </summary>
169  public int Cores { get; set; }
170 
171  /// <summary>
172  /// Size of RAM in GB of the Node
173  /// </summary>
174  public int Memory { get; set; }
175 
176  /// <summary>
177  /// Target environment for the node
178  /// </summary>
179  public NodeType Target { get; set; }
180 
181  /// <summary>
182  /// Constructs a SKU object out of the provided node configuration
183  /// </summary>
184  /// <param name="cores">Number of cores</param>
185  /// <param name="memory">Size of RAM in GBs</param>
186  /// <param name="target">Target Environment Live/Backtest/Research</param>
187  public SKU(int cores, int memory, NodeType target)
188  {
189  Cores = cores;
190  Memory = memory;
191  Target = target;
192  }
193 
194  /// <summary>
195  /// Generates the SKU string for API calls based on the specifications of the node
196  /// </summary>
197  /// <returns>String representation of the SKU</returns>
198  public override string ToString()
199  {
200  string result = "";
201 
202  switch (Target)
203  {
204  case NodeType.Backtest:
205  result += "B";
206  break;
207  case NodeType.Research:
208  result += "R";
209  break;
210  case NodeType.Live:
211  result += "L";
212  break;
213  }
214 
215  if (Cores == 0)
216  {
217  result += "-MICRO";
218  }
219  else
220  {
221  result += Cores + "-" + Memory;
222  }
223 
224  return result;
225  }
226  }
