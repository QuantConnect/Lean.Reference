23  /// <summary>
24  /// Provides a listing of pre-defined <see cref="OptionStrategyDefinition"/>
25  /// These definitions are blueprints for <see cref="OptionStrategy"/> instances.
26  /// Factory functions for those can be found at <see cref="OptionStrategies"/>

30  // lazy since 'AllDefinitions' is at top of file and static members are evaluated in order
31  private static readonly Lazy<ImmutableList<OptionStrategyDefinition>> All
32  = new Lazy<ImmutableList<OptionStrategyDefinition>>(() =>
33  typeof(OptionStrategyDefinitions)
34  .GetProperties(BindingFlags.Public | BindingFlags.Static)
35  .Where(property => property.PropertyType == typeof(OptionStrategyDefinition))
36  .Select(property => (OptionStrategyDefinition) property.GetValue(null))
37  .ToImmutableList()
38  );

61  /// <summary>
62  /// Bear Call Spread strategy consists of two calls with the same expiration but different strikes.
63  /// The strike price of the short call is below the strike of the long call. This is a credit spread.
64  /// </summary>
65  public static OptionStrategyDefinition BearCallSpread { get; }
66  = OptionStrategyDefinition.Create("Bear Call Spread",
67  OptionStrategyDefinition.CallLeg(-1),
68  OptionStrategyDefinition.CallLeg(+1, (legs, p) => p.Strike > legs[0].Strike,

128  /// <summary>
129  /// Short Butterfly Call strategy consists of two short calls at a middle strike, and one long call each at a lower
130  /// and upper strike. The upper and lower strikes must both be equidistant from the middle strike.
131  /// </summary>
132  public static OptionStrategyDefinition ButterflyCall { get; }
133  = OptionStrategyDefinition.Create("Butterfly Call",
134  OptionStrategyDefinition.CallLeg(+1),
135  OptionStrategyDefinition.CallLeg(-2, (legs, p) => p.Strike >= legs[0].Strike,
136  (legs, p) => p.Expiration == legs[0].Expiration),
137  OptionStrategyDefinition.CallLeg(+1, (legs, p) => p.Strike >= legs[1].Strike,
138  (legs, p) => p.Expiration == legs[0].Expiration,
139  (legs, p) => p.Strike - legs[1].Strike == legs[1].Strike - legs[0].Strike)
140  );

213  OptionStrategyDefinition.PutLeg(+1),

216  OptionStrategyDefinition.CallLeg(-1, (legs, p) => p.Strike > legs[1].Strike,
217  (legs, p) => p.Expiration == legs[0].Expiration),
