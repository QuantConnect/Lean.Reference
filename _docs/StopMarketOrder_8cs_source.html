16 using System;
17 using QuantConnect.Interfaces;
18 using QuantConnect.Securities;
19 
20 namespace QuantConnect.Orders
21 {
22  /// <summary>
23  /// Stop Market Order Type Definition
24  /// </summary>
25  public class StopMarketOrder : Order
26  {
27  /// <summary>
28  /// Stop price for this stop market order.
29  /// </summary>
30  public decimal StopPrice { get; internal set; }
31 
32  /// <summary>
33  /// StopMarket Order Type
34  /// </summary>
35  public override OrderType Type
36  {
37  get { return OrderType.StopMarket; }
38  }
39 
40  /// <summary>
41  /// Default constructor for JSON Deserialization:
42  /// </summary>
43  public StopMarketOrder()
44  {
45  }
46 
47  /// <summary>
48  /// New Stop Market Order constructor -
49  /// </summary>
50  /// <param name="symbol">Symbol asset we're seeking to trade</param>
51  /// <param name="quantity">Quantity of the asset we're seeking to trade</param>
52  /// <param name="time">Time the order was placed</param>
53  /// <param name="stopPrice">Price the order should be filled at if a limit order</param>
54  /// <param name="tag">User defined data tag for this order</param>
55  /// <param name="properties">The order properties for this order</param>
56  public StopMarketOrder(Symbol symbol, decimal quantity, decimal stopPrice, DateTime time, string tag = "", IOrderProperties properties = null)
57  : base(symbol, quantity, time, tag, properties)
58  {
59  StopPrice = stopPrice;
60  }
61 
62  /// <summary>
63  /// Gets the default tag for this order
64  /// </summary>
65  /// <returns>The default tag</returns>
66  public override string GetDefaultTag()
67  {
68  return Messages.StopMarketOrder.Tag(this);
69  }
70 
71  /// <summary>
72  /// Gets the order value in units of the security's quote currency
73  /// </summary>
74  /// <param name="security">The security matching this order's symbol</param>
75  protected override decimal GetValueImpl(Security security)
76  {
77  // selling, so higher price will be used
78  if (Quantity < 0)
79  {
80  return Quantity * Math.Max(StopPrice, security.Price);
81  }
82 
83  // buying, so lower price will be used
84  if (Quantity > 0)
85  {
86  return Quantity * Math.Min(StopPrice, security.Price);
87  }
88 
89  return 0m;
90  }
91 
92  /// <summary>
93  /// Modifies the state of this order to match the update request
94  /// </summary>
95  /// <param name="request">The request to update this order object</param>
96  public override void ApplyUpdateOrderRequest(UpdateOrderRequest request)
97  {
98  base.ApplyUpdateOrderRequest(request);
99  if (request.StopPrice.HasValue)
100  {
101  StopPrice = request.StopPrice.Value;
102  }
103  }
104 
105  /// <summary>
106  /// Returns a string that represents the current object.
107  /// </summary>
108  /// <returns>
109  /// A string that represents the current object.
110  /// </returns>
111  /// <filterpriority>2</filterpriority>
112  public override string ToString()
113  {
114  return Messages.StopMarketOrder.ToString(this);
115  }
116 
117  /// <summary>
118  /// Creates a deep-copy clone of this order
119  /// </summary>
120  /// <returns>A copy of this order</returns>
121  public override Order Clone()
122  {
123  var order = new StopMarketOrder { StopPrice = StopPrice };
124  CopyTo(order);
125  return order;
126  }
127  }
128 }
