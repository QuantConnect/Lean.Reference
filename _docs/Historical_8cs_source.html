22  /// <summary>
23  /// Class for fetching stock historical price from Yahoo Finance
24  /// </summary>
25  public class Historical

27  /// <summary>
28  /// Get stock historical price from Yahoo Finance
29  /// </summary>
30  /// <param name="symbol">Stock ticker symbol</param>
31  /// <param name="start">Starting datetime</param>
32  /// <param name="end">Ending datetime</param>
33  /// <returns>List of history price</returns>
34  public static List<HistoryPrice> Get(string symbol, DateTime start, DateTime end, string eventCode)
35  {
36  var historyPrices = new List<HistoryPrice>();
37 
38  try
39  {
40  var csvData = GetRaw(symbol, start, end, eventCode);
41  if (csvData != null)
42  {
43  historyPrices = Parse(csvData);
44  }
45  }
46  catch (Exception ex)
47  {
48  Debug.Print(ex.Message);
49  }
50 
51  return historyPrices;
52 
53  }

55  /// <summary>
56  /// Get raw stock historical price from Yahoo Finance
57  /// </summary>
58  /// <param name="symbol">Stock ticker symbol</param>
59  /// <param name="start">Starting datetime</param>
60  /// <param name="end">Ending datetime</param>
61  /// <returns>Raw history price string</returns>
62  public static string GetRaw(string symbol, DateTime start, DateTime end, string eventCode)
63  {
64 
65  string csvData = null;
66 
67  try
68  {
69  //if no token found, refresh it
70  if (string.IsNullOrEmpty(Token.Cookie) | string.IsNullOrEmpty(Token.Crumb))
71  {
72  if (!Token.Refresh(symbol))
73  {
74  return GetRaw(symbol, start, end, eventCode);
75  }
76  }
77 
78  var url = $"https://query1.finance.yahoo.com/v7/finance/download/{symbol}" +
79  $"?period1={Math.Round(Time.DateTimeToUnixTimeStamp(start), 0).ToStringInvariant()}" +
80  $"&period2={Math.Round(Time.DateTimeToUnixTimeStamp(end), 0).ToStringInvariant()}" +
81  $"&interval=1d&events={eventCode}&crumb={Token.Crumb}";
82 
83  using (var wc = new WebClient())
84  {
85  wc.Headers.Add(HttpRequestHeader.Cookie, Token.Cookie);
86  csvData = wc.DownloadString(url);
87  }
88 
89  }
90  catch (WebException webEx)
91  {
92  var response = (HttpWebResponse)webEx.Response;
93 
94  //Re-fecthing token
95  if (response.StatusCode == HttpStatusCode.Unauthorized)
96  {
97  Debug.Print(webEx.Message);
98  Token.Reset();
99  Debug.Print("Re-fetch");
100  return GetRaw(symbol, start, end, eventCode);
101  }
102  throw;
103 
104  }
105  catch (Exception ex)
106  {
107  Debug.Print(ex.Message);
108  }
109 
110  return csvData;
111 
112  }

114  /// <summary>
115  /// Parse raw historical price data into list
116  /// </summary>
117  /// <param name="csvData"></param>
118  /// <returns></returns>
119  private static List<HistoryPrice> Parse(string csvData)
120  {
121 
122  var hps = new List<HistoryPrice>();
123 
124  try
125  {
126  var rows = csvData.Split(Convert.ToChar(10));
127 
128  //row(0) was ignored because is column names
129  //data is read from oldest to latest
130  for (var i = 1; i <= rows.Length - 1; i++)
131  {
132 
133  var row = rows[i];
134  if (string.IsNullOrEmpty(row))
135  {
136  continue;
137  }
138 
139  var cols = row.Split(',');
140  if (cols[1] == "null")
141  {
142  continue;
143  }
144 
145  var hp = new HistoryPrice
146  {
147  Date = QuantConnect.Parse.DateTime(cols[0]),
148  Open = QuantConnect.Parse.Decimal(cols[1]),
149  High = QuantConnect.Parse.Decimal(cols[2]),
150  Low = QuantConnect.Parse.Decimal(cols[3]),
151  Close = QuantConnect.Parse.Decimal(cols[4]),
152  AdjClose = QuantConnect.Parse.Decimal(cols[5])
153  };
154 
155  //fixed issue in some currencies quote (e.g: SGDAUD=X)
156  if (cols[6] != "null")
157  {
158  hp.Volume = QuantConnect.Parse.Decimal(cols[6]);
159  }
160 
161  hps.Add(hp);
162 
163  }
164 
165  }
166  catch (Exception ex)
167  {
168  Debug.Print(ex.Message);
169  }
170 
171  return hps;
172 
173  }
174 
175  }
