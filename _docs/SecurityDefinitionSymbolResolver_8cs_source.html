29  /// <summary>
30  /// Resolves standardized security definitions such as FIGI, CUSIP, ISIN, SEDOL into
31  /// a properly mapped Lean <see cref="Symbol"/>.
32  /// </summary>
33  public class SecurityDefinitionSymbolResolver
34  {
35  private List<SecurityDefinition> _securityDefinitions;
36  private readonly IMapFileProvider _mapFileProvider;
37  private readonly string _securitiesDefinitionKey;
38  private readonly IDataProvider _dataProvider;

40  /// <summary>
41  /// Creates an instance of the symbol resolver
42  /// </summary>
43  /// <param name="dataProvider">Data provider used to obtain symbol mappings data</param>
44  /// <param name="securitiesDefinitionKey">Location to read the securities definition data from</param>
45  public SecurityDefinitionSymbolResolver(IDataProvider dataProvider = null, string securitiesDefinitionKey = null)
46  {
47  _securitiesDefinitionKey = securitiesDefinitionKey ?? Path.Combine(Globals.DataFolder, "symbol-properties", "security-database.csv");
48 
49  _dataProvider = dataProvider ??
50  Composer.Instance.GetExportedValueByTypeName<IDataProvider>(
51  Config.Get("data-provider", "QuantConnect.Lean.Engine.DataFeeds.DefaultDataProvider"));
52 
53  _mapFileProvider = Composer.Instance.GetExportedValueByTypeName<IMapFileProvider>(Config.Get("map-file-provider", "LocalDiskMapFileProvider"));
54  _mapFileProvider.Initialize(_dataProvider);
55  }

57  /// <summary>
58  /// Converts CUSIP into a Lean <see cref="Symbol"/>
59  /// </summary>
60  /// <param name="cusip">
61  /// The Committee on Uniform Securities Identification Procedures (CUSIP) number of a security
62  /// </param>
63  /// <param name="tradingDate">
64  /// The date that the stock was trading at with the CUSIP provided. This is used
65  /// to get the ticker of the symbol on this date.
66  /// </param>
67  /// <returns>The Lean Symbol corresponding to the CUSIP number on the trading date provided</returns>
68  public Symbol CUSIP(string cusip, DateTime tradingDate)
69  {
70  if (string.IsNullOrWhiteSpace(cusip))
71  {
72  return null;
73  }
74 
75  return SecurityDefinitionToSymbol(
76  GetSecurityDefinitions().FirstOrDefault(x => x.CUSIP != null && x.CUSIP.Equals(cusip, StringComparison.InvariantCultureIgnoreCase)),
77  tradingDate);
78  }

149  /// <summary>
150  /// Converts a SecurityDefinition to a <see cref="Symbol" />
151  /// </summary>
152  /// <param name="securityDefinition">Security definition</param>
153  /// <param name="tradingDate">
154  /// The date that the stock was being traded. This is used to resolve
155  /// the ticker that the stock was trading under on this date.
156  /// </param>
157  /// <returns>Symbol if matching Lean Symbol was found on the trading date, null otherwise</returns>
158  private Symbol SecurityDefinitionToSymbol(SecurityDefinition securityDefinition, DateTime tradingDate)
159  {
160  if (securityDefinition == null)
161  {
162  return null;
163  }
164 
165  var mapFileResolver = _mapFileProvider.Get(AuxiliaryDataKey.Create(securityDefinition.SecurityIdentifier));
166 
167  // Get the first ticker the symbol traded under, and then lookup the
168  // trading date to get the ticker on the trading date.
169  var mapFile = mapFileResolver
170  .ResolveMapFile(securityDefinition.SecurityIdentifier.Symbol, securityDefinition.SecurityIdentifier.Date);
171 
172  // The mapped ticker will be null if the map file is null or there's
173  // no entry found for the given trading date.
174  var mappedTicker = mapFile?.GetMappedSymbol(tradingDate, null);
175 
176  // If we're null, then try again; get the last entry of the map file and use
177  // it as the Symbol we return to the caller.
178  mappedTicker ??= mapFile?
179  .LastOrDefault()?
180  .MappedSymbol;
181 
182  return string.IsNullOrWhiteSpace(mappedTicker)
183  ? null
184  : new Symbol(securityDefinition.SecurityIdentifier, mappedTicker);
185  }
186 
187  /// <summary>
188  /// Get's the security definitions using a lazy initialization
189  /// </summary>
190  private IEnumerable<SecurityDefinition> GetSecurityDefinitions()
191  {
192  if (_securityDefinitions != null)
193  {
194  return _securityDefinitions;
195  }
196 
197  if (!SecurityDefinition.TryRead(_dataProvider, _securitiesDefinitionKey, out _securityDefinitions))
198  {
199  _securityDefinitions = new List<SecurityDefinition>();
200  Log.Error($"SecurityDefinitionSymbolResolver(): No security definitions data loaded from file: {_securitiesDefinitionKey}");
201  }
202  return _securityDefinitions;
203  }
204  }
