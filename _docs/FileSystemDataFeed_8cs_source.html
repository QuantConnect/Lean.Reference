100  private IEnumerator<BaseData> CreateDataEnumerator(SubscriptionRequest request, Resolution? fillForwardResolution)
101  {
102  // ReSharper disable once PossibleMultipleEnumeration
103  if (!request.TradableDaysInDataTimeZone.Any())
104  {
105  _algorithm.Error(
106  $"No data loaded for {request.Security.Symbol} because there were no tradeable dates for this security."
107  );
108  return null;
109  }
110 
111  // ReSharper disable once PossibleMultipleEnumeration
112  var enumerator = _subscriptionFactory.CreateEnumerator(request, _dataProvider);
113  enumerator = ConfigureEnumerator(request, false, enumerator, fillForwardResolution);
114 
115  return enumerator;
116  }

118  /// <summary>
119  /// Creates a new subscription to provide data for the specified security.
120  /// </summary>
121  /// <param name="request">Defines the subscription to be added, including start/end times the universe and security</param>
122  /// <returns>The created <see cref="Subscription"/> if successful, null otherwise</returns>
123  public virtual Subscription CreateSubscription(SubscriptionRequest request)
124  {
125  IEnumerator<BaseData> enumerator;
126  if(_algorithm.IsWarmingUp)
127  {
128  var pivotTimeUtc = _algorithm.StartDate.ConvertToUtc(_algorithm.TimeZone);
129 
130  var warmupRequest = new SubscriptionRequest(request, endTimeUtc: pivotTimeUtc,
131  configuration: new SubscriptionDataConfig(request.Configuration, resolution: _algorithm.Settings.WarmupResolution));
132  IEnumerator<BaseData> warmupEnumerator = null;
133  if (warmupRequest.TradableDaysInDataTimeZone.Any()
134  // since we change the resolution, let's validate it's still valid configuration (example daily equity quotes are not!)
135  && LeanData.IsValidConfiguration(warmupRequest.Configuration.SecurityType, warmupRequest.Configuration.Resolution, warmupRequest.Configuration.TickType))
136  {
137  // let them overlap a day if possible to avoid data gaps since each request will FFed it's own since they are different resolutions
138  pivotTimeUtc = Time.GetStartTimeForTradeBars(request.Security.Exchange.Hours,
139  _algorithm.StartDate.ConvertTo(_algorithm.TimeZone, request.Security.Exchange.TimeZone),
140  Time.OneDay,
141  1,
142  false,
143  warmupRequest.Configuration.DataTimeZone)
144  .ConvertToUtc(request.Security.Exchange.TimeZone);
145  if (pivotTimeUtc < warmupRequest.StartTimeUtc)
146  {
147  pivotTimeUtc = warmupRequest.StartTimeUtc;
148  }
149 
150  warmupEnumerator = CreateEnumerator(warmupRequest, _algorithm.Settings.WarmupResolution);
151  // don't let future data past
152  warmupEnumerator = new FilterEnumerator<BaseData>(warmupEnumerator, data => data == null || data.EndTime <= warmupRequest.EndTimeLocal);
153  }
154 
155  var normalEnumerator = CreateEnumerator(new SubscriptionRequest(request, startTimeUtc: pivotTimeUtc));
156  // don't let pre start data pass, since we adjust start so they overlap 1 day let's not let this data pass, we just want it for fill forwarding after the target start
157  // this is also useful to drop any initial selection point which was already emitted during warmup
158  normalEnumerator = new FilterEnumerator<BaseData>(normalEnumerator, data => data == null || data.EndTime >= warmupRequest.EndTimeLocal);
159 
160  // after the warmup enumerator we concatenate the 'normal' one
161  enumerator = new ConcatEnumerator(true, warmupEnumerator, normalEnumerator);
162  }
163  else
164  {
165  enumerator = CreateEnumerator(request);
166  }
167 
168  if (request.IsUniverseSubscription && request.Universe is UserDefinedUniverse)
169  {
170  // for user defined universe we do not use a worker task, since calls to AddData can happen in any moment
171  // and we have to be able to inject selection data points into the enumerator
172  return SubscriptionUtils.Create(request, enumerator);
173  }
174  return SubscriptionUtils.CreateAndScheduleWorker(request, enumerator, _factorFileProvider, true);
175  }

185  protected IEnumerator<BaseData> CreateUniverseEnumerator(SubscriptionRequest request, Func<SubscriptionRequest, Resolution?, IEnumerator<BaseData>> createUnderlyingEnumerator, Resolution? fillForwardResolution = null)
186  {
187  ISubscriptionEnumeratorFactory factory = _subscriptionFactory;
188  if (request.Universe is ITimeTriggeredUniverse)
189  {
190  factory = new TimeTriggeredUniverseSubscriptionEnumeratorFactory(request.Universe as ITimeTriggeredUniverse,
191  _marketHoursDatabase,
192  _timeProvider);
193 
194  if (request.Universe is UserDefinedUniverse)
195  {
196  return factory.CreateEnumerator(request, _dataProvider);
197  }
198  }
199  else if (request.Configuration.Type == typeof(FundamentalUniverse))
200  {
201  factory = new BaseDataCollectionSubscriptionEnumeratorFactory(_algorithm.ObjectStore);
202  }
203  else if (request.Configuration.Type == typeof(ZipEntryName))
204  {
205  // TODO: subscription should already come in correctly built
206  var resolution = request.Configuration.Resolution == Resolution.Tick ? Resolution.Second : request.Configuration.Resolution;
207 
208  // TODO: subscription should already come in as fill forward true
209  request = new SubscriptionRequest(request, configuration: new SubscriptionDataConfig(request.Configuration, fillForward: true, resolution: resolution));
210 
211  var result = new BaseDataSubscriptionEnumeratorFactory(_algorithm.OptionChainProvider, _algorithm.FutureChainProvider)
212  .CreateEnumerator(request, _dataProvider);
213  result = ConfigureEnumerator(request, true, result, fillForwardResolution);
214  return TryAppendUnderlyingEnumerator(request, result, createUnderlyingEnumerator, fillForwardResolution);
215  }
216 
217  // define our data enumerator
218  var enumerator = factory.CreateEnumerator(request, _dataProvider);
219  return enumerator;
220  }

222  /// <summary>
223  /// If required will add a new enumerator for the underlying symbol
224  /// </summary>
225  protected IEnumerator<BaseData> TryAppendUnderlyingEnumerator(SubscriptionRequest request, IEnumerator<BaseData> parent, Func<SubscriptionRequest, Resolution?, IEnumerator<BaseData>> createEnumerator, Resolution? fillForwardResolution)
226  {
227  if (request.Configuration.Symbol.SecurityType.IsOption() && request.Configuration.Symbol.HasUnderlying)
228  {
229  var underlyingSymbol = request.Configuration.Symbol.Underlying;
230  var underlyingMarketHours = _marketHoursDatabase.GetEntry(underlyingSymbol.ID.Market, underlyingSymbol, underlyingSymbol.SecurityType);
231 
232  // TODO: creating this subscription request/config is bad
233  var underlyingRequests = new SubscriptionRequest(request,
234  isUniverseSubscription: false,
235  configuration: new SubscriptionDataConfig(request.Configuration, symbol: underlyingSymbol, objectType: typeof(TradeBar), tickType: TickType.Trade,
236  // there's no guarantee the TZ are the same, specially the data timezone (index & index options)
237  dataTimeZone: underlyingMarketHours.DataTimeZone,
238  exchangeTimeZone: underlyingMarketHours.ExchangeHours.TimeZone));
239 
240  var underlying = createEnumerator(underlyingRequests, fillForwardResolution);
241  underlying = new FilterEnumerator<BaseData>(underlying, data => data.DataType != MarketDataType.Auxiliary);
242 
243  parent = new SynchronizingBaseDataEnumerator(parent, underlying);
244  // we aggregate both underlying and chain data
245  parent = new BaseDataCollectionAggregatorEnumerator(parent, request.Configuration.Symbol);
246  // only let through if underlying and chain data present
247  parent = new FilterEnumerator<BaseData>(parent, data => (data as BaseDataCollection).Underlying != null);
248  parent = ConfigureEnumerator(request, false, parent, fillForwardResolution);
249  }
250 
251  return parent;
252  }

269  /// <summary>
270  /// Configure the enumerator with aggregation/fill-forward/filter behaviors. Returns new instance if re-configured
271  /// </summary>
272  protected IEnumerator<BaseData> ConfigureEnumerator(SubscriptionRequest request, bool aggregate, IEnumerator<BaseData> enumerator, Resolution? fillForwardResolution)
273  {
274  if (aggregate)
275  {
276  enumerator = new BaseDataCollectionAggregatorEnumerator(enumerator, request.Configuration.Symbol);
277  }
278 
279  enumerator = TryAddFillForwardEnumerator(request, enumerator, request.Configuration.FillDataForward, fillForwardResolution);
280 
281  // optionally apply exchange/user filters
282  if (request.Configuration.IsFilteredSubscription)
283  {
284  enumerator = SubscriptionFilterEnumerator.WrapForDataFeed(_resultHandler, enumerator, request.Security,
285  request.EndTimeLocal, request.Configuration.ExtendedMarketHours, false, request.ExchangeHours);
286  }
287 
288  return enumerator;
289  }
290 
291  /// <summary>
292  /// Will add a fill forward enumerator if requested
293  /// </summary>
294  protected IEnumerator<BaseData> TryAddFillForwardEnumerator(SubscriptionRequest request, IEnumerator<BaseData> enumerator, bool fillForward, Resolution? fillForwardResolution)
295  {
296  // optionally apply fill forward logic, but never for tick data
297  if (fillForward && request.Configuration.Resolution != Resolution.Tick)
298  {
299  // copy forward Bid/Ask bars for QuoteBars
300  if (request.Configuration.Type == typeof(QuoteBar))
301  {
302  enumerator = new QuoteBarFillForwardEnumerator(enumerator);
303  }
304 
305  var fillForwardSpan = _subscriptions.UpdateAndGetFillForwardResolution(request.Configuration);
306  if (fillForwardResolution != null && fillForwardResolution != Resolution.Tick)
307  {
308  // if we are giving a FFspan we use it instead of the collection based one. This is useful during warmup when the warmup resolution has been set
309  fillForwardSpan = Ref.Create(fillForwardResolution.Value.ToTimeSpan());
310  }
311 
312  enumerator = new FillForwardEnumerator(enumerator, request.Security.Exchange, fillForwardSpan,
313  request.Configuration.ExtendedMarketHours, request.EndTimeLocal, request.Configuration.Resolution.ToTimeSpan(), request.Configuration.DataTimeZone);
314  }
315 
316  return enumerator;
317  }
