201  /// <summary>
202  /// Python implementation of GetFundamental, get fundamental data for input symbols or tickers
203  /// </summary>
204  /// <param name="input">The symbols or tickers to retrieve fundamental data for</param>
205  /// <param name="selector">Selects a value from the Fundamental data to filter the request output</param>
206  /// <param name="start">The start date of selected data</param>
207  /// <param name="end">The end date of selected data</param>
208  /// <returns>pandas DataFrame</returns>
209  public PyObject GetFundamental(PyObject input, string selector = null, DateTime? start = null, DateTime? end = null)
210  {
211  //Covert to symbols
212  var symbols = PythonUtil.ConvertToSymbols(input);
213 
214  //Fetch the data
215  var fundamentalData = GetAllFundamental(symbols, selector, start, end);
216 
217  using (Py.GIL())
218  {
219  var data = new PyDict();
220  foreach (var day in fundamentalData.OrderBy(x => x.Key))
221  {
222  var orderedValues = day.Value.OrderBy(x => x.Key.ID.ToString()).ToList();
223  var columns = orderedValues.Select(x => x.Key.ID.ToString());
224  var values = orderedValues.Select(x => x.Value);
225  var row = _pandas.Series(values, columns);
226  data.SetItem(day.Key.ToPython(), row);
227  }
228 
229  return _pandas.DataFrame.from_dict(data, orient:"index");
230  }
231  }

314  /// <summary>
315  /// Gets <see cref="OptionHistory"/> object for a given symbol, date and resolution
316  /// </summary>
317  /// <param name="symbol">The symbol to retrieve historical option data for</param>
318  /// <param name="start">The history request start time</param>
319  /// <param name="end">The history request end time. Defaults to 1 day if null</param>
320  /// <param name="resolution">The resolution to request</param>
321  /// <param name="fillForward">True to fill forward missing data, false otherwise</param>
322  /// <param name="extendedMarket">True to include extended market hours data, false otherwise</param>
323  /// <returns>A <see cref="OptionHistory"/> object that contains historical option data.</returns>
324  public OptionHistory GetOptionHistory(Symbol symbol, DateTime start, DateTime? end = null, Resolution? resolution = null,
325  bool fillForward = true, bool extendedMarket = false)
326  {
327  if (!end.HasValue || end.Value == start)
328  {
329  end = start.AddDays(1);
330  }
331 
332  // Load a canonical option Symbol if the user provides us with an underlying Symbol
333  if (!symbol.SecurityType.IsOption())
334  {
335  var option = AddOption(symbol, resolution, symbol.ID.Market, fillForward);
336 
337  // Allow 20 strikes from the money for futures. No expiry filter is applied
338  // so that any future contract provided will have data returned.
339  if (symbol.SecurityType == SecurityType.Future && symbol.IsCanonical())
340  {
341  throw new ArgumentException("The Future Symbol provided is a canonical Symbol (i.e. a Symbol representing all Futures), which is not supported at this time. " +
342  "If you are using the Symbol accessible from `AddFuture(...)`, use the Symbol from `AddFutureContract(...)` instead. " +
343  "You can use `qb.FutureOptionChainProvider(canonicalFuture, datetime)` to get a list of futures contracts for a given date, and add them to your algorithm with `AddFutureContract(symbol, Resolution)`.");
344  }
345  if (symbol.SecurityType == SecurityType.Future && !symbol.IsCanonical())
346  {
347  option.SetFilter(universe => universe.Strikes(-10, +10));
348  }
349 
350  symbol = option.Symbol;
351  }
352 
353  IEnumerable<Symbol> symbols;
354  if (symbol.IsCanonical())
355  {
356  // canonical symbol, lets find the contracts
357  var option = Securities[symbol] as Option;
358  var resolutionToUseForUnderlying = resolution ?? SubscriptionManager.SubscriptionDataConfigService
359  .GetSubscriptionDataConfigs(symbol)
360  .GetHighestResolution();
361  if (!Securities.ContainsKey(symbol.Underlying))
362  {
363  if (symbol.Underlying.SecurityType == SecurityType.Equity)
364  {
365  // only add underlying if not present
366  AddEquity(symbol.Underlying.Value, resolutionToUseForUnderlying, fillDataForward: fillForward,
367  extendedMarketHours: extendedMarket);
368  }
369  if (symbol.Underlying.SecurityType == SecurityType.Future && symbol.Underlying.IsCanonical())
370  {
371  AddFuture(symbol.Underlying.ID.Symbol, resolutionToUseForUnderlying, fillDataForward: fillForward,
372  extendedMarketHours: extendedMarket);
373  }
374  else if (symbol.Underlying.SecurityType == SecurityType.Future)
375  {
376  AddFutureContract(symbol.Underlying, resolutionToUseForUnderlying, fillDataForward: fillForward,
377  extendedMarketHours: extendedMarket);
378  }
379  }
380  var allSymbols = new List<Symbol>();
381  for (var date = start; date < end; date = date.AddDays(1))
382  {
383  if (option.Exchange.DateIsOpen(date))
384  {
385  allSymbols.AddRange(OptionChainProvider.GetOptionContractList(symbol.Underlying, date));
386  }
387  }
388 
389  var optionFilterUniverse = new OptionFilterUniverse();
390  var distinctSymbols = allSymbols.Distinct();
391  symbols = base.History(symbol.Underlying, start, end.Value, resolution)
392  .SelectMany(x =>
393  {
394  // the option chain symbols wont change so we can set 'exchangeDateChange' to false always
395  optionFilterUniverse.Refresh(distinctSymbols, x, exchangeDateChange:false);
396  return option.ContractFilter.Filter(optionFilterUniverse);
397  })
398  .Distinct().Concat(new[] { symbol.Underlying });
399  }
400  else
401  {
402  // the symbol is a contract
403  symbols = new List<Symbol>{ symbol };
404  }
405 
406  return new OptionHistory(History(symbols, start, end.Value, resolution, fillForward, extendedMarket));
407  }
408 
409  /// <summary>
410  /// Gets <see cref="FutureHistory"/> object for a given symbol, date and resolution
411  /// </summary>
412  /// <param name="symbol">The symbol to retrieve historical future data for</param>
413  /// <param name="start">The history request start time</param>
414  /// <param name="end">The history request end time. Defaults to 1 day if null</param>
415  /// <param name="resolution">The resolution to request</param>
416  /// <param name="fillForward">True to fill forward missing data, false otherwise</param>
417  /// <param name="extendedMarket">True to include extended market hours data, false otherwise</param>
418  /// <returns>A <see cref="FutureHistory"/> object that contains historical future data.</returns>
419  public FutureHistory GetFutureHistory(Symbol symbol, DateTime start, DateTime? end = null, Resolution? resolution = null,
420  bool fillForward = true, bool extendedMarket = false)
421  {
422  if (!end.HasValue || end.Value == start)
423  {
424  end = start.AddDays(1);
425  }
426 
427  var allSymbols = new HashSet<Symbol>();
428  if (symbol.IsCanonical())
429  {
430  // canonical symbol, lets find the contracts
431  var future = Securities[symbol] as Future;
432 
433  for (var date = start; date < end; date = date.AddDays(1))
434  {
435  if (future.Exchange.DateIsOpen(date))
436  {
437  var underlying = new Tick { Time = date };
438  var allList = FutureChainProvider.GetFutureContractList(future.Symbol, date);
439 
440  allSymbols.UnionWith(future.ContractFilter.Filter(new FutureFilterUniverse(allList, underlying)));
441  }
442  }
443  }
444  else
445  {
446  // the symbol is a contract
447  allSymbols.Add(symbol);
448  }
449 
450  return new FutureHistory(History(allSymbols, start, end.Value, resolution, fillForward, extendedMarket));
451  }

682  /// <summary>
683  /// Calculates the daily rate of change
684  /// </summary>
685  /// <param name="dictionary"><see cref="IDictionary{DateTime, Double}"/> with prices keyed by time</param>
686  /// <returns><see cref="List{Double}"/> with daily rate of change</returns>
687  private List<double> CalculateDailyRateOfChange(IDictionary<DateTime, double> dictionary)
688  {
689  var daily = dictionary.GroupBy(kvp => kvp.Key.Date)
690  .ToDictionary(x => x.Key, v => v.LastOrDefault().Value)
691  .Values.ToArray();
692 
693  var rocp = new double[daily.Length];
694  for (var i = 1; i < daily.Length; i++)
695  {
696  rocp[i] = (daily[i] - daily[i - 1]) / daily[i - 1];
697  }
698  rocp[0] = 0;
699 
700  return rocp.ToList();
701  }

719  .Where(x => x.PropertyType.IsGenericType)

736 
737  selector = selector ?? (x => x.Value);

817  private Dictionary<DateTime, DataDictionary<dynamic>> GetAllFundamental(IEnumerable<Symbol> symbols, string selector, DateTime? start = null, DateTime? end = null)
818  {
819  //SubscriptionRequest does not except nullable DateTimes, so set a startTime and endTime
820  var startTime = start.HasValue ? (DateTime)start : QuantConnect.Time.BeginningOfTime;
821  var endTime = end.HasValue ? (DateTime) end : DateTime.UtcNow.Date;
822 
823  //Collection to store our results

825 
826  //Build factory

828 
829  //Get all data for each symbol and fill our dictionary
830  var options = new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount };
831  Parallel.ForEach(symbols, options, symbol =>
832  {
833  var config = new SubscriptionDataConfig(
834  typeof(FineFundamental),
835  symbol,
836  Resolution.Daily,
837  TimeZones.NewYork,
838  TimeZones.NewYork,
839  false,
840  false,
841  false
842  );
843  var security = Securities.CreateSecurity(symbol, config);
844  var request = new SubscriptionRequest(false, null, security, config, startTime.ConvertToUtc(TimeZones.NewYork), endTime.ConvertToUtc(TimeZones.NewYork));
845  using (var enumerator = factory.CreateEnumerator(request, _dataProvider))
846  {
847  while (enumerator.MoveNext())
848  {

850  ? enumerator.Current

853  lock (data)

856  {
857  data[enumerator.Current.Time] = new DataDictionary<dynamic>(enumerator.Current.Time);
