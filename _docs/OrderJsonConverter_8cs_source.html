62  /// <summary>

64  /// </summary>

78  /// <summary>
79  /// Create an order from a simple JObject
80  /// </summary>
81  /// <param name="jObject"></param>
82  /// <returns>Order Object</returns>
83  public static Order CreateOrderFromJObject(JObject jObject)
84  {
85  // create order instance based on order type field
86  var orderType = (OrderType)jObject["Type"].Value<int>();
87  var order = CreateOrder(orderType, jObject);
88 
89  // populate common order properties
90  order.Id = jObject["Id"].Value<int>();
91 
92  var jsonStatus = jObject["Status"];
93  var jsonTime = jObject["Time"];
94  if (jsonStatus.Type == JTokenType.Integer)
95  {
96  order.Status = (OrderStatus)jsonStatus.Value<int>();
97  }
98  else if (jsonStatus.Type == JTokenType.Null)
99  {
100  order.Status = OrderStatus.Canceled;
101  }
102  else
103  {
104  // The `Status` tag can sometimes appear as a string of the enum value in the LiveResultPacket.
105  order.Status = (OrderStatus)Enum.Parse(typeof(OrderStatus), jsonStatus.Value<string>(), true);
106  }
107  if (jsonTime != null && jsonTime.Type != JTokenType.Null)
108  {
109  order.Time = jsonTime.Value<DateTime>();
110  }
111  else
112  {
113  // `Time` can potentially be null in some LiveResultPacket instances, but
114  // `CreatedTime` will always be there if `Time` is absent.
115  order.Time = jObject["CreatedTime"].Value<DateTime>();
116  }
117 
118  var orderSubmissionData = jObject["OrderSubmissionData"];
119  if (orderSubmissionData != null && orderSubmissionData.Type != JTokenType.Null)
120  {
121  var bidPrice = orderSubmissionData["BidPrice"].Value<decimal>();
122  var askPrice = orderSubmissionData["AskPrice"].Value<decimal>();
123  var lastPrice = orderSubmissionData["LastPrice"].Value<decimal>();
124  order.OrderSubmissionData = new OrderSubmissionData(bidPrice, askPrice, lastPrice);
125  }
126 
127  var priceAdjustmentMode = jObject["PriceAdjustmentMode"];
128  if (priceAdjustmentMode != null && priceAdjustmentMode.Type != JTokenType.Null)
129  {
130  var value = priceAdjustmentMode.Value<int>();
131  order.PriceAdjustmentMode = (DataNormalizationMode)value;
132  }
133 
134  var lastFillTime = jObject["LastFillTime"];
135  var lastUpdateTime = jObject["LastUpdateTime"];
136  var canceledTime = jObject["CanceledTime"];
137 
138  if (canceledTime != null && canceledTime.Type != JTokenType.Null)
139  {
140  order.CanceledTime = canceledTime.Value<DateTime>();
141  }
142  if (lastFillTime != null && lastFillTime.Type != JTokenType.Null)
143  {
144  order.LastFillTime = lastFillTime.Value<DateTime>();
145  }
146  if (lastUpdateTime != null && lastUpdateTime.Type != JTokenType.Null)
147  {
148  order.LastUpdateTime = lastUpdateTime.Value<DateTime>();
149  }
150  var tag = jObject["Tag"];
151  if (tag != null && tag.Type != JTokenType.Null)
152  {
153  order.Tag = tag.Value<string>();
154  }
155  else
156  {
157  order.Tag = string.Empty;
158  }
159 
160  order.Quantity = jObject["Quantity"].Value<decimal>();
161  var orderPrice = jObject["Price"];
162  if (orderPrice != null && orderPrice.Type != JTokenType.Null)
163  {
164  order.Price = orderPrice.Value<decimal>();
165  }
166  else
167  {
168  order.Price = default(decimal);
169  }
170 
171  var priceCurrency = jObject["PriceCurrency"];
172  if (priceCurrency != null && priceCurrency.Type != JTokenType.Null)
173  {
174  order.PriceCurrency = priceCurrency.Value<string>();
175  }
176  order.BrokerId = jObject["BrokerId"].Select(x => x.Value<string>()).ToList();
177  var jsonContingentId = jObject["ContingentId"];
178  if (jsonContingentId != null && jsonContingentId.Type != JTokenType.Null)
179  {
180  order.ContingentId = jsonContingentId.Value<int>();
181  }
182 
183  var timeInForce = jObject["Properties"]?["TimeInForce"] ?? jObject["TimeInForce"] ?? jObject["Duration"];
184  order.Properties.TimeInForce = (timeInForce != null)
185  ? CreateTimeInForce(timeInForce, jObject)
186  : TimeInForce.GoodTilCanceled;
187 
188  if (jObject.SelectTokens("Symbol.ID").Any())
189  {
190  var sid = SecurityIdentifier.Parse(jObject.SelectTokens("Symbol.ID").Single().Value<string>());
191  var ticker = jObject.SelectTokens("Symbol.Value").Single().Value<string>();
192  order.Symbol = new Symbol(sid, ticker);
193  }
194  else
195  {
196  string market = null;
197 
198  //does data have market?
199  var suppliedMarket = jObject.SelectTokens("Symbol.ID.Market");
200  if (suppliedMarket.Any())
201  {
202  market = suppliedMarket.Single().Value<string>();
203  }
204 
205  // we only get the security type if we need it, because it might not be there in other cases
206  var securityType = (SecurityType)jObject["SecurityType"].Value<int>();
207  if (jObject.SelectTokens("Symbol.Value").Any())
208  {
209  // provide for backwards compatibility
210  var ticker = jObject.SelectTokens("Symbol.Value").Single().Value<string>();
211 
212  if (market == null && !SymbolPropertiesDatabase.FromDataFolder().TryGetMarket(ticker, securityType, out market))
213  {
214  market = DefaultBrokerageModel.DefaultMarketMap[securityType];
215  }
216  order.Symbol = Symbol.Create(ticker, securityType, market);
217  }
218  else
219  {
220  var tickerstring = jObject["Symbol"].Value<string>();
221 
222  if (market == null && !SymbolPropertiesDatabase.FromDataFolder().TryGetMarket(tickerstring, securityType, out market))
223  {
224  market = DefaultBrokerageModel.DefaultMarketMap[securityType];
225  }
226  order.Symbol = Symbol.Create(tickerstring, securityType, market);
227  }
228  }
229 
230  return order;
231  }

233  /// <summary>
234  /// Creates an order of the correct type
235  /// </summary>
236  private static Order CreateOrder(OrderType orderType, JObject jObject)
237  {
238  Order order;
239  switch (orderType)
240  {
241  case OrderType.Market:
242  order = new MarketOrder();
243  break;
244 
245  case OrderType.Limit:
246  order = new LimitOrder { LimitPrice = jObject["LimitPrice"] == null ? default(decimal) : jObject["LimitPrice"].Value<decimal>() };
247  break;
248 
249  case OrderType.StopMarket:
250  order = new StopMarketOrder
251  {
252  StopPrice = jObject["StopPrice"] == null ? default(decimal) : jObject["StopPrice"].Value<decimal>()
253  };
254  break;
255 
256  case OrderType.StopLimit:
257  order = new StopLimitOrder
258  {
259  LimitPrice = jObject["LimitPrice"] == null ? default(decimal) : jObject["LimitPrice"].Value<decimal>(),
260  StopPrice = jObject["StopPrice"] == null ? default(decimal) : jObject["StopPrice"].Value<decimal>()
261  };
262  break;
263 
264  case OrderType.TrailingStop:
265  order = new TrailingStopOrder
266  {
267  StopPrice = jObject["StopPrice"] == null ? default(decimal) : jObject["StopPrice"].Value<decimal>(),
268  TrailingAmount = jObject["TrailingAmount"] == null ? default(decimal) : jObject["TrailingAmount"].Value<decimal>(),
269  TrailingAsPercentage = jObject["TrailingAsPercentage"] == null ? default(bool) : jObject["TrailingAsPercentage"].Value<bool>()
270  };
271  break;
272 
273  case OrderType.LimitIfTouched:
274  order = new LimitIfTouchedOrder
275  {
276  LimitPrice = jObject["LimitPrice"] == null ? default(decimal) : jObject["LimitPrice"].Value<decimal>(),
277  TriggerPrice = jObject["TriggerPrice"] == null ? default(decimal) : jObject["TriggerPrice"].Value<decimal>()
278  };
279  break;
280 
281  case OrderType.MarketOnOpen:
282  order = new MarketOnOpenOrder();
283  break;
284 
285  case OrderType.MarketOnClose:
286  order = new MarketOnCloseOrder();
287  break;
288 
289  case OrderType.OptionExercise:
290  order = new OptionExerciseOrder();
291  break;
292 
293  case OrderType.ComboMarket:
294  order = new ComboMarketOrder() { GroupOrderManager = DeserializeGroupOrderManager(jObject) };
295  break;
296 
297  case OrderType.ComboLimit:
298  order = new ComboLimitOrder() { GroupOrderManager = DeserializeGroupOrderManager(jObject) };
299  break;
300 
301  case OrderType.ComboLegLimit:
302  order = new ComboLegLimitOrder
303  {
304  GroupOrderManager = DeserializeGroupOrderManager(jObject),
305  LimitPrice = jObject["LimitPrice"] == null ? default(decimal) : jObject["LimitPrice"].Value<decimal>()
306  };
307  break;
308 
309  default:
310  throw new ArgumentOutOfRangeException();
311  }
312  return order;
313  }

315  /// <summary>
316  /// Creates a Time In Force of the correct type
317  /// </summary>
318  private static TimeInForce CreateTimeInForce(JToken timeInForce, JObject jObject)
319  {
320  // for backward-compatibility support deserialization of old JSON format
321  if (timeInForce is JValue)
322  {
323  var value = timeInForce.Value<int>();
324 
325  switch (value)
326  {
327  case 0:
328  return TimeInForce.GoodTilCanceled;
329 
330  case 1:
331  return TimeInForce.Day;
332 
333  case 2:
334  var expiry = jObject["DurationValue"].Value<DateTime>();
335  return TimeInForce.GoodTilDate(expiry);
336 
337  default:
338  throw new Exception($"Unknown time in force value: {value}");
339  }
340  }
341 
342  // convert with TimeInForceJsonConverter
343  return timeInForce.ToObject<TimeInForce>();
344  }

346  /// <summary>
347  /// Deserializes the GroupOrderManager from the JSON object
348  /// </summary>
349  private static GroupOrderManager DeserializeGroupOrderManager(JObject jObject)
350  {
351  var groupOrderManagerJObject = jObject["GroupOrderManager"];
352 
353  // this should never happen
354  if (groupOrderManagerJObject == null)
355  {
356  throw new ArgumentException("OrderJsonConverter.DeserializeGroupOrderManager(): JObject does not have a GroupOrderManager");
357  }
358 
359  var result = new GroupOrderManager(
360  groupOrderManagerJObject["Id"].Value<int>(),
361  groupOrderManagerJObject["Count"].Value<int>(),
362  groupOrderManagerJObject["Quantity"].Value<decimal>(),
363  groupOrderManagerJObject["LimitPrice"].Value<decimal>()
364  );
365 
366  foreach (var orderId in groupOrderManagerJObject["OrderIds"].Values<int>())
367  {
368  result.OrderIds.Add(orderId);
369  }
370 
371  return result;
372  }
373  }
