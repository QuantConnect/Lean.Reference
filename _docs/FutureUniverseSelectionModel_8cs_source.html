27  /// <summary>
28  /// Provides an implementation of <see cref="IUniverseSelectionModel"/> that subscribes to future chains
29  /// </summary>
30  public class FutureUniverseSelectionModel : UniverseSelectionModel
31  {
32  private DateTime _nextRefreshTimeUtc;
33 
34  private readonly TimeSpan _refreshInterval;
35  private readonly UniverseSettings _universeSettings;
36  private readonly Func<DateTime, IEnumerable<Symbol>> _futureChainSymbolSelector;
37 
38  /// <summary>
39  /// Gets the next time the framework should invoke the `CreateUniverses` method to refresh the set of universes.
40  /// </summary>
41  public override DateTime GetNextRefreshTimeUtc() => _nextRefreshTimeUtc;
42 
43  /// <summary>
44  /// Creates a new instance of <see cref="FutureUniverseSelectionModel"/>
45  /// </summary>
46  /// <param name="refreshInterval">Time interval between universe refreshes</param>
47  /// <param name="futureChainSymbolSelector">Selects symbols from the provided future chain</param>
48  public FutureUniverseSelectionModel(TimeSpan refreshInterval, Func<DateTime, IEnumerable<Symbol>> futureChainSymbolSelector)
49  : this(refreshInterval, futureChainSymbolSelector, null)
50  {
51  }
52 
53  /// <summary>
54  /// Creates a new instance of <see cref="FutureUniverseSelectionModel"/>
55  /// </summary>
56  /// <param name="refreshInterval">Time interval between universe refreshes</param>
57  /// <param name="futureChainSymbolSelector">Selects symbols from the provided future chain</param>
58  /// <param name="universeSettings">Universe settings define attributes of created subscriptions, such as their resolution and the minimum time in universe before they can be removed</param>
59  public FutureUniverseSelectionModel(
60  TimeSpan refreshInterval,
61  Func<DateTime, IEnumerable<Symbol>> futureChainSymbolSelector,
62  UniverseSettings universeSettings
63  )
64  {
65  _nextRefreshTimeUtc = DateTime.MinValue;
66 
67  _refreshInterval = refreshInterval;
68  _universeSettings = universeSettings;
69  _futureChainSymbolSelector = futureChainSymbolSelector;
70  }
71 
72  /// <summary>
73  /// Creates the universes for this algorithm. Called once after <see cref="IAlgorithm.Initialize"/>
74  /// </summary>
75  /// <param name="algorithm">The algorithm instance to create universes for</param>
76  /// <returns>The universes to be used by the algorithm</returns>
77  public override IEnumerable<Universe> CreateUniverses(QCAlgorithm algorithm)
78  {
79  _nextRefreshTimeUtc = algorithm.UtcTime + _refreshInterval;
80 
81  var uniqueSymbols = new HashSet<Symbol>();
82  foreach (var futureSymbol in _futureChainSymbolSelector(algorithm.UtcTime))
83  {
84  if (futureSymbol.SecurityType != SecurityType.Future)
85  {
86  throw new ArgumentException("FutureChainSymbolSelector must return future symbols.");
87  }
88 
89  // prevent creating duplicate future chains -- one per symbol
90  if (uniqueSymbols.Add(futureSymbol))
91  {
92  yield return CreateFutureChain(algorithm, futureSymbol);
93  }
94  }
95  }
96 
97  /// <summary>
98  /// Creates the canonical <see cref="Future"/> chain security for a given symbol
99  /// </summary>
100  /// <param name="subscriptionDataConfigService">The service used to create new <see cref="SubscriptionDataConfig"/></param>
101  /// <param name="symbol">Symbol of the future</param>
102  /// <param name="settings">Universe settings define attributes of created subscriptions, such as their resolution and the minimum time in universe before they can be removed</param>
103  /// <param name="securityManager">Used to create new <see cref="Security"/></param>
104  /// <returns><see cref="Future"/> for the given symbol</returns>
105  protected virtual Future CreateFutureChainSecurity(
106  ISubscriptionDataConfigService subscriptionDataConfigService,
107  Symbol symbol,
108  UniverseSettings settings,
109  SecurityManager securityManager)
110  {
111  var config = subscriptionDataConfigService.Add(
112  typeof(ZipEntryName),
113  symbol,
114  settings.Resolution,
115  settings.FillForward,
116  settings.ExtendedMarketHours,
117  isFilteredSubscription: false);
118  return (Future)securityManager.CreateSecurity(symbol, config, settings.Leverage, false);
119  }
120 
121  /// <summary>
122  /// Defines the future chain universe filter
123  /// </summary>
124  protected virtual FutureFilterUniverse Filter(FutureFilterUniverse filter)
125  {
126  // NOP
127  return filter;
128  }
129 
130  /// <summary>
131  /// Creates a <see cref="FuturesChainUniverse"/> for a given symbol
132  /// </summary>
133  /// <param name="algorithm">The algorithm instance to create universes for</param>
134  /// <param name="symbol">Symbol of the future</param>
135  /// <returns><see cref="FuturesChainUniverse"/> for the given symbol</returns>
136  private FuturesChainUniverse CreateFutureChain(QCAlgorithm algorithm, Symbol symbol)
137  {
138  if (symbol.SecurityType != SecurityType.Future)
139  {
140  throw new ArgumentException("CreateFutureChain requires a future symbol.");
141  }
142 
143  // rewrite non-canonical symbols to be canonical
144  var market = symbol.ID.Market;
145  if (!symbol.IsCanonical())
146  {
147  symbol = Symbol.Create(symbol.Value, SecurityType.Future, market, $"/{symbol.Value}");
148  }
149 
150  // resolve defaults if not specified
151  var settings = _universeSettings ?? algorithm.UniverseSettings;
152 
153  // create canonical security object, but don't duplicate if it already exists
154  Security security;
155  Future futureChain;
156  if (!algorithm.Securities.TryGetValue(symbol, out security))
157  {
158  futureChain = CreateFutureChainSecurity(
159  algorithm.SubscriptionManager.SubscriptionDataConfigService,
160  symbol,
161  settings,
162  algorithm.Securities);
163  }
164  else
165  {
166  futureChain = (Future)security;
167  }
168 
169  // set the future chain contract filter function
170  futureChain.SetFilter(Filter);
171 
172  // force future chain security to not be directly tradable AFTER it's configured to ensure it's not overwritten
173  futureChain.IsTradable = false;
174 
175  return new FuturesChainUniverse(futureChain, settings);
176  }
177  }
