119  /// <summary>
120  /// Converts a Lean symbol instance to a CoinAPI symbol id
121  /// </summary>
122  /// <param name="symbol">A Lean symbol instance</param>
123  /// <returns>The CoinAPI symbol id</returns>
124  public string GetBrokerageSymbol(Symbol symbol)
125  {
126  string symbolId;
127  if (!_symbolMap.TryGetValue(symbol, out symbolId))
128  {
129  throw new Exception($"CoinApiSymbolMapper.GetBrokerageSymbol(): Symbol not found: {symbol}");
130  }
131 
132  return symbolId;
133  }

135  /// <summary>
136  /// Converts a CoinAPI symbol id to a Lean symbol instance
137  /// </summary>
138  /// <param name="brokerageSymbol">The CoinAPI symbol id</param>
139  /// <param name="securityType">The security type</param>
140  /// <param name="market">The market</param>
141  /// <param name="expirationDate">Expiration date of the security (if applicable)</param>
142  /// <param name="strike">The strike of the security (if applicable)</param>
143  /// <param name="optionRight">The option right of the security (if applicable)</param>
144  /// <returns>A new Lean Symbol instance</returns>
145  public Symbol GetLeanSymbol(string brokerageSymbol, SecurityType securityType, string market,
146  DateTime expirationDate = new DateTime(), decimal strike = 0, OptionRight optionRight = OptionRight.Call)
147  {
148  var parts = brokerageSymbol.Split('_');
149  if (parts.Length != 4 || parts[1] != "SPOT")
150  {
151  throw new Exception($"CoinApiSymbolMapper.GetLeanSymbol(): Unsupported SymbolId: {brokerageSymbol}");
152  }
153 
154  string symbolMarket;
155  if (!MapExchangeIdsToMarkets.TryGetValue(parts[0], out symbolMarket))
156  {
157  throw new Exception($"CoinApiSymbolMapper.GetLeanSymbol(): Unsupported ExchangeId: {parts[0]}");
158  }
159 
160  var baseCurrency = ConvertCoinApiCurrencyToLeanCurrency(parts[2], symbolMarket);
161  var quoteCurrency = ConvertCoinApiCurrencyToLeanCurrency(parts[3], symbolMarket);
162 
163  var ticker = baseCurrency + quoteCurrency;
164 
165  return Symbol.Create(ticker, SecurityType.Crypto, symbolMarket);
166  }

181  private void LoadSymbolMap(string[] exchangeIds)
182  {
183  var list = string.Join(",", exchangeIds);
184  var json = string.Empty;
185 
186  if (_useLocalSymbolList)
187  {
188  if (!_coinApiSymbolsListFile.Exists)
189  {
190  throw new Exception($"CoinApiSymbolMapper.LoadSymbolMap(): File not found: {_coinApiSymbolsListFile.FullName}, please " +
191  $"download the latest symbol list from CoinApi.");
192  }
193  json = File.ReadAllText(_coinApiSymbolsListFile.FullName);
194  }
195  else
196  {
197  json = $"{RestUrl}/v1/symbols?filter_symbol_id={list}&apiKey={_apiKey}".DownloadData();
198  }
199 
200  var result = JsonConvert.DeserializeObject<List<CoinApiSymbol>>(json);
201 
202  // There were cases of entries in the CoinApiSymbols list with the following pattern:
203  // <Exchange>_SPOT_<BaseCurrency>_<QuoteCurrency>_<ExtraSuffix>
204  // Those cases should be ignored for SPOT prices.
205  foreach (var x in result
206  .Where(x => x.SymbolType == "SPOT" &&
207  x.SymbolId.Split('_').Length == 4 &&
208  // exclude Bitfinex BCH pre-2018-fork as for now we don't have historical mapping data
209  (x.ExchangeId != "BITFINEX" || x.AssetIdBase != "BCH" && x.AssetIdQuote != "BCH")
210  // solves the cases where we request 'binance' and get 'binanceus'
211  && MapExchangeIdsToMarkets.ContainsKey(x.ExchangeId)))
212  {
213  var market = MapExchangeIdsToMarkets[x.ExchangeId];
214  var symbol = GetLeanSymbol(x.SymbolId, SecurityType.Crypto, market);
215 
216  if (_symbolMap.ContainsKey(symbol))
217  {
218  // skipping duplicate symbols. Kraken has both USDC/AD & USD/CAD symbols
219  Log.Error($"CoinApiSymbolMapper(): Duplicate symbol found {symbol} will be skipped!");
220  continue;
221  }
222  _symbolMap[symbol] = x.SymbolId;
223  }
224  }

226  private static string ConvertCoinApiCurrencyToLeanCurrency(string currency, string market)
227  {
228  Dictionary<string, string> mappings;
229  if (CoinApiToLeanCurrencyMappings.TryGetValue(market, out mappings))
230  {
231  string mappedCurrency;
232  if (mappings.TryGetValue(currency, out mappedCurrency))
233  {
234  currency = mappedCurrency;
235  }
236  }
237 
238  return currency;
239  }
