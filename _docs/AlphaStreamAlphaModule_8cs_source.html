16 using System.Linq;
17 using QuantConnect.Data;
18 using System.Collections.Generic;
19 using QuantConnect.Data.UniverseSelection;
20 using QuantConnect.Data.Custom.AlphaStreams;
21 
22 namespace QuantConnect.Algorithm.Framework.Alphas
23 {
24  /// <summary>
25  /// Alpha model that will handle adding and removing securities from the algorithm based on the current portfolio of the different alphas
26  /// </summary>
27  public sealed class AlphaStreamAlphaModule : AlphaModel
28  {
29  private Dictionary<Symbol, HashSet<Symbol>> _symbolsPerAlpha = new Dictionary<Symbol, HashSet<Symbol>>();
30 
31  /// <summary>
32  /// Initialize new <see cref="AlphaStreamAlphaModule"/>
33  /// </summary>
34  public AlphaStreamAlphaModule(string name = null)
35  {
36  Name = name ?? "AlphaStreamAlphaModule";
37  }
38 
39  /// <summary>
40  /// Updates this alpha model with the latest data from the algorithm.
41  /// This is called each time the algorithm receives data for subscribed securities
42  /// </summary>
43  /// <param name="algorithm">The algorithm instance</param>
44  /// <param name="data">The new data available</param>
45  /// <returns>The new insights generated</returns>
46  public override IEnumerable<Insight> Update(QCAlgorithm algorithm, Slice data)
47  {
48  foreach (var portfolioState in data.Get<AlphaStreamsPortfolioState>().Values)
49  {
50  ProcessPortfolioState(algorithm, portfolioState);
51  }
52 
53  return Enumerable.Empty<Insight>();
54  }
55 
56  /// <summary>
57  /// Event fired each time the we add/remove securities from the data feed
58  /// </summary>
59  /// <param name="algorithm">The algorithm instance that experienced the change in securities</param>
60  /// <param name="changes">The security additions and removals from the algorithm</param>
61  public override void OnSecuritiesChanged(QCAlgorithm algorithm, SecurityChanges changes)
62  {
63  changes.FilterCustomSecurities = false;
64  foreach (var addedSecurity in changes.AddedSecurities)
65  {
66  if (addedSecurity.Symbol.IsCustomDataType<AlphaStreamsPortfolioState>())
67  {
68  if (!_symbolsPerAlpha.ContainsKey(addedSecurity.Symbol))
69  {
70  _symbolsPerAlpha[addedSecurity.Symbol] = new HashSet<Symbol>();
71  }
72  // warmup alpha state, adding target securities
73  ProcessPortfolioState(algorithm, addedSecurity.Cache.GetData<AlphaStreamsPortfolioState>());
74  }
75  }
76 
77  algorithm.Log($"OnSecuritiesChanged: {changes}");
78  }
79 
80  /// <summary>
81  /// Will handle adding and removing securities from the algorithm based on the current portfolio of the different alphas
82  /// </summary>
83  private void ProcessPortfolioState(QCAlgorithm algorithm, AlphaStreamsPortfolioState portfolioState)
84  {
85  if (portfolioState == null)
86  {
87  return;
88  }
89 
90  var alphaId = portfolioState.Symbol;
91  if (!_symbolsPerAlpha.TryGetValue(alphaId, out var currentSymbols))
92  {
93  _symbolsPerAlpha[alphaId] = currentSymbols = new HashSet<Symbol>();
94  }
95 
96  var newSymbols = new HashSet<Symbol>(currentSymbols.Count);
97  foreach (var symbol in portfolioState.PositionGroups?.SelectMany(positionGroup => positionGroup.Positions).Select(state => state.Symbol) ?? Enumerable.Empty<Symbol>())
98  {
99  // only add it if it's not used by any alpha (already added check)
100  if (newSymbols.Add(symbol) && !UsedBySomeAlpha(symbol))
101  {
102  algorithm.AddSecurity(symbol,
103  resolution: algorithm.UniverseSettings.Resolution,
104  extendedMarketHours: algorithm.UniverseSettings.ExtendedMarketHours);
105  }
106  }
107  _symbolsPerAlpha[alphaId] = newSymbols;
108 
109  foreach (var symbol in currentSymbols.Where(symbol => !UsedBySomeAlpha(symbol)))
110  {
111  algorithm.RemoveSecurity(symbol);
112  }
113  }
114 
115  private bool UsedBySomeAlpha(Symbol asset)
116  {
117  return _symbolsPerAlpha.Any(pair => pair.Value.Contains(asset));
118  }
119  }
120 }
