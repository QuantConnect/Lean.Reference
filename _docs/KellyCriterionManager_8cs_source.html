17 using System;
18 using System.Collections.Generic;
19 using MathNet.Numerics.Distributions;
20 using MathNet.Numerics.Statistics;
21 using QuantConnect.Logging;
22 
23 namespace QuantConnect.Statistics
24 {
25  /// <summary>
26  /// Class in charge of calculating the Kelly Criterion values.
27  /// Will use the sample values of the last year.
28  /// </summary>
29  /// <remarks>See https://www.quantconnect.com/forum/discussion/6194/insight-scoring-metric/p1 </remarks>
30  public class KellyCriterionManager
31  {
32  private bool _requiresRecalculation;
33  private double _average;
34  private readonly Normal _normalDistribution = new Normal();
35  /// <summary>
36  /// We keep both the value and the corresponding time in separate collections for performance
37  /// this way we can directly calculate the Mean() and Variance() on the values collection
38  /// with no need to select or create another temporary collection
39  /// </summary>
40  private readonly List<double> _insightValues = new List<double>();
41  private readonly List<DateTime> _insightTime = new List<DateTime>();
42 
43  /// <summary>
44  /// Score of the strategy's insights predictive power
45  /// </summary>
46  public decimal KellyCriterionEstimate { get; set; }
47 
48  /// <summary>
49  /// The p-value or probability value of the <see cref="KellyCriterionEstimate"/>
50  /// </summary>
51  public decimal KellyCriterionProbabilityValue { get; set; }
52 
53  /// <summary>
54  /// Adds a new value to the population.
55  /// Will remove values older than an year compared with the provided time.
56  /// For performance, will update the continuous average calculation
57  /// </summary>
58  /// <param name="newValue">The new value to add</param>
59  /// <param name="time">The new values time</param>
60  public void AddNewValue(decimal newValue, DateTime time)
61  {
62  _requiresRecalculation = true;
63 
64  // calculate new average, adding new value
65  _average = (_insightValues.Count * _average + (double)newValue)
66  / (_insightValues.Count + 1);
67 
68  _insightValues.Add((double)newValue);
69  _insightTime.Add(time);
70 
71  // clean up values older than a year
72  var firstTime = _insightTime[0];
73  while ((time - firstTime) >= Time.OneYear)
74  {
75  // calculate new average, removing a value
76  _average = (_insightValues.Count * _average - _insightValues[0])
77  / (_insightValues.Count - 1);
78 
79  _insightValues.RemoveAt(0);
80  _insightTime.RemoveAt(0);
81  // there will always be at least 1 item, the one we just added
82  firstTime = _insightTime[0];
83  }
84  }
85 
86  /// <summary>
87  /// Updates the Kelly Criterion values
88  /// </summary>
89  public void UpdateScores()
90  {
91  try
92  {
93  // need at least 2 samples
94  if (_requiresRecalculation && _insightValues.Count > 1)
95  {
96  _requiresRecalculation = false;
97 
98  var averagePowered = Math.Pow(_average, 2);
99 
100  var variance = _insightValues.Variance();
101  var denominator = averagePowered + variance;
102  var kellyCriterionEstimate = denominator.IsNaNOrZero() ? 0 : _average / denominator;
103 
104  KellyCriterionEstimate = kellyCriterionEstimate.SafeDecimalCast();
105 
106  var variancePowered = Math.Pow(variance, 2);
107  var kellyCriterionStandardDeviation = Math.Sqrt(
108  (1 / variance + 2 * averagePowered / variancePowered)
109  / _insightValues.Count - 1);
110 
111  KellyCriterionProbabilityValue = kellyCriterionStandardDeviation.IsNaNOrZero() ? 1 :
112  1 - _normalDistribution.CumulativeDistribution(kellyCriterionEstimate / kellyCriterionStandardDeviation)
113  .SafeDecimalCast();
114  }
115  }
116  catch (Exception exception)
117  {
118  // just in case...
119  Log.Error(exception);
120  }
121  }
122  }
123 }
