206  /// <summary>
207  /// Add a new point to this series
208  /// </summary>
209  /// <param name="chartPoint">The data point to add</param>
210  public void AddPoint(ChartPoint chartPoint)
211  {
212  if (Values.Count > 0 && Values[Values.Count - 1].x == chartPoint.x)
213  {
214  // duplicate points at the same time, overwrite the value
215  Values[Values.Count - 1] = chartPoint;
216  }
217  else
218  {
219  Values.Add(chartPoint);
220  }
221  }

223  /// <summary>
224  /// Get the updates since the last call to this function.
225  /// </summary>
226  /// <returns>List of the updates from the series</returns>
227  public Series GetUpdates()
228  {
229  var copy = new Series(Name, SeriesType, Index, Unit)
230  {
231  Color = Color,
232  ScatterMarkerSymbol = ScatterMarkerSymbol
233  };
234 
235  try
236  {
237  //Add the updates since the last
238  for (var i = _updatePosition; i < Values.Count; i++)
239  {
240  copy.Values.Add(Values[i]);
241  }
242  //Shuffle the update point to now:
243  _updatePosition = Values.Count;
244  }
245  catch (Exception err)
246  {
247  Log.Error(err);
248  }
249  return copy;
250  }

261  /// <summary>
262  /// Will sum up all chart points into a new single value, using the time of lastest point
263  /// </summary>
264  /// <returns>The new chart point</returns>
265  public ChartPoint ConsolidateChartPoints()
266  {
267  if (Values.Count <= 0) return null;
268 
269  var sum = 0m;
270  foreach (var point in Values)
271  {
272  sum += point.y;
273  }
274 
275  var lastPoint = Values.Last();
276  return new ChartPoint(lastPoint.x, sum);
277  }
