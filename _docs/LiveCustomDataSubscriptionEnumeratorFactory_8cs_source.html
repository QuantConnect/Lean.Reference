27  /// <summary>
28  /// Provides an implementation of <see cref="ISubscriptionEnumeratorFactory"/> to handle live custom data.
29  /// </summary>
30  public class LiveCustomDataSubscriptionEnumeratorFactory : ISubscriptionEnumeratorFactory
31  {
32  private readonly TimeSpan _minimumIntervalCheck;
33  private readonly ITimeProvider _timeProvider;
34  private readonly Func<DateTime, DateTime> _dateAdjustment;
35  private readonly IObjectStore _objectStore;
36 
37  /// <summary>
38  /// Initializes a new instance of the <see cref="LiveCustomDataSubscriptionEnumeratorFactory"/> class
39  /// </summary>
40  /// <param name="timeProvider">Time provider from data feed</param>
41  /// <param name="objectStore">The object store to use</param>
42  /// <param name="dateAdjustment">Func that allows adjusting the datetime to use</param>
43  /// <param name="minimumIntervalCheck">Allows specifying the minimum interval between each enumerator refresh and data check, default is 30 minutes</param>
44  public LiveCustomDataSubscriptionEnumeratorFactory(ITimeProvider timeProvider, IObjectStore objectStore,
45  Func<DateTime, DateTime> dateAdjustment = null, TimeSpan? minimumIntervalCheck = null)
46  {
47  _timeProvider = timeProvider;
48  _dateAdjustment = dateAdjustment;
49  _minimumIntervalCheck = minimumIntervalCheck ?? TimeSpan.FromMinutes(30);
50  _objectStore = objectStore;
51  }
52 
53  /// <summary>
54  /// Creates an enumerator to read the specified request.
55  /// </summary>
56  /// <param name="request">The subscription request to be read</param>
57  /// <param name="dataProvider">Provider used to get data when it is not present on disk</param>
58  /// <returns>An enumerator reading the subscription request</returns>
59  public IEnumerator<BaseData> CreateEnumerator(SubscriptionRequest request, IDataProvider dataProvider)
60  {
61  var config = request.Configuration;
62 
63  // frontier value used to prevent emitting duplicate time stamps between refreshed enumerators
64  // also provides some immediate fast-forward to handle spooling through remote files quickly
65  var frontier = Ref.Create(_dateAdjustment?.Invoke(request.StartTimeLocal) ?? request.StartTimeLocal);
66  var lastSourceRefreshTime = DateTime.MinValue;
67  var sourceFactory = config.GetBaseDataInstance();
68 
69  // this is refreshing the enumerator stack for each new source
70  var refresher = new RefreshEnumerator<BaseData>(() =>
71  {
72  // rate limit the refresh of this enumerator stack
73  var utcNow = _timeProvider.GetUtcNow();
74  var minimumTimeBetweenCalls = GetMinimumTimeBetweenCalls(config.Increment, _minimumIntervalCheck);
75  if (utcNow - lastSourceRefreshTime < minimumTimeBetweenCalls)
76  {
77  return Enumerable.Empty<BaseData>().GetEnumerator();
78  }
79 
80  lastSourceRefreshTime = utcNow;
81  var localDate = _dateAdjustment?.Invoke(utcNow.ConvertFromUtc(config.ExchangeTimeZone).Date) ?? utcNow.ConvertFromUtc(config.ExchangeTimeZone).Date;
82  var source = sourceFactory.GetSource(config, localDate, true);
83 
84  // fetch the new source and enumerate the data source reader
85  var enumerator = EnumerateDataSourceReader(config, dataProvider, frontier, source, localDate, sourceFactory);
86 
87  if (SourceRequiresFastForward(source))
88  {
89  // The FastForwardEnumerator implements these two features:
90  // (1) make sure we never emit past data
91  // (2) data filtering based on a maximum data age
92  // For custom data we don't want feature (2) because we would reject data points emitted later
93  // (e.g. Quandl daily data after a weekend), so we disable it using a huge maximum data age.
94 
95  // apply fast forward logic for file transport mediums
96  var maximumDataAge = GetMaximumDataAge(Time.MaxTimeSpan);
97  enumerator = new FastForwardEnumerator(enumerator, _timeProvider, config.ExchangeTimeZone, maximumDataAge);
98  }
99  else
100  {
101  // rate limit calls to this enumerator stack
102  enumerator = new RateLimitEnumerator<BaseData>(enumerator, _timeProvider, minimumTimeBetweenCalls);
103  }
104 
105  if (source.Format == FileFormat.UnfoldingCollection)
106  {
107  // unroll collections into individual data points after fast forward/rate limiting applied
108  enumerator = enumerator.SelectMany(data =>
109  {
110  var collection = data as BaseDataCollection;
111  IEnumerator<BaseData> collectionEnumerator;
112  if (collection != null)
113  {
114  if (source.TransportMedium == SubscriptionTransportMedium.Rest)
115  {
116  // we want to make sure the data points we *unroll* are not past
117  collectionEnumerator = collection.Data
118  .Where(baseData => baseData.EndTime > frontier.Value)
119  .GetEnumerator();
120  }
121  else
122  {
123  collectionEnumerator = collection.Data.GetEnumerator();
124  }
125  }
126  else
127  {
128  collectionEnumerator = new List<BaseData> { data }.GetEnumerator();
129  }
130  return collectionEnumerator;
131  });
132  }
133 
134  return enumerator;
135  });
136 
137  return refresher;
138  }
139 
140  private IEnumerator<BaseData> EnumerateDataSourceReader(SubscriptionDataConfig config, IDataProvider dataProvider, Ref<DateTime> localFrontier, SubscriptionDataSource source, DateTime localDate, BaseData baseDataInstance)
141  {
142  using (var dataCacheProvider = new SingleEntryDataCacheProvider(dataProvider))
143  {
144  var newLocalFrontier = localFrontier.Value;
145  var dataSourceReader = GetSubscriptionDataSourceReader(source, dataCacheProvider, config, localDate, baseDataInstance, dataProvider);
146  foreach (var datum in dataSourceReader.Read(source))
147  {
148  // always skip past all times emitted on the previous invocation of this enumerator
149  // this allows data at the same time from the same refresh of the source while excluding
150  // data from different refreshes of the source
151  if (datum != null && datum.EndTime > localFrontier.Value)
152  {
153  yield return datum;
154  }
155  else if (!SourceRequiresFastForward(source))
156  {
157  // if the 'source' is Rest and there is no new value,
158  // we *break*, else we will be caught in a tight loop
159  // because Rest source never ends!
160  // edit: we 'break' vs 'return null' so that the source is refreshed
161  // allowing date changes to impact the source value
162  // note it will respect 'minimumTimeBetweenCalls'
163  break;
164  }
165 
166  if (datum != null)
167  {
168  newLocalFrontier = Time.Max(datum.EndTime, newLocalFrontier);
169 
170  if (!SourceRequiresFastForward(source))
171  {
172  // if the 'source' is Rest we need to update the localFrontier here
173  // because Rest source never ends!
174  // Should be advance frontier for all source types here?
175  localFrontier.Value = newLocalFrontier;
176  }
177  }
178  }
179 
180  localFrontier.Value = newLocalFrontier;
181  }
182  }
183 
184  /// <summary>
185  /// Gets the <see cref="ISubscriptionDataSourceReader"/> for the specified source
186  /// </summary>
187  protected virtual ISubscriptionDataSourceReader GetSubscriptionDataSourceReader(SubscriptionDataSource source,
188  IDataCacheProvider dataCacheProvider,
189  SubscriptionDataConfig config,
190  DateTime date,
191  BaseData baseDataInstance,
192  IDataProvider dataProvider
193  )
194  {
195  return SubscriptionDataSourceReader.ForSource(source, dataCacheProvider, config, date, true, baseDataInstance, dataProvider, _objectStore);
196  }
197 
198  private bool SourceRequiresFastForward(SubscriptionDataSource source)
199  {
200  return source.TransportMedium == SubscriptionTransportMedium.LocalFile
201  || source.TransportMedium == SubscriptionTransportMedium.RemoteFile;
202  }
203 
204  private static TimeSpan GetMinimumTimeBetweenCalls(TimeSpan increment, TimeSpan minimumInterval)
205  {
206  return TimeSpan.FromTicks(Math.Min(increment.Ticks, minimumInterval.Ticks));
207  }
208 
209  private static TimeSpan GetMaximumDataAge(TimeSpan increment)
210  {
211  return TimeSpan.FromTicks(Math.Max(increment.Ticks, TimeSpan.FromSeconds(5).Ticks));
212  }
