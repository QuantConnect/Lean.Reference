16 using System;
17 
18 using QuantConnect.Interfaces;
19 using QuantConnect.Securities;
20 
21 namespace QuantConnect.Orders
22 {
23  /// <summary>
24  /// In effect, a LimitIfTouchedOrder behaves opposite to the <see cref="StopLimitOrder"/>;
25  /// after a trigger price is touched, a limit order is set for some user-defined value above (below)
26  /// the trigger when selling (buying).
27  /// https://www.interactivebrokers.ca/en/index.php?f=45318
28  /// </summary>
29  public class LimitIfTouchedOrder : Order
30  {
31  /// <summary>
32  /// Order Type
33  /// </summary>
34  public override OrderType Type => OrderType.LimitIfTouched;
35 
36  /// <summary>
37  /// The price which, when touched, will trigger the setting of a limit order at <see cref="LimitPrice"/>.
38  /// </summary>
39  public decimal TriggerPrice { get; internal set; }
40 
41  /// <summary>
42  /// The price at which to set the limit order following <see cref="TriggerPrice"/> being touched.
43  /// </summary>
44  public decimal LimitPrice { get; internal set; }
45 
46  /// <summary>
47  /// Whether or not the <see cref="TriggerPrice"/> has been touched.
48  /// </summary>
49  public bool TriggerTouched { get; internal set; }
50 
51  /// <summary>
52  /// New <see cref="LimitIfTouchedOrder"/> constructor.
53  /// </summary>
54  /// <param name="symbol">Symbol asset we're seeking to trade</param>
55  /// <param name="quantity">Quantity of the asset we're seeking to trade</param>
56  /// <param name="limitPrice">Maximum price to fill the order</param>
57  /// <param name="time">Time the order was placed</param>
58  /// <param name="triggerPrice">Price which must be touched in order to then set a limit order</param>
59  /// <param name="tag">User defined data tag for this order</param>
60  /// <param name="properties">The order properties for this order</param>
61  public LimitIfTouchedOrder(
62  Symbol symbol,
63  decimal quantity,
64  decimal? triggerPrice,
65  decimal limitPrice,
66  DateTime time,
67  string tag = "",
68  IOrderProperties properties = null
69  )
70  : base(symbol, quantity, time, tag, properties)
71  {
72  TriggerPrice = (decimal) triggerPrice;
73  LimitPrice = limitPrice;
74  }
75 
76  /// <summary>
77  /// Default constructor for JSON Deserialization:
78  /// </summary>
79  public LimitIfTouchedOrder()
80  {
81  }
82 
83  /// <summary>
84  /// Gets the default tag for this order
85  /// </summary>
86  /// <returns>The default tag</returns>
87  public override string GetDefaultTag()
88  {
89  return Messages.LimitIfTouchedOrder.Tag(this);
90  }
91 
92  /// <summary>
93  /// Modifies the state of this order to match the update request
94  /// </summary>
95  /// <param name="request">The request to update this order object</param>
96  public override void ApplyUpdateOrderRequest(UpdateOrderRequest request)
97  {
98  base.ApplyUpdateOrderRequest(request);
99  if (request.TriggerPrice.HasValue)
100  {
101  TriggerPrice = request.TriggerPrice.Value;
102  }
103 
104  if (request.LimitPrice.HasValue)
105  {
106  LimitPrice = request.LimitPrice.Value;
107  }
108  }
109 
110  /// <summary>
111  /// Creates a deep-copy clone of this order
112  /// </summary>
113  /// <returns>A copy of this order</returns>
114  public override Order Clone()
115  {
116  var order = new LimitIfTouchedOrder
117  {TriggerPrice = TriggerPrice, LimitPrice = LimitPrice, TriggerTouched = TriggerTouched};
118  CopyTo(order);
119  return order;
120  }
121 
122  /// <summary>
123  /// Returns a string that represents the current object.
124  /// </summary>
125  /// <returns>
126  /// A string that represents the current object.
127  /// </returns>
128  /// <filterpriority>2</filterpriority>
129  public override string ToString()
130  {
131  return Messages.LimitIfTouchedOrder.ToString(this);
132  }
133 
134  /// <summary>
135  /// Gets the order value in units of the security's quote currency for a single unit.
136  /// A single unit here is a single share of stock, or a single barrel of oil, or the
137  /// cost of a single share in an option contract.
138  /// </summary>
139  /// <param name="security">The security matching this order's symbol</param>
140  protected override decimal GetValueImpl(Security security)
141  {
142  // selling, so higher price will be used
143  if (Quantity < 0)
144  {
145  return Quantity * Math.Max(LimitPrice, security.Price);
146  }
147 
148  // buying, so lower price will be used
149  if (Quantity > 0)
150  {
151  return Quantity * Math.Min(LimitPrice, security.Price);
152  }
153 
154  return 0m;
155  }
156  }
157 }
