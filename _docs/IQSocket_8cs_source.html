65  public enum PortType { Level1 = 1, Lookup = 3, Level2 = 2, Admin = 0 }
66  public static class IQSocket
67  {
68  public static int GetPort(PortType portType)
69  {
70  var port = 0;
71  switch (portType)
72  {
73  case PortType.Level1:
74  port = 5009;
75  break;
76  case PortType.Lookup:
77  port = 9100;
78  break;
79  case PortType.Level2:
80  port = 9200;
81  break;
82  case PortType.Admin:
83  port = 9300;
84  break;
85  }
86  return port;
87  }
88  public static IPAddress GetIp(string host)
89  {
90  var ipAddresses = Dns.GetHostAddresses(host);
91  if (ipAddresses.Length == 0)
92  {
93  throw new ArgumentException("Unable to retrieve address from specified host name.", host);
94  }
95 
96  return ipAddresses[0];
97  }

117  public class Status
118  {
119  internal Status()
120  {
121  }
122  internal void Update(string line)
123  {
124  var fields = line.Split(',');
125  lock (this)
126  {
127  _serverIp = fields[2];
128  if (!int.TryParse(fields[3], out _serverPort)) _serverPort = 0;
129  if (!int.TryParse(fields[4], out _maxSymbols)) _maxSymbols = 0;
130  if (!int.TryParse(fields[5], out _numberOfSymbols)) _numberOfSymbols = 0;
131  if (!int.TryParse(fields[6], out _clientsConnected)) _clientsConnected = 0;
132  if (!int.TryParse(fields[7], out _secondsSinceLastUpdate)) _secondsSinceLastUpdate = 0;
133  if (!int.TryParse(fields[8], out _reconnections)) _reconnections = 0;
134  if (!int.TryParse(fields[9], out _attemptedReconnections)) _attemptedReconnections = 0;
135  if (!DateTime.TryParseExact(fields[10], "MMM dd hh':'mmtt", _enUS, DateTimeStyles.None, out _startTime)) _startTime = DateTime.MinValue;
136  if (!DateTime.TryParseExact(fields[11], "MMM dd hh':'mmtt", _enUS, DateTimeStyles.None, out _marketTime)) _marketTime = DateTime.MinValue;
137  _connected = false;
138  if (fields[12] == "Connected") { _connected = true; }
139  _iqFeedVersion = fields[13];
140  _loginId = fields[14];
141  if (!double.TryParse(fields[15], out _totalKbsRecv)) _totalKbsRecv = 0;
142  if (!double.TryParse(fields[16], out _kbsPerSecRecv)) _kbsPerSecRecv = 0;
143  if (!double.TryParse(fields[17], out _avgKbsPerSecRecv)) _avgKbsPerSecRecv = 0;
144  if (!double.TryParse(fields[18], out _totalKbsSent)) _totalKbsSent = 0;
145  if (!double.TryParse(fields[19], out _kbsPerSecSent)) _kbsPerSecSent = 0;
146  if (!double.TryParse(fields[20], out _avgKbsPerSecSent)) _avgKbsPerSecSent = 0;
147  }
148  }
