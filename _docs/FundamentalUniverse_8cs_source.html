51  /// <summary>
52  /// Return the URL string source of the file. This will be converted to a stream
53  /// </summary>
54  public override SubscriptionDataSource GetSource(SubscriptionDataConfig config, DateTime date, bool isLiveMode)
55  {
56  var path = _factory.GetSource(config, date, isLiveMode).Source;
57  return new SubscriptionDataSource(path, SubscriptionTransportMedium.LocalFile, FileFormat.FoldingCollection);
58  }
59 
60  /// <summary>
61  /// Will read a new instance from the given line
62  /// </summary>
63  /// <param name="config">The associated requested configuration</param>
64  /// <param name="line">The line to parse</param>
65  /// <param name="date">The current time</param>
66  /// <param name="isLiveMode">True if live mode</param>
67  /// <returns>A new instance or null</returns>
68  public override BaseData Reader(SubscriptionDataConfig config, string line, DateTime date, bool isLiveMode)
69  {
70  try
71  {
72  var csv = line.Split(',');
73  var symbol = new Symbol(SecurityIdentifier.Parse(csv[0]), csv[1]);
74  return new Fundamental(date, symbol);
75  }
76  catch (Exception)
77  {
78  return null;
79  }
80  }
81 
82  /// <summary>
83  /// Will clone the current instance
84  /// </summary>
85  /// <returns>The cloned instance</returns>
86  public override BaseData Clone()
87  {
88  return new FundamentalUniverse(Time, Symbol) { Data = Data, EndTime = EndTime };
89  }
90 
91  /// <summary>
92  /// Gets the default resolution for this data and security type
93  /// </summary>
94  /// <remarks>This is a method and not a property so that python
95  /// custom data types can override it</remarks>
96  public override Resolution DefaultResolution()
97  {
98  return Resolution.Daily;
99  }
100 
101  /// <summary>
102  /// Creates the universe symbol for the target market
103  /// </summary>
104  /// <returns>The universe symbol to use</returns>
105  public override Symbol UniverseSymbol(string market = null)
106  {
107  market ??= QuantConnect.Market.USA;
108  var ticker = $"{GetType().Name}-{market}-{Guid.NewGuid()}";
109  return Symbol.Create(ticker, SecurityType.Equity, market, baseDataType: GetType());
110  }
