16 using System;
17 using Python.Runtime;
18 using QuantConnect.Data;
19 using QuantConnect.Data.Consolidators;
20 
21 namespace QuantConnect.Python
22 {
23  /// <summary>
24  /// Provides an Data Consolidator that wraps a <see cref="PyObject"/> object that represents a custom Python consolidator
25  /// </summary>
26  public class DataConsolidatorPythonWrapper : IDataConsolidator
27  {
28  private readonly dynamic _consolidator;
29 
30  /// <summary>
31  /// Gets the most recently consolidated piece of data. This will be null if this consolidator
32  /// has not produced any data yet.
33  /// </summary>
34  public IBaseData Consolidated
35  {
36  get { using (Py.GIL()) {return _consolidator.Consolidated;} }
37  }
38 
39  /// <summary>
40  /// Gets a clone of the data being currently consolidated
41  /// </summary>
42  public IBaseData WorkingData
43  {
44  get { using (Py.GIL()) {return _consolidator.WorkingData;} }
45  }
46 
47  /// <summary>
48  /// Gets the type consumed by this consolidator
49  /// </summary>
50  public Type InputType
51  {
52  get { using (Py.GIL()) {return _consolidator.InputType;} }
53  }
54 
55  /// <summary>
56  /// Gets the type produced by this consolidator
57  /// </summary>
58  public Type OutputType
59  {
60  get { using (Py.GIL()) {return _consolidator.OutputType;} }
61  }
62 
63  /// <summary>
64  /// Event handler that fires when a new piece of data is produced
65  /// </summary>
66  public event DataConsolidatedHandler DataConsolidated
67  {
68  add { using (Py.GIL()) {_consolidator.DataConsolidated += value;} }
69  remove { using (Py.GIL()) {_consolidator.DataConsolidated -= value;} }
70  }
71 
72  /// <summary>
73  /// Constructor for initialising the <see cref="DataConsolidatorPythonWrapper"/> class with wrapped <see cref="PyObject"/> object
74  /// </summary>
75  /// <param name="consolidator">Represents a custom python consolidator</param>
76  public DataConsolidatorPythonWrapper(PyObject consolidator)
77  {
78  using (Py.GIL())
79  {
80  foreach (var attributeName in new[] { "InputType", "OutputType", "WorkingData", "Consolidated" })
81  {
82  if (!consolidator.HasAttr(attributeName))
83  {
84  throw new NotImplementedException(
85  Messages.PythonCommon.AttributeNotImplemented($"IDataConsolidator.{attributeName}", consolidator.GetPythonType()));
86  }
87  }
88  }
89  _consolidator = consolidator;
90  }
91 
92  /// <summary>
93  /// Scans this consolidator to see if it should emit a bar due to time passing
94  /// </summary>
95  /// <param name="currentLocalTime">The current time in the local time zone (same as <see cref="BaseData.Time"/>)</param>
96  public void Scan(DateTime currentLocalTime)
97  {
98  using (Py.GIL())
99  {
100  _consolidator.Scan(currentLocalTime);
101  }
102  }
103 
104  /// <summary>
105  /// Updates this consolidator with the specified data
106  /// </summary>
107  /// <param name="data">The new data for the consolidator</param>
108  public void Update(IBaseData data)
109  {
110  using (Py.GIL())
111  {
112  _consolidator.Update(data);
113  }
114  }
115 
116  /// <summary>Performs application-defined tasks associated with freeing, releasing, or resetting unmanaged resources.</summary>
117  /// <filterpriority>2</filterpriority>
118  public void Dispose()
119  {
120  }
121  }
122 }
