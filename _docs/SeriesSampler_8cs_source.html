16 using System;
17 using System.Collections.Generic;
18 using QuantConnect.Util;
19 
20 namespace QuantConnect
21 {
22  /// <summary>
23  /// A type capable of taking a chart and resampling using a linear interpolation strategy
24  /// </summary>
25  public class SeriesSampler
26  {
27  private readonly double _seconds;
28 
29  /// <summary>
30  /// Creates a new SeriesSampler to sample Series data on the specified resolution
31  /// </summary>
32  /// <param name="resolution">The desired sampling resolution</param>
33  public SeriesSampler(TimeSpan resolution)
34  {
35  _seconds = resolution.TotalSeconds;
36  }
37 
38  /// <summary>
39  /// Samples the given series
40  /// </summary>
41  /// <param name="series">The series to be sampled</param>
42  /// <param name="start">The date to start sampling, if before start of data then start of data will be used</param>
43  /// <param name="stop">The date to stop sampling, if after stop of data, then stop of data will be used</param>
44  /// <param name="truncateValues">True will truncate values to integers</param>
45  /// <returns>The sampled series</returns>
46  public Series Sample(Series series, DateTime start, DateTime stop, bool truncateValues = false)
47  {
48  var sampled = new Series(series.Name, series.SeriesType, series.Index, series.Unit);
49 
50  // chart point times are always in universal, so force it here as well
51  double nextSample = Time.DateTimeToUnixTimeStamp(start.ToUniversalTime());
52  double unixStopDate = Time.DateTimeToUnixTimeStamp(stop.ToUniversalTime());
53 
54  // we can't sample a single point and it doesn't make sense to sample scatter plots
55  // in this case just copy the raw data
56  if (series.Values.Count < 2 || series.SeriesType == SeriesType.Scatter)
57  {
58  // we can minimally verify we're within the start/stop interval
59  foreach (var point in series.Values)
60  {
61  if (point.x >= nextSample && point.x <= unixStopDate)
62  {
63  var samplePoint = point;
64  if (truncateValues)
65  {
66  // let's not modify the original
67  samplePoint = new ChartPoint(samplePoint) { y = Math.Truncate(samplePoint.y) };
68  }
69  sampled.Values.Add(samplePoint);
70  }
71  }
72  return sampled;
73  }
74 
75  var enumerator = series.Values.GetEnumerator();
76 
77  // initialize current/previous
78  enumerator.MoveNext();
79  ChartPoint previous = enumerator.Current;
80  enumerator.MoveNext();
81  ChartPoint current = enumerator.Current;
82 
83  // make sure we don't start sampling before the data begins
84  if (nextSample < previous.x)
85  {
86  nextSample = previous.x;
87  }
88 
89  // make sure to advance into the requestd time frame before sampling
90  while (current.x < nextSample && enumerator.MoveNext())
91  {
92  previous = current;
93  current = enumerator.Current;
94  }
95 
96  do
97  {
98  // advance our current/previous
99  if (nextSample > current.x)
100  {
101  if (enumerator.MoveNext())
102  {
103  previous = current;
104  current = enumerator.Current;
105  }
106  else
107  {
108  break;
109  }
110  }
111 
112  // iterate until we pass where we want our next point
113  while (nextSample <= current.x && nextSample <= unixStopDate)
114  {
115  var value = Interpolate(previous, current, (long) nextSample);
116  var point = new ChartPoint {x = (long) nextSample, y = value};
117  if (truncateValues)
118  {
119  point.y = Math.Truncate(point.y);
120  }
121  sampled.Values.Add(point);
122  nextSample += _seconds;
123  }
124 
125  // if we've passed our stop then we're finished sampling
126  if (nextSample > unixStopDate)
127  {
128  break;
129  }
130  }
131  while (true);
132 
133  enumerator.DisposeSafely();
134  return sampled;
135  }
136 
137  /// <summary>
138  /// Samples the given charts
139  /// </summary>
140  /// <param name="charts">The charts to be sampled</param>
141  /// <param name="start">The date to start sampling</param>
142  /// <param name="stop">The date to stop sampling</param>
143  /// <returns>The sampled charts</returns>
144  public Dictionary<string, Chart> SampleCharts(IDictionary<string, Chart> charts, DateTime start, DateTime stop)
145  {
146  var sampledCharts = new Dictionary<string, Chart>();
147  foreach (var chart in charts.Values)
148  {
149  var sampledChart = new Chart(chart.Name);
150  sampledCharts.Add(sampledChart.Name, sampledChart);
151  foreach (var series in chart.Series.Values)
152  {
153  var sampledSeries = Sample(series, start, stop);
154  sampledChart.AddSeries(sampledSeries);
155  }
156  }
157  return sampledCharts;
158  }
159 
160  /// <summary>
161  /// Linear interpolation used for sampling
162  /// </summary>
163  private static decimal Interpolate(ChartPoint previous, ChartPoint current, long target)
164  {
165  var deltaTicks = current.x - previous.x;
166 
167  // if they're at the same time return the current value
168  if (deltaTicks == 0)
169  {
170  return current.y;
171  }
172 
173  double percentage = (target - previous.x) / (double)deltaTicks;
174 
175  // y=mx+b
176  return (current.y - previous.y) * (decimal)percentage + previous.y;
177  }
178  }
