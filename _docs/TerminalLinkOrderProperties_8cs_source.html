37  /// <summary>
38  /// Custom user order notes 1

42  /// <summary>

45  public string CustomNotes2 { get; set; }

55  public string CustomNotes4 { get; set; }

61 
62  /// <summary>
63  /// The EMSX account
64  /// </summary>
65  public string Account { get; set; }
66 
67  /// <summary>

95  /// The strategy name

104  /// <summary>
105  /// Creates a new TerminalLink order strategy instance

108  /// <param name="fields">The strategy fields</param>

112  Fields = fields;
113  }
114  }
115 
116  /// <summary>
117  /// Models an EMSX order strategy field
118  /// </summary>
119  public class StrategyField
120  {
121  /// <summary>
122  /// The strategy field value
123  /// </summary>
124  public string Value { get; set; }
125 
126  /// <summary>
127  /// Whether the strategy field carries a value
128  /// </summary>
129  public bool HasValue { get; set; }
130 
131  /// <summary>
132  /// Creates a new TerminalLink order strategy field carrying a value.
133  /// </summary>
134  /// <param name="value">The strategy field value</param>

137  Value = value;

139  }
