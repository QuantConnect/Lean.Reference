27  /// <summary>
28  /// Provides an implementations of <see cref="ISubscriptionDataSourceReader"/> that uses the
29  /// <see cref="BaseData.Reader(SubscriptionDataConfig,string,DateTime,bool)"/>
30  /// method to read lines of text from a <see cref="SubscriptionDataSource"/>
31  /// </summary>
32  public class TextSubscriptionDataSourceReader : BaseSubscriptionDataSourceReader
33  {
34  private readonly bool _implementsStreamReader;
35  private readonly DateTime _date;
36  private BaseData _factory;
37  private bool _shouldCacheDataPoints;
38 
39  private static int CacheSize = 100;
40  private static volatile Dictionary<string, List<BaseData>> BaseDataSourceCache = new Dictionary<string, List<BaseData>>(100);
41  private static Queue<string> CacheKeys = new Queue<string>(100);
42 
43  /// <summary>
44  /// The requested subscription configuration
45  /// </summary>
46  protected SubscriptionDataConfig Config { get; set; }
47 
48  /// <summary>
49  /// Event fired when an exception is thrown during a call to
50  /// <see cref="BaseData.Reader(SubscriptionDataConfig,string,DateTime,bool)"/>
51  /// </summary>
52  public event EventHandler<ReaderErrorEventArgs> ReaderError;
53 
54  /// <summary>
55  /// Initializes a new instance of the <see cref="TextSubscriptionDataSourceReader"/> class
56  /// </summary>
57  /// <param name="dataCacheProvider">This provider caches files if needed</param>
58  /// <param name="config">The subscription's configuration</param>
59  /// <param name="date">The date this factory was produced to read data for</param>
60  /// <param name="isLiveMode">True if we're in live mode, false for backtesting</param>
61  public TextSubscriptionDataSourceReader(IDataCacheProvider dataCacheProvider, SubscriptionDataConfig config, DateTime date, bool isLiveMode,
62  IObjectStore objectStore)
63  : base(dataCacheProvider, isLiveMode, objectStore)
64  {
65  _date = date;
66  Config = config;
67  _shouldCacheDataPoints = !Config.IsCustomData && Config.Resolution >= Resolution.Hour
68  && Config.Type != typeof(FineFundamental) && Config.Type != typeof(CoarseFundamental) && Config.Type != typeof(Fundamental)
69  // don't cache universe data, doesn't make much sense and we don't want to change the symbol of the clone
70  && !Config.Type.IsAssignableTo(typeof(BaseDataCollection))
71  && !DataCacheProvider.IsDataEphemeral;
72 
73  _implementsStreamReader = Config.Type.ImplementsStreamReader();
74  }
75 
76  /// <summary>
77  /// Reads the specified <paramref name="source"/>
78  /// </summary>
79  /// <param name="source">The source to be read</param>
80  /// <returns>An <see cref="IEnumerable{BaseData}"/> that contains the data in the source</returns>
81  public override IEnumerable<BaseData> Read(SubscriptionDataSource source)
82  {
83  List<BaseData> cache = null;
84  _shouldCacheDataPoints = _shouldCacheDataPoints &&
85  // only cache local files
86  source.TransportMedium == SubscriptionTransportMedium.LocalFile;
87 
88  string cacheKey = null;
89  if (_shouldCacheDataPoints)
90  {
91  cacheKey = source.Source + Config.Type;
92  BaseDataSourceCache.TryGetValue(cacheKey, out cache);
93  }
94  if (cache == null)
95  {
96  cache = _shouldCacheDataPoints ? new List<BaseData>(30000) : null;
97  using (var reader = CreateStreamReader(source))
98  {
99  if (reader == null)
100  {
101  // if the reader doesn't have data then we're done with this subscription
102  yield break;
103  }
104 
105  if (_factory == null)
106  {
107  // only create a factory if the stream isn't null
108  _factory = Config.GetBaseDataInstance();
109  }
110  // while the reader has data
111  while (!reader.EndOfStream)
112  {
113  BaseData instance = null;
114  string line = null;
115  try
116  {
117  if (reader.StreamReader != null && _implementsStreamReader)
118  {
119  instance = _factory.Reader(Config, reader.StreamReader, _date, IsLiveMode);
120  }
121  else
122  {
123  // read a line and pass it to the base data factory
124  line = reader.ReadLine();
125  instance = _factory.Reader(Config, line, _date, IsLiveMode);
126  }
127  }
128  catch (Exception err)
129  {
130  OnReaderError(line ?? "StreamReader", err);
131  }
132 
133  if (instance != null && instance.EndTime != default(DateTime))
134  {
135  if (_shouldCacheDataPoints)
136  {
137  cache.Add(instance);
138  }
139  else
140  {
141  yield return instance;
142  }
143  }
144  else if (reader.ShouldBeRateLimited)
145  {
146  yield return instance;
147  }
148  }
149  }
150 
151  if (!_shouldCacheDataPoints)
152  {
153  yield break;
154  }
155 
156  lock (CacheKeys)
157  {
158  CacheKeys.Enqueue(cacheKey);
159  // we create a new dictionary, so we don't have to take locks when reading, and add our new item
160  var newCache = new Dictionary<string, List<BaseData>>(BaseDataSourceCache) { [cacheKey] = cache };
161 
162  if (BaseDataSourceCache.Count > CacheSize)
163  {
164  var removeCount = 0;
165  // we remove a portion of the first in entries
166  while (++removeCount < (CacheSize / 4))
167  {
168  newCache.Remove(CacheKeys.Dequeue());
169  }
170  // update the cache instance
171  BaseDataSourceCache = newCache;
172  }
173  else
174  {
175  // update the cache instance
176  BaseDataSourceCache = newCache;
177  }
178  }
179  }
180 
181  if (cache == null)
182  {
183  throw new InvalidOperationException($"Cache should not be null. Key: {cacheKey}");
184  }
185  // Find the first data point 10 days (just in case) before the desired date
186  // and subtract one item (just in case there was a time gap and data.Time is after _date)
187  var frontier = _date.AddDays(-10);
188  var index = cache.FindIndex(data => data.Time > frontier);
189  index = index > 0 ? (index - 1) : 0;
190  foreach (var data in cache.Skip(index))
191  {
192  var clone = data.Clone();
193  clone.Symbol = Config.Symbol;
194  yield return clone;
195  }
196  }
197 
198  /// <summary>
199  /// Event invocator for the <see cref="ReaderError"/> event
200  /// </summary>
201  /// <param name="line">The line that caused the exception</param>
202  /// <param name="exception">The exception that was caught</param>
203  private void OnReaderError(string line, Exception exception)
204  {
205  var handler = ReaderError;
206  if (handler != null) handler(this, new ReaderErrorEventArgs(line, exception));
207  }
208 
209  /// <summary>
210  /// Set the cache size to use
211  /// </summary>
212  /// <remarks>How to size this cache: Take worst case scenario, BTCUSD hour, 60k QuoteBar entries, which are roughly 200 bytes in size -> 11 MB * CacheSize</remarks>
213  public static void SetCacheSize(int megaBytesToUse)
214  {
215  if (megaBytesToUse != 0)
216  {
217  // we take worst case scenario, each entry is 12 MB
218  CacheSize = megaBytesToUse / 12;
219  Log.Trace($"TextSubscriptionDataSourceReader.SetCacheSize(): Setting cache size to {CacheSize} items");
220  }
221  }
222 
223  /// <summary>
224  /// Will clear the data cache.
225  /// Used for testing different time zones for the same data set and allow a clean fresh start for each backtest
226  /// </summary>
227  public static void ClearCache()
228  {
229  BaseDataSourceCache = new();
230  }
231  }
