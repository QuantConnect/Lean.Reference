16 using Python.Runtime;
17 using QuantConnect.Securities;

19 namespace QuantConnect.Python
20 {
21  /// <summary>
22  /// Wraps a <see cref="PyObject"/> object that represents a security's model of buying power
23  /// </summary>
24  public class BuyingPowerModelPythonWrapper : IBuyingPowerModel
25  {
26  private readonly dynamic _model;
27 
28  /// <summary>
29  /// Constructor for initializing the <see cref="BuyingPowerModelPythonWrapper"/> class with wrapped <see cref="PyObject"/> object
30  /// </summary>
31  /// <param name="model">Represents a security's model of buying power</param>
32  public BuyingPowerModelPythonWrapper(PyObject model)
33  {
34  _model = model.ValidateImplementationOf<IBuyingPowerModel>();
35  }
36 
37  /// <summary>
38  /// Gets the buying power available for a trade
39  /// </summary>
40  /// <param name="parameters">A parameters object containing the algorithm's potrfolio, security, and order direction</param>
41  /// <returns>The buying power available for the trade</returns>
42  public BuyingPower GetBuyingPower(BuyingPowerParameters parameters)
43  {
44  using (Py.GIL())
45  {
46  return (_model.GetBuyingPower(parameters) as PyObject).GetAndDispose<BuyingPower>();
47  }
48  }
49 
50  /// <summary>
51  /// Gets the current leverage of the security
52  /// </summary>
53  /// <param name="security">The security to get leverage for</param>
54  /// <returns>The current leverage in the security</returns>
55  public decimal GetLeverage(Security security)
56  {
57  using (Py.GIL())
58  {
59  return (_model.GetLeverage(security) as PyObject).GetAndDispose<decimal>();
60  }
61  }
62 
63  /// <summary>
64  /// Get the maximum market order quantity to obtain a position with a given buying power percentage.
65  /// Will not take into account free buying power.
66  /// </summary>
67  /// <param name="parameters">An object containing the portfolio, the security and the target signed buying power percentage</param>
68  /// <returns>Returns the maximum allowed market order quantity and if zero, also the reason</returns>
69  public GetMaximumOrderQuantityResult GetMaximumOrderQuantityForTargetBuyingPower(GetMaximumOrderQuantityForTargetBuyingPowerParameters parameters)
70  {
71  using (Py.GIL())
72  {
73  return (_model.GetMaximumOrderQuantityForTargetBuyingPower(parameters)
74  as PyObject).GetAndDispose<GetMaximumOrderQuantityResult>();
75  }
76  }
77 
78  /// <summary>

81  /// </summary>
82  /// <param name="parameters">An object containing the portfolio, the security and the delta buying power</param>
83  /// <returns>Returns the maximum allowed market order quantity and if zero, also the reason</returns>
84  public GetMaximumOrderQuantityResult GetMaximumOrderQuantityForDeltaBuyingPower(
85  GetMaximumOrderQuantityForDeltaBuyingPowerParameters parameters)
86  {
87  using (Py.GIL())
88  {
89  return (_model.GetMaximumOrderQuantityForDeltaBuyingPower(parameters)
90  as PyObject).GetAndDispose<GetMaximumOrderQuantityResult>();
91  }
92  }
93

100  {
101  using (Py.GIL())

122  /// <summary>
123  /// Sets the leverage for the applicable securities, i.e, equities
124  /// </summary>
125  /// <remarks>
126  /// This is added to maintain backwards compatibility with the old margin/leverage system
127  /// </remarks>
128  /// <param name="security">The security to set leverage for</param>
