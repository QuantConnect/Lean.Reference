28  /// <summary>
29  /// Supports data from http://TrueData.in
30  /// </summary>
31  public static void NseMarketDataConverter(string sourceDirectory, string destinationDirectory)
32  {
33  //Document the process:
34  Console.WriteLine("QuantConnect.ToolBox: NseMarketData Converter: ");
35  Console.WriteLine("==============================================");
36  Console.WriteLine("The NseMarketData converter transforms NseMarketData orders into the LEAN Algorithmic Trading Engine Data Format.");
37  Console.WriteLine("Parameters required: --source-dir= --destination-dir= ");
38  Console.WriteLine(" 1> Source Directory of Unzipped NSE Data.");
39  Console.WriteLine(" 2> Destination Directory of LEAN Data Folder. (Typically located under Lean/Data)");
40  Console.WriteLine(" ");
41  Console.WriteLine("NOTE: THIS WILL OVERWRITE ANY EXISTING FILES.");
42  if (sourceDirectory.IsNullOrEmpty() || destinationDirectory.IsNullOrEmpty())
43  {
44  Console.WriteLine("1. Source NSE source directory: ");
45  sourceDirectory = (Console.ReadLine() ?? "");
46  Console.WriteLine("2. Destination LEAN Data directory: ");
47  destinationDirectory = (Console.ReadLine() ?? "");
48  }
49 
50  //Validate the user input:
51  Validate(sourceDirectory, destinationDirectory);
52 
53  //Remove the final slash to make the path building easier:
54  sourceDirectory = StripFinalSlash(sourceDirectory);
55  destinationDirectory = StripFinalSlash(destinationDirectory);
56 
57  //iterate over all the dates
58 
59  //Count the total files to process:
60  Console.WriteLine("Counting Files..." + sourceDirectory);
61  var count = 0;
62  var totalCount = GetCount(sourceDirectory);
63  Console.WriteLine("Processing {0} Files ...", totalCount);
64 
65  //Enumerate folders by date as all folders are named as dates
66  foreach (var dateDirectory in Directory.EnumerateDirectories(sourceDirectory))
67  {
68  var date = GetDate(dateDirectory);
69  var dateDirectoryExtension = dateDirectory + "\\1\\Equity";
70  foreach (var file in Directory.EnumerateFiles(dateDirectoryExtension))
71  {
72  var symbol = GetSymbol(file);
73  var fileContents = File.ReadAllText(file);
74  string[] stringSeparators = new string[] { "\n" };
75  string[] lines = fileContents.Split(stringSeparators, StringSplitOptions.None);
76  var datawriter = new LeanDataWriter(Resolution.Minute, symbol, destinationDirectory);
77  IList<TradeBar> fileEnum = new List<TradeBar>();
78  foreach (string line in lines)
79  {
80  string[] separators = new string[] { "," };
81  string[] linearray = line.Split(separators, StringSplitOptions.None);
82  if (linearray.Length > 2)
83  {
84  String newline = linearray[0] + " ";
85  newline += linearray[1];
86  newline += ":00.0000";
87  var Time = Parse.DateTimeExact(newline, DateFormat.Forex);
88  var open = Parse.Decimal(linearray[2]);
89  var high = Parse.Decimal(linearray[3]);
90  var low = Parse.Decimal(linearray[4]);
91  var close = Parse.Decimal(linearray[5]);
92  var volume = linearray[6].LongCount();
93  var linedata = new TradeBar(Time, symbol, open, high, low, close, volume);
94  fileEnum.Add(linedata);
95  }
96  }
97  datawriter.Write(fileEnum);
98  count++;
99  }
100  }
101  Console.ReadKey();
102  }

116  /// <summary>
117  /// Get the count of the files to process
118  /// </summary>
119  /// <param name="sourceDirectory"></param>
120  /// <returns></returns>
121  private static int GetCount(string sourceDirectory)
122  {
123  var count = 0;
124  foreach (var date in Directory.EnumerateDirectories(sourceDirectory))
125  {
126  StripFinalSlash(date);
127  count += Directory.EnumerateFiles(date + "\\1\\Equity\\", "*").Count();
128  }
129  return count;
130  }

132  /// <summary>
133  /// Remove the final slash to make path building easier
134  /// </summary>
135  private static string StripFinalSlash(string directory)
136  {
137  return directory.Trim('/', '\\');
138  }
139 
140  /// <summary>
141  /// Get the date component of tie file path.
142  /// </summary>
143  /// <param name="date"></param>
144  /// <returns></returns>
145  private static DateTime GetDate(string date)
146  {
147  var splits = date.Split('/', '\\');
148  var dateString = splits[splits.Length - 1].Replace("allstocks_", "");
149  return Parse.DateTimeExact(dateString, "yyyyMMdd");
150  }

153  /// <summary>
154  /// Extract the symbol from the path
155  /// </summary>
156  private static Symbol GetSymbol(string filePath)
157  {
158  var splits = filePath.Split('/', '\\');
159  var file = splits[splits.Length - 1];
160  file = file.Trim('.', '/', '\\');
161  //This switch case is for the symbols names with spaces in their file names
162  switch (file)
163  {
164  case "NIFTY GS 4 8YR.csv":
165  file = "NIFTYGS4TO8YR";
166  break;
167  case "NIFTY GS 8 13YR.csv":
168  file = "NIFTYGS8TO13YR";
169  break;
170  case "NIFTY GS 10YR CLN.csv":
171  file = "NIFTYGS10YRCLN";
172  break;
173  case "NIFTY GS 10YR.csv":
174  file = "NIFTYGS10YR";
175  break;
176  case "NIFTY GS 11 15YR.csv":
177  file = "NIFTYGS11TO15YR";
178  break;
179  case "NIFTY GS 15YRPLUS.csv":
180  file = "NIFTYGS15YRPLUS";
181  break;
182  case "NIFTY GS COMPSITE.csv":
183  file = "NIFTYGSCOMPOSITE";
184  break;
185  default:
186  break;
187  }
188  return Symbol.Create(file.Replace(".csv", ""), SecurityType.Equity, Market.India);
189  }

191  /// <summary>
192  /// Validate the users input and throw error if not valid
193  /// </summary>
194  private static void Validate(string sourceDirectory, string destinationDirectory)
195  {
196  if (string.IsNullOrWhiteSpace(sourceDirectory))
197  {
198  Error("Error: Please enter a valid source directory.");
199  }
200  if (string.IsNullOrWhiteSpace(destinationDirectory))
201  {
202  Error("Error: Please enter a valid destination directory.");
203  }
204  if (!Directory.Exists(sourceDirectory))
205  {
206  Error("Error: Source directory does not exist.");
207  }
208  if (!Directory.Exists(destinationDirectory))
209  {
210  Error("Error: Destination directory does not exist.");
211  }
212  }
213  }
