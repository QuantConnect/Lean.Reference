16 using QuantConnect.Data.Market;
17 using QuantConnect.Interfaces;
18 using QuantConnect.Securities;
19 
20 namespace QuantConnect.Algorithm.Framework.Alphas.Analysis.Providers
21 {
22  /// <summary>
23  /// Provides an implementation of <see cref="ISecurityProvider"/> that uses the <see cref="SecurityManager"/>
24  /// to get the price for the specified symbols
25  /// </summary>
26  public class AlgorithmSecurityValuesProvider : ISecurityValuesProvider
27  {
28  private readonly IAlgorithm _algorithm;
29 
30  /// <summary>
31  /// Initializes a new instance of the <see cref="AlgorithmSecurityValuesProvider"/> class
32  /// </summary>
33  /// <param name="algorithm">The wrapped algorithm instance</param>
34  public AlgorithmSecurityValuesProvider(IAlgorithm algorithm)
35  {
36  _algorithm = algorithm;
37  }
38 
39  /// <summary>
40  /// Gets the current values for the specified symbol (price/volatility)
41  /// </summary>
42  /// <param name="symbol">The symbol to get price/volatility for</param>
43  /// <returns>The insight target values for the specified symbol</returns>
44  public SecurityValues GetValues(Symbol symbol)
45  {
46  var security = _algorithm.Securities[symbol];
47  var volume = security.Cache.GetData<TradeBar>()?.Volume ?? 0;
48  return new SecurityValues(symbol, _algorithm.UtcTime, security.Exchange.Hours, security.Price, security.VolatilityModel.Volatility, volume, security.QuoteCurrency.ConversionRate);
49  }
50 
51  /// <summary>
52  /// Gets the current values for all the algorithm securities (price/volatility)
53  /// </summary>
54  /// <returns>The insight target values for all the algorithm securities</returns>
55  public ReadOnlySecurityValuesCollection GetAllValues()
56  {
57  // lets be lazy creating the SecurityValues
58  return new ReadOnlySecurityValuesCollection(
59  symbol =>
60  {
61  var security = _algorithm.Securities[symbol];
62  var volume = security.Cache.GetData<TradeBar>()?.Volume ?? 0;
63  return new SecurityValues(security.Symbol, _algorithm.UtcTime, security.Exchange.Hours, security.Price, security.VolatilityModel.Volatility, volume, security.QuoteCurrency.ConversionRate);
64  });
65  }
66  }
67 }
