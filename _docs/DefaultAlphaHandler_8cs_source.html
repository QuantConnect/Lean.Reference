93  /// <summary>
94  /// Initializes this alpha handler to accept insights from the specified algorithm
95  /// </summary>
96  /// <param name="job">The algorithm job</param>
97  /// <param name="algorithm">The algorithm instance</param>
98  /// <param name="messagingHandler">Handler used for sending insights</param>
99  /// <param name="api">Api instance</param>
100  /// <param name="transactionHandler">Algorithms transaction handler</param>
101  public virtual void Initialize(AlgorithmNodePacket job, IAlgorithm algorithm, IMessagingHandler messagingHandler, IApi api, ITransactionHandler transactionHandler)
102  {
103  // initializing these properties just in case, doesn't hurt to have them populated
104  Job = job;
105  Algorithm = algorithm;
106  MessagingHandler = messagingHandler;
107 
108  _fitnessScore = new FitnessScoreManager();
109  _insights = new List<Insight>();
110  _securityValuesProvider = new AlgorithmSecurityValuesProvider(algorithm);
111 
112  InsightManager = CreateInsightManager();
113 
114  var statistics = new StatisticsInsightManagerExtension(algorithm);
115  RuntimeStatistics = statistics.Statistics;
116  InsightManager.AddExtension(statistics);
117 
118  AddInsightManagerCustomExtensions(statistics);
119 
120  var baseDirectory = Config.Get("results-destination-folder", Directory.GetCurrentDirectory());
121  var directory = Path.Combine(baseDirectory, AlgorithmId);
122  if (!Directory.Exists(directory))
123  {
124  Directory.CreateDirectory(directory);
125  }
126  _alphaResultsPath = Path.Combine(directory, "alpha-results.json");
127 
128  // when insight is generated, take snapshot of securities and place in queue for insight manager to process on alpha thread
129  algorithm.InsightsGenerated += (algo, collection) =>
130  {
131  lock (_insights)
132  {
133  _insights.AddRange(collection.Insights);
134  }
135  };
136  }

148  /// <summary>
149  /// Invoked after the algorithm's Initialize method was called allowing the alpha handler to check
150  /// other things, such as sampling period for backtests
151  /// </summary>
152  /// <param name="algorithm">The algorithm instance</param>
153  public void OnAfterAlgorithmInitialized(IAlgorithm algorithm)
154  {
155  _fitnessScore.Initialize(algorithm);
156  // send date ranges to extensions for initialization -- this data wasn't available when the handler was
157  // initialzied, so we need to invoke it here
158  InsightManager.InitializeExtensionsForRange(algorithm.StartDate, algorithm.EndDate, algorithm.UtcTime);
159 
160  if (LiveMode)
161  {
162  _storeTimer = new Timer(_ => StoreInsights(),
163  null,
164  _storePeriodMs,
165  Timeout.Infinite);
166  }
167  IsActive = true;
168  }

170  /// <summary>
171  /// Performs processing in sync with the algorithm's time loop to provide consisten reading of data
172  /// </summary>
173  public virtual void ProcessSynchronousEvents()
174  {
175  // check the last snap shot time, we may have already produced a snapshot via OnInsightsGenerated
176  if (_lastStepTime != Algorithm.UtcTime)
177  {
178  _lastStepTime = Algorithm.UtcTime;
179  lock (_insights)
180  {
181  InsightManager.Step(_lastStepTime,
182  _securityValuesProvider.GetAllValues(),
183  new GeneratedInsightsCollection(_lastStepTime, _insights.Count == 0 ? Enumerable.Empty<Insight>() : _insights, clone: false));
184  _insights.Clear();
185  }
186  }
187 
188  if (_lastFitnessScoreCalculation.Date != Algorithm.UtcTime.Date)
189  {
190  _lastFitnessScoreCalculation = Algorithm.UtcTime.Date;
191  _fitnessScore.UpdateScores();
192 
193  RuntimeStatistics.FitnessScore = _fitnessScore.FitnessScore;
194  RuntimeStatistics.PortfolioTurnover = _fitnessScore.PortfolioTurnover;
195  RuntimeStatistics.SortinoRatio = _fitnessScore.SortinoRatio;
196  RuntimeStatistics.ReturnOverMaxDrawdown = _fitnessScore.ReturnOverMaxDrawdown;
197  }
198  }

200  /// <summary>
201  /// Stops processing and stores insights
202  /// </summary>
203  public void Exit()
204  {
205  Log.Trace("DefaultAlphaHandler.Exit(): Exiting...");
206 
207  _storeTimer.DisposeSafely();
208  _storeTimer = null;
209 
210  // persist insights at exit
211  StoreInsights();
212 
213  InsightManager?.DisposeSafely();

216 
217  Log.Trace("DefaultAlphaHandler.Exit(): Ended");

220  /// <summary>
221  /// Save insight results to persistent storage
222  /// </summary>
223  /// <remarks>Method called by the storing timer and on exit</remarks>
224  protected virtual void StoreInsights()
225  {
226  try
227  {
228  // avoid reentrancy
229  if (Monitor.TryEnter(_lock))
230  {
231  try
232  {
233  if (InsightManager == null)
234  {
235  // could be null if we are not initialized and exit is called
236  return;
237  }
238  // default save all results to disk and don't remove any from memory
239  // this will result in one file with all of the insights/results in it
240  var insights = InsightManager.AllInsights.OrderBy(insight => insight.GeneratedTimeUtc).ToList();
241  if (insights.Count > 0)
242  {
243  var directory = Directory.GetParent(_alphaResultsPath);
244  if (!directory.Exists)
245  {
246  directory.Create();
247  }
248  File.WriteAllText(_alphaResultsPath, JsonConvert.SerializeObject(insights, Formatting.Indented));
249  }
250  }
251  finally
252  {
253  Monitor.Exit(_lock);
254  }
255  }
256  }
257  finally
258  {
259  try
260  {
261  // restart timer following end of persistence
262  _storeTimer?.Change(Time.GetSecondUnevenWait(_storePeriodMs), Timeout.Infinite);
263  }
264  catch (ObjectDisposedException)
265  {
266  // ignored disposed
267  }
268  }
269  }

281  /// <summary>
282  /// Encapsulates routing finalized insights to the messaging handler
283  /// </summary>
284  protected class AlphaResultPacketSender : IInsightManagerExtension, IDisposable
285  {
286  private readonly Timer _timer;
287  private readonly TimeSpan _interval;
288  private readonly int _maximumQueueLength;
289  private readonly AlgorithmNodePacket _job;
290  private readonly ConcurrentQueue<Insight> _insights;
291  private readonly IMessagingHandler _messagingHandler;
292  private readonly int _maximumNumberOfInsightsPerPacket;
293 
294  /// <summary>
295  /// Initialize a new instance of <see cref="AlphaResultPacketSender"/>
296  /// </summary>
297  /// <param name="job">Algorithm job</param>
298  /// <param name="messagingHandler">Message handler to use</param>
299  /// <param name="interval">Timespan interval</param>
300  /// <param name="maximumNumberOfInsightsPerPacket">Limit on insights per packet</param>
301  public AlphaResultPacketSender(AlgorithmNodePacket job, IMessagingHandler messagingHandler, TimeSpan interval, int maximumNumberOfInsightsPerPacket)
302  {
303  _job = job;
304  _interval = interval;
305  _messagingHandler = messagingHandler;
306  _insights = new ConcurrentQueue<Insight>();
307  _maximumNumberOfInsightsPerPacket = maximumNumberOfInsightsPerPacket;
308 
309  _timer = new Timer(MessagingUpdateIntervalElapsed);
310  _timer.Change(interval, Timeout.InfiniteTimeSpan);
311 
312  // don't bother holding on more than makes sense. this makes the maximum
313  // number of insights we'll hold in the queue equal to one hour's worth of
314  // processing. For 50 insights/message @ 1message/sec this is 90K
315  _maximumQueueLength = (int) (TimeSpan.FromMinutes(30).Ticks / interval.Ticks * maximumNumberOfInsightsPerPacket);
316  }
317 
318  private void MessagingUpdateIntervalElapsed(object state)
319  {
320  try
321  {
322  Insight insight;
323  var insights = new List<Insight>();
324  while (insights.Count < _maximumNumberOfInsightsPerPacket && _insights.TryDequeue(out insight))
325  {
326  insights.Add(insight);
327  }
328 
329  if (insights.Count > 0)
330  {
331  _messagingHandler.Send(new AlphaResultPacket(_job.AlgorithmId, _job.UserId, insights));
332  }
333  }
334  catch (Exception err)
335  {
336  Log.Error(err);
337  }
338  finally
339  {
340  try
341  {
342  var nextDueTime = Time.GetSecondUnevenWait((int)Math.Ceiling(_interval.TotalMilliseconds));
343  _timer.Change(nextDueTime, Timeout.Infinite);
344  }
345  catch (ObjectDisposedException)
346  {
347  // ignored disposed
348  }
349  }
350  }
