16 using Newtonsoft.Json;
17 using QuantConnect.Interfaces;
18 using QuantConnect.Brokerages;
19 
20 namespace QuantConnect
21 {
22  /// <summary>
23  /// This class includes algorithm configuration settings and parameters.
24  /// This is used to include configuration parameters in the result packet to be used for report generation.
25  /// </summary>
26  public class AlgorithmConfiguration
27  {
28  /// <summary>
29  /// The algorithm's account currency
30  /// </summary>
31  [JsonProperty(PropertyName = "AccountCurrency", NullValueHandling = NullValueHandling.Ignore)]
32  public string AccountCurrency;
33 
34  /// <summary>
35  /// The algorithm's brokerage model
36  /// </summary>
37  /// <remarks> Required to set the correct brokerage model on report generation.</remarks>
38  [JsonProperty(PropertyName = "Brokerage")]
39  public BrokerageName BrokerageName;
40 
41  /// <summary>
42  /// The algorithm's account type
43  /// </summary>
44  /// <remarks> Required to set the correct brokerage model on report generation.</remarks>
45  [JsonProperty(PropertyName = "AccountType")]
46  public AccountType AccountType;
47 
48  /// <summary>
49  /// Initializes a new instance of the <see cref="AlgorithmConfiguration"/> class
50  /// </summary>
51  public AlgorithmConfiguration(string accountCurrency, BrokerageName brokerageName, AccountType accountType)
52  {
53  AccountCurrency = accountCurrency;
54  BrokerageName = brokerageName;
55  AccountType = accountType;
56  }
57 
58  /// <summary>
59  /// Initializes a new empty instance of the <see cref="AlgorithmConfiguration"/> class
60  /// </summary>
61  public AlgorithmConfiguration()
62  {
63  }
64 
65  /// <summary>
66  /// Provides a convenience method for creating a <see cref="AlgorithmConfiguration"/> for a given algorithm.
67  /// </summary>
68  /// <param name="algorithm">Algorithm for which the configuration object is being created</param>
69  /// <returns>A new AlgorithmConfiguration object for the specified algorithm</returns>
70  public static AlgorithmConfiguration Create(IAlgorithm algorithm)
71  {
72  return new AlgorithmConfiguration(
73  algorithm.AccountCurrency,
74  BrokerageModel.GetBrokerageName(algorithm.BrokerageModel),
75  algorithm.BrokerageModel.AccountType);
76  }
77  }
78 }
