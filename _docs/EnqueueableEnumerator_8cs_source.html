17 using System;
18 using System.Collections;
19 using System.Collections.Concurrent;
20 using System.Collections.Generic;
21 using System.Threading;
22 
23 namespace QuantConnect.Lean.Engine.DataFeeds.Enumerators
24 {
25  /// <summary>
26  /// An implementation of <see cref="IEnumerator{T}"/> that relies on the
27  /// <see cref="Enqueue"/> method being called and only ends when <see cref="Stop"/>
28  /// is called
29  /// </summary>
30  /// <typeparam name="T">The item type yielded by the enumerator</typeparam>
31  public class EnqueueableEnumerator<T> : IEnumerator<T>
32  {
33  private T _current;
34  private T _lastEnqueued;
35  private volatile bool _end;
36  private volatile bool _disposed;
37 
38  private readonly bool _isBlocking;
39  private readonly int _timeout;
40  private readonly object _lock = new object();
41  private readonly BlockingCollection<T> _blockingCollection;
42 
43  /// <summary>
44  /// Gets the current number of items held in the internal queue
45  /// </summary>
46  public int Count
47  {
48  get
49  {
50  lock (_lock)
51  {
52  if (_end) return 0;
53  return _blockingCollection.Count;
54  }
55  }
56  }
57 
58  /// <summary>
59  /// Gets the last item that was enqueued
60  /// </summary>
61  public T LastEnqueued
62  {
63  get { return _lastEnqueued; }
64  }
65 
66  /// <summary>
67  /// Returns true if the enumerator has finished and will not accept any more data
68  /// </summary>
69  public bool HasFinished
70  {
71  get { return _end || _disposed; }
72  }
73 
74  /// <summary>
75  /// Initializes a new instance of the <see cref="EnqueueableEnumerator{T}"/> class
76  /// </summary>
77  /// <param name="blocking">Specifies whether or not to use the blocking behavior</param>
78  public EnqueueableEnumerator(bool blocking = false)
79  {
80  _blockingCollection = new BlockingCollection<T>();
81  _isBlocking = blocking;
82  _timeout = blocking ? Timeout.Infinite : 0;
83  }
84 
85  /// <summary>
86  /// Enqueues the new data into this enumerator
87  /// </summary>
88  /// <param name="data">The data to be enqueued</param>
89  public void Enqueue(T data)
90  {
91  lock (_lock)
92  {
93  if (_end) return;
94  _blockingCollection.Add(data);
95  _lastEnqueued = data;
96  }
97  }
98 
99  /// <summary>
100  /// Signals the enumerator to stop enumerating when the items currently
101  /// held inside are gone. No more items will be added to this enumerator.
102  /// </summary>
103  public void Stop()
104  {
105  lock (_lock)
106  {
107  if (_end) return;
108  // no more items can be added, so no need to wait anymore
109  _blockingCollection.CompleteAdding();
110  _end = true;
111  }
112  }
113 
114  /// <summary>
115  /// Advances the enumerator to the next element of the collection.
116  /// </summary>
117  /// <returns>
118  /// true if the enumerator was successfully advanced to the next element; false if the enumerator has passed the end of the collection.
119  /// </returns>
120  /// <exception cref="T:System.InvalidOperationException">The collection was modified after the enumerator was created. </exception><filterpriority>2</filterpriority>
121  public bool MoveNext()
122  {
123  T current;
124  if (!_blockingCollection.TryTake(out current, _timeout))
125  {
126  _current = default(T);
127 
128  // if the enumerator has blocking behavior and there is no more data, it has ended
129  if (_isBlocking)
130  {
131  lock (_lock)
132  {
133  _end = true;
134  }
135  }
136 
137  return !_end;
138  }
139 
140  _current = current;
141 
142  // even if we don't have data to return, we haven't technically
143  // passed the end of the collection, so always return true until
144  // the enumerator is explicitly disposed or ended
145  return true;
146  }
147 
148  /// <summary>
149  /// Sets the enumerator to its initial position, which is before the first element in the collection.
150  /// </summary>
151  /// <exception cref="T:System.InvalidOperationException">The collection was modified after the enumerator was created. </exception><filterpriority>2</filterpriority>
152  public void Reset()
153  {
154  throw new NotImplementedException("EnqueableEnumerator.Reset() has not been implemented yet.");
155  }
156 
157  /// <summary>
158  /// Gets the element in the collection at the current position of the enumerator.
159  /// </summary>
160  /// <returns>
161  /// The element in the collection at the current position of the enumerator.
162  /// </returns>
163  public T Current
164  {
165  get { return _current; }
166  }
167 
168  /// <summary>
169  /// Gets the current element in the collection.
170  /// </summary>
171  /// <returns>
172  /// The current element in the collection.
173  /// </returns>
174  /// <filterpriority>2</filterpriority>
175  object IEnumerator.Current
176  {
177  get { return Current; }
178  }
179 
180  /// <summary>
181  /// Performs application-defined tasks associated with freeing, releasing, or resetting unmanaged resources.
182  /// </summary>
183  /// <filterpriority>2</filterpriority>
184  public void Dispose()
185  {
186  lock (_lock)
187  {
188  if (_disposed) return;
189  Stop();
190  if (_blockingCollection != null) _blockingCollection.Dispose();
191  _disposed = true;
192  }
193  }
194  }
195 }
