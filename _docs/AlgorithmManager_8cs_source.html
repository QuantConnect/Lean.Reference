43  /// <summary>
44  /// Algorithm manager class executes the algorithm and generates and passes through the algorithm events.
45  /// </summary>
46  public class AlgorithmManager
47  {
48  private IAlgorithm _algorithm;
49  private readonly object _lock;
50  private readonly bool _liveMode;
51 
52  /// <summary>
53  /// Publicly accessible algorithm status
54  /// </summary>
55  public AlgorithmStatus State => _algorithm?.Status ?? AlgorithmStatus.Running;
56 
57  /// <summary>
58  /// Public access to the currently running algorithm id.
59  /// </summary>
60  public string AlgorithmId { get; private set; }
61 
62  /// <summary>
63  /// Provides the isolator with a function for verifying that we're not spending too much time in each
64  /// algorithm manager time loop
65  /// </summary>
66  public AlgorithmTimeLimitManager TimeLimit { get; }
67 
68  /// <summary>
69  /// Quit state flag for the running algorithm. When true the user has requested the backtest stops through a Quit() method.
70  /// </summary>
71  /// <seealso cref="QCAlgorithm.Quit(String)"/>
72  public bool QuitState => State == AlgorithmStatus.Deleted;
73 
74  /// <summary>
75  /// Gets the number of data points processed per second
76  /// </summary>
77  public long DataPoints { get; private set; }
78 
79  /// <summary>
80  /// Gets the number of data points of algorithm history provider
81  /// </summary>
82  public int AlgorithmHistoryDataPoints => _algorithm?.HistoryProvider?.DataPointCount ?? 0;
83 
84  /// <summary>
85  /// Initializes a new instance of the <see cref="AlgorithmManager"/> class
86  /// </summary>
87  /// <param name="liveMode">True if we're running in live mode, false for backtest mode</param>
88  /// <param name="job">Provided by LEAN when creating a new algo manager. This is the job
89  /// that the algo manager is about to execute. Research and other consumers can provide the
90  /// default value of null</param>
91  public AlgorithmManager(bool liveMode, AlgorithmNodePacket job = null)
92  {
93  AlgorithmId = "";
94  _liveMode = liveMode;
95  _lock = new object();
96 
97  // initialize the time limit manager
98  TimeLimit = new AlgorithmTimeLimitManager(
99  CreateTokenBucket(job?.Controls?.TrainingLimits),
100  TimeSpan.FromMinutes(Config.GetDouble("algorithm-manager-time-loop-maximum", 20))
101  );
102  }
103 
104  /// <summary>
105  /// Launch the algorithm manager to run this strategy
106  /// </summary>
107  /// <param name="job">Algorithm job</param>
108  /// <param name="algorithm">Algorithm instance</param>
109  /// <param name="synchronizer">Instance which implements <see cref="ISynchronizer"/>. Used to stream the data</param>
110  /// <param name="transactions">Transaction manager object</param>
111  /// <param name="results">Result handler object</param>
112  /// <param name="realtime">Realtime processing object</param>
113  /// <param name="leanManager">ILeanManager implementation that is updated periodically with the IAlgorithm instance</param>
114  /// <param name="token">Cancellation token</param>
115  /// <remarks>Modify with caution</remarks>
116  public void Run(AlgorithmNodePacket job, IAlgorithm algorithm, ISynchronizer synchronizer, ITransactionHandler transactions, IResultHandler results, IRealTimeHandler realtime, ILeanManager leanManager, CancellationToken token)
117  {
118  //Initialize:
119  DataPoints = 0;
120  _algorithm = algorithm;
121 
122  var backtestMode = (job.Type == PacketType.BacktestNode);
123  var methodInvokers = new Dictionary<Type, MethodInvoker>();
124  var marginCallFrequency = TimeSpan.FromMinutes(5);
125  var nextMarginCallTime = DateTime.MinValue;
126  var nextSecurityModelScan = algorithm.UtcTime.RoundDown(Time.OneHour) + Time.OneHour;
127  var time = algorithm.StartDate.Date;
128 
129  var pendingDelistings = new List<Delisting>();
130  var splitWarnings = new List<Split>();
131 
132  //Initialize Properties:
133  AlgorithmId = job.AlgorithmId;
134 
135  //Create the method accessors to push generic types into algorithm: Find all OnData events:
136 
137  // Algorithm 2.0 data accessors
138  var hasOnDataTradeBars = AddMethodInvoker<TradeBars>(algorithm, methodInvokers);
139  var hasOnDataQuoteBars = AddMethodInvoker<QuoteBars>(algorithm, methodInvokers);
140  var hasOnDataOptionChains = AddMethodInvoker<OptionChains>(algorithm, methodInvokers);
141  var hasOnDataTicks = AddMethodInvoker<Ticks>(algorithm, methodInvokers);
142 
143  // dividend and split events
144  var hasOnDataDividends = AddMethodInvoker<Dividends>(algorithm, methodInvokers);
145  var hasOnDataSplits = AddMethodInvoker<Splits>(algorithm, methodInvokers);
146  var hasOnDataDelistings = AddMethodInvoker<Delistings>(algorithm, methodInvokers);
147  var hasOnDataSymbolChangedEvents = AddMethodInvoker<SymbolChangedEvents>(algorithm, methodInvokers);
148 
149  //Go through the subscription types and create invokers to trigger the event handlers for each custom type:
150  foreach (var config in algorithm.SubscriptionManager.Subscriptions)
151  {
152  //If type is a custom feed, check for a dedicated event handler
153  if (config.IsCustomData)
154  {
155  //Get the matching method for this event handler - e.g. public void OnData(Quandl data) { .. }
156  var genericMethod = (algorithm.GetType()).GetMethod("OnData", new[] { config.Type });
157 
158  //If we already have this Type-handler then don't add it to invokers again.
159  if (methodInvokers.ContainsKey(config.Type)) continue;
160 
161  if (genericMethod != null)
162  {
163  methodInvokers.Add(config.Type, genericMethod.DelegateForCallMethod());
164  }
165  }
166  }
167 
168  // Schedule a daily event for sampling at midnight every night
169  algorithm.Schedule.On("Daily Sampling", algorithm.Schedule.DateRules.EveryDay(),
170  algorithm.Schedule.TimeRules.Midnight, () =>
171  {
172  results.Sample(algorithm.UtcTime);
173  });
174 
175  //Loop over the queues: get a data collection, then pass them all into relevent methods in the algorithm.
176  Log.Trace($"AlgorithmManager.Run(): Begin DataStream - Start: {algorithm.StartDate} Stop: {algorithm.EndDate} Time: {algorithm.Time} Warmup: {algorithm.IsWarmingUp}");
177  foreach (var timeSlice in Stream(algorithm, synchronizer, results, token))
178  {
179  // reset our timer on each loop
180  TimeLimit.StartNewTimeStep();
181 
182  //Check this backtest is still running:
183  if (_algorithm.Status != AlgorithmStatus.Running && _algorithm.RunTimeError == null)
184  {
185  Log.Error($"AlgorithmManager.Run(): Algorithm state changed to {_algorithm.Status} at {timeSlice.Time.ToStringInvariant()}");
186  break;
187  }
188 
189  //Execute with TimeLimit Monitor:
190  if (token.IsCancellationRequested)
191  {
192  Log.Error($"AlgorithmManager.Run(): CancellationRequestion at {timeSlice.Time.ToStringInvariant()}");
193  return;
194  }
195 
196  // Update the ILeanManager
197  leanManager.Update();
198 
199  time = timeSlice.Time;
200  DataPoints += timeSlice.DataPointCount;
201 
202  if (backtestMode && algorithm.Portfolio.TotalPortfolioValue <= 0)
203  {
204  var logMessage = "AlgorithmManager.Run(): Portfolio value is less than or equal to zero, stopping algorithm.";
205  Log.Error(logMessage);
206  results.SystemDebugMessage(logMessage);
207  break;
208  }
209 
210  // If backtesting/warmup, we need to check if there are realtime events in the past
211  // which didn't fire because at the scheduled times there was no data (i.e. markets closed)
212  // and fire them with the correct date/time.
213  realtime.ScanPastEvents(time);
214 
215  // will scan registered consolidators for which we've past the expected scan call
216  algorithm.SubscriptionManager.ScanPastConsolidators(time, algorithm);
217 
218  //Set the algorithm and real time handler's time
219  algorithm.SetDateTime(time);
220 
221  // the time pulse are just to advance algorithm time, lets shortcut the loop here
222  if (timeSlice.IsTimePulse)
223  {
224  continue;
225  }
226 
227  // Update the current slice before firing scheduled events or any other task
228  algorithm.SetCurrentSlice(timeSlice.Slice);
229 
230  if (timeSlice.Slice.SymbolChangedEvents.Count != 0)
231  {
232  if (hasOnDataSymbolChangedEvents)
233  {
234  methodInvokers[typeof(SymbolChangedEvents)](algorithm, timeSlice.Slice.SymbolChangedEvents);
235  }
236  foreach (var symbol in timeSlice.Slice.SymbolChangedEvents.Keys)
237  {
238  // cancel all orders for the old symbol
239  foreach (var ticket in transactions.GetOpenOrderTickets(x => x.Symbol == symbol))
240  {
241  ticket.Cancel("Open order cancelled on symbol changed event");
242  }
243  }
244  }
245 
246  if (timeSlice.SecurityChanges != SecurityChanges.None)
247  {
248  algorithm.ProcessSecurityChanges(timeSlice.SecurityChanges);
249 
250  leanManager.OnSecuritiesChanged(timeSlice.SecurityChanges);
251  realtime.OnSecuritiesChanged(timeSlice.SecurityChanges);
252  results.OnSecuritiesChanged(timeSlice.SecurityChanges);
253  }
254 
255  //Update the securities properties: first before calling user code to avoid issues with data
256  foreach (var update in timeSlice.SecuritiesUpdateData)
257  {
258  var security = update.Target;
259 
260  security.Update(update.Data, update.DataType, update.ContainsFillForwardData);
261 
262  // Send market price updates to the TradeBuilder
263  algorithm.TradeBuilder.SetMarketPrice(security.Symbol, security.Price);
264  }
265 
266  // TODO: potentially push into a scheduled event
267  if (time >= nextSecurityModelScan)
268  {
269  foreach (var security in algorithm.Securities.Values)
270  {
271  security.MarginInterestRateModel.ApplyMarginInterestRate(new MarginInterestRateParameters(security, time));
272 
273  // perform check for settlement of unsettled funds
274  security.SettlementModel.Scan(new ScanSettlementModelParameters(algorithm.Portfolio, security, time));
275  }
276  nextSecurityModelScan = time.RoundDown(Time.OneHour) + Time.OneHour;
277  }
278 
279  //Update the securities properties with any universe data
280  if (timeSlice.UniverseData.Count > 0)
281  {
282  foreach (var dataCollection in timeSlice.UniverseData.Values)
283  {
284  if (!dataCollection.ShouldCacheToSecurity()) continue;
285 
286  foreach (var data in dataCollection.Data)
287  {
288  if (algorithm.Securities.TryGetValue(data.Symbol, out var security))
289  {
290  security.Cache.StoreData(new[] { data }, data.GetType());
291  }
292  }
293  }
294  }
295 
296  // poke each cash object to update from the recent security data
297  foreach (var cash in algorithm.Portfolio.CashBook.Values.Where(x => x.CurrencyConversion != null))
298  {
299  cash.Update();
300  }
301 
302  // security prices got updated
303  algorithm.Portfolio.InvalidateTotalPortfolioValue();
304 
305  // process fill models on the updated data before entering algorithm, applies to all non-market orders
306  transactions.ProcessSynchronousEvents();
307 
308  // fire real time events after we've updated based on the new data
309  realtime.SetTime(timeSlice.Time);
310 
311  // process split warnings for options
312  ProcessSplitSymbols(algorithm, splitWarnings, pendingDelistings);
313 
314  //Check if the user's signalled Quit: loop over data until day changes.
315  if (_algorithm.Status != AlgorithmStatus.Running && _algorithm.RunTimeError == null)
316  {
317  Log.Error($"AlgorithmManager.Run(): Algorithm state changed to {_algorithm.Status} at {timeSlice.Time.ToStringInvariant()}");
318  break;
319  }
320  if (algorithm.RunTimeError != null)
321  {
322  Log.Error($"AlgorithmManager.Run(): Stopping, encountered a runtime error at {algorithm.UtcTime} UTC.");
323  return;
324  }
325 
326  // perform margin calls, in live mode we can also use realtime to emit these
327  if (time >= nextMarginCallTime || (_liveMode && nextMarginCallTime > DateTime.UtcNow))
328  {
329  // determine if there are possible margin call orders to be executed
330  bool issueMarginCallWarning;
331  var marginCallOrders = algorithm.Portfolio.MarginCallModel.GetMarginCallOrders(out issueMarginCallWarning);
332  if (marginCallOrders.Count != 0)
333  {
334  var executingMarginCall = false;
335  try
336  {
337  // tell the algorithm we're about to issue the margin call
338  algorithm.OnMarginCall(marginCallOrders);
339 
340  executingMarginCall = true;
341 
342  // execute the margin call orders
343  var executedTickets = algorithm.Portfolio.MarginCallModel.ExecuteMarginCall(marginCallOrders);
344  foreach (var ticket in executedTickets)
345  {
346  algorithm.Error($"{algorithm.Time.ToStringInvariant()} - Executed MarginCallOrder: {ticket.Symbol} - " +
347  $"Quantity: {ticket.Quantity.ToStringInvariant()} @ {ticket.AverageFillPrice.ToStringInvariant()}"
348  );
349  }
350  }
351  catch (Exception err)
352  {
353  algorithm.SetRuntimeError(err, executingMarginCall ? "Portfolio.MarginCallModel.ExecuteMarginCall" : "OnMarginCall");
354  return;
355  }
356  }
357  // we didn't perform a margin call, but got the warning flag back, so issue the warning to the algorithm
358  else if (issueMarginCallWarning)
359  {
360  try
361  {
362  algorithm.OnMarginCallWarning();
363  }
364  catch (Exception err)
365  {
366  algorithm.SetRuntimeError(err, "OnMarginCallWarning");
367  return;
368  }
369  }
370 
371  nextMarginCallTime = time + marginCallFrequency;
372  }
373 
374  // before we call any events, let the algorithm know about universe changes
375  if (timeSlice.SecurityChanges != SecurityChanges.None)
376  {
377  try
378  {
379  var algorithmSecurityChanges = new SecurityChanges(timeSlice.SecurityChanges)
380  {
381  // by default for user code we want to filter out custom securities
382  FilterCustomSecurities = true,
383  // by default for user code we want to filter out internal securities
384  FilterInternalSecurities = true
385  };
386 
387  algorithm.OnSecuritiesChanged(algorithmSecurityChanges);
388  algorithm.OnFrameworkSecuritiesChanged(algorithmSecurityChanges);
389  }
390  catch (Exception err)
391  {
392  algorithm.SetRuntimeError(err, "OnSecuritiesChanged");
393  return;
394  }
395  }
396 
397  // apply dividends
398  HandleDividends(timeSlice, algorithm, _liveMode);
399 
400  // apply splits
401  HandleSplits(timeSlice, algorithm, _liveMode);
402 
403  //Update registered consolidators for this symbol index
404  try
405  {
406  if (timeSlice.ConsolidatorUpdateData.Count > 0)
407  {
408  var timeKeeper = algorithm.TimeKeeper;
409  foreach (var update in timeSlice.ConsolidatorUpdateData)
410  {
411  var localTime = timeKeeper.GetLocalTimeKeeper(update.Target.ExchangeTimeZone).LocalTime;
412  var consolidators = update.Target.Consolidators;
413  foreach (var consolidator in consolidators)
414  {
415  foreach (var dataPoint in update.Data)
416  {
417  // only push data into consolidators on the native, subscribed to resolution
418  if (EndTimeIsInNativeResolution(update.Target, dataPoint.EndTime))
419  {
420  consolidator.Update(dataPoint);
421  }
422  }
423 
424  // scan for time after we've pumped all the data through for this consolidator
425  consolidator.Scan(localTime);
426  }
427  }
428  }
429  }
430  catch (Exception err)
431  {
432  algorithm.SetRuntimeError(err, "Consolidators update");
433  return;
434  }
435 
436  // fire custom event handlers
437  foreach (var update in timeSlice.CustomData)
438  {
439  MethodInvoker methodInvoker;
440  if (!methodInvokers.TryGetValue(update.DataType, out methodInvoker))
441  {
442  continue;
443  }
444 
445  try
446  {
447  foreach (var dataPoint in update.Data)
448  {
449  if (update.DataType.IsInstanceOfType(dataPoint))
450  {
451  methodInvoker(algorithm, dataPoint);
452  }
453  }
454  }
455  catch (Exception err)
456  {
457  algorithm.SetRuntimeError(err, "Custom Data");
458  return;
459  }
460  }
461 
462  try
463  {
464  // fire off the dividend and split events before pricing events
465  if (hasOnDataDividends && timeSlice.Slice.Dividends.Count != 0)
466  {
467  methodInvokers[typeof(Dividends)](algorithm, timeSlice.Slice.Dividends);
468  }
469  if (hasOnDataSplits && timeSlice.Slice.Splits.Count != 0)
470  {
471  methodInvokers[typeof(Splits)](algorithm, timeSlice.Slice.Splits);
472  }
473  if (hasOnDataDelistings && timeSlice.Slice.Delistings.Count != 0)
474  {
475  methodInvokers[typeof(Delistings)](algorithm, timeSlice.Slice.Delistings);
476  }
477  }
478  catch (Exception err)
479  {
480  algorithm.SetRuntimeError(err, "Dividends/Splits/Delistings");
481  return;
482  }
483 
484  // Only track pending delistings in non-live mode.
485  if (!algorithm.LiveMode)
486  {
487  // Keep this up to date even though we don't process delistings here anymore
488  foreach (var delisting in timeSlice.Slice.Delistings.Values)
489  {
490  if (delisting.Type == DelistingType.Warning)
491  {
492  // Store our delistings warnings because they are still used by ProcessSplitSymbols above
493  pendingDelistings.Add(delisting);
494  }
495  else
496  {
497  // If we have an actual delisting event, remove it from pending delistings
498  var index = pendingDelistings.FindIndex(x => x.Symbol == delisting.Symbol);
499  if (index != -1)
500  {
501  pendingDelistings.RemoveAt(index);
502  }
503  }
504  }
505  }
506 
507  // run split logic after firing split events
508  HandleSplitSymbols(timeSlice.Slice.Splits, splitWarnings);
509 
510  //After we've fired all other events in this second, fire the pricing events:
511  try
512  {
513  if (hasOnDataTradeBars && timeSlice.Slice.Bars.Count > 0) methodInvokers[typeof(TradeBars)](algorithm, timeSlice.Slice.Bars);
514  if (hasOnDataQuoteBars && timeSlice.Slice.QuoteBars.Count > 0) methodInvokers[typeof(QuoteBars)](algorithm, timeSlice.Slice.QuoteBars);
515  if (hasOnDataOptionChains && timeSlice.Slice.OptionChains.Count > 0) methodInvokers[typeof(OptionChains)](algorithm, timeSlice.Slice.OptionChains);
516  if (hasOnDataTicks && timeSlice.Slice.Ticks.Count > 0) methodInvokers[typeof(Ticks)](algorithm, timeSlice.Slice.Ticks);
517  }
518  catch (Exception err)
519  {
520  algorithm.SetRuntimeError(err, "methodInvokers");
521  return;
522  }
523 
524  try
525  {
526  if (timeSlice.Slice.HasData)
527  {
528  // EVENT HANDLER v3.0 -- all data in a single event
529  algorithm.OnData(timeSlice.Slice);
530  }
531 
532  // always turn the crank on this method to ensure universe selection models function properly on day changes w/out data
533  algorithm.OnFrameworkData(timeSlice.Slice);
534  }
535  catch (Exception err)
536  {
537  algorithm.SetRuntimeError(err, "OnData");
538  return;
539  }
540 
541  //If its the historical/paper trading models, wait until market orders have been "filled"
542  // Manually trigger the event handler to prevent thread switch.
543  transactions.ProcessSynchronousEvents();
544 
545  // Process any required events of the results handler such as sampling assets, equity, or stock prices.
546  results.ProcessSynchronousEvents();
547 
548  // poke the algorithm at the end of each time step
549  algorithm.OnEndOfTimeStep();
550 
551  } // End of ForEach feed.Bridge.GetConsumingEnumerable
552 
553  // stop timing the loops
554  TimeLimit.StopEnforcingTimeLimit();
555 
556  //Stream over:: Send the final packet and fire final events:
557  Log.Trace("AlgorithmManager.Run(): Firing On End Of Algorithm...");
558  try
559  {
560  algorithm.OnEndOfAlgorithm();
561  }
562  catch (Exception err)
563  {
564  algorithm.SetRuntimeError(err, "OnEndOfAlgorithm");
565  return;
566  }
567 
568  // Process any required events of the results handler such as sampling assets, equity, or stock prices.
569  results.ProcessSynchronousEvents(forceProcess: true);
570 
571  //Liquidate Holdings for Calculations:
572  if (_algorithm.Status == AlgorithmStatus.Liquidated && _liveMode)
573  {
574  Log.Trace("AlgorithmManager.Run(): Liquidating algorithm holdings...");
575  algorithm.Liquidate();
576  results.LogMessage("Algorithm Liquidated");
577  results.SendStatusUpdate(AlgorithmStatus.Liquidated);
578  }
579 
580  //Manually stopped the algorithm
581  if (_algorithm.Status == AlgorithmStatus.Stopped)
582  {
583  Log.Trace("AlgorithmManager.Run(): Stopping algorithm...");
584  results.LogMessage("Algorithm Stopped");
585  results.SendStatusUpdate(AlgorithmStatus.Stopped);
586  }
587 
588  //Backtest deleted.
589  if (_algorithm.Status == AlgorithmStatus.Deleted)
590  {
591  Log.Trace("AlgorithmManager.Run(): Deleting algorithm...");
592  results.DebugMessage("Algorithm Id:(" + job.AlgorithmId + ") Deleted by request.");
593  results.SendStatusUpdate(AlgorithmStatus.Deleted);
594  }
595 
596  //Algorithm finished, send regardless of commands:
597  results.SendStatusUpdate(AlgorithmStatus.Completed);
598  SetStatus(AlgorithmStatus.Completed);
599 
600  //Take final samples:
601  results.Sample(time);
602 
603  } // End of Run();

605  /// <summary>
606  /// Set the quit state.
607  /// </summary>
608  public void SetStatus(AlgorithmStatus state)
609  {
610  lock (_lock)
611  {
612  //We don't want anyone else to set our internal state to "Running".
613  //This is controlled by the algorithm private variable only.
614  //Algorithm could be null after it's initialized and they call Run on us
615  if (state != AlgorithmStatus.Running && _algorithm != null)
616  {
617  _algorithm.SetStatus(state);
618  }
619  }
620  }

622  private IEnumerable<TimeSlice> Stream(IAlgorithm algorithm, ISynchronizer synchronizer, IResultHandler results, CancellationToken cancellationToken)
623  {
624  var nextWarmupStatusTime = DateTime.MinValue;
625  var warmingUp = algorithm.IsWarmingUp;
626  var warmingUpPercent = 0;
627  if (warmingUp)
628  {
629  nextWarmupStatusTime = DateTime.UtcNow.AddSeconds(1);
630  algorithm.Debug("Algorithm starting warm up...");
631  results.SendStatusUpdate(AlgorithmStatus.History, $"{warmingUpPercent}");
632  }
633  else
634  {
635  results.SendStatusUpdate(AlgorithmStatus.Running);
636  // let's be polite, and call warmup finished even though there was no warmup period and avoid algorithms having to handle it instead.
637  // we trigger this callback here and not internally in the algorithm so that we can go through python if required
638  algorithm.OnWarmupFinished();
639  }
640 
641  // bellow we compare with slice.Time which is in UTC
642  var startTimeTicks = algorithm.UtcTime.Ticks;
643  var warmupEndTicks = algorithm.StartDate.ConvertToUtc(algorithm.TimeZone).Ticks;
644 
645  // fulfilling history requirements of volatility models in live mode
646  if (algorithm.LiveMode)
647  {
648  warmupEndTicks = DateTime.UtcNow.Ticks;
649  ProcessVolatilityHistoryRequirements(algorithm, _liveMode);
650  }
651 
652  foreach (var timeSlice in synchronizer.StreamData(cancellationToken))
653  {
654  if (algorithm.IsWarmingUp)
655  {
656  var now = DateTime.UtcNow;
657  if (now > nextWarmupStatusTime)
658  {
659  // send some status to the user letting them know we're done history, but still warming up,
660  // catching up to real time data
661  nextWarmupStatusTime = now.AddSeconds(2);
662  var newPercent = (int)(100 * (timeSlice.Time.Ticks - startTimeTicks) / (double)(warmupEndTicks - startTimeTicks));
663  // if there isn't any progress don't send the same update many times
664  if (newPercent != warmingUpPercent)
665  {
666  warmingUpPercent = newPercent;
667  algorithm.Debug($"Processing algorithm warm-up request {warmingUpPercent}%...");
668  results.SendStatusUpdate(AlgorithmStatus.History, $"{warmingUpPercent}");
669  }
670  }
671  }
672  else if (warmingUp)
673  {
674  // warmup finished, send an update
675  warmingUp = false;
676  // we trigger this callback here and not internally in the algorithm so that we can go through python if required
677  algorithm.OnWarmupFinished();
678  algorithm.Debug("Algorithm finished warming up.");
679  results.SendStatusUpdate(AlgorithmStatus.Running, "100");
680  }
681  yield return timeSlice;
682  }
683  }

685  /// <summary>

704  /// <summary>
705  /// Helper method to apply a split to an algorithm instance
706  /// </summary>
707  public static void HandleSplits(TimeSlice timeSlice, IAlgorithm algorithm, bool liveMode)
708  {
709  foreach (var split in timeSlice.Slice.Splits.Values)
710  {
711  try
712  {
713  // only process split occurred events (ignore warnings)
714  if (split.Type != SplitType.SplitOccurred)
715  {
716  continue;
717  }
718 
719  if (liveMode && algorithm.IsWarmingUp)
720  {
721  // skip past split during live warmup, the algorithms position already reflects them
722  Log.Trace($"AlgorithmManager.Run(): {algorithm.Time}: Skip Split during live warmup: {split}");
723  continue;
724  }
725 
726  if (Log.DebuggingEnabled)
727  {
728  Log.Debug($"AlgorithmManager.Run(): {algorithm.Time}: Applying Split for {split.Symbol}");
729  }
730 
731  Security security = null;
732  if (algorithm.Securities.TryGetValue(split.Symbol, out security) && liveMode)
733  {
734  Log.Trace($"AlgorithmManager.Run(): {algorithm.Time}: Pre-Split for {split}. Security Price: {security.Price} Holdings: {security.Holdings.Quantity}");
735  }
736 
737  var mode = algorithm.SubscriptionManager.SubscriptionDataConfigService
738  .GetSubscriptionDataConfigs(split.Symbol)
739  .DataNormalizationMode();
740 
741  // apply the split event to the portfolio
742  algorithm.Portfolio.ApplySplit(split, security, liveMode, mode);
743 
744  // apply the split event to the trade builder
745  algorithm.TradeBuilder.ApplySplit(split, liveMode, mode);
746 
747  // apply the split event to the security volatility model
748  ApplySplitOrDividendToVolatilityModel(algorithm, security, liveMode, mode);
749 
750  if (liveMode && security != null)
751  {
752  Log.Trace($"AlgorithmManager.Run(): {algorithm.Time}: Post-Split for {split}. Security Price: {security.Price} Holdings: {security.Holdings.Quantity}");
753  }
754 
755  // apply the split to open orders as well in raw mode, all other modes are split adjusted
756  if (liveMode || mode == DataNormalizationMode.Raw)
757  {
758  // in live mode we always want to have our order match the order at the brokerage, so apply the split to the orders
759  var openOrders = algorithm.Transactions.GetOpenOrderTickets(ticket => ticket.Symbol == split.Symbol);
760  algorithm.BrokerageModel.ApplySplit(openOrders.ToList(), split);
761  }
762  }
763  catch (Exception err)
764  {
765  algorithm.SetRuntimeError(err, "Split event");
766  return;
767  }
768  }
769  }

864  /// <summary>
865  /// Liquidate option contact holdings who's underlying security has split
866  /// </summary>
867  private void ProcessSplitSymbols(IAlgorithm algorithm, List<Split> splitWarnings, List<Delisting> pendingDelistings)
868  {
869  // NOTE: This method assumes option contracts have the same core trading hours as their underlying contract
870  // This is a small performance optimization to prevent scanning every contract on every time step,
871  // instead we scan just the underlyings, thereby reducing the time footprint of this methods by a factor
872  // of N, the number of derivative subscriptions
873  for (int i = splitWarnings.Count - 1; i >= 0; i--)
874  {
875  var split = splitWarnings[i];
876  var security = algorithm.Securities[split.Symbol];
877 
878  if (!security.IsTradable
879  && !algorithm.UniverseManager.ActiveSecurities.Keys.Contains(split.Symbol))
880  {
881  Log.Debug($"AlgorithmManager.ProcessSplitSymbols(): {_algorithm.Time} - Removing split warning for {security.Symbol}");
882 
883  // remove the warning from out list
884  splitWarnings.RemoveAt(i);
885  // Since we are storing the split warnings for a loop
886  // we need to check if the security was removed.
887  // When removed, it will be marked as non tradable but just in case
888  // we expect it not to be an active security either
889  continue;
890  }
891 
892  var nextMarketClose = security.Exchange.Hours.GetNextMarketClose(security.LocalTime, false);
893 
894  // determine the latest possible time we can submit a MOC order
895  var configs = algorithm.SubscriptionManager.SubscriptionDataConfigService
896  .GetSubscriptionDataConfigs(security.Symbol);
897 
898  if (configs.Count == 0)
899  {
900  // should never happen at this point, if it does let's give some extra info
901  throw new Exception(
902  $"AlgorithmManager.ProcessSplitSymbols(): {_algorithm.Time} - No subscriptions found for {security.Symbol}" +
903  $", IsTradable: {security.IsTradable}" +
904  $", Active: {algorithm.UniverseManager.ActiveSecurities.Keys.Contains(split.Symbol)}");
905  }
906 
907  var latestMarketOnCloseTimeRoundedDownByResolution = nextMarketClose.Subtract(MarketOnCloseOrder.SubmissionTimeBuffer)
908  .RoundDownInTimeZone(configs.GetHighestResolution().ToTimeSpan(), security.Exchange.TimeZone, configs.First().DataTimeZone);
909 
910  // we don't need to do anyhing until the market closes
911  if (security.LocalTime < latestMarketOnCloseTimeRoundedDownByResolution) continue;
912 
913  // fetch all option derivatives of the underlying with holdings (excluding the canonical security)
914  var derivatives = algorithm.Securities.Values.Where(potentialDerivate =>
915  potentialDerivate.Symbol.SecurityType.IsOption() &&
916  potentialDerivate.Symbol.Underlying == security.Symbol &&
917  !potentialDerivate.Symbol.Underlying.IsCanonical() &&
918  potentialDerivate.HoldStock
919  );
920 
921  foreach (var derivative in derivatives)
922  {
923  var optionContractSymbol = derivative.Symbol;
924  var optionContractSecurity = (Option)derivative;
925 
926  if (pendingDelistings.Any(x => x.Symbol == optionContractSymbol
927  && x.Time.Date == optionContractSecurity.LocalTime.Date))
928  {
929  // if the option is going to be delisted today we skip sending the market on close order
930  continue;
931  }
932 
933  // close any open orders
934  algorithm.Transactions.CancelOpenOrders(optionContractSymbol, "Canceled due to impending split. Separate MarketOnClose order submitted to liquidate position.");
935 
936  var request = new SubmitOrderRequest(OrderType.MarketOnClose, optionContractSecurity.Type, optionContractSymbol,
937  -optionContractSecurity.Holdings.Quantity, 0, 0, algorithm.UtcTime,
938  "Liquidated due to impending split. Option splits are not currently supported."
939  );
940 
941  // send MOC order to liquidate option contract holdings
942  algorithm.Transactions.AddOrder(request);
943 
944  // mark option contract as not tradable
945  optionContractSecurity.IsTradable = false;
946 
947  algorithm.Debug($"MarketOnClose order submitted for option contract '{optionContractSymbol}' due to impending {split.Symbol.Value} split event. "
948  + "Option splits are not currently supported.");
949  }
950 
951  // remove the warning from out list
952  splitWarnings.RemoveAt(i);
953  }
954  }

969  /// <summary>
970  /// Determines if a data point is in it's native, configured resolution
971  /// </summary>
972  private static bool EndTimeIsInNativeResolution(SubscriptionDataConfig config, DateTime dataPointEndTime)
973  {
974  if (config.Resolution == Resolution.Tick
975  ||
976  // time zones don't change seconds or milliseconds so we can
977  // shortcut timezone conversions
978  (config.Resolution == Resolution.Second
979  || config.Resolution == Resolution.Minute)
980  && dataPointEndTime.Ticks % config.Increment.Ticks == 0)
981  {
982  return true;
983  }
984 
985  var roundedDataPointEndTime = dataPointEndTime.RoundDownInTimeZone(config.Increment, config.ExchangeTimeZone, config.DataTimeZone);
986  return dataPointEndTime == roundedDataPointEndTime;
987  }
