16 using System;
17 using QuantConnect.Interfaces;
18 using QuantConnect.Securities;
19 
20 namespace QuantConnect.Orders
21 {
22  /// <summary>
23  /// Stop Market Order Type Definition
24  /// </summary>
25  public class StopLimitOrder : Order
26  {
27  /// <summary>
28  /// Stop price for this stop market order.
29  /// </summary>
30  public decimal StopPrice { get; internal set; }
31 
32  /// <summary>
33  /// Signal showing the "StopLimitOrder" has been converted into a Limit Order
34  /// </summary>
35  public bool StopTriggered { get; internal set; }
36 
37  /// <summary>
38  /// Limit price for the stop limit order
39  /// </summary>
40  public decimal LimitPrice { get; internal set; }
41 
42  /// <summary>
43  /// StopLimit Order Type
44  /// </summary>
45  public override OrderType Type
46  {
47  get { return OrderType.StopLimit; }
48  }
49 
50  /// <summary>
51  /// Default constructor for JSON Deserialization:
52  /// </summary>
53  public StopLimitOrder()
54  {
55  }
56 
57  /// <summary>
58  /// New Stop Market Order constructor -
59  /// </summary>
60  /// <param name="symbol">Symbol asset we're seeking to trade</param>
61  /// <param name="quantity">Quantity of the asset we're seeking to trade</param>
62  /// <param name="limitPrice">Maximum price to fill the order</param>
63  /// <param name="time">Time the order was placed</param>
64  /// <param name="stopPrice">Price the order should be filled at if a limit order</param>
65  /// <param name="tag">User defined data tag for this order</param>
66  /// <param name="properties">The order properties for this order</param>
67  public StopLimitOrder(Symbol symbol, decimal quantity, decimal stopPrice, decimal limitPrice, DateTime time, string tag = "", IOrderProperties properties = null)
68  : base(symbol, quantity, time, tag, properties)
69  {
70  StopPrice = stopPrice;
71  LimitPrice = limitPrice;
72  }
73 
74  /// <summary>
75  /// Gets the default tag for this order
76  /// </summary>
77  /// <returns>The default tag</returns>
78  public override string GetDefaultTag()
79  {
80  return Messages.StopLimitOrder.Tag(this);
81  }
82 
83  /// <summary>
84  /// Gets the order value in units of the security's quote currency
85  /// </summary>
86  /// <param name="security">The security matching this order's symbol</param>
87  protected override decimal GetValueImpl(Security security)
88  {
89  // selling, so higher price will be used
90  if (Quantity < 0)
91  {
92  return Quantity * Math.Max(LimitPrice, security.Price);
93  }
94 
95  // buying, so lower price will be used
96  if (Quantity > 0)
97  {
98  return Quantity * Math.Min(LimitPrice, security.Price);
99  }
100 
101  return 0m;
102  }
103 
104  /// <summary>
105  /// Modifies the state of this order to match the update request
106  /// </summary>
107  /// <param name="request">The request to update this order object</param>
108  public override void ApplyUpdateOrderRequest(UpdateOrderRequest request)
109  {
110  base.ApplyUpdateOrderRequest(request);
111  if (request.StopPrice.HasValue)
112  {
113  StopPrice = request.StopPrice.Value;
114  }
115  if (request.LimitPrice.HasValue)
116  {
117  LimitPrice = request.LimitPrice.Value;
118  }
119  }
120 
121  /// <summary>
122  /// Returns a string that represents the current object.
123  /// </summary>
124  /// <returns>
125  /// A string that represents the current object.
126  /// </returns>
127  /// <filterpriority>2</filterpriority>
128  public override string ToString()
129  {
130  return Messages.StopLimitOrder.ToString(this);
131  }
132 
133  /// <summary>
134  /// Creates a deep-copy clone of this order
135  /// </summary>
136  /// <returns>A copy of this order</returns>
137  public override Order Clone()
138  {
139  var order = new StopLimitOrder { StopPrice = StopPrice, LimitPrice = LimitPrice, StopTriggered = StopTriggered };
140  CopyTo(order);
141  return order;
142  }
143  }
144 }
