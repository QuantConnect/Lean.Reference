103  /// <summary>
104  /// Adds the specified symbols to the subscription: new IQLevel1WatchItem("IBM", true)
105  /// </summary>
106  /// <param name="symbols">The symbols to be added keyed by SecurityType</param>
107  public void Subscribe(IEnumerable<Symbol> symbols)
108  {
109  try
110  {
111  foreach (var symbol in symbols)
112  {
113  lock (_sync)
114  {
115  Log.Trace("IQFeed.Subscribe(): Subscribe Request: " + symbol.ToString());
116 
117  if (_symbols.Add(symbol))
118  {
119  // processing canonical option symbol to subscribe to underlying prices
120  var subscribeSymbol = symbol;
121 
122  if (symbol.ID.SecurityType == SecurityType.Option && symbol.IsCanonical())
123  {
124  subscribeSymbol = symbol.Underlying;
125  _underlyings.Add(subscribeSymbol, symbol);
126  }
127 
128  if (symbol.ID.SecurityType == SecurityType.Future && symbol.IsCanonical())
129  {
130  // do nothing for now. Later might add continuous contract symbol.
131  return;
132  }
133 
134  var ticker = _symbolUniverse.GetBrokerageSymbol(subscribeSymbol);
135 
136  if (!string.IsNullOrEmpty(ticker))
137  {
138  _level1Port.Subscribe(ticker);
139  Log.Trace("IQFeed.Subscribe(): Subscribe Processed: {0} ({1})", symbol.Value, ticker);
140  }
141  else
142  {
143  Log.Error("IQFeed.Subscribe(): Symbol {0} was not found in IQFeed symbol universe", symbol.Value);
144  }
145  }
146  }
147  }
148  }
149  catch (Exception err)
150  {
151  Log.Error("IQFeed.Subscribe(): " + err.Message);
152  }
153  }

427  public Level1Port(IDataAggregator aggregator, IQFeedDataQueueUniverseProvider symbolUniverse)
428  : base(IQFeedDefault.BufferSize)
429  {
430  start = DateTime.Now;
431  _prices = new ConcurrentDictionary<string, double>();
432  _openInterests = new ConcurrentDictionary<string, int>();
433 
434  _aggregator = aggregator;
435  _symbolUniverse = symbolUniverse;
436  Level1SummaryUpdateEvent += OnLevel1SummaryUpdateEvent;
437  Level1TimerEvent += OnLevel1TimerEvent;
438  Level1ServerDisconnectedEvent += OnLevel1ServerDisconnected;
439  Level1ServerReconnectFailed += OnLevel1ServerReconnectFailed;
440  Level1UnknownEvent += OnLevel1UnknownEvent;
441  Level1FundamentalEvent += OnLevel1FundamentalEvent;
442 
443  _timer = new Timer(1000);
444  _timer.Enabled = false;
445  _timer.AutoReset = true;
446  _timer.Elapsed += (sender, args) =>
447  {
448  var ticksPerSecond = count / (DateTime.Now - start).TotalSeconds;
449  int dataQueueCount = Interlocked.Exchange(ref _dataQueueCount, 0);
450  if (ticksPerSecond > 1000 || dataQueueCount > 31)
451  {
452  Log.Trace($"IQFeed.OnSecond(): Ticks/sec: {ticksPerSecond.ToStringInvariant("0000.00")} " +
453  $"Engine.Ticks.Count: {dataQueueCount} CPU%: {OS.CpuUsage.ToStringInvariant("0.0") + "%"}"
454  );
455  }
456 
457  count = 0;
458  start = DateTime.Now;
459  };
460 
461  _timer.Enabled = true;
462  }

469  private void OnLevel1FundamentalEvent(object sender, Level1FundamentalEventArgs e)
470  {
471  // handle split data, they're only valid today, they'll show up around 4:45am EST
472  if (e.SplitDate1.Date == DateTime.Today && DateTime.Now.TimeOfDay.TotalHours <= 8) // they will always be sent premarket
473  {
474  // get the last price, if it doesn't exist then we'll just issue the split claiming the price was zero
475  // this should (ideally) never happen, but sending this without the price is much better then not sending
476  // it at all
477  double referencePrice;
478  _prices.TryGetValue(e.Symbol, out referencePrice);
479 
480  var symbol = GetLeanSymbol(e.Symbol);
481  var split = new Split(symbol, FeedTime, (decimal)referencePrice, (decimal)e.SplitFactor1, SplitType.SplitOccurred);
482  Emit(split);
483  }

486  /// <summary>
487  /// Handle a new price update packet:
488  /// </summary>
489  private void OnLevel1SummaryUpdateEvent(object sender, Level1SummaryUpdateEventArgs e)
490  {
491  // if ticker is not found, unsubscribe
492  if (e.NotFound) Unsubscribe(e.Symbol);
493 
494  // only update if we have a value
495  if (e.Last == 0) return;
496 
497  // only accept trade and B/A updates
498  if (e.TypeOfUpdate != Level1SummaryUpdateEventArgs.UpdateType.ExtendedTrade
499  && e.TypeOfUpdate != Level1SummaryUpdateEventArgs.UpdateType.Trade
500  && e.TypeOfUpdate != Level1SummaryUpdateEventArgs.UpdateType.Bid
501  && e.TypeOfUpdate != Level1SummaryUpdateEventArgs.UpdateType.Ask) return;
502 
503  count++;
504  var time = FeedTime;
505  var last = (decimal)(e.TypeOfUpdate == Level1SummaryUpdateEventArgs.UpdateType.ExtendedTrade ? e.ExtendedTradingLast : e.Last);
506 
507  var symbol = GetLeanSymbol(e.Symbol);
508 
509  TickType tradeType;
510 
511  switch (symbol.ID.SecurityType)
512  {
513  // the feed time is in NYC/EDT, convert it into EST
514  case SecurityType.Forex:
515 
516  time = FeedTime.ConvertTo(TimeZones.NewYork, TimeZones.EasternStandard);
517  // TypeOfUpdate always equal to UpdateType.Trade for FXCM, but the message contains B/A and last data
518  tradeType = TickType.Quote;
519 
520  break;
521 
522  // for all other asset classes we leave it as is (NYC/EDT)
523  default:
524 
525  time = FeedTime;
526  tradeType = e.TypeOfUpdate == Level1SummaryUpdateEventArgs.UpdateType.Bid ||
527  e.TypeOfUpdate == Level1SummaryUpdateEventArgs.UpdateType.Ask ?
528  TickType.Quote :
529  TickType.Trade;
530  break;
531  }
532 
533  var tick = new Tick(time, symbol, last, (decimal)e.Bid, (decimal)e.Ask)
534  {
535  AskSize = e.AskSize,
536  BidSize = e.BidSize,
537  Quantity = e.IncrementalVolume,
538  TickType = tradeType,
539  DataType = MarketDataType.Tick
540  };
541  Emit(tick);
542  _prices[e.Symbol] = e.Last;
543 
544  if (symbol.ID.SecurityType == SecurityType.Option || symbol.ID.SecurityType == SecurityType.Future)
545  {
546  if (!_openInterests.ContainsKey(e.Symbol) || _openInterests[e.Symbol] != e.OpenInterest)
547  {
548  var oi = new OpenInterest(time, symbol, e.OpenInterest);
549  Emit(oi);
550 
551  _openInterests[e.Symbol] = e.OpenInterest;
552  }
553  }
554  }

599  // this type is expected to be used for exactly one job at a time
600  public class HistoryPort : IQLookupHistorySymbolClient
601  {
602  private bool _inProgress;
603  private ConcurrentDictionary<string, HistoryRequest> _requestDataByRequestId;
604  private ConcurrentDictionary<string, List<BaseData>> _currentRequest;
605  private readonly string DataDirectory = Config.Get("data-directory", "../../../Data");
606  private readonly double MaxHistoryRequestMinutes = Config.GetDouble("max-history-minutes", 5);
607  private readonly IQFeedDataQueueUniverseProvider _symbolUniverse;
608 
609  /// <summary>
610  /// ...
611  /// </summary>
612  public HistoryPort(IQFeedDataQueueUniverseProvider symbolUniverse)
613  : base(IQFeedDefault.BufferSize)
614  {
615  _symbolUniverse = symbolUniverse;
616  _requestDataByRequestId = new ConcurrentDictionary<string, HistoryRequest>();
617  _currentRequest = new ConcurrentDictionary<string, List<BaseData>>();
618  }
619 
620  /// <summary>
621  /// ...
622  /// </summary>
623  public HistoryPort(IQFeedDataQueueUniverseProvider symbolUniverse, int maxDataPoints, int dataPointsPerSend)
624  : this(symbolUniverse)
625  {
626  MaxDataPoints = maxDataPoints;
627  DataPointsPerSend = dataPointsPerSend;
628  }
629 
630  /// <summary>
631  /// Populate request data
632  /// </summary>
633  public IEnumerable<Slice> ProcessHistoryRequests(HistoryRequest request)
634  {
635  // skipping universe and canonical symbols
636  if (!CanHandle(request.Symbol) ||
637  (request.Symbol.ID.SecurityType == SecurityType.Option && request.Symbol.IsCanonical()) ||
638  (request.Symbol.ID.SecurityType == SecurityType.Future && request.Symbol.IsCanonical()))
639  {
640  yield break;
641  }
642 
643  // Set this process status
644  _inProgress = true;
645 
646  var ticker = _symbolUniverse.GetBrokerageSymbol(request.Symbol);
647  var start = request.StartTimeUtc.ConvertFromUtc(TimeZones.NewYork);
648  DateTime? end = request.EndTimeUtc.ConvertFromUtc(TimeZones.NewYork);
649  var exchangeTz = request.ExchangeHours.TimeZone;
650  // if we're within a minute of now, don't set the end time
651  if (request.EndTimeUtc >= DateTime.UtcNow.AddMinutes(-1))
652  {
653  end = null;
654  }
655 
656  Log.Trace($"HistoryPort.ProcessHistoryJob(): Submitting request: {request.Symbol.SecurityType.ToStringInvariant()}-{ticker}: " +
657  $"{request.Resolution.ToStringInvariant()} {start.ToStringInvariant()}->{(end ?? DateTime.UtcNow.AddMinutes(-1)).ToStringInvariant()}"
658  );
659 
660  int id;
661  var reqid = string.Empty;
662 
663  switch (request.Resolution)
664  {
665  case Resolution.Tick:
666  id = RequestTickData(ticker, start, end, true);
667  reqid = CreateRequestID(LookupType.REQ_HST_TCK, id);
668  break;
669  case Resolution.Daily:
670  id = RequestDailyData(ticker, start, end, true);
671  reqid = CreateRequestID(LookupType.REQ_HST_DWM, id);
672  break;
673  default:
674  var interval = new Interval(GetPeriodType(request.Resolution), 1);
675  id = RequestIntervalData(ticker, interval, start, end, true);
676  reqid = CreateRequestID(LookupType.REQ_HST_INT, id);
677  break;
678  }
679 
680  _requestDataByRequestId[reqid] = request;
681 
682  while (_inProgress)
683  {
684  continue;
685  }
686 
687  // After all data arrive, we pass it to the algorithm through memory and write to a file
688  foreach (var key in _currentRequest.Keys)
689  {
690  List<BaseData> tradeBars;
691  if (_currentRequest.TryRemove(key, out tradeBars))
692  {
693  foreach (var tradeBar in tradeBars)
694  {
695  // Returns IEnumerable<Slice> object
696  yield return new Slice(tradeBar.EndTime, new[] { tradeBar }, tradeBar.EndTime.ConvertToUtc(exchangeTz));
697  }
698  }
699  }
700  }

702  /// <summary>
703  /// Returns true if this data provide can handle the specified symbol
704  /// </summary>
705  /// <param name="symbol">The symbol to be handled</param>
706  /// <returns>True if this data provider can get data for the symbol, false otherwise</returns>
707  private bool CanHandle(Symbol symbol)

710  var securityType = symbol.ID.SecurityType;
711  return
712  (securityType == SecurityType.Equity && market == Market.USA) ||
713  (securityType == SecurityType.Forex && market == Market.FXCM) ||
714  (securityType == SecurityType.Option && market == Market.USA) ||
715  (securityType == SecurityType.Future && IQFeedDataQueueUniverseProvider.FuturesExchanges.Values.Contains(market));

729  /// <summary>
730  /// Method called when a new Lookup event is fired
731  /// </summary>
732  /// <param name="e">Received data</param>
733  protected override void OnLookupEvent(LookupEventArgs e)
734  {
735  try
736  {
737  switch (e.Sequence)
738  {
739  case LookupSequence.MessageStart:
740  _currentRequest.AddOrUpdate(e.Id, new List<BaseData>());
741  break;
742  case LookupSequence.MessageDetail:
743  List<BaseData> current;
744  if (_currentRequest.TryGetValue(e.Id, out current))
745  {
746  HandleMessageDetail(e, current);
747  }
748  break;
749  case LookupSequence.MessageEnd:
750  _inProgress = false;
751  break;
752  default:
753  throw new ArgumentOutOfRangeException();
754  }
755  }
756  catch (Exception err)
757  {
758  Log.Error(err);
759  }
760  }

764  /// </summary>

777  /// <summary>
778  /// Transform received data into BaseData object
779  /// </summary>
780  /// <param name="e">Received data</param>
781  /// <param name="requestData">Request information</param>
782  /// <returns>BaseData object</returns>
783  private BaseData GetData(LookupEventArgs e, HistoryRequest requestData)
784  {
785  var isEquity = requestData.Symbol.SecurityType == SecurityType.Equity;
786  try
787  {
788  switch (e.Type)
789  {
790  case LookupType.REQ_HST_TCK:
791  var t = (LookupTickEventArgs)e;
792  var time = isEquity ? t.DateTimeStamp : t.DateTimeStamp.ConvertTo(TimeZones.NewYork, TimeZones.EasternStandard);
793  return new Tick(time, requestData.Symbol, (decimal)t.Last, (decimal)t.Bid, (decimal)t.Ask) { Quantity = t.LastSize };
794  case LookupType.REQ_HST_INT:
795  var i = (LookupIntervalEventArgs)e;
796  if (i.DateTimeStamp == DateTime.MinValue) return null;
797  var istartTime = i.DateTimeStamp - requestData.Resolution.ToTimeSpan();
798  if (!isEquity) istartTime = istartTime.ConvertTo(TimeZones.NewYork, TimeZones.EasternStandard);
799  return new TradeBar(istartTime, requestData.Symbol, (decimal)i.Open, (decimal)i.High, (decimal)i.Low, (decimal)i.Close, i.PeriodVolume);
800  case LookupType.REQ_HST_DWM:
801  var d = (LookupDayWeekMonthEventArgs)e;
802  if (d.DateTimeStamp == DateTime.MinValue) return null;
803  var dstartTime = d.DateTimeStamp.Date;
804  if (!isEquity) dstartTime = dstartTime.ConvertTo(TimeZones.NewYork, TimeZones.EasternStandard);
805  return new TradeBar(dstartTime, requestData.Symbol, (decimal)d.Open, (decimal)d.High, (decimal)d.Low, (decimal)d.Close, d.PeriodVolume, requestData.Resolution.ToTimeSpan());
806 
807  // we don't need to handle these other types
808  case LookupType.REQ_SYM_SYM:
809  case LookupType.REQ_SYM_SIC:
810  case LookupType.REQ_SYM_NAC:
811  case LookupType.REQ_TAB_MKT:
812  case LookupType.REQ_TAB_SEC:
813  case LookupType.REQ_TAB_MKC:
814  case LookupType.REQ_TAB_SIC:
815  case LookupType.REQ_TAB_NAC:
816  default:
817  return null;
818  }
819  }
820  catch (Exception err)
821  {
822  Log.Error("Encountered error while processing request: " + e.Id);
823  Log.Error(err);
824  return null;
825  }
826  }
