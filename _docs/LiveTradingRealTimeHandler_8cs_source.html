31  /// <summary>
32  /// Live trading realtime event processing.
33  /// </summary>
34  public class LiveTradingRealTimeHandler : BacktestingRealTimeHandler
35  {
36  private Thread _realTimeThread;
37  private CancellationTokenSource _cancellationTokenSource = new();
38  private readonly bool _forceExchangeAlwaysOpen = Config.GetBool("force-exchange-always-open");
39 
40  /// <summary>
41  /// Gets the current market hours database instance
42  /// </summary>
43  protected MarketHoursDatabase MarketHoursDatabase { get; set; } = MarketHoursDatabase.FromDataFolder();
44 
45  /// <summary>
46  /// Gets the time provider
47  /// </summary>
48  /// <remarks>
49  /// This should be fixed to RealTimeHandler, but made a protected property for testing purposes
50  /// </remarks>
51  protected virtual ITimeProvider TimeProvider { get; } = RealTimeProvider.Instance;
52 
53  /// <summary>
54  /// Boolean flag indicating thread state.
55  /// </summary>
56  public override bool IsActive { get; protected set; }
57 
58  /// <summary>
59  /// Initializes the real time handler for the specified algorithm and job
60  /// </summary>
61  public override void Setup(IAlgorithm algorithm, AlgorithmNodePacket job, IResultHandler resultHandler, IApi api, IIsolatorLimitResultProvider isolatorLimitProvider)
62  {
63  base.Setup(algorithm, job, resultHandler, api, isolatorLimitProvider);
64 
65  var todayInAlgorithmTimeZone = TimeProvider.GetUtcNow().ConvertFromUtc(Algorithm.TimeZone).Date;
66 
67  // refresh the market hours for today explicitly, and then set up an event to refresh them each day at midnight
68  RefreshMarketHoursToday(todayInAlgorithmTimeZone);
69 
70  // every day at midnight from tomorrow until the end of time
71  var times =
72  from date in Time.EachDay(todayInAlgorithmTimeZone.AddDays(1), Time.EndOfTime)
73  select date.ConvertToUtc(Algorithm.TimeZone);
74 
75  Add(new ScheduledEvent("RefreshMarketHours", times, (name, triggerTime) =>
76  {
77  // refresh market hours from api every day
78  RefreshMarketHoursToday(triggerTime.ConvertFromUtc(Algorithm.TimeZone).Date);
79  }));
80  }
81 
82  /// <summary>
83  /// Get's the timeout the scheduled task time monitor should use
84  /// </summary>
85  protected override int GetTimeMonitorTimeout()
86  {
87  return 500;
88  }
89 
90  /// <summary>
91  /// Execute the live realtime event thread montioring.
92  /// It scans every second monitoring for an event trigger.
93  /// </summary>
94  private void Run()
95  {
96  IsActive = true;
97 
98  // continue thread until cancellation is requested
99  while (!_cancellationTokenSource.IsCancellationRequested)
100  {
101  var time = TimeProvider.GetUtcNow();
102 
103  // pause until the next second
104  var nextSecond = time.RoundUp(TimeSpan.FromSeconds(1));
105  var delay = Convert.ToInt32((nextSecond - time).TotalMilliseconds);
106  Thread.Sleep(delay < 0 ? 1 : delay);
107 
108  // poke each event to see if it should fire, we order by unique id to be deterministic
109  foreach (var kvp in ScheduledEvents.OrderBySafe(pair => pair.Value))
110  {
111  var scheduledEvent = kvp.Key;
112  try
113  {
114  IsolatorLimitProvider.Consume(scheduledEvent, time, TimeMonitor);
115  }
116  catch (Exception exception)
117  {
118  Algorithm.SetRuntimeError(exception, $"Scheduled event: '{scheduledEvent.Name}' at {time}");
119  }
120  }
121  }
122 
123  IsActive = false;
124  Log.Trace("LiveTradingRealTimeHandler.Run(): Exiting thread... Exit triggered: " + _cancellationTokenSource.IsCancellationRequested);
125  }
126 
127  /// <summary>
128  /// Refresh the market hours for each security in the given date
129  /// </summary>
130  /// <remarks>Each time this method is called, the MarketHoursDatabase is reset</remarks>
131  protected void RefreshMarketHoursToday(DateTime date)
132  {
133  date = date.Date;
134  ResetMarketHoursDatabase();
135 
136  // update market hours for each security
137  foreach (var kvp in Algorithm.Securities)
138  {
139  var security = kvp.Value;
140  UpdateMarketHours(security);
141 
142  var localMarketHours = security.Exchange.Hours.GetMarketHours(date);
143  Log.Trace($"LiveTradingRealTimeHandler.RefreshMarketHoursToday({security.Type}): Market hours set: Symbol: {security.Symbol} {localMarketHours} ({security.Exchange.Hours.TimeZone})");
144  }
145  }

147  /// <summary>
148  /// Set the current time. If the date changes re-start the realtime event setup routines.
149  /// </summary>
150  /// <param name="time"></param>
151  public override void SetTime(DateTime time)
152  {
153  if (Algorithm.IsWarmingUp)
154  {
155  base.SetTime(time);
156  }
157  else if (_realTimeThread == null)
158  {
159  // in live mode we use current time for our time keeping
160  // this method is used by backtesting to set time based on the data
161  _realTimeThread = new Thread(Run) { IsBackground = true, Name = "RealTime Thread" };
162  _realTimeThread.Start(); // RealTime scan time for time based events
163  }
164  }
165

167  /// Scan for past events that didn't fire because there was no data at the scheduled time.
168  /// </summary>

170  public override void ScanPastEvents(DateTime time)
171  {
172  if (Algorithm.IsWarmingUp)
173  {
174  base.ScanPastEvents(time);
175  }
176  // in live mode we use current time for our time keeping
177  // this method is used by backtesting to scan for past events based on the data
178  }

180  /// <summary>
181  /// Stop the real time thread
182  /// </summary>
183  public override void Exit()
184  {
185  _realTimeThread.StopSafely(TimeSpan.FromMinutes(5), _cancellationTokenSource);
186  _cancellationTokenSource.DisposeSafely();
187  base.Exit();
188  }
189 
190  /// <summary>
191  /// Updates the market hours for the specified security.
192  /// </summary>
193  /// <remarks>This is done after a MHDB refresh</remarks>
194  protected virtual void UpdateMarketHours(Security security)
195  {
196  var hours = _forceExchangeAlwaysOpen
197  ? SecurityExchangeHours.AlwaysOpen(security.Exchange.TimeZone)
198  : MarketHoursDatabase.GetExchangeHours(security.Symbol.ID.Market, security.Symbol, security.Symbol.ID.SecurityType);
199 
200  // Use Update method to avoid replacing the reference
201  security.Exchange.Hours.Update(hours);
202  }
203 
204  /// <summary>
205  /// Resets the market hours database, forcing a reload when reused.
206  /// Called in tests where multiple algorithms are run sequentially,
207  /// and we need to guarantee that every test starts with the same environment.
208  /// </summary>
209  protected virtual void ResetMarketHoursDatabase()
210  {
211  MarketHoursDatabase.ReloadEntries();
212  }
213  }
