15 using System;

22 namespace QuantConnect.Data.UniverseSelection

24  /// <summary>
25  /// Provides an implementation of <see cref="Universe"/> that wraps a <see cref="PyObject"/> object
26  /// </summary>
27  public class UniversePythonWrapper : Universe
28  {
29  private readonly dynamic _universe;
30 
31  /// <summary>
32  /// Gets the settings used for subscriptions added for this universe
33  /// </summary>
34  public override UniverseSettings UniverseSettings
35  {
36  get
37  {
38  using (Py.GIL())
39  {
40  return _universe.UniverseSettings;
41  }
42  }
43  set
44  {
45  using (Py.GIL())
46  {
47  _universe.UniverseSettings = value;
48  }
49  }
50  }
51 
52  /// <summary>
53  /// Flag indicating if disposal of this universe has been requested
54  /// </summary>
55  public override bool DisposeRequested
56  {
57  get
58  {
59  using (Py.GIL())
60  {
61  return _universe.DisposeRequested;
62  }
63  }
64  protected set
65  {
66  using (Py.GIL())
67  {
68  _universe.DisposeRequested = value;
69  }
70  }
71  }
72 
73  /// <summary>
74  /// Gets the configuration used to get universe data
75  /// </summary>
76  public override SubscriptionDataConfig Configuration
77  {
78  get
79  {
80  using (Py.GIL())
81  {
82  return _universe.Configuration;
83  }
84  }
85  }
86 
87  /// <summary>
88  /// Gets the internal security collection used to define membership in this universe
89  /// </summary>
90  public override ConcurrentDictionary<Symbol, Member> Securities
91  {
92  get
93  {
94  using (Py.GIL())
95  {
96  return _universe.Securities;
97  }
98  }
99  }
100 
101  /// <summary>
102  /// Initializes a new instance of the <see cref="UniversePythonWrapper"/> class
103  /// </summary>
104  public UniversePythonWrapper(PyObject universe) : base(null)
105  {
106  _universe = universe;
107  }
108 
109  /// <summary>
110  /// Performs universe selection using the data specified
111  /// </summary>
112  /// <param name="utcTime">The current utc time</param>
113  /// <param name="data">The symbols to remain in the universe</param>
114  /// <returns>The data that passes the filter</returns>
115  public override IEnumerable<Symbol> SelectSymbols(DateTime utcTime, BaseDataCollection data)
116  {
117  using (Py.GIL())
118  {
119  var symbols = _universe.SelectSymbols(utcTime, data) as PyObject;
120  var iterator = symbols.GetIterator();
121  foreach (PyObject symbol in iterator)
122  {
123  yield return symbol.GetAndDispose<Symbol>();
124  }
125  iterator.Dispose();
126  symbols.Dispose();
127  }
128  }
129 
130  /// <summary>
131  /// Gets the subscription requests to be added for the specified security
132  /// </summary>
133  /// <param name="security">The security to get subscriptions for</param>
134  /// <param name="currentTimeUtc">The current time in utc. This is the frontier time of the algorithm</param>
135  /// <param name="maximumEndTimeUtc">The max end time</param>
136  /// <param name="subscriptionService">Instance which implements <see cref="ISubscriptionDataConfigService"/> interface</param>
137  /// <returns>All subscriptions required by this security</returns>
138  public override IEnumerable<SubscriptionRequest> GetSubscriptionRequests(Security security, DateTime currentTimeUtc, DateTime maximumEndTimeUtc,
139  ISubscriptionDataConfigService subscriptionService)
140  {
141  using (Py.GIL())
142  {
143  var subscriptionRequests = _universe.GetSubscriptionRequests(security, currentTimeUtc, maximumEndTimeUtc, subscriptionService) as PyObject;
144  var iterator = subscriptionRequests.GetIterator();
145  foreach (PyObject request in iterator)
146  {
147  var subscriptionRequest = request.GetAndDispose<SubscriptionRequest>();
148  yield return new SubscriptionRequest(subscriptionRequest, universe:this);
149  }
150  iterator.Dispose();
151  subscriptionRequests.Dispose();
152  }
153  }
154  }
155 }
