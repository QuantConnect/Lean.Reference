60  /// <summary>
61  /// Creates a new instance of the DataManager
62  /// </summary>
63  public DataManager(
64  IDataFeed dataFeed,
65  UniverseSelection universeSelection,
66  IAlgorithm algorithm,
67  ITimeKeeper timeKeeper,
68  MarketHoursDatabase marketHoursDatabase,
69  bool liveMode,
70  IRegisteredSecurityDataTypesProvider registeredTypesProvider,
71  IDataPermissionManager dataPermissionManager)
72  {
73  _dataFeed = dataFeed;
74  UniverseSelection = universeSelection;
75  UniverseSelection.SetDataManager(this);
76  _algorithmSettings = algorithm.Settings;
77  AvailableDataTypes = SubscriptionManager.DefaultDataTypes();
78  _timeKeeper = timeKeeper;
79  _marketHoursDatabase = marketHoursDatabase;
80  _liveMode = liveMode;
81  _registeredTypesProvider = registeredTypesProvider;
82  _dataPermissionManager = dataPermissionManager;
83 
84  // wire ourselves up to receive notifications when universes are added/removed
85  algorithm.UniverseManager.CollectionChanged += (sender, args) =>
86  {
87  switch (args.Action)
88  {
89  case NotifyCollectionChangedAction.Add:
90  foreach (var universe in args.NewItems.OfType<Universe>())
91  {
92  var config = universe.Configuration;
93  var start = algorithm.UtcTime;
94 
95  var end = algorithm.LiveMode ? Time.EndOfTime
96  : algorithm.EndDate.ConvertToUtc(algorithm.TimeZone);
97 
98  Security security;
99  if (!algorithm.Securities.TryGetValue(config.Symbol, out security))
100  {
101  // create a canonical security object if it doesn't exist
102  security = new Security(
103  _marketHoursDatabase.GetExchangeHours(config),
104  config,
105  algorithm.Portfolio.CashBook[algorithm.AccountCurrency],
106  SymbolProperties.GetDefault(algorithm.AccountCurrency),
107  algorithm.Portfolio.CashBook,
108  RegisteredSecurityDataTypesProvider.Null,
109  new SecurityCache()
110  );
111  }
112 
113  // Let's adjust the start time to the previous tradable date
114  // so universe selection always happens right away at the start of the algorithm.
115  var universeType = universe.GetType();
116  if (
117  // We exclude the OptionChainUniverse because their selection in live trading is based on having a full bar
118  // of the underlying. In the future, option chain universe file-based selection will be improved
119  // in order to avoid this.
120  universeType != typeof(OptionChainUniverse) &&
121  // We exclude the UserDefinedUniverse because their selection already happens at the algorithm start time.
122  // For instance, ETFs universe selection depends its first trigger time to be before the equity universe
123  // (the UserDefinedUniverse), because the ETFs are EndTime-indexed and that would make their first selection
124  // time to be before the algorithm start time, with the EndTime being the algorithms's start date,
125  // and both the Equity and the ETFs constituents first selection to happen together.
126  !universeType.IsAssignableTo(typeof(UserDefinedUniverse)) &&
127  // We exclude the ScheduledUniverse because it's already scheduled to run at a specific time.
128  // Adjusting the start time would cause the first selection trigger time to be before the algorithm start time,
129  // making the selection to be triggered at the first algorithm time, which would be the exact StartDate.
130  universeType != typeof(ScheduledUniverse))
131  {
132  start = Time.GetStartTimeForTradeBars(security.Exchange.Hours, start.ConvertFromUtc(security.Exchange.TimeZone),
133  Time.OneDay, 1, true, config.DataTimeZone);
134  start = start.ConvertToUtc(security.Exchange.TimeZone);
135  }
136 
137  AddSubscription(
138  new SubscriptionRequest(true,
139  universe,
140  security,
141  config,
142  start,
143  end));
144  }
145  break;
146 
147  case NotifyCollectionChangedAction.Remove:
148  foreach (var universe in args.OldItems.OfType<Universe>())
149  {
150  // removing the subscription will be handled by the SubscriptionSynchronizer
151  // in the next loop as well as executing a UniverseSelection one last time.
152  if (!universe.DisposeRequested)
153  {
154  universe.Dispose();
155  }
156  }
157  break;
158 
159  default:
160  throw new NotImplementedException("The specified action is not implemented: " + args.Action);
161  }
162  };
163 
164  DataFeedSubscriptions = new SubscriptionCollection();
165  if (!_liveMode)
166  {
167  DataFeedSubscriptions.FillForwardResolutionChanged += (object sender, FillForwardResolutionChangedEvent changedEvent) =>
168  {
169  var requests = DataFeedSubscriptions
170  // we don't fill forward tick resolution so we don't need to touch their subscriptions
171  .Where(subscription => subscription.Configuration.FillDataForward && subscription.Configuration.Resolution != Resolution.Tick)
172  .SelectMany(subscription => subscription.SubscriptionRequests)
173  .ToList();
174 
175  if(requests.Count > 0)
176  {
177  Log.Trace($"DataManager(): Fill forward resolution has changed from {changedEvent.Old} to {changedEvent.New} at utc: {algorithm.UtcTime}. " +
178  $"Restarting {requests.Count} subscriptions...");
179 
180  // disable reentry while we remove and re add
181  DataFeedSubscriptions.FreezeFillForwardResolution(true);
182 
183  // remove
184  foreach (var request in requests)
185  {
186  RemoveSubscription(request.Configuration, request.Universe);
187  }
188 
189  // re add
190  foreach (var request in requests)
191  {
192  AddSubscription(new SubscriptionRequest(request, startTimeUtc: algorithm.UtcTime));
193  }
194 
195  DataFeedSubscriptions.FreezeFillForwardResolution(false);
196  }
197  };
198  }
199  }

208  /// <summary>
209  /// Will remove all current <see cref="Subscription"/>
210  /// </summary>
211  public void RemoveAllSubscriptions()
212  {
213  // remove each subscription from our collection
214  foreach (var subscription in DataFeedSubscriptions)
215  {
216  try
217  {
218  RemoveSubscription(subscription.Configuration);
219  }
220  catch (Exception err)
221  {
222  Log.Error(err, "DataManager.RemoveAllSubscriptions():" +
223  $"Error removing: {subscription.Configuration}");
224  }
225  }
226  }

228  /// <summary>
229  /// Adds a new <see cref="Subscription"/> to provide data for the specified security.
230  /// </summary>
231  /// <param name="request">Defines the <see cref="SubscriptionRequest"/> to be added</param>
232  /// <returns>True if the subscription was created and added successfully, false otherwise</returns>
233  public bool AddSubscription(SubscriptionRequest request)
234  {
235  lock (_subscriptionManagerSubscriptions)
236  {
237  // guarantee the configuration is present in our config collection
238  // this is related to GH issue 3877: where we added a configuration which we also removed
239  if(_subscriptionManagerSubscriptions.TryAdd(request.Configuration, request.Configuration))
240  {
241  _subscriptionDataConfigsEnumerator = null;
242  }
243  }
244 
245  Subscription subscription;
246  if (DataFeedSubscriptions.TryGetValue(request.Configuration, out subscription))
247  {
248  // duplicate subscription request
249  subscription.AddSubscriptionRequest(request);
250  // only result true if the existing subscription is internal, we actually added something from the users perspective
251  return subscription.Configuration.IsInternalFeed;
252  }
253 
254  if (request.Configuration.DataNormalizationMode == DataNormalizationMode.ScaledRaw)
255  {
256  throw new InvalidOperationException($"{DataNormalizationMode.ScaledRaw} normalization mode only intended for history requests.");
257  }
258 
259  if (request.Configuration.SecurityType == SecurityType.IndexOption && request.Configuration.Resolution == Resolution.Daily)
260  {
261  throw new InvalidOperationException($"{Resolution.Daily} resolution is currently not supported for {SecurityType.IndexOption}, please use {Resolution.Hour} instead");
262  }
263 
264  // before adding the configuration to the data feed let's assert it's valid
265  _dataPermissionManager.AssertConfiguration(request.Configuration, request.StartTimeLocal, request.EndTimeLocal);
266 
267  subscription = _dataFeed.CreateSubscription(request);
268 
269  if (subscription == null)
270  {
271  Log.Trace($"DataManager.AddSubscription(): Unable to add subscription for: {request.Configuration}");
272  // subscription will be null when there's no tradeable dates for the security between the requested times, so
273  // don't even try to load the data
274  return false;
275  }
276 
277  if (_liveMode)
278  {
279  OnSubscriptionAdded(subscription);
280  Log.Trace($"DataManager.AddSubscription(): Added {request.Configuration}." +
281  $" Start: {request.StartTimeUtc}. End: {request.EndTimeUtc}");
282  }
283  else if(Log.DebuggingEnabled)
284  {
285  // for performance lets not create the message string if debugging is not enabled
286  // this can be executed many times and its in the algorithm thread
287  Log.Debug($"DataManager.AddSubscription(): Added {request.Configuration}." +
288  $" Start: {request.StartTimeUtc}. End: {request.EndTimeUtc}");
289  }
290 
291  return DataFeedSubscriptions.TryAdd(subscription);
292  }
293 
294  /// <summary>
295  /// Removes the <see cref="Subscription"/>, if it exists
296  /// </summary>
297  /// <param name="configuration">The <see cref="SubscriptionDataConfig"/> of the subscription to remove</param>
298  /// <param name="universe">Universe requesting to remove <see cref="Subscription"/>.
299  /// Default value, null, will remove all universes</param>
300  /// <returns>True if the subscription was successfully removed, false otherwise</returns>
301  public bool RemoveSubscription(SubscriptionDataConfig configuration, Universe universe = null)
302  {
303  // remove the subscription from our collection, if it exists
304  Subscription subscription;
305 
306  if (DataFeedSubscriptions.TryGetValue(configuration, out subscription))
307  {
308  // we remove the subscription when there are no other requests left
309  if (subscription.RemoveSubscriptionRequest(universe))
310  {
311  if (!DataFeedSubscriptions.TryRemove(configuration, out subscription))
312  {
313  Log.Error($"DataManager.RemoveSubscription(): Unable to remove {configuration}");
314  return false;
315  }
316 
317  _dataFeed.RemoveSubscription(subscription);
318 
319  if (_liveMode)
320  {
321  OnSubscriptionRemoved(subscription);
322  }
323 
324  subscription.Dispose();
325 
326  RemoveSubscriptionDataConfig(subscription);
327 
328  if (_liveMode)
329  {
330  Log.Trace($"DataManager.RemoveSubscription(): Removed {configuration}");
331  }
332  else if(Log.DebuggingEnabled)
333  {
334  // for performance lets not create the message string if debugging is not enabled
335  // this can be executed many times and its in the algorithm thread
336  Log.Debug($"DataManager.RemoveSubscription(): Removed {configuration}");
337  }
338  return true;
339  }
340  }
341  else if (universe != null)
342  {
343  // a universe requested removal of a subscription which wasn't present anymore, this can happen when a subscription ends
344  // it will get removed from the data feed subscription list, but the configuration will remain until the universe removes it
345  // why? the effect I found is that the fill models are using these subscriptions to determine which data they could use
346  lock (_subscriptionManagerSubscriptions)
347  {
348  if (_subscriptionManagerSubscriptions.Remove(configuration))
349  {
350  _subscriptionDataConfigsEnumerator = null;
351  }
352  }
353  }
354  return false;
355  }

397  /// <summary>
398  /// Gets existing or adds new <see cref="SubscriptionDataConfig" />
399  /// </summary>
400  /// <returns>Returns the SubscriptionDataConfig instance used</returns>
401  public SubscriptionDataConfig SubscriptionManagerGetOrAdd(SubscriptionDataConfig newConfig)
402  {
403  SubscriptionDataConfig config;
404  lock (_subscriptionManagerSubscriptions)
405  {
406  if (!_subscriptionManagerSubscriptions.TryGetValue(newConfig, out config))
407  {
408  _subscriptionManagerSubscriptions[newConfig] = config = newConfig;
409  _subscriptionDataConfigsEnumerator = null;
410  }
411  }
412 
413  // if the reference is not the same, means it was already there and we did not add anything new
414  if (!ReferenceEquals(config, newConfig))
415  {
416  // for performance lets not create the message string if debugging is not enabled
417  // this can be executed many times and its in the algorithm thread
418  if (Log.DebuggingEnabled)
419  {
420  Log.Debug("DataManager.SubscriptionManagerGetOrAdd(): subscription already added: " + config);
421  }
422  }
423  else
424  {
425  // add the time zone to our time keeper
426  _timeKeeper.AddTimeZone(newConfig.ExchangeTimeZone);
427  }
428 
429  return config;
430  }

432  /// <summary>
433  /// Will try to remove a <see cref="SubscriptionDataConfig"/> and update the corresponding
434  /// consumers accordingly
435  /// </summary>
436  /// <param name="subscription">The <see cref="Subscription"/> owning the configuration to remove</param>
437  private void RemoveSubscriptionDataConfig(Subscription subscription)
438  {
439  // the subscription could of ended but might still be part of the universe
440  if (subscription.RemovedFromUniverse.Value)
441  {
442  lock (_subscriptionManagerSubscriptions)
443  {
444  if (_subscriptionManagerSubscriptions.Remove(subscription.Configuration))
445  {
446  _subscriptionDataConfigsEnumerator = null;
447  }
448  }
449  }

470  /// <summary>
471  /// Creates and adds a list of <see cref="SubscriptionDataConfig" /> for a given symbol and configuration.
472  /// Can optionally pass in desired subscription data type to use.
473  /// If the config already existed will return existing instance instead
474  /// </summary>
475  public SubscriptionDataConfig Add(
476  Type dataType,
477  Symbol symbol,
478  Resolution? resolution = null,
479  bool fillForward = true,
480  bool extendedMarketHours = false,
481  bool isFilteredSubscription = true,
482  bool isInternalFeed = false,
483  bool isCustomData = false,
484  DataNormalizationMode dataNormalizationMode = DataNormalizationMode.Adjusted,
485  DataMappingMode dataMappingMode = DataMappingMode.OpenInterest,
486  uint contractDepthOffset = 0
487  )
488  {
489  return Add(symbol, resolution, fillForward, extendedMarketHours, isFilteredSubscription, isInternalFeed, isCustomData,
490  new List<Tuple<Type, TickType>> { new Tuple<Type, TickType>(dataType, LeanData.GetCommonTickTypeForCommonDataTypes(dataType, symbol.SecurityType))},
491  dataNormalizationMode, dataMappingMode, contractDepthOffset)
492  .First();
493  }

495  /// <summary>
496  /// Creates and adds a list of <see cref="SubscriptionDataConfig" /> for a given symbol and configuration.
497  /// Can optionally pass in desired subscription data types to use.
498  /// If the config already existed will return existing instance instead
499  /// </summary>
500  public List<SubscriptionDataConfig> Add(
501  Symbol symbol,
502  Resolution? resolution = null,
503  bool fillForward = true,
504  bool extendedMarketHours = false,
505  bool isFilteredSubscription = true,
506  bool isInternalFeed = false,
507  bool isCustomData = false,
508  List<Tuple<Type, TickType>> subscriptionDataTypes = null,
509  DataNormalizationMode dataNormalizationMode = DataNormalizationMode.Adjusted,
510  DataMappingMode dataMappingMode = DataMappingMode.OpenInterest,
511  uint contractDepthOffset = 0
512  )
513  {
514  var dataTypes = subscriptionDataTypes;
515  if(dataTypes == null)
516  {
517  if (symbol.SecurityType == SecurityType.Base && SecurityIdentifier.TryGetCustomDataTypeInstance(symbol.ID.Symbol, out var type))
518  {
519  // we've detected custom data request if we find a type let's use it
520  dataTypes = new List<Tuple<Type, TickType>> { new Tuple<Type, TickType>(type, TickType.Trade) };
521  }
522  else
523  {
524  dataTypes = LookupSubscriptionConfigDataTypes(symbol.SecurityType, resolution ?? Resolution.Minute, symbol.IsCanonical());
525  }
526  }
527 
528  if (!dataTypes.Any())
529  {
530  throw new ArgumentNullException(nameof(dataTypes), "At least one type needed to create new subscriptions");
531  }
532 
533  var resolutionWasProvided = resolution.HasValue;
534  foreach (var typeTuple in dataTypes)
535  {
536  var baseInstance = typeTuple.Item1.GetBaseDataInstance();
537  baseInstance.Symbol = symbol;
538  if (!resolutionWasProvided)
539  {
540  var defaultResolution = baseInstance.DefaultResolution();
541  if (resolution.HasValue && resolution != defaultResolution)
542  {
543  // we are here because there are multiple 'dataTypes'.
544  // if we get different default resolutions lets throw, this shouldn't happen
545  throw new InvalidOperationException(
546  $"Different data types ({string.Join(",", dataTypes.Select(tuple => tuple.Item1))})" +
547  $" provided different default resolutions {defaultResolution} and {resolution}, this is an unexpected invalid operation.");
548  }
549  resolution = defaultResolution;
550  }
551  else
552  {
553  // only assert resolution in backtesting, live can use other data source
554  // for example daily data for options
555  if (!_liveMode)
556  {
557  var supportedResolutions = baseInstance.SupportedResolutions();
558  if (supportedResolutions.Contains(resolution.Value))
559  {
560  continue;
561  }
562 
563  throw new ArgumentException($"Sorry {resolution.ToStringInvariant()} is not a supported resolution for {typeTuple.Item1.Name}" +
564  $" and SecurityType.{symbol.SecurityType.ToStringInvariant()}." +
565  $" Please change your AddData to use one of the supported resolutions ({string.Join(",", supportedResolutions)}).");
566  }
567  }
568  }
569  var marketHoursDbEntry = _marketHoursDatabase.GetEntry(symbol, dataTypes.Select(tuple => tuple.Item1));
570 
571  var exchangeHours = marketHoursDbEntry.ExchangeHours;
572  if (symbol.ID.SecurityType.IsOption() ||
573  symbol.ID.SecurityType == SecurityType.Index)
574  {
575  dataNormalizationMode = DataNormalizationMode.Raw;
576  }
577 
578  if (marketHoursDbEntry.DataTimeZone == null)
579  {
580  throw new ArgumentNullException(nameof(marketHoursDbEntry.DataTimeZone),
581  "DataTimeZone is a required parameter for new subscriptions. Set to the time zone the raw data is time stamped in.");
582  }
583 
584  if (exchangeHours.TimeZone == null)
585  {
586  throw new ArgumentNullException(nameof(exchangeHours.TimeZone),
587  "ExchangeTimeZone is a required parameter for new subscriptions. Set to the time zone the security exchange resides in.");
588  }
589 
590  var result = (from subscriptionDataType in dataTypes
591  let dataType = subscriptionDataType.Item1
592  let tickType = subscriptionDataType.Item2
593  select new SubscriptionDataConfig(
594  dataType,
595  symbol,
596  resolution.Value,
597  marketHoursDbEntry.DataTimeZone,
598  exchangeHours.TimeZone,
599  fillForward,
600  extendedMarketHours,
601  // if the subscription data types were not provided and the tick type is OpenInterest we make it internal
602  subscriptionDataTypes == null && tickType == TickType.OpenInterest || isInternalFeed,
603  isCustomData,
604  isFilteredSubscription: isFilteredSubscription,
605  tickType: tickType,
606  dataNormalizationMode: dataNormalizationMode,
607  dataMappingMode: dataMappingMode,
608  contractDepthOffset: contractDepthOffset)).ToList();
609 
610  for (int i = 0; i < result.Count; i++)
611  {
612  result[i] = SubscriptionManagerGetOrAdd(result[i]);
613 
614  // track all registered data types
615  _registeredTypesProvider.RegisterType(result[i].Type);
616  }
617  return result;
618  }

620  /// <summary>
621  /// Get the data feed types for a given <see cref="SecurityType" /> <see cref="Resolution" />
622  /// </summary>
623  /// <param name="symbolSecurityType">The <see cref="SecurityType" /> used to determine the types</param>
624  /// <param name="resolution">The resolution of the data requested</param>
625  /// <param name="isCanonical">Indicates whether the security is Canonical (future and options)</param>
626  /// <returns>Types that should be added to the <see cref="SubscriptionDataConfig" /></returns>
627  /// <remarks>TODO: data type additions are very related to ticktype and should be more generic/independent of each other</remarks>
628  public List<Tuple<Type, TickType>> LookupSubscriptionConfigDataTypes(
629  SecurityType symbolSecurityType,
630  Resolution resolution,
631  bool isCanonical
632  )
633  {
634  if (isCanonical)
635  {
636  return new List<Tuple<Type, TickType>> { new Tuple<Type, TickType>(typeof(ZipEntryName), TickType.Quote) };
637  }
638 
639  IEnumerable<TickType> availableDataType = AvailableDataTypes[symbolSecurityType]
640  // Equities will only look for trades in case of low resolutions.
641  .Where(tickType => LeanData.IsValidConfiguration(symbolSecurityType, resolution, tickType));
642 
643  var result = availableDataType
644  .Select(tickType => new Tuple<Type, TickType>(LeanData.GetDataType(resolution, tickType), tickType)).ToList();
645 
646  if(symbolSecurityType == SecurityType.CryptoFuture)
647  {
648  result.Add(new Tuple<Type, TickType>(typeof(MarginInterestRate), TickType.Quote));
649  }
650  return result;
651  }

654  /// Gets a list of all registered <see cref="SubscriptionDataConfig"/> for a given <see cref="Symbol"/>
655  /// </summary>
