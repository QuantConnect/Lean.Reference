16 using System;
17 using QuantConnect.Data;
18 using QuantConnect.Securities;
19 
20 namespace QuantConnect.Lean.Engine.DataFeeds
21 {
22  /// <summary>
23  /// Store data (either raw or adjusted) and the time at which it should be synchronized
24  /// </summary>
25  public class SubscriptionData
26  {
27  /// <summary>
28  /// Data
29  /// </summary>
30  protected BaseData _data;
31 
32  /// <summary>
33  /// Gets the data
34  /// </summary>
35  public virtual BaseData Data => _data;
36 
37  /// <summary>
38  /// Gets the UTC emit time for this data
39  /// </summary>
40  public DateTime EmitTimeUtc { get; }
41 
42  /// <summary>
43  /// Initializes a new instance of the <see cref="SubscriptionData"/> class
44  /// </summary>
45  /// <param name="data">The base data</param>
46  /// <param name="emitTimeUtc">The emit time for the data</param>
47  public SubscriptionData(BaseData data, DateTime emitTimeUtc)
48  {
49  _data = data;
50  EmitTimeUtc = emitTimeUtc;
51  }
52 
53  /// <summary>
54  /// Clones the data, computes the utc emit time and performs exchange round down behavior, storing the result in a new <see cref="SubscriptionData"/> instance
55  /// </summary>
56  /// <param name="configuration">The subscription's configuration</param>
57  /// <param name="exchangeHours">The exchange hours of the security</param>
58  /// <param name="offsetProvider">The subscription's offset provider</param>
59  /// <param name="data">The data being emitted</param>
60  /// <param name="normalizationMode">Specifies how data is normalized</param>
61  /// <param name="factor">price scale factor</param>
62  /// <returns>A new <see cref="SubscriptionData"/> containing the specified data</returns>
63  public static SubscriptionData Create(SubscriptionDataConfig configuration, SecurityExchangeHours exchangeHours, TimeZoneOffsetProvider offsetProvider, BaseData data, DataNormalizationMode normalizationMode, decimal? factor = null)
64  {
65  if (data == null)
66  {
67  return null;
68  }
69 
70  data = data.Clone(data.IsFillForward);
71  var emitTimeUtc = offsetProvider.ConvertToUtc(data.EndTime);
72 
73  // Let's round down for any data source that implements a time delta between
74  // the start of the data and end of the data (usually used with Bars).
75  // The time delta ensures that the time collected from `EndTime` has
76  // no look-ahead bias, and is point-in-time.
77  // When fill forwarding time and endtime might not respect the original ends times, here we will enforce it
78  // note we do this after fetching the 'emitTimeUtc' which should use the end time set by the fill forward enumerator
79  var barSpan = data.EndTime - data.Time;
80  if (barSpan != TimeSpan.Zero)
81  {
82  if (barSpan != configuration.Increment)
83  {
84  // when we detect a difference let's refetch the span in utc using noda time 'ConvertToUtc' that will not take into account day light savings difference
85  // we don't do this always above because it's expensive, only do it if we need to.
86  // Behavior asserted by tests 'FillsForwardBarsAroundDaylightMovementForDifferentResolutions_Algorithm' && 'ConvertToUtcAndDayLightSavings'.
87  // Note: we don't use 'configuration.Increment' because during warmup, if the warmup resolution is set, we will emit data respecting it instead of the 'configuration'
88  barSpan = data.EndTime.ConvertToUtc(configuration.ExchangeTimeZone) - data.Time.ConvertToUtc(configuration.ExchangeTimeZone);
89  }
90  data.Time = data.Time.ExchangeRoundDownInTimeZone(barSpan, exchangeHours, configuration.DataTimeZone, configuration.ExtendedMarketHours);
91  }
92 
93  if (factor.HasValue && (configuration.SecurityType != SecurityType.Equity || (factor.Value != 1 || configuration.SumOfDividends != 0)))
94  {
95  var normalizedData = data.Clone(data.IsFillForward).Normalize(factor.Value, normalizationMode, configuration.SumOfDividends);
96 
97  return new PrecalculatedSubscriptionData(configuration, data, normalizedData, normalizationMode, emitTimeUtc);
98  }
99 
100  return new SubscriptionData(data, emitTimeUtc);
101  }
102  }
103 }
