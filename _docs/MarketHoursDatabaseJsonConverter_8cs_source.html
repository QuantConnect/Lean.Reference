66  /// <summary>
67  /// Defines the json structure of the market-hours-database.json file
68  /// </summary>
69  [JsonObject(MemberSerialization.OptIn)]
70  public class MarketHoursDatabaseJson
71  {
72  /// <summary>
73  /// The entries in the market hours database, keyed by <see cref="SecurityDatabaseKey"/>
74  /// </summary>
75  [JsonProperty("entries")]
76  public Dictionary<string, MarketHoursDatabaseEntryJson> Entries;
77 
78  /// <summary>
79  /// Initializes a new instance of the <see cref="MarketHoursDatabaseJson"/> class
80  /// </summary>
81  /// <param name="database">The database instance to copy</param>
82  public MarketHoursDatabaseJson(MarketHoursDatabase database)
83  {
84  if (database == null) return;
85  Entries = new Dictionary<string, MarketHoursDatabaseEntryJson>();
86  foreach (var kvp in database.ExchangeHoursListing)
87  {
88  var key = kvp.Key;
89  var entry = kvp.Value;
90  Entries[key.ToString()] = new MarketHoursDatabaseEntryJson(entry);
91  }
92  }
93 
94  /// <summary>
95  /// Converts this json representation to the <see cref="MarketHoursDatabase"/> type
96  /// </summary>
97  /// <returns>A new instance of the <see cref="MarketHoursDatabase"/> class</returns>
98  public MarketHoursDatabase Convert()
99  {
100  var entries = new Dictionary<SecurityDatabaseKey, MarketHoursDatabase.Entry>();
101  foreach (var entry in Entries)
102  {
103  try
104  {
105  var key = SecurityDatabaseKey.Parse(entry.Key);
106  if (key == null)
107  {
108  continue;
109  }
110 
111  entries[key] = entry.Value.Convert();
112  }
113  catch (Exception err)
114  {
115  Log.Error(err);
116  }
117  }
118  return new MarketHoursDatabase(entries);
119  }
120  }

200  /// <summary>
201  /// Initializes a new instance of the <see cref="MarketHoursDatabaseEntryJson"/> class
202  /// </summary>
203  /// <param name="entry">The entry instance to copy</param>
204  public MarketHoursDatabaseEntryJson(MarketHoursDatabase.Entry entry)
205  {
206  if (entry == null) return;
207  DataTimeZone = entry.DataTimeZone.Id;
208  var hours = entry.ExchangeHours;
209  ExchangeTimeZone = hours.TimeZone.Id;
210  SetSegmentsForDay(hours, DayOfWeek.Sunday, out Sunday);
211  SetSegmentsForDay(hours, DayOfWeek.Monday, out Monday);
212  SetSegmentsForDay(hours, DayOfWeek.Tuesday, out Tuesday);
213  SetSegmentsForDay(hours, DayOfWeek.Wednesday, out Wednesday);
214  SetSegmentsForDay(hours, DayOfWeek.Thursday, out Thursday);
215  SetSegmentsForDay(hours, DayOfWeek.Friday, out Friday);
216  SetSegmentsForDay(hours, DayOfWeek.Saturday, out Saturday);
217  Holidays = hours.Holidays.Select(x => x.ToString("M/d/yyyy", CultureInfo.InvariantCulture)).ToList();
218  EarlyCloses = entry.ExchangeHours.EarlyCloses.ToDictionary(pair => pair.Key.ToString("M/d/yyyy", CultureInfo.InvariantCulture), pair => pair.Value);
219  LateOpens = entry.ExchangeHours.LateOpens.ToDictionary(pair => pair.Key.ToString("M/d/yyyy", CultureInfo.InvariantCulture), pair => pair.Value);
220  }

222  /// <summary>
223  /// Converts this json representation to the <see cref="MarketHoursDatabase.Entry"/> type
224  /// </summary>
225  /// <returns>A new instance of the <see cref="MarketHoursDatabase.Entry"/> class</returns>
226  public MarketHoursDatabase.Entry Convert()
227  {
228  var hours = new Dictionary<DayOfWeek, LocalMarketHours>
229  {
230  { DayOfWeek.Sunday, new LocalMarketHours(DayOfWeek.Sunday, Sunday) },
231  { DayOfWeek.Monday, new LocalMarketHours(DayOfWeek.Monday, Monday) },
232  { DayOfWeek.Tuesday, new LocalMarketHours(DayOfWeek.Tuesday, Tuesday) },
233  { DayOfWeek.Wednesday, new LocalMarketHours(DayOfWeek.Wednesday, Wednesday) },
234  { DayOfWeek.Thursday, new LocalMarketHours(DayOfWeek.Thursday, Thursday) },
235  { DayOfWeek.Friday, new LocalMarketHours(DayOfWeek.Friday, Friday) },
236  { DayOfWeek.Saturday, new LocalMarketHours(DayOfWeek.Saturday, Saturday) }
237  };
238  var holidayDates = Holidays.Select(x => DateTime.ParseExact(x, "M/d/yyyy", CultureInfo.InvariantCulture)).ToHashSet();
239  var earlyCloses = EarlyCloses.ToDictionary(x => DateTime.ParseExact(x.Key, "M/d/yyyy", CultureInfo.InvariantCulture), x => x.Value);
240  var lateOpens = LateOpens.ToDictionary(x => DateTime.ParseExact(x.Key, "M/d/yyyy", CultureInfo.InvariantCulture), x => x.Value);
241  var exchangeHours = new SecurityExchangeHours(DateTimeZoneProviders.Tzdb[ExchangeTimeZone], holidayDates, hours, earlyCloses, lateOpens);
242  return new MarketHoursDatabase.Entry(DateTimeZoneProviders.Tzdb[DataTimeZone], exchangeHours);
243  }

245  private void SetSegmentsForDay(SecurityExchangeHours hours, DayOfWeek day, out List<MarketHoursSegment> segments)
246  {
247  LocalMarketHours local;
248  if (hours.MarketHours.TryGetValue(day, out local))
249  {
250  segments = local.Segments.ToList();
251  }
252  else
253  {
254  segments = new List<MarketHoursSegment>();
255  }
256  }
257  }
