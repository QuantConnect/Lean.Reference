28 namespace QuantConnect.Lean.Engine.DataFeeds.Enumerators.Factories
29 {
30  /// <summary>
31  /// Provides an implementation of <see cref="ISubscriptionEnumeratorFactory"/> to emit
32  /// ticks based on <see cref="UserDefinedUniverse.GetTriggerTimes"/>, allowing universe
33  /// selection to fire at planned times.
34  /// </summary>
35  public class TimeTriggeredUniverseSubscriptionEnumeratorFactory : ISubscriptionEnumeratorFactory
36  {
37  private readonly ITimeProvider _timeProvider;
38  private readonly ITimeTriggeredUniverse _universe;
39  private readonly MarketHoursDatabase _marketHoursDatabase;
40 
41  /// <summary>
42  /// Initializes a new instance of the <see cref="TimeTriggeredUniverseSubscriptionEnumeratorFactory"/> class
43  /// </summary>
44  /// <param name="universe">The user defined universe</param>
45  /// <param name="marketHoursDatabase">The market hours database</param>
46  /// <param name="timeProvider">The time provider</param>
47  public TimeTriggeredUniverseSubscriptionEnumeratorFactory(ITimeTriggeredUniverse universe, MarketHoursDatabase marketHoursDatabase, ITimeProvider timeProvider)
48  {
49  _universe = universe;
50  _timeProvider = timeProvider;
51  _marketHoursDatabase = marketHoursDatabase;
52  }
53 
54  /// <summary>
55  /// Creates an enumerator to read the specified request
56  /// </summary>
57  /// <param name="request">The subscription request to be read</param>
58  /// <param name="dataProvider">Provider used to get data when it is not present on disk</param>
59  /// <returns>An enumerator reading the subscription request</returns>
60  public IEnumerator<BaseData> CreateEnumerator(SubscriptionRequest request, IDataProvider dataProvider)
61  {
62  var enumerator = (IEnumerator<BaseData>) _universe.GetTriggerTimes(request.StartTimeUtc, request.EndTimeUtc, _marketHoursDatabase)
63  .Select(x => new Tick { Time = x, Symbol = request.Configuration.Symbol })
64  .GetEnumerator();
65 
66  var universe = request.Universe as UserDefinedUniverse;
67  if (universe != null)
68  {
69  enumerator = new InjectionEnumerator(enumerator);
70 
71  // Trigger universe selection when security added/removed after Initialize
72  universe.CollectionChanged += (sender, args) =>
73  {
74  // If it is an add we will set time 1 tick ahead to properly sync data
75  // with next timeslice, if it is a remove then we will set time to now
76  IList items;
77  DateTime time;
78  if (args.Action == NotifyCollectionChangedAction.Add)
79  {
80  items = args.NewItems;
81  time = _timeProvider.GetUtcNow().AddTicks(1);
82  }
83  else if (args.Action == NotifyCollectionChangedAction.Remove)
84  {
85  items = args.OldItems;
86  time = _timeProvider.GetUtcNow();
87  }
88  else
89  {
90  items = null;
91  time = DateTime.MinValue;
92  }
93 
94  // Check that we have our items and time
95  if (items == null || time == DateTime.MinValue) return;
96 
97  var symbol = items.OfType<Symbol>().FirstOrDefault();
98 
99  if(symbol == null) return;
100 
101  // the data point time should always be in exchange timezone
102  time = time.ConvertFromUtc(request.Configuration.ExchangeTimeZone);
103 
104  var collection = new BaseDataCollection(time, symbol);
105  ((InjectionEnumerator) enumerator).InjectDataPoint(collection);
106  };
107  }
108 
109  return enumerator;
110  }
111 
112  private class InjectionEnumerator : IEnumerator<BaseData>
113  {
114  private volatile bool _wasInjected;
115  private readonly IEnumerator<BaseData> _underlyingEnumerator;
116 
117  public BaseData Current { get; private set; }
118 
119  object IEnumerator.Current => Current;
120 
121  public InjectionEnumerator(IEnumerator<BaseData> underlyingEnumerator)
122  {
123  _underlyingEnumerator = underlyingEnumerator;
124  }
125 
126  public void InjectDataPoint(BaseData baseData)
127  {
128  // we use a lock because the main algorithm thread is the one injecting and the base exchange is the thread pulling MoveNext()
129  lock (_underlyingEnumerator)
130  {
131  _wasInjected = true;
132  Current = baseData;
133  }
134  }
135 
136  public void Dispose()
137  {
138  _underlyingEnumerator.Dispose();
139  }
140 
141  public bool MoveNext()
142  {
143  lock (_underlyingEnumerator)
144  {
145  if (_wasInjected)
146  {
147  _wasInjected = false;
148  return true;
149  }
150  _underlyingEnumerator.MoveNext();
151  Current = _underlyingEnumerator.Current;
152  return true;
153  }
154  }
155 
156  public void Reset()
157  {
158  _underlyingEnumerator.Reset();
159  }
160  }
161  }
