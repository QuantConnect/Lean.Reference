86  /// <summary>

88  /// </summary>
89  [JsonProperty("label")]

91

95  [JsonProperty("productId")]
96  public int ProductId { get; private set; }

103  public string ContractId { get; private set; }
104 
105  /// <summary>

110  }

123  /// <summary>
124  /// Year of expiry
125  /// </summary>
126  [JsonProperty("year")]

128 
129  /// <summary>
