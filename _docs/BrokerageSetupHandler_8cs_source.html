91  /// <summary>
92  /// Create a new instance of an algorithm from a physical dll path.
93  /// </summary>
94  /// <param name="assemblyPath">The path to the assembly's location</param>
95  /// <param name="algorithmNodePacket">Details of the task required</param>
96  /// <returns>A new instance of IAlgorithm, or throws an exception if there was an error</returns>
97  public IAlgorithm CreateAlgorithmInstance(AlgorithmNodePacket algorithmNodePacket, string assemblyPath)
98  {
99  string error;
100  IAlgorithm algorithm;
101 
102  // limit load times to 10 seconds and force the assembly to have exactly one derived type
103  var loader = new Loader(false, algorithmNodePacket.Language, BaseSetupHandler.AlgorithmCreationTimeout, names => names.SingleOrAlgorithmTypeName(Config.Get("algorithm-type-name", algorithmNodePacket.AlgorithmId)), WorkerThread);
104  var complete = loader.TryCreateAlgorithmInstanceWithIsolator(assemblyPath, algorithmNodePacket.RamAllocation, out algorithm, out error);
105  if (!complete) throw new AlgorithmSetupException($"During the algorithm initialization, the following exception has occurred: {error}");
106 
107  return algorithm;
108  }

110  /// <summary>
111  /// Creates the brokerage as specified by the job packet
112  /// </summary>
113  /// <param name="algorithmNodePacket">Job packet</param>
114  /// <param name="uninitializedAlgorithm">The algorithm instance before Initialize has been called</param>
115  /// <param name="factory">The brokerage factory</param>
116  /// <returns>The brokerage instance, or throws if error creating instance</returns>
117  public IBrokerage CreateBrokerage(AlgorithmNodePacket algorithmNodePacket, IAlgorithm uninitializedAlgorithm, out IBrokerageFactory factory)
118  {
119  var liveJob = algorithmNodePacket as LiveNodePacket;
120  if (liveJob == null)
121  {
122  throw new ArgumentException("BrokerageSetupHandler.CreateBrokerage requires a live node packet");
123  }
124 
125  Log.Trace($"BrokerageSetupHandler.CreateBrokerage(): creating brokerage '{liveJob.Brokerage}'");
126 
127  // find the correct brokerage factory based on the specified brokerage in the live job packet
128  _factory = Composer.Instance.Single<IBrokerageFactory>(brokerageFactory => brokerageFactory.BrokerageType.MatchesTypeName(liveJob.Brokerage));
129  factory = _factory;
130 
131  PreloadDataQueueHandler(liveJob, uninitializedAlgorithm, factory);
132 
133  // initialize the correct brokerage using the resolved factory
134  var brokerage = _factory.CreateBrokerage(liveJob, uninitializedAlgorithm);
135 
136  return brokerage;
137  }

139  /// <summary>
140  /// Primary entry point to setup a new algorithm
141  /// </summary>
142  /// <param name="parameters">The parameters object to use</param>
143  /// <returns>True on successfully setting up the algorithm state, or false on error.</returns>
144  public bool Setup(SetupHandlerParameters parameters)
145  {
146  var algorithm = parameters.Algorithm;
147  var brokerage = parameters.Brokerage;
148  // verify we were given the correct job packet type
149  var liveJob = parameters.AlgorithmNodePacket as LiveNodePacket;
150  if (liveJob == null)
151  {
152  AddInitializationError("BrokerageSetupHandler requires a LiveNodePacket");
153  return false;
154  }
155 
156  algorithm.Name = liveJob.GetAlgorithmName();
157 
158  // verify the brokerage was specified
159  if (string.IsNullOrWhiteSpace(liveJob.Brokerage))
160  {
161  AddInitializationError("A brokerage must be specified");
162  return false;
163  }
164 
165 
166  // attach to the message event to relay brokerage specific initialization messages
167  EventHandler<BrokerageMessageEvent> brokerageOnMessage = (sender, args) =>
168  {
169  if (args.Type == BrokerageMessageType.Error)
170  {
171  AddInitializationError($"Brokerage Error Code: {args.Code} - {args.Message}");
172  }
173  };
174 
175  try
176  {
177  // let the world know what we're doing since logging in can take a minute
178  parameters.ResultHandler.SendStatusUpdate(AlgorithmStatus.LoggingIn, "Logging into brokerage...");
179 
180  brokerage.Message += brokerageOnMessage;
181 
182  Log.Trace("BrokerageSetupHandler.Setup(): Connecting to brokerage...");
183  try
184  {
185  // this can fail for various reasons, such as already being logged in somewhere else
186  brokerage.Connect();
187  }
188  catch (Exception err)
189  {
190  Log.Error(err);
191  AddInitializationError(
192  $"Error connecting to brokerage: {err.Message}. " +
193  "This may be caused by incorrect login credentials or an unsupported account type.", err);
194  return false;
195  }
196 
197  if (!brokerage.IsConnected)
198  {
199  // if we're reporting that we're not connected, bail
200  AddInitializationError("Unable to connect to brokerage.");
201  return false;
202  }
203 
204  var message = $"{brokerage.Name} account base currency: {brokerage.AccountBaseCurrency ?? algorithm.AccountCurrency}";
205 
206 
207  var accountCurrency = brokerage.AccountBaseCurrency;
208  if (liveJob.BrokerageData.ContainsKey(MaxAllocationLimitConfig))
209  {
210  accountCurrency = Currencies.USD;
211  message += ". Allocation limited, will use 'USD' account currency";
212  }
213 
214  Log.Trace($"BrokerageSetupHandler.Setup(): {message}");
215 
216  algorithm.Debug(message);
217  if (accountCurrency != null && accountCurrency != algorithm.AccountCurrency)
218  {
219  algorithm.SetAccountCurrency(accountCurrency);
220  }
221 
222  Log.Trace("BrokerageSetupHandler.Setup(): Initializing algorithm...");
223 
224  parameters.ResultHandler.SendStatusUpdate(AlgorithmStatus.Initializing, "Initializing algorithm...");
225 
226  //Execute the initialize code:
227  var controls = liveJob.Controls;
228  var isolator = new Isolator();
229  var initializeComplete = isolator.ExecuteWithTimeLimit(TimeSpan.FromSeconds(300), () =>
230  {
231  try
232  {
233  //Set the default brokerage model before initialize
234  algorithm.SetBrokerageModel(_factory.GetBrokerageModel(algorithm.Transactions));
235 
236  //Margin calls are disabled by default in live mode
237  algorithm.Portfolio.MarginCallModel = MarginCallModel.Null;
238 
239  //Set our parameters
240  algorithm.SetParameters(liveJob.Parameters);
241  algorithm.SetAvailableDataTypes(BaseSetupHandler.GetConfiguredDataFeeds());
242 
243  //Algorithm is live, not backtesting:
244  algorithm.SetAlgorithmMode(liveJob.AlgorithmMode);
245  algorithm.SetDeploymentTarget(liveJob.DeploymentTarget);
246 
247  //Initialize the algorithm's starting date
248  algorithm.SetDateTime(DateTime.UtcNow);
249 
250  //Set the source impl for the event scheduling
251  algorithm.Schedule.SetEventSchedule(parameters.RealTimeHandler);
252 
253  var optionChainProvider = Composer.Instance.GetPart<IOptionChainProvider>();
254  if (optionChainProvider == null)
255  {
256  optionChainProvider = new CachingOptionChainProvider(new LiveOptionChainProvider(parameters.DataCacheProvider, parameters.MapFileProvider));
257  Composer.Instance.AddPart(optionChainProvider);
258  }
259  // set the option chain provider
260  algorithm.SetOptionChainProvider(optionChainProvider);
261 
262  var futureChainProvider = Composer.Instance.GetPart<IFutureChainProvider>();
263  if (futureChainProvider == null)
264  {
265  futureChainProvider = new CachingFutureChainProvider(new LiveFutureChainProvider(parameters.DataCacheProvider));
266  Composer.Instance.AddPart(futureChainProvider);
267  }
268  // set the future chain provider
269  algorithm.SetFutureChainProvider(futureChainProvider);
270 
271  //Initialise the algorithm, get the required data:
272  algorithm.Initialize();
273 
274  if (liveJob.Brokerage != "PaperBrokerage")
275  {
276  //Zero the CashBook - we'll populate directly from brokerage
277  foreach (var kvp in algorithm.Portfolio.CashBook)
278  {
279  kvp.Value.SetAmount(0);
280  }
281  }
282  }
283  catch (Exception err)
284  {
285  AddInitializationError(err.ToString(), err);
286  }
287  }, controls.RamAllocation,
288  sleepIntervalMillis: 100); // entire system is waiting on this, so be as fast as possible
289 
290  if (Errors.Count != 0)
291  {
292  // if we already got an error just exit right away
293  return false;
294  }
295 
296  if (!initializeComplete)
297  {
298  AddInitializationError("Initialization timed out.");
299  return false;
300  }
301 
302  if (!LoadCashBalance(brokerage, algorithm))
303  {
304  return false;
305  }
306 
307  if (!LoadExistingHoldingsAndOrders(brokerage, algorithm, parameters))
308  {
309  return false;
310  }
311 
312  // after algorithm was initialized, should set trading days per year for our great portfolio statistics
313  BaseSetupHandler.SetBrokerageTradingDayPerYear(algorithm);
314 
315  //Finalize Initialization
316  algorithm.PostInitialize();
317 
318  BaseSetupHandler.SetupCurrencyConversions(algorithm, parameters.UniverseSelection);
319 
320  if (algorithm.Portfolio.TotalPortfolioValue == 0)
321  {
322  algorithm.Debug("Warning: No cash balances or holdings were found in the brokerage account.");
323  }
324 
325  string maxCashLimitStr;
326  if (liveJob.BrokerageData.TryGetValue(MaxAllocationLimitConfig, out maxCashLimitStr))
327  {
328  var maxCashLimit = decimal.Parse(maxCashLimitStr, NumberStyles.Any, CultureInfo.InvariantCulture);
329 
330  // If allocation exceeded by more than $10,000; block deployment
331  if (algorithm.Portfolio.TotalPortfolioValue > (maxCashLimit + 10000m))
332  {
333  var exceptionMessage = $"TotalPortfolioValue '{algorithm.Portfolio.TotalPortfolioValue}' exceeds allocation limit '{maxCashLimit}'";
334  algorithm.Debug(exceptionMessage);
335  throw new ArgumentException(exceptionMessage);
336  }
337  }
338 
339  //Set the starting portfolio value for the strategy to calculate performance:
340  StartingPortfolioValue = algorithm.Portfolio.TotalPortfolioValue;
341  StartingDate = DateTime.Now;
342  }
343  catch (Exception err)
344  {
345  AddInitializationError(err.ToString(), err);
346  }
347  finally
348  {
349  if (brokerage != null)
350  {
351  brokerage.Message -= brokerageOnMessage;
352  }
353  }
354 
355  return Errors.Count == 0;
356  }

358  private bool LoadCashBalance(IBrokerage brokerage, IAlgorithm algorithm)
359  {
360  Log.Trace("BrokerageSetupHandler.Setup(): Fetching cash balance from brokerage...");
361  try
362  {
363  // set the algorithm's cash balance for each currency
364  var cashBalance = brokerage.GetCashBalance();
365  foreach (var cash in cashBalance)
366  {
367  Log.Trace($"BrokerageSetupHandler.Setup(): Setting {cash.Currency} cash to {cash.Amount}");
368 
369  algorithm.Portfolio.SetCash(cash.Currency, cash.Amount, 0);
370  }
371  }
372  catch (Exception err)
373  {
374  Log.Error(err);
375  AddInitializationError("Error getting cash balance from brokerage: " + err.Message, err);
376  return false;
377  }
378  return true;
379  }

381  protected bool LoadExistingHoldingsAndOrders(IBrokerage brokerage, IAlgorithm algorithm, SetupHandlerParameters parameters)
382  {
383  Log.Trace("BrokerageSetupHandler.Setup(): Fetching open orders from brokerage...");
384  try
385  {
386  GetOpenOrders(algorithm, parameters.ResultHandler, parameters.TransactionHandler, brokerage);
387  }
388  catch (Exception err)
389  {
390  Log.Error(err);
391  AddInitializationError("Error getting open orders from brokerage: " + err.Message, err);
392  return false;
393  }
394 
395  Log.Trace("BrokerageSetupHandler.Setup(): Fetching holdings from brokerage...");
396  try
397  {
398  var utcNow = DateTime.UtcNow;
399 
400  // populate the algorithm with the account's current holdings
401  var holdings = brokerage.GetAccountHoldings();
402 
403  // add options first to ensure raw data normalization mode is set on the equity underlyings
404  foreach (var holding in holdings.OrderByDescending(x => x.Type))
405  {
406  Log.Trace("BrokerageSetupHandler.Setup(): Has existing holding: " + holding);
407 
408  // verify existing holding security type
409  Security security;
410  if (!GetOrAddUnrequestedSecurity(algorithm, holding.Symbol, holding.Type, out security))
411  {
412  continue;
413  }
414 
415  var exchangeTime = utcNow.ConvertFromUtc(security.Exchange.TimeZone);
416 
417  security.Holdings.SetHoldings(holding.AveragePrice, holding.Quantity);
418 
419  if (holding.MarketPrice == 0)
420  {
421  // try warming current market price
422  holding.MarketPrice = algorithm.GetLastKnownPrice(security)?.Price ?? 0;
423  }
424 
425  if (holding.MarketPrice != 0)
426  {
427  security.SetMarketPrice(new TradeBar
428  {
429  Time = exchangeTime,
430  Open = holding.MarketPrice,
431  High = holding.MarketPrice,
432  Low = holding.MarketPrice,
433  Close = holding.MarketPrice,
434  Volume = 0,
435  Symbol = holding.Symbol,
436  DataType = MarketDataType.TradeBar
437  });
438  }
439  }
440  }
441  catch (Exception err)
442  {
443  Log.Error(err);
444  AddInitializationError("Error getting account holdings from brokerage: " + err.Message, err);
445  return false;
446  }
447 
448  return true;
449  }

451  private bool GetOrAddUnrequestedSecurity(IAlgorithm algorithm, Symbol symbol, SecurityType securityType, out Security security)
452  {
453  return algorithm.GetOrAddUnrequestedSecurity(symbol, out security,
454  onError: (supportedSecurityTypes) => AddInitializationError(
455  "Found unsupported security type in existing brokerage holdings: " + securityType + ". " +
456  "QuantConnect currently supports the following security types: " + string.Join(",", supportedSecurityTypes)));
457  }

459  /// <summary>
460  /// Get the open orders from a brokerage. Adds <see cref="Orders.Order"/> and <see cref="Orders.OrderTicket"/> to the transaction handler
461  /// </summary>
462  /// <param name="algorithm">Algorithm instance</param>
463  /// <param name="resultHandler">The configured result handler</param>
464  /// <param name="transactionHandler">The configurated transaction handler</param>
465  /// <param name="brokerage">Brokerage output instance</param>
466  protected void GetOpenOrders(IAlgorithm algorithm, IResultHandler resultHandler, ITransactionHandler transactionHandler, IBrokerage brokerage)
467  {
468  // populate the algorithm with the account's outstanding orders
469  var openOrders = brokerage.GetOpenOrders();
470 
471  // add options first to ensure raw data normalization mode is set on the equity underlyings
472  foreach (var order in openOrders.OrderByDescending(x => x.SecurityType))
473  {
474  // verify existing holding security type
475  Security security;
476  if (!GetOrAddUnrequestedSecurity(algorithm, order.Symbol, order.SecurityType, out security))
477  {
478  continue;
479  }
480 
481  transactionHandler.AddOpenOrder(order, algorithm);
482  order.PriceCurrency = security?.SymbolProperties.QuoteCurrency;
483 
484  Log.Trace($"BrokerageSetupHandler.Setup(): Has open order: {order}");
485  resultHandler.DebugMessage($"BrokerageSetupHandler.Setup(): Open order detected. Creating order tickets for open order {order.Symbol.Value} with quantity {order.Quantity}. Beware that this order ticket may not accurately reflect the quantity of the order if the open order is partially filled.");
486  }
487  }

523  else
524  {

526  }

530  private void PreloadDataQueueHandler(LiveNodePacket liveJob, IAlgorithm algorithm, IBrokerageFactory factory)
531  {
532  // preload the data queue handler using custom BrokerageFactory attribute
533  var dataQueueHandlerType = Assembly.GetAssembly(typeof(Brokerage))
534  .GetTypes()
535  .FirstOrDefault(x =>
536  x.FullName != null &&
537  x.FullName.EndsWith(liveJob.DataQueueHandler) &&
538  x.HasAttribute(typeof(BrokerageFactoryAttribute)));
539 
540  if (dataQueueHandlerType != null)
541  {
542  var attribute = dataQueueHandlerType.GetCustomAttribute<BrokerageFactoryAttribute>();
543 
544  // only load the data queue handler if the factory is different from our brokerage factory
545  if (attribute.Type != factory.GetType())
546  {
547  var brokerageFactory = (BrokerageFactory)Activator.CreateInstance(attribute.Type);
548 
549  // copy the brokerage data (usually credentials)
550  foreach (var kvp in brokerageFactory.BrokerageData)
551  {
552  if (!liveJob.BrokerageData.ContainsKey(kvp.Key))
553  {
554  liveJob.BrokerageData.Add(kvp.Key, kvp.Value);
555  }
556  }
557 
558  // create the data queue handler and add it to composer
559  _dataQueueHandlerBrokerage = brokerageFactory.CreateBrokerage(liveJob, algorithm);
560 
561  // open connection for subscriptions
562  _dataQueueHandlerBrokerage.Connect();
563  }
564  }
565  }
