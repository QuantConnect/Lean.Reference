26 namespace QuantConnect.ToolBox.IEX
27 {
28  /// <summary>
29  /// Class wraps a collection of clients for getting data on SSE.
30  /// SSE endpoints are limited to 50 symbols per connection. To consume more than 50 symbols we need multiple connections .
31  /// </summary>
32  public class IEXEventSourceCollection : IDisposable
33  {
34  private static readonly TimeSpan TimeoutToUpdate = TimeSpan.FromSeconds(30);
35  private const int SymbolsPerConnectionLimit = 50;
36  private readonly string _apiKey;
37  private readonly EventHandler<MessageReceivedEventArgs> _messageAction;
38  protected readonly ConcurrentDictionary<EventSource, string[]> ClientSymbolsDictionary = new ConcurrentDictionary<EventSource, string[]>();
39  protected readonly CountdownEvent Counter = new CountdownEvent(1);
40 
41  // IEX API documentation says:
42  // "We limit requests to 100 per second per IP measured in milliseconds, so no more than 1 request per 10 milliseconds."
43  private readonly RateGate _rateGate = new RateGate(1, TimeSpan.FromMilliseconds(10));
44 
45  /// <summary>
46  /// Indicates whether a client is connected - i.e delivers any data.
47  /// </summary>
48  public bool IsConnected { get; private set; }
49 
50  /// <summary>
51  /// Creates a new instance of <see cref="IEXEventSourceCollection"/>
52  /// </summary>
53  public IEXEventSourceCollection(EventHandler<MessageReceivedEventArgs> messageAction, string apiKey)
54  {
55  _messageAction = messageAction;
56  _apiKey = apiKey;
57  }
58 
59  /// <summary>
60  /// Updates the data subscription to reflect the current user-subscribed symbols set.
61  /// </summary>
62  /// <param name="symbols">Symbols that user is currently subscribed to</param>
63  /// <returns></returns>
64  public void UpdateSubscription(string[] symbols)
65  {
66  Log.Debug("IEXEventSourceCollection.UpdateSubscription(): Subscription update started");
67 
68  var remainingSymbols = new List<string>(symbols);
69  var clientsToRemove = new List<EventSource>();
70 
71  foreach (var kvp in ClientSymbolsDictionary)
72  {
73  var clientSymbols = kvp.Value;
74 
75  // Need to perform no changes if all client symbols are relevant
76  // and subscription is fully loaded (otherwise it is also a replacement)
77  if (clientSymbols.All(symbols.Contains) && clientSymbols.Length == SymbolsPerConnectionLimit)
78  {
79  Log.Debug($"IEXEventSourceCollection.UpdateSubscription(): Leave unchanged subscription for: {string.Join(",", clientSymbols)}");
80 
81  // Just remove symbols from collection of remaining
82  remainingSymbols.RemoveAll(i => clientSymbols.Contains(i));
83  continue;
84  }
85 
86  clientsToRemove.Add(kvp.Key);
87  }
88 
89  Log.Debug($"IEXEventSourceCollection.UpdateSubscription(): {clientsToRemove.Count} old clients to remove");
90 
91  if (!remainingSymbols.Any())
92  {
93  throw new Exception("IEXEventSourceCollection.UpdateSubscription(): Invalid logic, remaining symbols can't be an empty set.");
94  }
95 
96  // Group all remaining symbols in a smaller packages to comply with per-connection-limits
97  var packagedSymbolsList = new List<string[]>();
98  do
99  {
100  if (remainingSymbols.Count > SymbolsPerConnectionLimit)
101  {
102  var firstFifty = remainingSymbols.Take(SymbolsPerConnectionLimit).ToArray();
103  remainingSymbols.RemoveAll(i => firstFifty.Contains(i));
104  packagedSymbolsList.Add(firstFifty);
105  }
106  else
107  {
108  // Add all remaining symbols as a last package
109  packagedSymbolsList.Add(remainingSymbols.ToArray());
110  break;
111  }
112  }
113  while (remainingSymbols.Any());
114 
115  // Create new client for every package (make sure that we do not exceed the rate-gate-limit while creating)
116  packagedSymbolsList.DoForEach(package =>
117  {
118  Log.Debug($"IEXEventSourceCollection.CreateNewSubscription(): Creating new subscription for: {string.Join(",", package)}");
119 
120  var client = CreateNewSubscription(package);
121 
122  // Add to the dictionary
123  ClientSymbolsDictionary.TryAdd(client, package);
124  });
125 
126  Counter.Signal();
127  if (!Counter.Wait(TimeoutToUpdate))
128  {
129  throw new Exception("IEXEventSourceCollection.UpdateSubscription(): Could not update subscription within a timeout");
130  }
131 
132  clientsToRemove.DoForEach(RemoveOldClient);
133 
134  // Reset counter
135  Log.Debug($"IEXEventSourceCollection.CreateNewSubscription(): Updated successfully. Resetting the counter.");
136  Counter.Reset(1);
137 
138  IsConnected = true;
139  }
140 
141  protected virtual EventSource CreateNewSubscription(string[] symbols)
142  {
143  Counter.AddCount(); // Increment
144  _rateGate.WaitToProceed();
145 
146  var client = CreateNewClient(symbols);
147 
148  // Set up the handlers
149  client.Opened += (sender, args) =>
150  {
151  Log.Debug($"ClientOnOpened(): Sender's hashcode is {sender.GetHashCode()}");
152 
153  Counter.Signal(); // Decrement
154 
155  Log.Debug($"ClientOnOpened(): Counter count after decrement: {Counter.CurrentCount}");
156  };
157 
158  client.MessageReceived += _messageAction;
159 
160  client.Error += (sender, args) =>
161  {
162  var exception = args.Exception;
163  Log.Debug($"ClientOnError(): EventSource Error Occurred. Details: {exception.Message} " +
164  $"ErrorType: {exception.GetType().FullName}");
165  };
166 
167  client.Closed += (sender, args) =>
168  {
169  Log.Debug("ClientOnClosed(): Closing a client");
170  };
171 
172  // Client start call will block until Stop() is called (!) - runs continuously in a background
173  Task.Run(async () => await client.StartAsync().ConfigureAwait(false));
174 
175  return client;
176  }
177 
178  protected EventSource CreateNewClient(string[] symbols)
179  {
180  var url = BuildUrlString(symbols);
181  var client = new EventSource(LaunchDarkly.EventSource.Configuration.Builder(new Uri(url)).Build());
182  return client;
183  }
184 
185  protected virtual void RemoveOldClient(EventSource client)
186  {
187  Log.Debug($"IEXEventSourceCollection.UpdateSubscription(): Remove subscription for: {string.Join(",", ClientSymbolsDictionary[client])}");
188 
189  string[] stub;
190  ClientSymbolsDictionary.TryRemove(client, out stub);
191 
192  client.DisposeSafely();
193  }
194 
195  private string BuildUrlString(IEnumerable<string> symbols)
196  {
197  var url = "https://cloud-sse.iexapis.com/stable/stocksUSNoUTP1Second?token=" + _apiKey;
198  url += "&symbols=" + string.Join(",", symbols);
199  return url;
200  }
201 
202  public void Dispose()
203  {
204  foreach (var client in ClientSymbolsDictionary.Keys)
205  {
206  client.Close();
207  client.DisposeSafely();
208  }
209 
210  IsConnected = false;
211  }
212  }
