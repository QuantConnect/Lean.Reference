17 using System;
18 using System.Collections.Generic;
19 using System.IO;
20 using QuantConnect.Interfaces;
21 
22 namespace QuantConnect.Lean.Engine.DataFeeds.Transport
23 {
24  /// <summary>
25  /// Represents a stream reader capabable of downloading a remote file and then
26  /// reading it from disk
27  /// </summary>
28  public class RemoteFileSubscriptionStreamReader : IStreamReader
29  {
30  private readonly IStreamReader _streamReader;
31  private static IDownloadProvider _downloader;
32  // lock for multi thread scenarios where we are sharing the same cached file
33  private static readonly object _fileSystemLock = new object();
34 
35  /// <summary>
36  /// Gets whether or not this stream reader should be rate limited
37  /// </summary>
38  public bool ShouldBeRateLimited => false;
39 
40  /// <summary>
41  /// Direct access to the StreamReader instance
42  /// </summary>
43  public StreamReader StreamReader => _streamReader.StreamReader;
44 
45  /// <summary>
46  /// Initializes a new instance of the <see cref="RemoteFileSubscriptionStreamReader"/> class.
47  /// </summary>
48  /// <param name="dataCacheProvider">The <see cref="IDataCacheProvider"/> used to retrieve a stream of data</param>
49  /// <param name="source">The remote url to be downloaded via web client</param>
50  /// <param name="downloadDirectory">The local directory and destination of the download</param>
51  /// <param name="headers">Defines header values to add to the request</param>
52  public RemoteFileSubscriptionStreamReader(IDataCacheProvider dataCacheProvider, string source, string downloadDirectory, IEnumerable<KeyValuePair<string, string>> headers)
53  {
54  // don't use cache if data is ephemeral
55  // will be false for live history requests and live subscriptions
56  var useCache = !dataCacheProvider.IsDataEphemeral;
57 
58  // create a hash for a new filename
59  var filename = (useCache ? source.ToMD5() : Guid.NewGuid().ToString()) + source.GetExtension();
60  var destination = Path.Combine(downloadDirectory, filename);
61 
62  string contents = null;
63  if (useCache)
64  {
65  lock (_fileSystemLock)
66  {
67  if (!File.Exists(destination))
68  {
69  contents = _downloader.Download(source, headers, null, null);
70  File.WriteAllText(destination, contents);
71  }
72  }
73  }
74  else
75  {
76  contents = _downloader.Download(source, headers, null, null);
77  File.WriteAllText(destination, contents);
78  }
79 
80  if (contents != null)
81  {
82  // Send the file to the dataCacheProvider so it is available when the streamReader asks for it
83  dataCacheProvider.Store(destination, System.Text.Encoding.UTF8.GetBytes(contents));
84  }
85 
86  // now we can just use the local file reader
87  _streamReader = new LocalFileSubscriptionStreamReader(dataCacheProvider, destination);
88  }
89 
90  /// <summary>
91  /// Gets <see cref="SubscriptionTransportMedium.RemoteFile"/>
92  /// </summary>
93  public SubscriptionTransportMedium TransportMedium
94  {
95  get { return SubscriptionTransportMedium.RemoteFile; }
96  }
97 
98  /// <summary>
99  /// Gets whether or not there's more data to be read in the stream
100  /// </summary>
101  public bool EndOfStream
102  {
103  get { return _streamReader.EndOfStream; }
104  }
105 
106  /// <summary>
107  /// Gets the next line/batch of content from the stream
108  /// </summary>
109  public string ReadLine()
110  {
111  return _streamReader.ReadLine();
112  }
113 
114  /// <summary>
115  /// Disposes of the stream
116  /// </summary>
117  public void Dispose()
118  {
119  _streamReader.Dispose();
120  }
121 
122  /// <summary>
123  /// Save reference to the download system.
124  /// </summary>
125  /// <param name="downloader">Downloader provider for the remote file fetching.</param>
126  public static void SetDownloadProvider(IDownloadProvider downloader)
127  {
128  _downloader = downloader;
129  }
130  }
131 }
