138  /// <summary>
139  /// Tries to update the stop price for a trailing stop order given the current market price
140  /// </summary>
141  /// <param name="currentMarketPrice">The current market price</param>
142  /// <param name="currentStopPrice">The current trailing stop order stop price</param>
143  /// <param name="trailingAmount">The trailing amount to be used to update the stop price</param>
144  /// <param name="trailingAsPercentage">Whether the <paramref name="trailingAmount"/> is a percentage or an absolute currency value</param>
145  /// <param name="direction">The order direction</param>
146  /// <param name="updatedStopPrice">The updated stop price</param>
147  /// <returns>
148  /// Whether the stop price was updated.
149  /// This only happens when the distance between the current stop price and the current market price is greater than the trailing amount,
150  /// which will happen when the market price raises/falls for sell/buy orders respectively.
151  /// </returns>
152  public static bool TryUpdateStopPrice(decimal currentMarketPrice, decimal currentStopPrice, decimal trailingAmount,
153  bool trailingAsPercentage, OrderDirection direction, out decimal updatedStopPrice)
154  {
155  updatedStopPrice = 0m;
156  var distanceToMarketPrice = direction == OrderDirection.Sell
157  ? currentMarketPrice - currentStopPrice
158  : currentStopPrice - currentMarketPrice;
159  var stopReference = trailingAsPercentage ? currentMarketPrice * trailingAmount : trailingAmount;
160 
161  if (distanceToMarketPrice <= stopReference)
162  {
163  return false;
164  }
165 
166  updatedStopPrice = CalculateStopPrice(currentMarketPrice, trailingAmount, trailingAsPercentage, direction);
167  return true;
168  }
169 
170  /// <summary>
171  /// Calculates the stop price for a trailing stop order given the current market price
172  /// </summary>
173  /// <param name="currentMarketPrice">The current market price</param>
174  /// <param name="trailingAmount">The trailing amount to be used to update the stop price</param>
175  /// <param name="trailingAsPercentage">Whether the <paramref name="trailingAmount"/> is a percentage or an absolute currency value</param>
176  /// <param name="direction">The order direction</param>
177  /// <returns>The stop price for the order given the current market price</returns>
178  public static decimal CalculateStopPrice(decimal currentMarketPrice, decimal trailingAmount, bool trailingAsPercentage,
179  OrderDirection direction)
180  {
181  if (trailingAsPercentage)
182  {
183  return direction == OrderDirection.Buy
184  ? currentMarketPrice * (1 + trailingAmount)
185  : currentMarketPrice * (1 - trailingAmount);
186  }
187 
188  return direction == OrderDirection.Buy
189  ? currentMarketPrice + trailingAmount
190  : currentMarketPrice - trailingAmount;
191  }
192  }
