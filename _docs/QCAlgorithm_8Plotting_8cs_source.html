169  /// <summary>
170  /// Plot a value to a chart of string-chart name, with string series name, and decimal value. If chart does not exist, create it.
171  /// </summary>
172  /// <param name="chart">Chart name</param>
173  /// <param name="series">Series name</param>
174  /// <param name="value">Value of the point</param>
175  [DocumentationAttribute(Charting)]
176  public void Plot(string chart, string series, decimal value)
177  {
178  // Check if chart/series names are reserved
179  List<string> reservedSeriesNames;
180  if (ReservedChartSeriesNames.TryGetValue(chart, out reservedSeriesNames))
181  {
182  if (reservedSeriesNames.Count == 0)
183  {
184  throw new Exception($"Algorithm.Plot(): '{chart}' is a reserved chart name.");
185  }
186  if (reservedSeriesNames.Contains(series))
187  {
188  throw new Exception($"Algorithm.Plot(): '{series}' is a reserved series name for chart '{chart}'.");
189  }
190  }
191 
192  // If we don't have the chart, create it:
193  _charts.TryAdd(chart, new Chart(chart));
194 
195  var thisChart = _charts[chart];
196  if (!thisChart.Series.ContainsKey(series))
197  {
198  //Number of series in total, excluding reserved charts
199  var seriesCount = _charts.Select(x => x.Value)
200  .Aggregate(0, (i, c) => ReservedChartSeriesNames.TryGetValue(c.Name, out reservedSeriesNames)
201  ? i + c.Series.Values.Count(s => reservedSeriesNames.Count > 0 && !reservedSeriesNames.Contains(s.Name))
202  : i + c.Series.Count);
203 
204  if (seriesCount > 10)
205  {
206  Error("Exceeded maximum series count: Each backtest can have up to 10 series in total.");
207  return;
208  }
209 
210  //If we don't have the series, create it:
211  thisChart.AddSeries(new Series(series, SeriesType.Line, 0, "$"));
212  }
213 
214  if (LiveMode && IsWarmingUp)
215  {
216  if (!_isEmitWarmupPlotWarningSet)
217  {
218  _isEmitWarmupPlotWarningSet = true;
219  Debug("Plotting is disabled during algorithm warmup in live trading.");
220  }
221  return;
222  }
223  thisChart.Series[series].AddPoint(UtcTime, value);
224  }

226  /// <summary>
227  /// Add a series object for charting. This is useful when initializing charts with
228  /// series other than type = line. If a series exists in the chart with the same name,
229  /// then it is replaced.
230  /// </summary>
231  /// <param name="chart">The chart name</param>
232  /// <param name="series">The series name</param>
233  /// <param name="seriesType">The type of series, i.e, Scatter</param>
234  /// <param name="unit">The unit of the y axis, usually $</param>
235  [DocumentationAttribute(Charting)]
236  public void AddSeries(string chart, string series, SeriesType seriesType, string unit = "$")
237  {
238  Chart c;
239  if (!_charts.TryGetValue(chart, out c))
240  {
241  _charts[chart] = c = new Chart(chart);
242  }
243 
244  c.Series[series] = new Series(series, seriesType, unit);
245  }

272  /// <summary>
273  /// Automatically plots each indicator when a new value is available, optionally waiting for indicator.IsReady to return true
274  /// </summary>
275  [DocumentationAttribute(Charting)]
276  [DocumentationAttribute(Indicators)]
277  public void PlotIndicator(string chart, bool waitForReady, params IndicatorBase[] indicators)
278  {
279  foreach (var i in indicators)
280  {
281  if (i == null) continue;
282 
283  // copy loop variable for usage in closure
284  var ilocal = i;
285  i.Updated += (sender, args) =>
286  {
287  if (!waitForReady || ilocal.IsReady)
288  {
289  Plot(chart, ilocal);
290  }
291  };
292  }
293  }

340  /// <summary>
341  /// Get the chart updates by fetch the recent points added and return for dynamic Charting.
342  /// </summary>
343  /// <param name="clearChartData"></param>
344  /// <returns>List of chart updates since the last request</returns>
345  /// <remarks>GetChartUpdates returns the latest updates since previous request.</remarks>
346  [DocumentationAttribute(Charting)]
347  public List<Chart> GetChartUpdates(bool clearChartData = false)
348  {
349  var updates = _charts.Select(x => x.Value).Select(chart => chart.GetUpdates()).ToList();
350 
351  if (clearChartData)
352  {
353  // we can clear this data out after getting updates to prevent unnecessary memory usage
354  foreach (var chart in _charts)
355  {
356  foreach (var series in chart.Value.Series)
357  {
358  series.Value.Purge();
359  }
360  }
361  }
362  return updates;
363  }
364  }
