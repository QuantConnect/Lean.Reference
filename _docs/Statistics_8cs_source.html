101  /// Run a full set of orders and return a Dictionary of statistics.
102  /// </summary>
103  /// <param name="pointsEquity">Equity value over time.</param>
104  /// <param name="profitLoss">profit loss from trades</param>
105  /// <param name="pointsPerformance"> Daily performance</param>
106  /// <param name="unsortedBenchmark"> Benchmark data as dictionary. Data does not need to be ordered</param>
107  /// <param name="startingCash">Amount of starting cash in USD </param>
108  /// <param name="totalFees">The total fees incurred over the life time of the algorithm</param>
109  /// <param name="totalTrades">Total number of orders executed.</param>

112  public static Dictionary<string, string> Generate(IEnumerable<ChartPoint> pointsEquity,
113  SortedDictionary<DateTime, decimal> profitLoss,
114  IEnumerable<ChartPoint> pointsPerformance,
115  Dictionary<DateTime, decimal> unsortedBenchmark,

117  decimal totalFees,
118  decimal totalTrades,

122  //Initialise the response:

124  decimal totalClosedTrades = 0;
125  decimal totalWins = 0;
126  decimal totalLosses = 0;
127  decimal averageWin = 0;
128  decimal averageLoss = 0;
129  decimal averageWinRatio = 0;
130  decimal winRate = 0;
131  decimal lossRate = 0;
132  decimal totalNetProfit = 0;
133  double fractionOfYears = 1;
134  decimal profitLossValue = 0, runningCash = startingCash;
135  decimal algoCompoundingPerformance = 0;
136  decimal finalBenchmarkCash = 0;
137  decimal benchCompoundingPerformance = 0;
138  var years = new List<int>();
139  var annualTrades = new SortedDictionary<int, int>();
140  var annualWins = new SortedDictionary<int, int>();
141  var annualLosses = new SortedDictionary<int, int>();
142  var annualLossTotal = new SortedDictionary<int, decimal>();
143  var annualWinTotal = new SortedDictionary<int, decimal>();
144  var annualNetProfit = new SortedDictionary<int, decimal>();
145  var statistics = new Dictionary<string, string>();
146  var dtPrevious = new DateTime();
147  var listPerformance = new List<double>();
148  var listBenchmark = new List<double>();
149  var equity = new SortedDictionary<DateTime, decimal>();
150  var performance = new SortedDictionary<DateTime, decimal>();
151  SortedDictionary<DateTime, decimal> benchmark = null;
152  try
153  {
154  //Get array versions of the performance:
155  performance = ChartPointToDictionary(pointsPerformance);
156  equity = ChartPointToDictionary(pointsEquity);
157  performance.Values.ToList().ForEach(i => listPerformance.Add((double)(i / 100)));
158  benchmark = new SortedDictionary<DateTime, decimal>(unsortedBenchmark);
159 
160  // to find the delta in benchmark for first day, we need to know the price at the opening
161  // moment of the day, but since we cannot find this, we cannot find the first benchmark's delta,
162  // so we pad it with Zero. If running a short backtest this will skew results, longer backtests
163  // will not be affected much
164  listBenchmark.Add(0);
165 
166  //Get benchmark performance array for same period:
167  benchmark.Keys.ToList().ForEach(dt =>
168  {
169  if (dt >= equity.Keys.FirstOrDefault().AddDays(-1) && dt < equity.Keys.LastOrDefault())
170  {
171  decimal previous;
172  if (benchmark.TryGetValue(dtPrevious, out previous) && previous != 0)
173  {
174  var deltaBenchmark = (benchmark[dt] - previous)/previous;
175  listBenchmark.Add((double)(deltaBenchmark));
176  }
177  else
178  {
179  listBenchmark.Add(0);
180  }
181  dtPrevious = dt;
182  }
183  });
184 
185  // TODO : if these lists are required to be the same length then we should create structure to pair the values, this way, by contract it will be enforced.
186 
187  //THIS SHOULD NEVER HAPPEN --> But if it does, log it and fail silently.
188  while (listPerformance.Count < listBenchmark.Count)
189  {
190  listPerformance.Add(0);
191  Log.Error("Statistics.Generate(): Padded Performance");
192  }
193  while (listPerformance.Count > listBenchmark.Count)
194  {
195  listBenchmark.Add(0);
196  Log.Error("Statistics.Generate(): Padded Benchmark");
197  }
198  }
199  catch (Exception err)
200  {
201  Log.Error(err, "Dic-Array Convert:");
202  }
203 
204  try
205  {
206  //Number of years in this dataset:
207  fractionOfYears = (equity.Keys.LastOrDefault() - equity.Keys.FirstOrDefault()).TotalDays / 365;
208  }
209  catch (Exception err)
210  {
211  Log.Error(err, "Fraction of Years:");
212  }
213 
214  try
215  {
216  if (benchmark != null)
217  {
218  algoCompoundingPerformance = CompoundingAnnualPerformance(startingCash, equity.Values.LastOrDefault(), (decimal) fractionOfYears);
219  finalBenchmarkCash = ((benchmark.Values.Last() - benchmark.Values.First())/benchmark.Values.First())*startingCash;
220  benchCompoundingPerformance = CompoundingAnnualPerformance(startingCash, finalBenchmarkCash, (decimal) fractionOfYears);
221  }
222  }
223  catch (Exception err)
224  {
225  Log.Error(err, "Compounding:");
226  }
227 
228  try
229  {
230  //Run over each equity day:
231  foreach (var closedTrade in profitLoss.Keys)
232  {
233  profitLossValue = profitLoss[closedTrade];
234 
235  //Check if this date is in the "years" array:
236  var year = closedTrade.Year;
237  if (!years.Contains(year))
238  {
239  //Initialise a new year holder:
240  years.Add(year);
241  annualTrades.Add(year, 0);
242  annualWins.Add(year, 0);
243  annualWinTotal.Add(year, 0);
244  annualLosses.Add(year, 0);
245  annualLossTotal.Add(year, 0);
246  }
247 
248  //Add another trade:
249  annualTrades[year]++;
250 
251  //Profit loss tracking:
252  if (profitLossValue > 0)
253  {
254  annualWins[year]++;
255  annualWinTotal[year] += profitLossValue / runningCash;
256  }
257  else
258  {
259  annualLosses[year]++;
260  annualLossTotal[year] += profitLossValue / runningCash;
261  }
262 
263  //Increment the cash:
264  runningCash += profitLossValue;
265  }
266 
267  //Get the annual percentage of profit and loss:
268  foreach (var year in years)
269  {
270  annualNetProfit[year] = (annualWinTotal[year] + annualLossTotal[year]);
271  }
272 
273  //Sum the totals:
274  try
275  {
276  if (profitLoss.Keys.Count > 0)
277  {
278  totalClosedTrades = annualTrades.Values.Sum();
279  totalWins = annualWins.Values.Sum();
280  totalLosses = annualLosses.Values.Sum();
281  totalNetProfit = (equity.Values.LastOrDefault() / startingCash) - 1;
282 
283  //-> Handle Div/0 Errors
284  if (totalWins == 0)
285  {
286  averageWin = 0;
287  }
288  else
289  {
290  averageWin = annualWinTotal.Values.Sum() / totalWins;
291  }
292  if (totalLosses == 0)
293  {
294  averageLoss = 0;
295  averageWinRatio = 0;
296  }
297  else
298  {
299  averageLoss = annualLossTotal.Values.Sum() / totalLosses;
300  averageWinRatio = Math.Abs(averageWin / averageLoss);
301  }
302  if (totalTrades == 0)
303  {
304  winRate = 0;
305  lossRate = 0;
306  }
307  else
308  {
309  winRate = Math.Round(totalWins / totalClosedTrades, 5);
310  lossRate = Math.Round(totalLosses / totalClosedTrades, 5);
311  }
312  }
313 
314  }
315  catch (Exception err)
316  {
317  Log.Error(err, "Second Half:");
318  }
319 
320  var profitLossRatio = ProfitLossRatio(averageWin, averageLoss);
321  var profitLossRatioHuman = profitLossRatio.ToString(CultureInfo.InvariantCulture);
322  if (profitLossRatio == -1) profitLossRatioHuman = "0";
323 
324  //Add the over all results first, break down by year later:
325  statistics = new Dictionary<string, string> {
326  { "Total Trades", Math.Round(totalTrades, 0).ToStringInvariant() },
327  { "Average Win", Math.Round(averageWin * 100, 2).ToStringInvariant() + "%" },
328  { "Average Loss", Math.Round(averageLoss * 100, 2).ToStringInvariant() + "%" },
329  { "Compounding Annual Return", Math.Round(algoCompoundingPerformance * 100, 3).ToStringInvariant() + "%" },
330  { "Drawdown", (DrawdownPercent(equity, 3) * 100).ToStringInvariant() + "%" },
331  { "Expectancy", Math.Round((winRate * averageWinRatio) - (lossRate), 3).ToStringInvariant() },
332  { "Net Profit", Math.Round(totalNetProfit * 100, 3).ToStringInvariant() + "%"},
333  { "Sharpe Ratio", Math.Round(SharpeRatio(listPerformance, riskFreeRate), 3).ToStringInvariant() },
334  { "Loss Rate", Math.Round(lossRate * 100).ToStringInvariant() + "%" },
335  { "Win Rate", Math.Round(winRate * 100).ToStringInvariant() + "%" },
336  { "Profit-Loss Ratio", profitLossRatioHuman },
337  { "Alpha", Math.Round(Alpha(listPerformance, listBenchmark, riskFreeRate), 3).ToStringInvariant() },
338  { "Beta", Math.Round(Beta(listPerformance, listBenchmark), 3).ToStringInvariant() },
339  { "Annual Standard Deviation", Math.Round(AnnualStandardDeviation(listPerformance, tradingDaysPerYear), 3).ToStringInvariant() },
340  { "Annual Variance", Math.Round(AnnualVariance(listPerformance, tradingDaysPerYear), 3).ToStringInvariant() },
341  { "Information Ratio", Math.Round(InformationRatio(listPerformance, listBenchmark), 3).ToStringInvariant() },
342  { "Tracking Error", Math.Round(TrackingError(listPerformance, listBenchmark), 3).ToStringInvariant() },
343  { "Treynor Ratio", Math.Round(TreynorRatio(listPerformance, listBenchmark, riskFreeRate), 3).ToStringInvariant() },
344  { "Total Fees", "$" + totalFees.ToStringInvariant("0.00") }
345  };
346  }
347  catch (Exception err)
348  {
349  Log.Error(err);
350  }
351  return statistics;
352  }

366  /// <summary>
367  /// Drawdown maximum percentage.
368  /// </summary>
369  /// <param name="equityOverTime"></param>
370  /// <param name="rounding"></param>
371  /// <returns></returns>
372  public static decimal DrawdownPercent(SortedDictionary<DateTime, decimal> equityOverTime, int rounding = 2)
373  {
374  var dd = 0m;
375  try
376  {
377  var lPrices = equityOverTime.Values.ToList();
378  var lDrawdowns = new List<decimal>();
379  var high = lPrices[0];
380  foreach (var price in lPrices)
381  {
382  if (price >= high) high = price;
383  lDrawdowns.Add((price/high) - 1);
384  }
385  dd = Math.Round(Math.Abs(lDrawdowns.Min()), rounding);
386  }
387  catch (Exception err)
388  {
389  Log.Error(err);
390  }
391  return dd;

394  /// <summary>
395  /// Drawdown maximum value
396  /// </summary>
397  /// <param name="equityOverTime">Array of portfolio value over time.</param>
398  /// <param name="rounding">Round the drawdown statistics.</param>
399  /// <returns>Draw down percentage over period.</returns>
400  public static decimal DrawdownValue(SortedDictionary<DateTime, decimal> equityOverTime, int rounding = 2)
401  {
402  //Initialise:
403  var priceMaximum = 0;
404  var previousMinimum = 0;
405  var previousMaximum = 0;
406 
407  try
408  {
409  var lPrices = equityOverTime.Values.ToList();
410 
411  for (var id = 0; id < lPrices.Count; id++)
412  {
413  if (lPrices[id] >= lPrices[priceMaximum])
414  {
415  priceMaximum = id;
416  }
417  else
418  {
419  if ((lPrices[priceMaximum] - lPrices[id]) > (lPrices[previousMaximum] - lPrices[previousMinimum]))
420  {
421  previousMaximum = priceMaximum;
422  previousMinimum = id;
423  }
424  }
425  }
426  return Math.Round((lPrices[previousMaximum] - lPrices[previousMinimum]), rounding);
427  }
428  catch (Exception err)
429  {
430  Log.Error(err);
431  }
432  return 0;
433  } // End Drawdown:

436  /// <summary>
437  /// Annual compounded returns statistic based on the final-starting capital and years.
438  /// </summary>
439  /// <param name="startingCapital">Algorithm starting capital</param>
440  /// <param name="finalCapital">Algorithm final capital</param>
441  /// <param name="years">Years trading</param>
442  /// <returns>Decimal fraction for annual compounding performance</returns>
443  public static decimal CompoundingAnnualPerformance(decimal startingCapital, decimal finalCapital, decimal years)
444  {
445  if (years == 0 || startingCapital == 0)
446  {
447  return 0;
448  }
449  var power = 1 / (double)years;
450  var baseNumber = (double)finalCapital / (double)startingCapital;
451  var result = Math.Pow(baseNumber, power) - 1;
452  return result.IsNaNOrInfinity() ? 0 : result.SafeDecimalCast();
453  }

455  /// <summary>
456  /// Annualized return statistic calculated as an average of daily trading performance multiplied by the number of trading days per year.
457  /// </summary>
458  /// <param name="performance">Dictionary collection of double performance values</param>
459  /// <param name="tradingDaysPerYear">Trading days per year for the assets in portfolio</param>
460  /// <remarks>May be unaccurate for forex algorithms with more trading days in a year</remarks>
461  /// <returns>Double annual performance percentage</returns>
462  public static double AnnualPerformance(List<double> performance, double tradingDaysPerYear = 252)
463  {
464  return Math.Pow((performance.Average() + 1), tradingDaysPerYear) - 1;
465  }

479  /// <summary>
480  /// Annualized standard deviation
481  /// </summary>
482  /// <param name="performance">Collection of double values for daily performance</param>
483  /// <param name="tradingDaysPerYear">Number of trading days for the assets in portfolio to get annualize standard deviation.</param>
484  /// <remarks>
485  /// Invokes the variance extension in the MathNet Statistics class.
486  /// Feasibly the trading days per year can be fetched from the dictionary of performance which includes the date-times to get the range; if is more than 1 year data.

494  /// <summary>
495  /// Algorithm "beta" statistic - the covariance between the algorithm and benchmark performance, divided by benchmark's variance
496  /// </summary>
497  /// <param name="algoPerformance">Collection of double values for algorithm daily performance.</param>
498  /// <param name="benchmarkPerformance">Collection of double benchmark daily performance values.</param>
499  /// <remarks>Invokes the variance and covariance extensions in the MathNet Statistics class</remarks>
500  /// <returns>Value for beta</returns>
501  public static double Beta(List<double> algoPerformance, List<double> benchmarkPerformance)
502  {
503  return algoPerformance.Covariance(benchmarkPerformance) / benchmarkPerformance.Variance();
504  }

506  /// <summary>
507  /// Algorithm "Alpha" statistic - abnormal returns over the risk free rate and the relationshio (beta) with the benchmark returns.
508  /// </summary>
509  /// <param name="algoPerformance">Collection of double algorithm daily performance values.</param>
510  /// <param name="benchmarkPerformance">Collection of double benchmark daily performance values.</param>
511  /// <param name="riskFreeRate">Risk free rate of return for the T-Bonds.</param>
512  /// <returns>Value for alpha</returns>
513  public static double Alpha(List<double> algoPerformance, List<double> benchmarkPerformance, double riskFreeRate)
514  {
515  return AnnualPerformance(algoPerformance) - (riskFreeRate + Beta(algoPerformance, benchmarkPerformance) * (AnnualPerformance(benchmarkPerformance) - riskFreeRate));
516  }

518  /// <summary>
519  /// Tracking error volatility (TEV) statistic - a measure of how closely a portfolio follows the index to which it is benchmarked
520  /// </summary>
521  /// <remarks>If algo = benchmark, TEV = 0</remarks>
522  /// <param name="algoPerformance">Double collection of algorithm daily performance values</param>
523  /// <param name="benchmarkPerformance">Double collection of benchmark daily performance values</param>
524  /// <param name="tradingDaysPerYear">Number of trading days per year</param>
525  /// <returns>Value for tracking error</returns>
526  public static double TrackingError(List<double> algoPerformance, List<double> benchmarkPerformance, double tradingDaysPerYear = 252)
527  {
528  // Un-equal lengths will blow up other statistics, but this will handle the case here
529  if (algoPerformance.Count() != benchmarkPerformance.Count())
530  {
531  return 0.0;
532  }
533 
534  var performanceDifference = new List<double>();
535  for (var i = 0; i < algoPerformance.Count(); i++)
536  {
537  performanceDifference.Add(algoPerformance[i] - benchmarkPerformance[i]);
538  }
539 
540  return Math.Sqrt(AnnualVariance(performanceDifference, tradingDaysPerYear));
541  }

544  /// <summary>
545  /// Information ratio - risk adjusted return
546  /// </summary>
547  /// <param name="algoPerformance">Collection of doubles for the daily algorithm daily performance</param>
548  /// <param name="benchmarkPerformance">Collection of doubles for the benchmark daily performance</param>
549  /// <remarks>(risk = tracking error volatility, a volatility measures that considers the volatility of both algo and benchmark)</remarks>
550  /// <seealso cref="TrackingError"/>
551  /// <returns>Value for information ratio</returns>
552  public static double InformationRatio(List<double> algoPerformance, List<double> benchmarkPerformance)
553  {
554  return (AnnualPerformance(algoPerformance) - AnnualPerformance(benchmarkPerformance)) / (TrackingError(algoPerformance, benchmarkPerformance));
555  }

557  /// <summary>
558  /// Sharpe ratio with respect to risk free rate: measures excess of return per unit of risk.
559  /// </summary>
560  /// <remarks>With risk defined as the algorithm's volatility</remarks>
561  /// <param name="algoPerformance">Collection of double values for the algorithm daily performance</param>
562  /// <param name="riskFreeRate"></param>
563  /// <returns>Value for sharpe ratio</returns>
564  public static double SharpeRatio(List<double> algoPerformance, double riskFreeRate)
565  {
566  return (AnnualPerformance(algoPerformance) - riskFreeRate) / (AnnualStandardDeviation(algoPerformance));
567  }

569  /// <summary>
570  /// Treynor ratio statistic is a measurement of the returns earned in excess of that which could have been earned on an investment that has no diversifiable risk
571  /// </summary>
572  /// <param name="algoPerformance">Collection of double algorithm daily performance values</param>
573  /// <param name="benchmarkPerformance">Collection of double benchmark daily performance values</param>
574  /// <param name="riskFreeRate">Risk free rate of return</param>
575  /// <returns>double Treynor ratio</returns>
576  public static double TreynorRatio(List<double> algoPerformance, List<double> benchmarkPerformance, double riskFreeRate)
577  {
578  return (AnnualPerformance(algoPerformance) - riskFreeRate) / (Beta(algoPerformance, benchmarkPerformance));

581  /// <summary>
582  /// Helper method to calculate the probabilistic sharpe ratio
583  /// </summary>
584  /// <param name="listPerformance">The list of algorithm performance values</param>
585  /// <param name="benchmarkSharpeRatio">The benchmark sharpe ratio to use</param>
586  /// <returns>Probabilistic Sharpe Ratio</returns>
587  public static double ProbabilisticSharpeRatio(List<double> listPerformance,
588  double benchmarkSharpeRatio)
589  {
590  var observedSharpeRatio = ObservedSharpeRatio(listPerformance);
591 
592  var skewness = listPerformance.Skewness();
593  var kurtosis = listPerformance.Kurtosis();
594 
595  var operandA = skewness * observedSharpeRatio;
596  var operandB = ((kurtosis - 1) / 4) * (Math.Pow(observedSharpeRatio, 2));
597 
598  // Calculated standard deviation of point estimate
599  var estimateStandardDeviation = Math.Pow((1 - operandA + operandB) / (listPerformance.Count - 1), 0.5);
600 
601  if (double.IsNaN(estimateStandardDeviation))
602  {
603  return 0;
604  }
605 
606  // Calculate PSR(benchmark)
607  var value = estimateStandardDeviation.IsNaNOrZero() ? 0 : (observedSharpeRatio - benchmarkSharpeRatio) / estimateStandardDeviation;
608  return (new Normal()).CumulativeDistribution(value);
609  }

611  /// <summary>
612  /// Calculates the observed sharpe ratio
613  /// </summary>
614  /// <param name="listPerformance">The performance samples to use</param>
615  /// <returns>The observed sharpe ratio</returns>
616  public static double ObservedSharpeRatio(List<double> listPerformance)
617  {
618  var performanceAverage = listPerformance.Average();
619  var standardDeviation = listPerformance.StandardDeviation();
620  // we don't annualize it
621  return standardDeviation.IsNaNOrZero() ? 0 : performanceAverage / standardDeviation;
622  }
