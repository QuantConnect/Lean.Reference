33  /// <summary>
34  /// Data provider which downloads data using an <see cref="IDataDownloader"/> or <see cref="IBrokerage"/> implementation
35  /// </summary>
36  public class DownloaderDataProvider : BaseDownloaderDataProvider
37  {
38  /// <summary>
39  /// Synchronizer in charge of guaranteeing a single operation per file path
40  /// </summary>
41  private readonly static KeyStringSynchronizer DiskSynchronizer = new();
42 
43  private bool _customDataDownloadError;
44  private readonly ConcurrentDictionary<Symbol, Symbol> _marketHoursWarning = new();
45  private readonly MarketHoursDatabase _marketHoursDatabase = MarketHoursDatabase.FromDataFolder();
46  private readonly IDataDownloader _dataDownloader;
47  private readonly IDataCacheProvider _dataCacheProvider = new DiskDataCacheProvider(DiskSynchronizer);
48  private readonly IMapFileProvider _mapFileProvider = Composer.Instance.GetPart<IMapFileProvider>();
49 
50  /// <summary>
51  /// Creates a new instance
52  /// </summary>
53  public DownloaderDataProvider()
54  {
55  var dataDownloaderConfig = Config.Get("data-downloader");
56  if (!string.IsNullOrEmpty(dataDownloaderConfig))
57  {
58  _dataDownloader = Composer.Instance.GetExportedValueByTypeName<IDataDownloader>(dataDownloaderConfig);
59  }
60  else
61  {
62  throw new ArgumentException("DownloaderDataProvider(): requires 'data-downloader' to be set with a valid type name");
63  }
64  }

67  /// Creates a new instance using a target data downloader used for testing
68  /// </summary>
69  public DownloaderDataProvider(IDataDownloader dataDownloader)
70  {
71  _dataDownloader = dataDownloader;

74  /// <summary>
75  /// Determines if it should downloads new data and retrieves data from disc
76  /// </summary>
77  /// <param name="key">A string representing where the data is stored</param>
78  /// <returns>A <see cref="Stream"/> of the data requested</returns>
79  public override Stream Fetch(string key)
80  {
81  return DownloadOnce(key, s =>
82  {
83  if (LeanData.TryParsePath(key, out var symbol, out var date, out var resolution, out var tickType, out var dataType))
84  {
85  if (symbol.SecurityType == SecurityType.Base)
86  {
87  if (!_customDataDownloadError)
88  {
89  _customDataDownloadError = true;
90  // lean data writter doesn't support it
91  Log.Trace($"DownloaderDataProvider.Get(): custom data is not supported, requested: {symbol}");
92  }
93  return;
94  }
95 
96  MarketHoursDatabase.Entry entry;
97  try
98  {
99  entry = _marketHoursDatabase.GetEntry(symbol.ID.Market, symbol, symbol.SecurityType);
100  }
101  catch
102  {
103  // this could happen for some sources using the data provider but with not market hours data base entry, like interest rates
104  if (_marketHoursWarning.TryAdd(symbol, symbol))
105  {
106  // log once
107  Log.Trace($"DownloaderDataProvider.Get(): failed to find market hours for {symbol}, skipping");
108  }
109  // this shouldn't happen for data we want can download
110  return;
111  }
112 
113  var dataTimeZone = entry.DataTimeZone;
114  var exchangeTimeZone = entry.ExchangeHours.TimeZone;
115  DateTime startTimeUtc;
116  DateTime endTimeUtc;
117  // we will download until yesterday so we are sure we don't get partial data
118  var endTimeUtcLimit = DateTime.UtcNow.Date.AddDays(-1);
119  if (resolution < Resolution.Hour)
120  {
121  // we can get the date from the path
122  startTimeUtc = date.ConvertToUtc(dataTimeZone);
123  // let's get the whole day
124  endTimeUtc = date.AddDays(1).ConvertToUtc(dataTimeZone);
125  if (endTimeUtc > endTimeUtcLimit)
126  {
127  // we are at the limit, avoid getting partial data
128  return;
129  }
130  }
131  else
132  {
133  // since hourly & daily are a single file we fetch the whole file
134  try
135  {
136  // we don't really know when Futures, FutureOptions, Cryptos, etc, start date so let's give it a good guess
137  if (symbol.SecurityType == SecurityType.Crypto)
138  {
139  // bitcoin start
140  startTimeUtc = new DateTime(2009, 1, 1);
141  }
142  else if (symbol.SecurityType.IsOption() && symbol.Underlying != null && symbol.SecurityType == SecurityType.Option)
143  {
144  // the underlying equity start date
145  startTimeUtc = symbol.Underlying.ID.Date;
146  }
147  else
148  {
149  startTimeUtc = symbol.ID.Date;
150  }
151  }
152  catch (InvalidOperationException)
153  {
154  startTimeUtc = Time.Start;
155  }
156 
157  if (startTimeUtc < Time.Start)
158  {
159  startTimeUtc = Time.Start;
160  }
161 
162  endTimeUtc = endTimeUtcLimit;
163  }
164 
165  try
166  {
167  LeanDataWriter writer = null;
168  var getParams = new DataDownloaderGetParameters(symbol, resolution, startTimeUtc, endTimeUtc, tickType);
169 
170  var downloaderDataParameters = getParams.GetDataDownloaderParameterForAllMappedSymbols(_mapFileProvider, exchangeTimeZone);
171 
172  var downloadedData = GetDownloadedData(downloaderDataParameters, symbol, exchangeTimeZone, dataTimeZone, dataType);
173 
174  foreach (var dataPerSymbol in downloadedData)
175  {
176  if (writer == null)
177  {
178  writer = new LeanDataWriter(resolution, symbol, Globals.DataFolder, tickType, mapSymbol: true, dataCacheProvider: _dataCacheProvider);
179  }
180  // Save the data
181  writer.Write(dataPerSymbol);
182  }
183  }
184  catch (Exception e)
185  {
186  Log.Error(e);
187  }
188  }
189  });
190  }

192  /// <summary>
193  /// Retrieves downloaded data grouped by symbol based on <see cref="IDownloadProvider"/>.
194  /// </summary>
195  /// <param name="downloaderDataParameters">Parameters specifying the data to be retrieved.</param>
196  /// <param name="symbol">Represents a unique security identifier, generate by ticker name.</param>
197  /// <param name="exchangeTimeZone">The time zone of the exchange where the symbol is traded.</param>
198  /// <param name="dataTimeZone">The time zone in which the data is represented.</param>
199  /// <param name="dataType">The type of data to be retrieved. (e.g. <see cref="Data.Market.TradeBar"/>)</param>
200  /// <returns>An IEnumerable containing groups of data grouped by symbol. Each group contains data related to a specific symbol.</returns>
201  /// <exception cref="ArgumentException"> Thrown when the downloaderDataParameters collection is null or empty.</exception>
202  public IEnumerable<IGrouping<Symbol, BaseData>> GetDownloadedData(
203  IEnumerable<DataDownloaderGetParameters> downloaderDataParameters,
204  Symbol symbol,
205  DateTimeZone exchangeTimeZone,
206  DateTimeZone dataTimeZone,
207  Type dataType)
208  {
209  if (downloaderDataParameters.IsNullOrEmpty())
210  {
211  throw new ArgumentException($"{nameof(DownloaderDataProvider)}.{nameof(GetDownloadedData)}: DataDownloaderGetParameters are empty or equal to null.");
212  }
213 
214  foreach (var downloaderDataParameter in downloaderDataParameters)
215  {
216  var downloadedData = _dataDownloader.Get(downloaderDataParameter);
217 
218  if (downloadedData == null)
219  {
220  // doesn't support this download request, that's okay
221  continue;
222  }
223  var startDateTimeInExchangeTimeZone = downloaderDataParameter.StartUtc.ConvertFromUtc(exchangeTimeZone);
224  var endDateTimeInExchangeTimeZone = downloaderDataParameter.EndUtc.ConvertFromUtc(exchangeTimeZone);
225 
226  var groupedData = downloadedData
227  .Where(baseData =>
228  {
229  // Sometimes, external Downloader provider returns excess data
230  if (baseData.Time < startDateTimeInExchangeTimeZone || baseData.Time > endDateTimeInExchangeTimeZone)
231  {
232  return false;
233  }
234 
235  if (symbol.SecurityType == SecurityType.Base || baseData.GetType() == dataType)
236  {
237  // we need to store the data in data time zone
238  baseData.Time = baseData.Time.ConvertTo(exchangeTimeZone, dataTimeZone);
239  baseData.EndTime = baseData.EndTime.ConvertTo(exchangeTimeZone, dataTimeZone);
240  return true;
241  }
242  return false;
243  })
244  // for canonical symbols, downloader will return data for all of the chain
245  .GroupBy(baseData => baseData.Symbol);
246 
247  foreach (var data in groupedData)
248  {
249  yield return data;
250  }
251  }
252  }

254  /// <summary>
255  /// Get's the stream for a given file path
256  /// </summary>
257  protected override Stream GetStream(string key)
258  {
259  if (LeanData.TryParsePath(key, out var symbol, out var date, out var resolution) && resolution > Resolution.Minute && symbol.RequiresMapping())
260  {
261  // because the file could be updated even after it's created because of symbol mapping we can't stream from disk
262  return DiskSynchronizer.Execute(key, () =>
263  {
264  var baseStream = base.Fetch(key);
265  if (baseStream != null)
266  {
267  var result = new MemoryStream();
268  baseStream.CopyTo(result);
269  baseStream.Dispose();
270  // move position back to the start
271  result.Position = 0;
272 
273  return result;
274  }
275  return null;
276  });
277  }
278 
279  return base.Fetch(key);
280  }
281 
282  /// <summary>
283  /// Main filter to determine if this file needs to be downloaded
284  /// </summary>
285  /// <param name="filePath">File we are looking at</param>
286  /// <returns>True if should download</returns>
287  protected override bool NeedToDownload(string filePath)
288  {
289  // Ignore null and invalid data requests
290  if (filePath == null
291  || filePath.Contains("fine", StringComparison.InvariantCultureIgnoreCase) && filePath.Contains("fundamental", StringComparison.InvariantCultureIgnoreCase)
292  || filePath.Contains("map_files", StringComparison.InvariantCultureIgnoreCase)
293  || filePath.Contains("factor_files", StringComparison.InvariantCultureIgnoreCase)
294  || filePath.Contains("margins", StringComparison.InvariantCultureIgnoreCase) && filePath.Contains("future", StringComparison.InvariantCultureIgnoreCase))
295  {
296  return false;
297  }
298 
299  // Only download if it doesn't exist or is out of date.
300  // Files are only "out of date" for non date based files (hour, daily, margins, etc.) because this data is stored all in one file
301  return !File.Exists(filePath) || filePath.IsOutOfDate();
302  }
303  }
