17 using System;
18 using System.IO;
19 using System.Linq;
20 using QuantConnect.Util;
21 using QuantConnect.Data;
22 using QuantConnect.Logging;
23 using QuantConnect.Securities;

27 namespace QuantConnect.Lean.Engine.DataFeeds
28 {
29  /// <summary>
30  /// Data provider which downloads data using an <see cref="IDataDownloader"/> or <see cref="IBrokerage"/> implementation
31  /// </summary>
32  public class DownloaderDataProvider : BaseDownloaderDataProvider
33  {
34  private readonly IDataDownloader _dataDownloader;
35 
36  /// <summary>
37  /// Creates a new instance
38  /// </summary>
39  public DownloaderDataProvider()
40  {
41  var dataDownloaderConfig = Config.Get("data-downloader");
42  if (!string.IsNullOrEmpty(dataDownloaderConfig))
43  {
44  _dataDownloader = Composer.Instance.GetExportedValueByTypeName<IDataDownloader>(dataDownloaderConfig);
45  }
46  else
47  {
48  throw new ArgumentException("DownloaderDataProvider(): requires 'data-downloader' to be set with a valid type name");
49  }
50  }
51 
52  /// <summary>
53  /// Determines if it should downloads new data and retrieves data from disc
54  /// </summary>
55  /// <param name="key">A string representing where the data is stored</param>
56  /// <returns>A <see cref="Stream"/> of the data requested</returns>
57  public override Stream Fetch(string key)
58  {
59  return DownloadOnce(key, s =>
60  {
61  if (LeanData.TryParsePath(key, out var symbol, out var date, out var resolution, out var tickType, out var dataType))
62  {
63  var dataTimeZone = MarketHoursDatabase.FromDataFolder().GetDataTimeZone(symbol.ID.Market, symbol, symbol.SecurityType);
64 
65  DateTime startTimeUtc;
66  DateTime endTimeUtc;
67  if (resolution < Resolution.Hour)
68  {
69  // we can get the date from the path
70  startTimeUtc = date.ConvertToUtc(dataTimeZone);
71  endTimeUtc = date.Add(resolution.ToTimeSpan()).ConvertToUtc(dataTimeZone);
72  }
73  else
74  {
75  try
76  {
77  startTimeUtc = symbol.ID.Date;
78  }
79  catch (InvalidOperationException)
80  {
81  startTimeUtc = Time.BeginningOfTime;
82  }
83  endTimeUtc = DateTime.UtcNow;
84  }
85 
86  // Save the data
87  var writer = new LeanDataWriter(resolution, symbol, Globals.DataFolder, tickType);
88  try
89  {
90  var getParams = new DataDownloaderGetParameters(symbol, resolution, startTimeUtc, endTimeUtc, tickType);
91 
92  var data = _dataDownloader.Get(getParams)
93  .Where(baseData => symbol.SecurityType == SecurityType.Base || baseData.GetType() == dataType)
94  // for canonical symbols, downloader will return data for all of the chain
95  .GroupBy(baseData => baseData.Symbol);
96 
97  foreach (var dataPerSymbol in data)
98  {
99  writer.Write(dataPerSymbol);
100  }
101  }
102  catch (Exception e)
103  {
104  Log.Error(e);
105  }
106  }
107  });
108  }
109 
110  /// <summary>
111  /// Main filter to determine if this file needs to be downloaded
112  /// </summary>
113  /// <param name="filePath">File we are looking at</param>
114  /// <returns>True if should download</returns>
115  protected override bool NeedToDownload(string filePath)
116  {
117  // Ignore null and invalid data requests
118  if (filePath == null
119  || filePath.Contains("fine", StringComparison.InvariantCultureIgnoreCase) && filePath.Contains("fundamental", StringComparison.InvariantCultureIgnoreCase)
120  || filePath.Contains("map_files", StringComparison.InvariantCultureIgnoreCase)
121  || filePath.Contains("factor_files", StringComparison.InvariantCultureIgnoreCase)
122  || filePath.Contains("margins", StringComparison.InvariantCultureIgnoreCase) && filePath.Contains("future", StringComparison.InvariantCultureIgnoreCase))
123  {
124  return false;
125  }
126 
127  // Only download if it doesn't exist or is out of date.
128  // Files are only "out of date" for non date based files (hour, daily, margins, etc.) because this data is stored all in one file
129  return !File.Exists(filePath) || filePath.IsOutOfDate();
130  }
131  }
132 }
