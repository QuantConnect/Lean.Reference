97  /// <summary>
98  /// Applies universe selection the the data feed and algorithm
99  /// </summary>
100  /// <param name="universe">The universe to perform selection on</param>
101  /// <param name="dateTimeUtc">The current date time in utc</param>
102  /// <param name="universeData">The data provided to perform selection with</param>
103  public SecurityChanges ApplyUniverseSelection(Universe universe, DateTime dateTimeUtc, BaseDataCollection universeData)
104  {
105  var algorithmEndDateUtc = _algorithm.EndDate.ConvertToUtc(_algorithm.TimeZone);
106  if (dateTimeUtc > algorithmEndDateUtc)
107  {
108  return SecurityChanges.None;
109  }
110 
111  IEnumerable<Symbol> selectSymbolsResult;
112 
113  // check if this universe must be filtered with fine fundamental data
114  Universe fineFiltered = (universe as FineFundamentalFilteredUniverse)?.FineFundamentalUniverse;
115  fineFiltered ??= (universe as FundamentalFilteredUniverse)?.FundamentalUniverse;
116 
117  if (fineFiltered != null
118  // if the universe has been disposed we don't perform selection. This us handled bellow by 'Universe.PerformSelection'
119  // but in this case we directly call 'SelectSymbols' because we want to perform fine selection even if coarse returns the same
120  // symbols, see 'Universe.PerformSelection', which detects this and returns 'Universe.Unchanged'
121  && !universe.DisposeRequested)
122  {
123  // perform initial filtering and limit the result
124  selectSymbolsResult = universe.SelectSymbols(dateTimeUtc, universeData);
125 
126  if (!ReferenceEquals(selectSymbolsResult, Universe.Unchanged))
127  {
128  // prepare a BaseDataCollection of FineFundamental instances
129  var fineCollection = new BaseDataCollection();
130 
131  // if the input is already fundamental data we just need to filter it and pass it through
132  var hasFundamentalData = universeData.Data.Count > 0 && universeData.Data[0] is Fundamental;
133  if(hasFundamentalData)
134  {
135  // Remove selected symbols that does not have fine fundamental data
136  var anyDoesNotHaveFundamentalData = false;
137 
138  // only pre filter selected symbols if there actually is any fundamental data. This way we can support custom universe filtered by fine fundamental data
139  // which do not use coarse data as underlying, in which case it could happen that we try to load fine fundamental data that is missing, but no problem,
140  // 'FineFundamentalSubscriptionEnumeratorFactory' won't emit it
141  var set = selectSymbolsResult.ToHashSet();
142  fineCollection.Data.AddRange(universeData.Data.OfType<Fundamental>().Where(fundamental => {
143  // we remove to we distict by symbol
144  if (set.Remove(fundamental.Symbol))
145  {
146  if (!fundamental.HasFundamentalData)
147  {
148  anyDoesNotHaveFundamentalData = true;
149  return false;
150  }
151  return true;
152  }
153  return false;
154  }));
155 
156  if (!_anyDoesNotHaveFundamentalDataWarningLogged && anyDoesNotHaveFundamentalData)
157  {
158  _algorithm.Debug("Note: Your coarse selection filter was updated to exclude symbols without fine fundamental data. Make sure your coarse filter excludes symbols where HasFundamental is false.");
159  _anyDoesNotHaveFundamentalDataWarningLogged = true;
160  }
161  }
162  else
163  {
164  // we need to load the fundamental data
165  var currentTime = dateTimeUtc.ConvertFromUtc(TimeZones.NewYork);
166  foreach (var symbol in selectSymbolsResult)
167  {
168  fineCollection.Data.Add(new Fundamental(currentTime, symbol));
169  }
170  }
171 
172  universeData.Data = fineCollection.Data;
173  // perform the fine fundamental universe selection
174  selectSymbolsResult = fineFiltered.PerformSelection(dateTimeUtc, fineCollection);
175  }
176  }
177  else
178  {
179  // perform initial filtering and limit the result
180  selectSymbolsResult = universe.PerformSelection(dateTimeUtc, universeData);
181  }
182 
183  // materialize the enumerable into a set for processing
184  var selections = selectSymbolsResult.ToHashSet();
185 
186  // first check for no pending removals, even if the universe selection
187  // didn't change we might need to remove a security because a position was closed
188  RemoveSecurityFromUniverse(
189  _pendingRemovalsManager.CheckPendingRemovals(selections, universe),
190  dateTimeUtc,
191  algorithmEndDateUtc);
192 
193  // check for no changes second
194  if (ReferenceEquals(selectSymbolsResult, Universe.Unchanged))
195  {
196  return SecurityChanges.None;
197  }
198 
199  // determine which data subscriptions need to be removed from this universe
200  foreach (var member in universe.Securities.Values.OrderBy(member => member.Security.Symbol.SecurityType))
201  {
202  var security = member.Security;
203  // if we've selected this subscription again, keep it
204  if (selections.Contains(security.Symbol)) continue;
205 
206  // don't remove if the universe wants to keep him in
207  if (!universe.CanRemoveMember(dateTimeUtc, security)) continue;
208 
209  if (!member.Security.IsDelisted)
210  {
211  // TODO: here we are not checking if other universes have this security still selected
212  _securityChangesConstructor.Remove(member.Security, member.IsInternal);
213  }
214 
215  RemoveSecurityFromUniverse(_pendingRemovalsManager.TryRemoveMember(security, universe),
216  dateTimeUtc,
217  algorithmEndDateUtc);
218  }
219 
220  Dictionary<Symbol, Security> pendingAdditions;
221  if (!_pendingSecurityAdditions.TryGetValue(dateTimeUtc, out pendingAdditions))
222  {
223  // if the frontier moved forward then we've added these securities to the algorithm
224  _pendingSecurityAdditions.Clear();
225 
226  // keep track of created securities so we don't create the same security twice, leads to bad things :)
227  pendingAdditions = new Dictionary<Symbol, Security>();
228  _pendingSecurityAdditions[dateTimeUtc] = pendingAdditions;
229  }
230 
231  // find new selections and add them to the algorithm
232  foreach (var symbol in selections)
233  {
234  if (universe.Securities.ContainsKey(symbol))
235  {
236  // if its already part of the universe no need to re add it
237  continue;
238  }
239 
240  Security underlying = null;
241  if (symbol.HasUnderlying)
242  {
243  underlying = GetOrCreateSecurity(pendingAdditions, symbol.Underlying, universe.UniverseSettings);
244  }
245  // create the new security, the algorithm thread will add this at the appropriate time
246  var security = GetOrCreateSecurity(pendingAdditions, symbol, universe.UniverseSettings, underlying);
247 
248  var addedSubscription = false;
249  var dataFeedAdded = false;
250  var internalFeed = true;
251  foreach (var request in universe.GetSubscriptionRequests(security, dateTimeUtc, algorithmEndDateUtc,
252  _algorithm.SubscriptionManager.SubscriptionDataConfigService))
253  {
254  if (security.Symbol == request.Configuration.Symbol // Just in case check its the same symbol, else AddData will throw.
255  && !security.Subscriptions.Contains(request.Configuration))
256  {
257  // For now this is required for retro compatibility with usages of security.Subscriptions
258  security.AddData(request.Configuration);
259  }
260 
261  var toRemove = _currencySubscriptionDataConfigManager.GetSubscriptionDataConfigToRemove(request.Configuration.Symbol);
262  if (toRemove != null)
263  {
264  Log.Trace($"UniverseSelection.ApplyUniverseSelection(): Removing internal currency data feed {toRemove}");
265  _dataManager.RemoveSubscription(toRemove);
266  }
267 
268  // 'dataFeedAdded' will help us notify the user for security changes only once per non internal subscription
269  // for example two universes adding the sample configuration, we don't want two notifications
270  dataFeedAdded = _dataManager.AddSubscription(request);
271 
272  // only update our security changes if we actually added data
273  if (!request.IsUniverseSubscription)
274  {
275  addedSubscription = true;
276  // if any config isn't internal then it's not internal
277  internalFeed &= request.Configuration.IsInternalFeed;
278  _internalSubscriptionManager.AddedSubscriptionRequest(request);
279  }
280  }
281 
282  if (addedSubscription)
283  {
284  var addedMember = universe.AddMember(dateTimeUtc, security, internalFeed);
285 
286  if (addedMember && dataFeedAdded)
287  {
288  _securityChangesConstructor.Add(security, internalFeed);
289  }
290  }
291  }
292 
293  var securityChanges = _securityChangesConstructor.Flush();
294 
295  // Add currency data feeds that weren't explicitly added in Initialize
296  if (securityChanges.AddedSecurities.Count > 0)
297  {
298  EnsureCurrencyDataFeeds(securityChanges);
299  }
300 
301  if (securityChanges != SecurityChanges.None && Log.DebuggingEnabled)
302  {
303  // for performance lets not create the message string if debugging is not enabled
304  // this can be executed many times and its in the algorithm thread
305  Log.Debug("UniverseSelection.ApplyUniverseSelection(): " + dateTimeUtc + ": " + securityChanges);
306  }
307 
308  return securityChanges;
309  }

311  /// <summary>
312  /// Will add any pending internal currency subscriptions
313  /// </summary>
314  /// <param name="utcStart">The current date time in utc</param>
315  /// <returns>Will return true if any subscription was added</returns>
316  public bool AddPendingInternalDataFeeds(DateTime utcStart)
317  {
318  var added = false;
319  if (!_initializedSecurityBenchmark)
320  {
321  _initializedSecurityBenchmark = true;
322 
323  var securityBenchmark = _algorithm.Benchmark as SecurityBenchmark;
324  if (securityBenchmark != null)
325  {
326  var resolution = _algorithm.LiveMode ? Resolution.Minute : Resolution.Hour;
327 
328  // Check that the tradebar subscription we are using can support this resolution GH #5893
329  var subscriptionType = _algorithm.SubscriptionManager.SubscriptionDataConfigService.LookupSubscriptionConfigDataTypes(securityBenchmark.Security.Type, resolution, securityBenchmark.Security.Symbol.IsCanonical()).First();
330  var symbol = securityBenchmark.Security.Symbol;
331  var isCustomData = false;
332 
333  // Check if the benchmark security is a custom data in order to make sure we get the correct
334  // type
335  if (symbol.SecurityType == SecurityType.Base)
336  {
337  var symbolDataConfigs = _algorithm.SubscriptionManager.SubscriptionDataConfigService.GetSubscriptionDataConfigs(symbol);
338  if (symbolDataConfigs.Any())
339  {
340  subscriptionType = new Tuple<Type, TickType>(symbolDataConfigs.First().Type, TickType.Trade);
341  isCustomData = true;
342  }
343  }
344 
345  var baseInstance = subscriptionType.Item1.GetBaseDataInstance();
346  baseInstance.Symbol = securityBenchmark.Security.Symbol;
347  var supportedResolutions = baseInstance.SupportedResolutions();
348  if (!supportedResolutions.Contains(resolution))
349  {
350  resolution = supportedResolutions.OrderByDescending(x => x).First();
351  }
352 
353  var subscriptionList = new List<Tuple<Type, TickType>>() {subscriptionType};
354  var dataConfig = _algorithm.SubscriptionManager.SubscriptionDataConfigService.Add(
355  securityBenchmark.Security.Symbol,
356  resolution,
357  isInternalFeed: true,
358  fillForward: false,
359  isCustomData: isCustomData,
360  subscriptionDataTypes: subscriptionList
361  ).First();
362 
363  // we want to start from the previous tradable bar so the benchmark security
364  // never has 0 price
365  var previousTradableBar = Time.GetStartTimeForTradeBars(
366  securityBenchmark.Security.Exchange.Hours,
367  utcStart.ConvertFromUtc(securityBenchmark.Security.Exchange.TimeZone),
368  _algorithm.LiveMode ? Time.OneMinute : Time.OneDay,
369  1,
370  false,
371  dataConfig.DataTimeZone).ConvertToUtc(securityBenchmark.Security.Exchange.TimeZone);
372 
373  if (dataConfig != null)
374  {
375  added |= _dataManager.AddSubscription(new SubscriptionRequest(
376  false,
377  null,
378  securityBenchmark.Security,
379  dataConfig,
380  previousTradableBar,
381  _algorithm.EndDate.ConvertToUtc(_algorithm.TimeZone)));
382 
383  Log.Trace($"UniverseSelection.AddPendingInternalDataFeeds(): Adding internal benchmark data feed {dataConfig}");
384  }
385  }
386  }
387 
388  if (_currencySubscriptionDataConfigManager.UpdatePendingSubscriptionDataConfigs(_algorithm.BrokerageModel))
389  {
390  foreach (var subscriptionDataConfig in _currencySubscriptionDataConfigManager
391  .GetPendingSubscriptionDataConfigs())
392  {
393  var security = _algorithm.Securities[subscriptionDataConfig.Symbol];
394  added |= _dataManager.AddSubscription(new SubscriptionRequest(
395  false,
396  null,
397  security,
398  subscriptionDataConfig,
399  utcStart,
400  _algorithm.EndDate.ConvertToUtc(_algorithm.TimeZone)));
401  }
402  }
403  return added;
404  }

414  public SecurityChanges HandleDelisting(BaseData data, bool isInternalFeed)
415  {
416  if (_algorithm.Securities.TryGetValue(data.Symbol, out var security))
417  {
418  // don't allow users to open a new position once delisted
419  security.IsDelisted = true;
420  security.IsTradable = false;
421 
422  if (_algorithm.Securities.Remove(data.Symbol))
423  {
424  _securityChangesConstructor.Remove(security, isInternalFeed);
425 
426  return _securityChangesConstructor.Flush();
427  }
428  }
429 
430  return SecurityChanges.None;
431  }

433  private void RemoveSecurityFromUniverse(
434  List<PendingRemovalsManager.RemovedMember> removedMembers,
435  DateTime dateTimeUtc,
436  DateTime algorithmEndDateUtc)
437  {
438  if (removedMembers == null)
439  {
440  return;
441  }
442  foreach (var removedMember in removedMembers)
443  {
444  var universe = removedMember.Universe;
445  var member = removedMember.Security;
446 
447  // safe to remove the member from the universe
448  universe.RemoveMember(dateTimeUtc, member);
449 
450  var isActive = _algorithm.UniverseManager.ActiveSecurities.ContainsKey(member.Symbol);
451  foreach (var subscription in universe.GetSubscriptionRequests(member, dateTimeUtc, algorithmEndDateUtc,
452  _algorithm.SubscriptionManager.SubscriptionDataConfigService))
453  {
454  if (_dataManager.RemoveSubscription(subscription.Configuration, universe))
455  {
456  _internalSubscriptionManager.RemovedSubscriptionRequest(subscription);
457 
458  // if not used by any universe
459  if (!isActive)
460  {
461  member.IsTradable = false;
462  // We need to mark this security as untradeable while it has no data subscription
463  // it is expected that this function is called while in sync with the algo thread,
464  // so we can make direct edits to the security here.
465  // We only clear the cache once the subscription is removed from the data stack
466  member.Cache.Reset();
467 
468  _algorithm.Securities.Remove(member.Symbol);
469  }
470  }
471  }
472  }

475  private Security GetOrCreateSecurity(Dictionary<Symbol, Security> pendingAdditions, Symbol symbol, UniverseSettings universeSettings, Security underlying = null)
476  {
477  // create the new security, the algorithm thread will add this at the appropriate time
478  Security security;
479  if (!pendingAdditions.TryGetValue(symbol, out security) && !_algorithm.Securities.TryGetValue(symbol, out security))
480  {
481  security = _securityService.CreateSecurity(symbol, new List<SubscriptionDataConfig>(), universeSettings.Leverage, symbol.ID.SecurityType.IsOption(), underlying);
482 
483  pendingAdditions.Add(symbol, security);
484  }
485 
486  return security;
487  }
